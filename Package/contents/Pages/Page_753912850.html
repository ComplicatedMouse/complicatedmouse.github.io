Example 5: Sync to a Running Application where the Profiler Library is Integrated
# Connect to a running application, allowing for both in-process and out-of-process profiling.
#   1. Connect to the dev kit.
#   2. Set the sync scenario.
#   3. Sync to the dev kit.
# Note: The application will need to have integrated the profiler library.


# 1. Connect to the dev kit.
Profiler-WaitFor CanSync
$connected = Profiler-Connect
if ($false -eq $connected)
{
    Write-Error "Could not connect to a dev kit."
    exit
}

# 2. Set the sync scenario.
# Since we are assuming we will support in-process, use autofill.
$nssFile = [string]::Empty
Profiler-NX-WaitFor InProcessAvailable
Profiler-NX-Scenario AttachToProgramId -NssFile $nssFile -ProgramId "0x0000000000000000" -Autofill

# 3. Sync to the dev kit.
$synced = Profiler-Sync
if ($false -eq $synced)
{
    Write-Error "Could not sync to the dev kit."
    exit
}
Profiler-WaitFor ReadyToProfile
