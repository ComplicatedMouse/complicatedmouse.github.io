/*
* ScalableAOMipmapPass Constructor
* --------------------------------
* Sets up default member data values.
*/
ScalableAOMipmapPass::ScalableAOMipmapPass() : m_ViewSpaceDepthTexHandle(0)
{
m_Name = "ScalableAOMipmapPass";
}
/*
* ScalableAOMipmapPass Destructor
* -------------------------------
* Empty destructor.
*/
ScalableAOMipmapPass::~ScalableAOMipmapPass()
{
}
/*
* ScalableAOMipmapPass::Init
* --------------------------
* Sets the asset data to use for drawing.
*/
void ScalableAOMipmapPass::Init(AssetFileDataHolder* pData)
{
m_pAssetData = pData;
}
/*
* ScalableAOMipmapPass::SetCurrentMipLevel
* ----------------------------------------
* Sets the current mip level that will be used
* to generate the new mip level.
*/
void ScalableAOMipmapPass::SetCurrentMipLevel(int currentMipLevel)
{
m_CurrentMipLevel = currentMipLevel;
}
/*
* ScalableAOMipmapPass::SetupUniforms
* -----------------------------------
* Sets up the uniform buffer.
*/
void ScalableAOMipmapPass::SetupUniforms(UniformBufferManager* pUniformBufferManager)
{
for (int i = 0; i < g_MipLevels; ++i)
{
m_pBlockSamplerFS[i] = pUniformBufferManager->CreateUniformBuffer(sizeof(ScalableAOMipmap::BlockSamplerFSUniformBlockData));
}
}
/*
* ScalableAOMipmapPass::SetTextureHandle
* --------------------------------------
* Sets up the texture handle for the texture needed to generate SAO texture.
*/
void ScalableAOMipmapPass::SetTextureHandle(const NVNtextureHandle& viewSpaceDepthHandle)
{
m_ViewSpaceDepthTexHandle = viewSpaceDepthHandle;
}
/*
* ScalableAOMipmapPass::UpdateUniforms
* ------------------------------------
* Updates the uniform buffer.
*/
void ScalableAOMipmapPass::UpdateUniforms()
{
for (int i = 0; i < g_MipLevels; ++i)
{
ScalableAOMipmap::BlockSamplerFSUniformBlockData* blockSamplerFS = reinterpret_cast<ScalableAOMipmap::BlockSamplerFSUniformBlockData*>(m_pBlockSamplerFS[i]->GetMappedPointer());
blockSamplerFS->SetUniform_u_ViewSpaceDepthTex(m_ViewSpaceDepthTexHandle);
blockSamplerFS->SetUniform_u_PreviousMipLevel(i);
}
}
/*
* ScalableAOMipmapPass::Draw
* --------------------------
* Draws a screen space quad that converts dapth values into a view space z buffer
*/
void ScalableAOMipmapPass::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
ScalableAOMipmap::BlockSamplerFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockSamplerFS[m_CurrentMipLevel - 1]->GetCurrentBufferAddress(),
sizeof(ScalableAOMipmap::BlockSamplerFSUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
