#include <cstdio>
#include <nn/settings/settings_Language.h>
namespace {
struct UserInfo
{
nn::settings::LanguageCode languageCode;
};
} // Namespace.
extern "C" void nnMain()
{
// You can create language codes using the member functions of the LanguageCode structure.
nn::settings::LanguageCode needle = nn::settings::LanguageCode::EnglishUk();
// The internal expression of the language code can be used as an IETF language tag.
std::printf("IETF Language Tag: %s\n", needle.string);
const int count = nn::settings::GetAvailableLanguageCodeCount();
nn::settings::LanguageCode* const
haystack = new nn::settings::LanguageCode[count];
// Enumerates the language code values that can be obtained.
nn::settings::GetAvailableLanguageCodes(haystack, count);
if (0 < count)
{
// Language codes can be directly compared by the operator.
std::printf("%s %s %s\n",
haystack[0].string,
haystack[0] == needle ? "==" : "!=",
needle.string);
}
delete[] haystack;
// Use the language code when handling language settings.
UserInfo userInfo = {};
userInfo.languageCode = needle;
std::printf("User Language: %s\n", userInfo.languageCode.string);
}
