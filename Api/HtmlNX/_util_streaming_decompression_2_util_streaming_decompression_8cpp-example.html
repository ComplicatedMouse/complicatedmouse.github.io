//
// ZLIB streaming decompression occurs with smaller input and output buffers, and
// the decompression results are written out to a log as a string in this sample function.
//
void DisplayWithStreamingDecompressionZlib(const uint8_t* pComressedData,
size_t compressedDataSize)
{
// Pointer to the original compressed data.
const uint8_t* pCompressedDataPtr = pComressedData;
// Remaining size of the compressed data.
size_t compressedDataRemainSize = compressedDataSize;
const size_t SrcBufSize = 32; // Size of the input buffer.
char srcBuf[SrcBufSize]; // The input buffer that temporarily stores compressed data.
char* pSrcBufCurrent = nullptr; // Pointer indicating the position within the input buffer.
size_t remainSrcBufSize = 0; // Remaining size in the input buffer.
const size_t DstBufSize = 64; // Size of the output buffer.
char dstBuf[DstBufSize]; // The output buffer that temporarily stores decompressed data.
size_t dstConsumedSize; // The amount of output data.
size_t srcConsumedSize; // The amount of data used for decompression.
// Get the context for streaming decompression from the heap.
std::unique_ptr<nn::util::StreamingDecompressZlibContext>
pContext(new nn::util::StreamingDecompressZlibContext);
NN_ABORT_UNLESS_NOT_NULL(pContext);
// Initialize the context for streaming decompression.
nn::util::InitializeStreamingDecompressZlibContext(pContext.get());
// Streaming decompression loop.
for(;;)
{
// After the input buffer is empty, continue loading the compressed data.
if(remainSrcBufSize == 0)
{
// Load either just the size of the input buffer or the size of the remaining compressed data.
size_t inputSize = std::min(compressedDataRemainSize, SrcBufSize);
// Because this sample is simple, load the compressed data placed in memory from the beginning.
std::memcpy(srcBuf, pCompressedDataPtr, inputSize);
pCompressedDataPtr += inputSize;
compressedDataRemainSize -= inputSize;
pSrcBufCurrent = srcBuf;
remainSrcBufSize = inputSize;
}
// After either the input buffer is empty or the output buffer is full, pass the content of the output buffer
// minus the terminating character as a streaming decompression string to be written to the log.
bool isSuccess = nn::util::StreamingDecompressZlib(&dstConsumedSize, &srcConsumedSize,
dstBuf, (DstBufSize - sizeof(dstBuf[0])),
pSrcBufCurrent, remainSrcBufSize, pContext.get());
// Check the return value from the library.
if(!isSuccess)
{
// When the library returns false, either the input data is corrupted or the compression format is not GZIP.
NN_LOG("Streaming Decompression (ZLIB): NG.\n");
NN_LOG(" (The input data is corrupted or in an unsupported format.)\n");
return ;
}
// Processing ends if neither input nor output is being used for decompression.
if(dstConsumedSize == 0 && srcConsumedSize == 0)
{
return ;
}
// If there is decompression data in the output buffer, write it to the log.
if(dstConsumedSize > 0)
{
// Insert a terminating character to the end of the derived decompression results because the string will be written to the log.
NN_ABORT_UNLESS(dstConsumedSize < DstBufSize);
dstBuf[dstConsumedSize] = '\0';
// Write the string derived from this episode of streaming decompression to the log.
NN_LOG("< Streaming decompressed data >: %s\n", dstBuf);
}
// Update the pointer to the input buffer and the size, because there will continue to be input.
pSrcBufCurrent += srcConsumedSize;
remainSrcBufSize -= srcConsumedSize;
}
}
} // Namespace.
//-----------------------------------------------------------------------------
//
// Main function.
//
extern "C" void nnMain()
{
// Write the size of the compressed data to the log.
NN_LOG("\nCompressed data size ( %d Bytes )\n\n", SampleDataSize);
// Log output of the results of the ZLIB streaming decompression.
DisplayWithStreamingDecompressionZlib(SampleData, SampleDataSize);
}
