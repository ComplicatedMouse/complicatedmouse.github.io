#include <nn/time.h>
#include <nn/nn_Abort.h>
#include <nn/nn_Assert.h>
#include <nn/nn_Log.h>
#include <nn/os/os_ThreadApi.h>
extern "C" void nnMain()
{
// Initialize the time library.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::time::Initialize() );
NN_LOG("nn::time::Initialize\n");
NN_LOG("---- Sample of calculating nn::time::StandardUserSystemClock difference by user.\n");
// Create the original context for the comparison.
// Use ClockSnapshot::CreateWithAdjustableSystemClock to initialize the object if using AdjustableUserSystemClock and AdjustableNetworkSystemClock.
nn::time::ClockSnapshot oldContext;
nn::time::ClockSnapshot::CreateWithStandardSystemClock(&oldContext);
// Saving the initial ClockSnapshot object that you created in a location such as save data makes it possible to reload that object later to calculate
// the amount of time by which the clock has been changed even spanning across events such as applications being closed or the system being shut down.
// ClockSnapshot objects satisfy the standard C++ requirements for being trivially copyable and can be copied with std::memcpy().
NN_LOG("Waiting three seconds.\n");
nn::os::SleepThread(nn::TimeSpan::FromSeconds(3));
// During this sleep period, press the HOME Button to display DevMenu or the HOME Menu, and then change the time or the time zone setting.
// This step triggers the clock operation detection process and satisfies the conditions for the branch for detecting changes to the time zone.
// Create the current context.
nn::time::ClockSnapshot latestContext;
nn::time::ClockSnapshot::CreateWithStandardSystemClock(&latestContext);
// CalculateStandardUserSystemClockDifferenceByUser() calculates the amount by which
// the user intentionally changed the user clock (nn::time::StandardUserSystemClock)
// between oldContext and latestContext.
nn::TimeSpan differenceByUser = nn::time::CalculateStandardUserSystemClockDifferenceByUser(oldContext, latestContext);
if(differenceByUser == nn::TimeSpan(0))
{
// If the user has not intentionally changed the clock during this period, the function returns nn::TimeSpan(0).
NN_LOG("StandardUserSystemClock was not operated by the user.\n");
// Even if CalculateStandardUserSystemClockDifferenceByUser() returns nn::TimeSpan(0), it only indicates the user has not intentionally changed the clock.
// It does not guarantee that the time value that can be obtained with
// StandardUserSystemClock::GetCurrentTime() has not changed by a large amount.
}
else
{
// The user has intentionally changed the clock.
NN_LOG("StandardUserSystemClock was operated by the user: %lld [sec]\n", differenceByUser.GetSeconds());
}
// CalculateStandardUserSystemClockDifferenceByUser() does not check whether the time zone has been changed.
// Use ClockSnapshot::GetLocationName() to make this comparison if necessary.
if(oldContext.GetLocationName() == latestContext.GetLocationName())
{
// The time zone has not been changed in system settings.
NN_LOG("LocationName was not changed.\n");
}
else
{
// The time zone has been changed in system settings.
NN_LOG("LocationName was changed.\n");
// If you need to calculate the magnitude of the time difference to apply to the local time, do the following.
// Note, however, that the magnitude of this time difference may change even if the time zone has not been changed in system settings, such as when daylight saving time begins or ends.
auto oldUtcOffsetSeconds = oldContext.GetStandardUserSystemClockCalendarAdditionalInfo().timeZone.utcOffsetSeconds;
auto latestUtcOffsetSeconds = latestContext.GetStandardUserSystemClockCalendarAdditionalInfo().timeZone.utcOffsetSeconds;
auto diffOfUtcOffsetSeconds = latestUtcOffsetSeconds - oldUtcOffsetSeconds;
NN_LOG("UTC offset seconds at local time: %d seconds -> %d seconds; difference: %d seconds\n",
oldUtcOffsetSeconds, latestUtcOffsetSeconds, diffOfUtcOffsetSeconds);
}
// Finalize the time library.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::time::Finalize() );
NN_LOG("nn::time::Finalize\n");
}
