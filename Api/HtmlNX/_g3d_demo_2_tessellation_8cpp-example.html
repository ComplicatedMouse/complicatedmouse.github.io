// Initialize the rasterizer state.
void InitializeRasterizerState(nn::gfx::Device* pDevice, nn::gfx::RasterizerState& rasterizerState, bool isWireframeEnabled) NN_NOEXCEPT
{
nn::gfx::RasterizerStateInfo rasterizerStateInfo;
rasterizerStateInfo.SetDefault();
if (isWireframeEnabled)
{
rasterizerStateInfo.SetFillMode(nn::gfx::FillMode_Wireframe);
rasterizerStateInfo.SetDepthBias(-1);
rasterizerStateInfo.SetSlopeScaledDepthBias(-.5f);
}
// Make the topology type PrimitiveTopologyType_Patch.
rasterizerStateInfo.SetPrimitiveTopologyType(nn::gfx::PrimitiveTopologyType_Patch);
rasterizerStateInfo.SetScissorEnabled(true);
rasterizerState.Initialize(pDevice, rasterizerStateInfo);
}
// Destroy the rasterizer state.
void FinalizeRasterizerState(nn::gfx::Device* pDevice, nn::gfx::RasterizerState& rasterizerState) NN_NOEXCEPT
{
rasterizerState.Finalize(pDevice);
}
// Initialize the blend state for the wireframe.
void InitializeBlendState(nn::gfx::Device* pDevice, nn::gfx::BlendState& blendState) NN_NOEXCEPT
{
nn::gfx::BlendState::InfoType info;
info.SetDefault();
nn::gfx::BlendTargetStateInfo targetStateInfo;
{
targetStateInfo.SetChannelMask(nn::gfx::ChannelMask_All);
targetStateInfo.SetBlendEnabled(true);
targetStateInfo.SetColorBlendFunction(nn::gfx::BlendFunction_Add);
targetStateInfo.SetSourceColorBlendFactor(nn::gfx::BlendFactor_Zero);
targetStateInfo.SetDestinationColorBlendFactor(nn::gfx::BlendFactor_Zero);
targetStateInfo.SetAlphaBlendFunction(nn::gfx::BlendFunction_Add);
targetStateInfo.SetSourceAlphaBlendFactor(nn::gfx::BlendFactor_Zero);
targetStateInfo.SetDestinationAlphaBlendFactor(nn::gfx::BlendFactor_One);
}
info.SetBlendTargetStateInfoArray(&targetStateInfo, 1);
size_t size = nn::gfx::BlendState::GetRequiredMemorySize(info);
if (size > 0)
{
void* pBuffer = g3ddemo::AllocateMemory(size, nn::gfx::BlendState::RequiredMemoryInfo_Alignment);
blendState.SetMemory(pBuffer, size);
}
blendState.Initialize(pDevice, info);
}
// Destroy the blend state for the wireframe.
void FinalizeBlendState(nn::gfx::Device* pDevice, nn::gfx::BlendState& blendState) NN_NOEXCEPT
{
void* pBuffer = g_WireframeBlendState.GetMemory();
if (pBuffer)
{
g3ddemo::FreeMemory(pBuffer);
}
blendState.Finalize(pDevice);
}
// Initialize the tessellation state.
static void InitializeTessellationState(nn::gfx::Device* pDevice, nn::gfx::TessellationState& tessellationState) NN_NOEXCEPT
{
nn::gfx::TessellationState::InfoType info;
info.SetDefault();
info.SetPatchControlPointCount(3);
tessellationState.Initialize(pDevice, info);
}
// Destroy the tessellation state.
void FinalizeTessellationState(nn::gfx::Device* pDevice, nn::gfx::TessellationState& rasterizerState) NN_NOEXCEPT
{
rasterizerState.Finalize(pDevice);
}
// Update the camera.
void UpdateCamera() NN_NOEXCEPT
{
nn::util::Vector3fType& pos = g_CameraPosition;
nn::util::Vector3fType& target = g_CameraTarget;
nn::util::Vector3fType& up = g_CameraUp;
bool isUpdated = !(g3ddemo::IsMouseUsed() && ImGui::IsWindowFocused(ImGuiFocusedFlags_AnyWindow));
if (isUpdated)
{
nn::util::Vector3fType targetDir;
float length;
{
// Calculate the view vector.
VectorSubtract(&targetDir, target, pos);
float lengthSq = VectorLengthSquared(targetDir);
length = std::sqrt(lengthSq);
nn::util::VectorDivide(&targetDir, targetDir, length);
}
// Get the camera attitude in world space.
nn::util::Vector3fType viewRight;
nn::util::Vector3fType viewUp;
nn::util::Vector3fType viewForward;
{
const nn::util::Matrix4x3fType& viewMtx = g_RenderView.GetViewMtx(0);
g3ddemo::MatrixGetColumn0(&viewRight, viewMtx);
g3ddemo::MatrixGetColumn1(&viewUp, viewMtx);
nn::util::VectorSubtract(&viewForward, nn::util::Vector3f::Zero(), targetDir);
}
// Update the position using the current camera attitude and the stick input value.
{
const float AtMoveSpeed = 0.8f;
nn::util::Vector3fType dirRight;
nn::util::Vector3fType dirUp;
nn::util::Vector3fType dirForward;
VectorCross(&dirForward, viewRight, viewUp);
VectorNormalize(&dirForward, dirForward);
VectorCross(&dirUp, dirForward, viewRight);
VectorNormalize(&dirUp, dirUp);
const nn::util::Float2& leftStick = g3ddemo::GetLeftStick();
VectorMultiply(&dirRight, viewRight, AtMoveSpeed * (leftStick.x));
VectorAdd(&pos, dirRight, pos);
VectorMultiply(&dirForward, dirForward, AtMoveSpeed * (-leftStick.y));
VectorAdd(&pos, dirForward, pos);
}
// Set the camera rotation amount from the stick input value.
nn::util::Vector4fType quatanion;
{
const float RotSpeed = 0.015f;
const nn::util::Float2& rightStick = g3ddemo::GetRightStick();
nn::util::Vector4fType quatanionX;
{
float angle = rightStick.y * RotSpeed * 0.5f;
float cos = std::cos(angle);
float sin = std::sin(angle);
VectorSet(&quatanionX, sin * VectorGetX(viewRight), sin * VectorGetY(viewRight), sin * VectorGetZ(viewRight), cos);
}
nn::util::Vector4fType quatanionY;
{
float angle = rightStick.x * -RotSpeed * 0.5f;
VectorSet(&quatanionY, 0.0f, std::sin(angle), 0.0f, std::cos(angle));
}
QuaternionMultiply(&quatanion, quatanionY, quatanionX);
}
{
// Apply the rotation amount calculated from the view vector.
{
nn::util::Matrix4x3fType matrix;
MatrixSetRotate(&matrix, quatanion);
VectorTransformNormal(&targetDir, targetDir, matrix);
VectorNormalize(&targetDir, targetDir);
}
// Update the target position.
VectorMultiply(&target, targetDir, length);
VectorAdd(&target, target, pos);
// Create the upward direction from the position and target.
VectorCross(&up, viewRight, targetDir);
VectorSet(&up, 0.0f, (VectorGetY(up) >= 0) ? 1.0f : -1.0f, 0.0f);
}
}
// Apply to the constant buffer.
{
nn::util::Matrix4x3fType matrix;
MatrixLookAtRightHanded(&matrix, pos, target, up);
g_RenderView.SetViewMtx(0, matrix);
}
}
void Calculate(nns::gfx::GraphicsFramework* pGfxFramework, void* pUserData) NN_NOEXCEPT
{
NN_UNUSED(pGfxFramework);
int uniformBufferIndex = *reinterpret_cast<int*>(pUserData);
g3ddemo::DbGuiNewFrame();
ImGui::SetNextWindowPos(ImVec2(20.0f, 20.0f));
if (ImGui::Begin("[Tessellation]", nullptr, ImGuiWindowFlags_AlwaysAutoResize))
{
ImGui::Checkbox("Wireframe", &g_WireframeEnabled);
if (ImGui::Checkbox("MeshLod", &g_MeshLodEnabled))
{
nn::g3d::ModelObj* pModelObj = g_pSquareRenderModelObj->GetModelObj();
int shapeCount = pModelObj->GetShapeCount();
for (int shapeIndex = 0; shapeIndex < shapeCount; ++shapeIndex)
{
nns::g3d::RenderUnitObj* pRenderUnitObj = g_pSquareRenderModelObj->GetRenderUnitObj(shapeIndex);
nn::g3d::ShaderSelector* pShaderSelector = pRenderUnitObj->GetShaderSelector();
int optionIndex = pShaderSelector->FindDynamicOptionIndex("enable_mesh_lod");
if (optionIndex != nn::util::ResDic::Npos)
{
const nn::g3d::ResShaderOption* pResShaderOption = pShaderSelector->GetDynamicOption(optionIndex);
int choiceIndex = pResShaderOption->FindChoiceIndex(g_MeshLodEnabled ? "1" : "0");
pShaderSelector->WriteDynamicKey(optionIndex, choiceIndex);
}
}
g_pSquareRenderModelObj->UpdateShader(pGfxFramework->GetDevice());
}
ImGui::SliderFloat("TessLevelMax", &g_TessLevelMax, 1.f, 25.f, "%.2f");
}
ImGui::End();
// Camera position calculation.
UpdateCamera();
// Projection matrix.
{
nn::util::Matrix4x4fType matrix;
MatrixPerspectiveFieldOfViewRightHanded(&matrix, nn::util::DegreeToRadian(45.0f), 16.0f / 9.0f, 1.0f, 4000.0f);
g_RenderView.SetProjectionMtx(0, matrix);
}
// Update the tessellation parameter.
TessellationParam* pMapped = g_TessellationParamBuffer[uniformBufferIndex].Map<TessellationParam>();
pMapped->tessLevelMax = g_TessLevelMax;
g_TessellationParamBuffer[uniformBufferIndex].Unmap();
// View update processing.
g_RenderView.Calculate(uniformBufferIndex);
// Update the models.
g_pSquareModelAnimObj->CalculateView(0, uniformBufferIndex, g_RenderView.GetViewMtx(0));
g_pSquareModelAnimObj->Calculate();
g_pSquareModelAnimObj->CalculateUniformBlock(uniformBufferIndex);
}
void MakeCommand(nns::gfx::GraphicsFramework* pGfxFramework, int bufferIndex, void* pUserData) NN_NOEXCEPT
{
int uniformBufferIndex = *reinterpret_cast<int*>(pUserData);
// Generate the commands.
pGfxFramework->BeginFrame(bufferIndex);
{
nn::gfx::CommandBuffer* pCommandBuffer = pGfxFramework->GetRootCommandBuffer(bufferIndex);
// Clear the color target.
nn::gfx::ColorTargetView* pColor = pGfxFramework->GetColorTargetView();
pCommandBuffer->ClearColor(pColor, 0.3f, 0.3f, 0.3f, 1.0f, nullptr);
// Clear the depth stencil.
nn::gfx::DepthStencilView* pDepth = pGfxFramework->GetDepthStencilView();
pCommandBuffer->ClearDepthStencil(pDepth, 1.0f, 0, nn::gfx::DepthStencilClearMode_DepthStencil, nullptr);
// Set render targets.
pCommandBuffer->SetRenderTargets(1, &pColor, pDepth);
// Set a viewport scissor state.
pCommandBuffer->SetViewportScissorState(pGfxFramework->GetViewportScissorState());
// Set the tessellation state.
pCommandBuffer->SetTessellationState(&g_TessellationState);
g_pSquareRenderModelObj->GetRenderModel()->GetRenderMaterial(0)->SetRasterizerState(&g_RasterizerState);
g_pSquareRenderModelObj->GetRenderModel()->GetRenderMaterial(0)->SetBlendState(nns::g3d::RenderMaterial::GetDefaultBlendState());
g_pSquareRenderModelObj->Draw(pCommandBuffer, 0, uniformBufferIndex);
// Wireframe drawing of the terrain model.
if (g_WireframeEnabled)
{
g_pSquareRenderModelObj->GetRenderModel()->GetRenderMaterial(0)->SetRasterizerState(&g_WireframeRasterizerState);
g_pSquareRenderModelObj->GetRenderModel()->GetRenderMaterial(0)->SetBlendState(&g_WireframeBlendState);
pCommandBuffer->SetLineWidth(0.8f);
g_pSquareRenderModelObj->Draw(pCommandBuffer, 0, uniformBufferIndex);
}
// Generate text display commands.
nns::dbgui::Render(pCommandBuffer);
}
pGfxFramework->EndFrame(bufferIndex);
}

int TessellationMain() NN_NOEXCEPT
{
nns::gfx::DebugGraphicsFramework* pGfxFramework = g3ddemo::GetGfxFramework();
pGfxFramework->SetFrameworkMode(nns::gfx::GraphicsFramework::FrameworkMode_DeferredSubmission);
nn::gfx::Device* pDevice = pGfxFramework->GetDevice();
// Initialize the load measurement feature.
pGfxFramework->InitializePerf();
#if defined( NN_BUILD_CONFIG_OS_WIN )
NN_PERF_SET_GET_CORE_NUMBER_FUNCTION(g3ddemo::GetFixedCoreNumber);
#endif
// Assign the current main thread to core 0.
nn::os::SetThreadCoreMask(nn::os::GetCurrentThread(), 0, 1);
g_Holder.Initialize();
// Initialize the tessellation state.
InitializeRasterizerState(pDevice, g_RasterizerState, false);
InitializeTessellationState(pDevice, g_TessellationState);
// Initialize the wireframe state.
InitializeRasterizerState(pDevice, g_WireframeRasterizerState, true);
InitializeBlendState(pDevice, g_WireframeBlendState);
// Initialize the shader archive.
nn::g3d::ResShaderArchive* pDemoShaderArchive;
{
nn::g3d::ResShaderFile* pFile = g3ddemo::LoadResource<nn::g3d::ResShaderFile>(g_EnvBfshaPath);
pFile->Setup(pDevice);
g_Holder.shaderFiles.push_back(pFile);
pDemoShaderArchive = pFile->GetResShaderArchive();
}
nn::g3d::ResShaderArchive* pTessellationShaderArchive;
{
nn::g3d::ResShaderFile* pFile = g3ddemo::LoadResource<nn::g3d::ResShaderFile>(g_TessellationBfshaPath);
pFile->Setup(pDevice);
g_Holder.shaderFiles.push_back(pFile);
pTessellationShaderArchive = pFile->GetResShaderArchive();
}
// Initialize model resources.
nn::g3d::ResFile* pEnvModelFile;
{
pEnvModelFile = g3ddemo::LoadResource<nn::g3d::ResFile>(g_EnvBfresPath);
pEnvModelFile->Setup(pDevice);
g_Holder.files.push_back(pEnvModelFile);
}
nn::g3d::ResFile* pSquareModelFile;
{
pSquareModelFile = g3ddemo::LoadResource<nn::g3d::ResFile>(g_SquareBfresPath);
pSquareModelFile->Setup(pDevice);
g3ddemo::SetupTexture(pDevice, pSquareModelFile, g3ddemo::TextureBindCallback);
g3ddemo::RegisterSamplerToDescriptorPool(pSquareModelFile);
g_Holder.files.push_back(pSquareModelFile);
}
// Create the RenderModel that manages the vertex state, depth stencil state, blend state, rasterizer state, and shading model of the model.
nns::g3d::RenderModel* pSquareRenderModel;
{
nn::g3d::ResModel* pResModel = pSquareModelFile->FindModel("Square");
pSquareRenderModel = nns::g3d::CreateRenderModel(pDevice, pResModel);
pSquareRenderModel->CreateDrawPass(pDevice, pTessellationShaderArchive);
g_Holder.renderModels.push_back(pSquareRenderModel);
}
// Create a ModelAnimObj for simplifying model and animation calculation processing.
{
nn::g3d::ModelObj::Builder builder(pSquareModelFile->FindModel("Square"));
builder.ShapeBufferingCount(g_BufferingCount);
builder.SkeletonBufferingCount(g_BufferingCount);
builder.MaterialBufferingCount(g_BufferingCount);
g_pSquareModelAnimObj = nns::g3d::CreateModelAnimObj(pDevice, builder);
g_Holder.modelAnimObjs.push_back(g_pSquareModelAnimObj);
}
// Combine RenderModel and ModelObj to create the RenderModelObj that performs rendering.
{
g_pSquareRenderModelObj = nns::g3d::CreateRenderModelObj(g_pSquareModelAnimObj->GetModelObj(), pSquareRenderModel);
g_Holder.renderModelObjs.push_back(g_pSquareRenderModelObj);
g_pSquareRenderModelObj->UpdateShader(pDevice);
}
// Create the uniform block outside of g3d control that is required when rendering, and set it to the RenderModelObj.
nn::g3d::ModelObj* pEnvModelObj;
{
nn::g3d::ResModel* pResModel = pEnvModelFile->GetModel(0);
nn::g3d::ShaderUtility::BindShaderParam(pResModel->GetMaterial(0), pDemoShaderArchive->FindShadingModel("env"));
nn::g3d::ModelObj::Builder builder(pResModel);
builder.MaterialBufferingCount(g_BufferingCount);
pEnvModelObj = nns::g3d::CreateModelObj(pDevice, builder);
// This demo does not need to be updated every frame, so calculation is performed only once here.
pEnvModelObj->CalculateMaterial(0);
pEnvModelObj->CalculateMaterial(1);
g_Holder.modelObjs.push_back(pEnvModelObj);
const nn::gfx::Buffer* buffers[g_BufferingCount];
buffers[0] = pEnvModelObj->GetMaterial(0)->GetMaterialBlock(0);
buffers[1] = pEnvModelObj->GetMaterial(0)->GetMaterialBlock(1);
g_pSquareRenderModelObj->BindUniformBlock("env", buffers, g_BufferingCount);
}
// Set the uniform block for the view.
g_RenderView.Initialize(pDevice, 1, g_BufferingCount);
{
const nn::gfx::Buffer* buffers[g_BufferingCount];
buffers[0] = g_RenderView.GetViewUniformBlock(0, 0);
buffers[1] = g_RenderView.GetViewUniformBlock(0, 1);
g_pSquareRenderModelObj->BindUniformBlock("view", buffers, g_BufferingCount);
}
// Set the tessellation uniform block.
{
nn::gfx::Buffer::InfoType info;
info.SetDefault();
info.SetGpuAccessFlags(nn::gfx::GpuAccess_ConstantBuffer);
info.SetSize(g_TessellationParamBufferSize);
nn::gfx::MemoryPool* pMemoryPool = pGfxFramework->GetMemoryPool(nns::gfx::GraphicsFramework::MemoryPoolType_ConstantBuffer);
for (int bufferIndex = 0; bufferIndex < g_BufferingCount; ++bufferIndex)
{
g_TessellationParamBufferOffset[bufferIndex] = g3ddemo::GetGfxFramework()->AllocatePoolMemory(nns::gfx::GraphicsFramework::MemoryPoolType_ConstantBuffer,
info.GetSize(), nn::gfx::Buffer::GetBufferAlignment(pDevice, info));
g_TessellationParamBuffer[bufferIndex].Initialize(pDevice, info, pMemoryPool, g_TessellationParamBufferOffset[bufferIndex], info.GetSize());
TessellationParam* pMapped = g_TessellationParamBuffer[bufferIndex].Map<TessellationParam>();
pMapped->tessLevelMax = g_TessLevelMax;
g_TessellationParamBuffer[bufferIndex].Unmap();
}
const nn::gfx::Buffer* buffers[g_BufferingCount];
buffers[0] = &g_TessellationParamBuffer[0];
buffers[1] = &g_TessellationParamBuffer[1];
g_pSquareRenderModelObj->BindUniformBlock("TessellationParam", buffers, g_BufferingCount);
}
// Initialize the camera.
{
VectorSet(&g_CameraPosition, 0.0f, 25.f, 55.f);
VectorSet(&g_CameraUp, 0.0f, 1.0f, 0.0f);
VectorSet(&g_CameraTarget, 0.0f, 0.f, 0.0f);
nn::util::Matrix4x3fType matrix;
MatrixLookAtRightHanded(&matrix,
g_CameraPosition,
g_CameraTarget,
g_CameraUp);
g_RenderView.SetViewMtx(0, matrix);
}
// Set the scratch memory needed for the shader.
g3ddemo::SetShaderScratchMemory(pGfxFramework, g_Holder.shaderFiles, g_Holder.files);
int uniformBufferIndex = 0;
pGfxFramework->SetCalculateCallback(Calculate, &uniformBufferIndex);
pGfxFramework->SetMakeCommandCallback(MakeCommand, &uniformBufferIndex);
pGfxFramework->ResetFrame();
// Main loop.
while (NN_STATIC_CONDITION(1))
{
g3ddemo::UpdateController();
if (g3ddemo::IsButtonTriggered<nns::hid::Button::Start>())
{
break;
}
// Execute frame processing.
pGfxFramework->ProcessFrame();
uniformBufferIndex ^= 1;
}
pGfxFramework->QueueFinish();
// Finalize.
g3ddemo::DestroyAll(pDevice, &g_Holder);
// Free the tessellation uniform block buffer.
for (int bufferIndex = 0; bufferIndex < g_BufferingCount; ++bufferIndex)
{
g_TessellationParamBuffer[bufferIndex].Finalize(pDevice);
pGfxFramework->FreePoolMemory(nns::gfx::GraphicsFramework::MemoryPoolType_ConstantBuffer, g_TessellationParamBufferOffset[bufferIndex]);
}
// Free the uniform block for the view.
{
g_RenderView.Finalize(pDevice);
}
// Free the load measurement meter.
pGfxFramework->FinalizePerf();
pGfxFramework->SetCalculateCallback(nullptr, nullptr);
pGfxFramework->SetMakeCommandCallback(nullptr, nullptr);
FinalizeRasterizerState(pDevice, g_RasterizerState);
FinalizeRasterizerState(pDevice, g_WireframeRasterizerState);
FinalizeTessellationState(pDevice, g_TessellationState);
FinalizeBlendState(pDevice, g_WireframeBlendState);
return EXIT_SUCCESS;
} // NOLINT
