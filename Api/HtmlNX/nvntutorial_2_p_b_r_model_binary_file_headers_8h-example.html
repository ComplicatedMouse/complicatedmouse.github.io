#pragma once
#include <cstdint>
namespace PBRModelBinary
{
struct FileHeader
{
uint64_t m_MeshBlockOffset;
uint64_t m_MaterialBlockOffset;
};
struct MeshBlockHeader
{
uint64_t m_NumMeshes;
uint64_t* m_pMeshOffsets;
};
struct MeshHeader
{
uint32_t m_NumIndices;
uint32_t m_NumVertices;
uint32_t m_MaterialIndex;
uint32_t m_MagicPadding;
uint64_t m_PositionOffset;
uint64_t m_NormalOffset;
uint64_t m_TexCoordOffset;
uint64_t m_TangentOffset;
uint64_t m_BinormalOffset;
uint64_t m_IndexOffset;
};
struct MaterialBlockHeader
{
uint64_t m_NumMaterials;
uint64_t* m_pMaterialOffsets;
};
struct MaterialHeader
{
float m_Diffuse[3];
float m_Emissive[3];
float m_Specular;
float m_Roughness;
float m_Metallic;
float m_DiffuseMapMod[2];
float m_SpecularMapMod[2];
float m_EmissiveMapMod[2];
float m_RoughnessMapMod[2];
float m_MetallicMapMod[2];
float m_NormalMapMod[2];
float m_AlphaMapMod[2];
float m_DiffuseValMod[3];
float m_SpecularValMod;
float m_EmissiveValMod[3];
float m_RoughnessValMod;
float m_MetallicValMod;
float m_NormalValMod[3];
float m_AlphaValMod;
uint64_t m_DiffuseTexNameOffset;
uint64_t m_SpecularTexNameOffset;
uint64_t m_EmissiveTexNameOffset;
uint64_t m_RoughnessTexNameOffset;
uint64_t m_MetallicTexNameOffset;
uint64_t m_NormalTexNameOffset;
uint64_t m_AlphaTexNameOffset;
};
struct DataHeader
{
uint64_t m_DataSize;
const void* m_pData;
uint32_t m_PaddingBytes;
};
}
