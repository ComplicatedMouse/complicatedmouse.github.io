/*
* SkyBox Constructor
* ------------------
* Sets up default member data values.
*/
SkyBox::SkyBox() : m_Scale(1.0f),
m_Geometry(nullptr),
m_CameraXRotation(0.0f),
m_CameraYRotation(0.0f),
m_pBlockVS(nullptr),
m_pBlockFS(nullptr)
{
m_Name = "SkyBox";
nn::util::MatrixIdentity(&m_ProjMatrix);
/* Enable depth testing but disable writing */
nvnDepthStencilStateSetDepthTestEnable(&m_DepthStencilState, NVN_TRUE);
nvnDepthStencilStateSetDepthWriteEnable(&m_DepthStencilState, NVN_FALSE);
}
/*
* SkyBox Destructor
* -----------------
* Empty destructor.
*/
SkyBox::~SkyBox()
{
}
/*
* SkyBox::UpdateUniforms
* ----------------------
* Updates the sky box uniform buffers.
*/
void SkyBox::UpdateUniforms()
{
SkyBoxShader::BlockFSUniformBlockData* uniformMapBlockFS = reinterpret_cast<SkyBoxShader::BlockFSUniformBlockData*>(m_pBlockFS->GetMappedPointer());
uniformMapBlockFS->SetUniform_u_bindlessTex(m_SkyTexture);
SkyBoxShader::BlockVSUniformBlockData* uniformMapBlockVS = reinterpret_cast<SkyBoxShader::BlockVSUniformBlockData*>(m_pBlockVS->GetMappedPointer());
uniformMapBlockVS->SetUniform_u_scale(m_Scale);
nn::util::Matrix4x3f temp4x3;
nn::util::Matrix4x4f temp4x4;
nn::util::MatrixCameraRotateZxy(&temp4x3,
nn::util::Vector3f(0.0f, 0.0f, 0.0f),
nn::util::Vector3f(m_CameraXRotation,
m_CameraYRotation,
0.0f));
nn::util::MatrixConvert(&temp4x4, temp4x3);
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, temp4x4);
uniformMapBlockVS->SetUniform_u_viewMtx(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_ProjMatrix);
uniformMapBlockVS->SetUniform_u_projMtx(*reinterpret_cast<float(*)[16]>(&temp));
}
/*
* SkyBox::Init
* ------------
* Sets the shader used to draw the sky box.
*/
void SkyBox::Init(AssetFileDataHolder* pShader)
{
m_pAssetData = pShader;
}
/*
* SkyBox::SetupUniforms
* ---------------------
* Sets the shader used to draw the sky box.
*/
void SkyBox::SetupUniforms(UniformBufferManager* pUniformBufferManager)
{
m_pBlockVS = pUniformBufferManager->CreateUniformBuffer(sizeof(SkyBoxShader::BlockVSUniformBlockData));
m_pBlockFS = pUniformBufferManager->CreateUniformBuffer(sizeof(SkyBoxShader::BlockFSUniformBlockData));
}
/*
* SkyBox::Draw
* ------------
* Binds the uniforms for the sky box and uses nvnCommandBufferCallCommands
* to draw the model.
*/
void SkyBox::Draw(NVNcommandBuffer* pCommandBuffer) const
{
/* Bind the uniform buffers. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_VERTEX,
SkyBoxShader::BlockVSUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX),
m_pBlockVS->GetCurrentBufferAddress(),
sizeof(SkyBoxShader::BlockVSUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
SkyBoxShader::BlockFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockFS->GetCurrentBufferAddress(),
sizeof(SkyBoxShader::BlockFSUniformBlockData));
/* Draw the model via call commands */
NVNcommandHandle drawHandle = m_Geometry->GetCommandHandle(m_pAssetData->GetProgramData()[0]->m_pProgramName);
nvnCommandBufferCallCommands(pCommandBuffer, 1, &drawHandle);
}

}
