void FileCache::Initialize(nn::mem::StandardAllocator* pAllocator) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pAllocator = pAllocator;
m_IsInitialized = true;
}
}
void FileCache::LoadFile(const char* filePath) NN_NOEXCEPT
{
// loads a single file into the cache
std::map<std::string, FileInfo>::const_iterator iter = m_FileContentMap.find(std::string(filePath));
if (iter != m_FileContentMap.end())
{
NN_ASSERT(false, "%s already in FileCache.\n", filePath);
}
FileInfo fileInfo;
int64_t fileSize = 0;
fileInfo.addr = ReadFile(&fileSize, filePath, m_pAllocator);
fileInfo.size = fileSize;
m_FileContentMap[filePath] = fileInfo;
}
void FileCache::LoadDirectory(const char* directoryPath) NN_NOEXCEPT
{
// loads all files within the specified directory into the cache
// subdirectories are ignored
nn::fs::DirectoryHandle dirHandle;
size_t numDirEntries = 0;
nn::fs::DirectoryEntry* dirEntries = ReadDirectoryEntries(&dirHandle, &numDirEntries, directoryPath);
std::string filePath;
std::string baseDirPath;
PrepareBaseDirectoryPath(baseDirPath, directoryPath);
for (size_t i = 0; i < numDirEntries; ++i)
{
nn::fs::DirectoryEntry* pCurEntry = &dirEntries[i];
if (pCurEntry->directoryEntryType == 1) // skip over (sub)directories, which have a type value of 0
{
filePath.clear();
filePath.append(baseDirPath).append(pCurEntry->name); // use the entire file path for lookup purposes
LoadFile(filePath.c_str());
}
}
delete[] dirEntries;
nn::fs::CloseDirectory(dirHandle);
}
void FileCache::LoadDirectory(const char* directoryPath, const char** includeExtensions, size_t numIncludeExtensions) NN_NOEXCEPT
{
// loads files that have specific file extensions within the specified directory into the cache
// includeExtensions is an array of extensions with periods in front (e.g. ".txt")
// subdirectories are ignored
nn::fs::DirectoryHandle dirHandle;
size_t numDirEntries = 0;
nn::fs::DirectoryEntry* dirEntries = ReadDirectoryEntries(&dirHandle, &numDirEntries, directoryPath);
std::string filePath;
std::string baseDirPath;
PrepareBaseDirectoryPath(baseDirPath, directoryPath);
for (size_t i = 0; i < numDirEntries; ++i)
{
nn::fs::DirectoryEntry* pCurEntry = &dirEntries[i];
if (pCurEntry->directoryEntryType == 1) // skip over (sub)directories, which have a type value of 0
{
filePath.clear();
filePath.append(baseDirPath).append(pCurEntry->name); // use the entire file path for lookup purposes
// only add the entry if its extension is one of the specified includeExtensions
size_t index = filePath.find_last_of(".");
if (index != std::string::npos)
{
std::string extension = filePath.substr(index);
for (size_t j = 0; j < numIncludeExtensions; ++j)
{
if (extension.compare(includeExtensions[j]) == 0)
{
LoadFile(filePath.c_str());
break;
}
}
}
}
}
delete[] dirEntries;
nn::fs::CloseDirectory(dirHandle);
}
void FileCache::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
for (auto& entry : m_FileContentMap)
{
m_pAllocator->Free(entry.second.addr);
}
m_FileContentMap.clear();
m_IsInitialized = false;
}
}
void* FileCache::LookupFile(int64_t* pOutFileSize, const char* filePath) const NN_NOEXCEPT
{
// use this method to get the file content stored in a buffer
// filePath contains the absolute path to the file (which begins with rom FS archive name and ":/")
std::map<std::string, FileInfo>::const_iterator iter;
iter = m_FileContentMap.find(std::string(filePath));
if (iter == m_FileContentMap.end())
{
NN_ASSERT(false, "%s does not exist in FileCache.", filePath);
}
const FileInfo& fileInfo = iter->second;
if (pOutFileSize != nullptr)
{
*pOutFileSize = fileInfo.size;
}
return fileInfo.addr;
}
int64_t FileCache::CalculateTotalUsedSize() const NN_NOEXCEPT
{
int64_t totalSize = 0;
for (auto& entry : m_FileContentMap)
{
totalSize += entry.second.size;
}
return totalSize;
}
size_t FileCache::GetNumFiles() const NN_NOEXCEPT
{
return m_FileContentMap.size();
}

nn::fs::DirectoryEntry* FileCache::ReadDirectoryEntries(nn::fs::DirectoryHandle* pOutDirHandle, size_t* pOutNumEntries, const char* directoryPath) NN_NOEXCEPT
{
NN_ASSERT(directoryPath != nullptr);
NN_ASSERT(pOutNumEntries != nullptr);
NN_ASSERT(pOutDirHandle != nullptr);
nn::fs::DirectoryHandle dirHandle;
nn::Result result = nn::fs::OpenDirectory(&dirHandle, directoryPath, nn::fs::OpenDirectoryMode_All);
NN_ASSERT(result.IsSuccess());
NN_UNUSED(result);
int64_t numEntries = 0;
nn::fs::GetDirectoryEntryCount(&numEntries, dirHandle);
nn::fs::DirectoryEntry* entries = new nn::fs::DirectoryEntry[static_cast<size_t>(numEntries)];
int64_t numStoredEntries = 0;
nn::fs::ReadDirectory(&numStoredEntries, entries, dirHandle, numEntries);
NN_ASSERT(numStoredEntries == numEntries);
*pOutNumEntries = static_cast<size_t>(numEntries);
*pOutDirHandle = dirHandle;
return entries;
}
void FileCache::PrepareBaseDirectoryPath(std::string& baseDir, const char* baseDirPath) NN_NOEXCEPT
{
baseDir = baseDirPath;
char lastChar = baseDirPath[baseDir.length() - 1];
if (lastChar != '/' && lastChar != '\\')
{
baseDir.append("/");
}
}
