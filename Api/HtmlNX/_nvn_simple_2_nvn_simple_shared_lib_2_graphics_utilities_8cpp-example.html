/**************************************************************************************************
* The following routines are intended to help set up NVN bindings of vertex data to vertex
* attributes through NVN.
*
* Simple pictorial representation of the NVN data structures involved with binding vertex data:
*
* NVN Application | Vertex Shader
* --------------- | -------------
* StreamIndex |
* +---------------+ +-------------------+ | +-------------------+ | +------------------+
* | vertex buffer | --> | VertexStreamState | --- | VertexAttribState | <--- | vertex attribute |
* +---------------+ +-------------------+ +-------------------+ | +------------------+
**************************************************************************************************/
void SetupVertexAttribAndStreamStateSeparate(const char** vertexAttribNames, int numVertexAttribs, const GraphicsShader& shader,
nvn::VertexAttribState* vertexAttribStates, nvn::VertexStreamState* vertexStreamStates) NN_NOEXCEPT
{
// Set up NVN VertexStreamState and VertexAttribState objects based on a list of attribute names.
// One VertexStreamState per vertex attribute is used (i.e. uses separate streams per vertex attribute).
// for each attribute
for (int i = 0; i < numVertexAttribs; ++i)
{
// look up the VertexAttributeInfo based on the name of the attribute to bind
const VertexAttributeInfo& attributeInfo = shader.LookupVertexAttributeInfo(vertexAttribNames[i]);
// 1) Initialize VertexAttribState:
//
// The VertexAttribState holds the format of a vertex attribute, and is further linked with a VertexStreamState object
// through the StreamIndex value. The StreamIndex is the index of the VertexStreamState element to associate the VertexAttribState
// with, and this index is based on the ordering of the VertexStreamState array that is bound via nvn::CommandBuffer::BindVertexStreamState.
//
// Note that the ordering of the nvn::VertexAttribState objects passed into nvn::CommandBuffer::BindVertexAttribState must
// match the ordering of the vertex attributes in the vertex shader code based on their location, so we lookup the location
// of the current vertex attribute and use that as an index into vertexAttribStates. The location of the
// vertex attribute in the shader can be either explicitly declared or implicitly allocated by the compiler.
int location = attributeInfo.GetLocationInShader();
NN_ASSERT(location >= 0 && location < numVertexAttribs);
vertexAttribStates[location].SetDefaults();
vertexAttribStates[location].SetFormat(attributeInfo.GetFormat(), 0);
vertexAttribStates[location].SetStreamIndex(i);
// 2) Initialize VertexStreamState:
//
// The VertexStreamState holds the stride for a vertex in a vertex buffer.
//
// Here we make the logical ordering of the nvn::VertexStreamState elements in our array match the ordering of the
// vertexAttribNames array that was passed in. This ordering is relevant towards determining the way to call
// nvn::CommandBuffer::BindVertexBuffer and nvn::CommandBuffer::BindVertexStreamState later when generating rendering commands.
vertexStreamStates[i].SetDefaults();
vertexStreamStates[i].SetStride(attributeInfo.GetElementWidth());
// SetDivisor is not used (it is only necessary when using instanced draw calls)
}
}
void SetupVertexAttribAndStreamStateInterleaved(
const char** vertexAttribNames, int numVertexAttribs, const GraphicsShader& shader,
nvn::VertexAttribState* vertexAttribStates, nvn::VertexStreamState* vertexStreamState) NN_NOEXCEPT
{
// Set up NVN VertexStreamState and VertexAttribState objects based on a list of attribute names.
// One common VertexStreamState is used for all vertex attributes (i.e. uses a single interleaved stream of vertex attributes).
ptrdiff_t curAttributeOffset = 0;
// for each attribute
for (int i = 0; i < numVertexAttribs; ++i)
{
// look up the VertexAttributeInfo based on the name of the attribute to bind
const VertexAttributeInfo& attributeInfo = shader.LookupVertexAttributeInfo(vertexAttribNames[i]);
// initialize VertexAttribState
int location = attributeInfo.GetLocationInShader();
NN_ASSERT(location >= 0 && location < numVertexAttribs);
vertexAttribStates[location].SetDefaults();
vertexAttribStates[location].SetFormat(attributeInfo.GetFormat(), curAttributeOffset);
vertexAttribStates[location].SetStreamIndex(0);
curAttributeOffset += attributeInfo.GetElementWidth();
}
// initialize VertexStreamState
vertexStreamState->SetDefaults();
vertexStreamState->SetStride(curAttributeOffset); // here, curAttributeOffset holds the sum of the widths of each attribute
}
void BindVertexBuffer(int vertexStreamIndex, nvn::Buffer* pVertexDataBuffer, nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
// Note that vertexStreamIndex (i.e. the index passed into nvn::CommandBuffer::BindVertexBuffer) needs to be the index of the nvn::VertexStreamState
// object to associate this vertex buffer with, and this index is determined by the ordering of the nvn::VertexStreamState array that is passed in to
// nvn::CommandBuffer::BindVertexStreamState.
pCommandBuffer->BindVertexBuffer(vertexStreamIndex, pVertexDataBuffer->GetAddress(), pVertexDataBuffer->GetSize());
}
void BindUniformBuffer(nvn::ShaderStage uniformBlockShaderStage, int32_t uniformBlockBindingLocation, nvn::Buffer* pUniformBlockBuffer, nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
pCommandBuffer->BindUniformBuffer(uniformBlockShaderStage, uniformBlockBindingLocation, pUniformBlockBuffer->GetAddress(), pUniformBlockBuffer->GetSize());
}
void ComputeNormal(nn::util::Vector3f* pOutNormal, const nn::util::Vector3f& p1, const nn::util::Vector3f& p2, const nn::util::Vector3f& p3) NN_NOEXCEPT
{
nn::util::Vector3f u = p2 - p1;
nn::util::Vector3f v = p3 - p1;
nn::util::VectorCross(pOutNormal, v, u);
nn::util::VectorNormalize(pOutNormal, *pOutNormal);
}
