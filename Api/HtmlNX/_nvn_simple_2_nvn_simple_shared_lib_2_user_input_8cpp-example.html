namespace
{
nn::hid::DebugPadState g_DebugPadState[2];
nn::hid::NpadHandheldState g_NpadHandheldState[2];
nn::hid::NpadJoyDualState g_NpadJoyDualState[2];
nn::hid::NpadFullKeyState g_NpadFullKeyState[2];
nn::hid::NpadIdType g_NpadIdTypes[2] = { nn::hid::NpadId::Handheld, nn::hid::NpadId::No1 };
int g_CurPadIndex = 0;
int g_NumNpads = 0;
}
//-------------------------------------------------------------------------------------------------
// Private helpers
//-------------------------------------------------------------------------------------------------
int GetDebugPadButton(PadButton padButton) NN_NOEXCEPT
{
// the ordering of PadButton enum matches that of nn::hid::DebugPadButton
return static_cast<int>(padButton);
}
int GetNpadButton(PadButton padButton) NN_NOEXCEPT
{
int buttonIndex = static_cast<int>(padButton);
if (buttonIndex > 3)
{
// The nn::hid::NpadButton has StickL = 4 and StickR = 5 which nn::hid::DebugPadButton hasn't.
// So we match PadButton enum to nn::hid::NpadButton by an offset of 2 when buttonIndex of PadButton is greater than 3.
// compare nn::hid::NpadButton to nn::hid::DebugPadButton to see how the following works.
buttonIndex += 2;
}
return buttonIndex;
}

void InitPad() NN_NOEXCEPT
{
// setup mapping so that the keyboard can be used as a DebugPad for Windows builds
nn::settings::DebugPadKeyboardMap map;
nn::settings::GetDebugPadKeyboardMap(&map);
map.buttonA = nn::hid::KeyboardKey::A::Index;
map.buttonB = nn::hid::KeyboardKey::B::Index;
map.buttonX = nn::hid::KeyboardKey::X::Index;
map.buttonY = nn::hid::KeyboardKey::Y::Index;
map.buttonL = nn::hid::KeyboardKey::L::Index;
map.buttonR = nn::hid::KeyboardKey::R::Index;
map.buttonLeft = nn::hid::KeyboardKey::LeftArrow::Index;
map.buttonRight = nn::hid::KeyboardKey::RightArrow::Index;
map.buttonUp = nn::hid::KeyboardKey::UpArrow::Index;
map.buttonDown = nn::hid::KeyboardKey::DownArrow::Index;
map.buttonStart = nn::hid::KeyboardKey::Space::Index;
nn::settings::SetDebugPadKeyboardMap(map);
g_NumNpads = sizeof(g_NpadIdTypes) / sizeof(*g_NpadIdTypes);
nn::hid::InitializeDebugPad();
nn::hid::InitializeNpad();
nn::hid::SetSupportedNpadStyleSet(nn::hid::NpadStyleFullKey::Mask | nn::hid::NpadStyleJoyDual::Mask | nn::hid::NpadStyleHandheld::Mask);
nn::hid::SetSupportedNpadIdType(g_NpadIdTypes, g_NumNpads);
g_CurPadIndex = 0;
}
void UpdatePad() NN_NOEXCEPT
{
g_CurPadIndex = (g_CurPadIndex + 1) % 2;
nn::hid::GetDebugPadState(&g_DebugPadState[g_CurPadIndex]);
for (int i = 0; i < g_NumNpads; ++i)
{
nn::hid::NpadStyleSet style = nn::hid::GetNpadStyleSet(g_NpadIdTypes[i]);
if (style.Test<nn::hid::NpadStyleHandheld>())
{
nn::hid::GetNpadState(&g_NpadHandheldState[g_CurPadIndex], g_NpadIdTypes[i]);
}
else if (style.Test<nn::hid::NpadStyleJoyDual>())
{
nn::hid::GetNpadState(&g_NpadJoyDualState[g_CurPadIndex], g_NpadIdTypes[i]);
}
else if (style.Test<nn::hid::NpadStyleFullKey>())
{
nn::hid::GetNpadState(&g_NpadFullKeyState[g_CurPadIndex], g_NpadIdTypes[i]);
}
}
}
bool IsTriggered(PadButton padButton) NN_NOEXCEPT
{
int cur = g_CurPadIndex;
int prev = 1 - cur;
// debug pad
int buttonIndex = GetDebugPadButton(padButton);
if (!g_DebugPadState[prev].buttons[buttonIndex] && g_DebugPadState[cur].buttons[buttonIndex])
{
return true;
}
// npad
buttonIndex = GetNpadButton(padButton);
for (int i = 0; i < g_NumNpads; ++i)
{
nn::hid::NpadStyleSet style = nn::hid::GetNpadStyleSet(g_NpadIdTypes[i]);
if (style.Test<nn::hid::NpadStyleHandheld>())
{
if (!g_NpadHandheldState[prev].buttons[buttonIndex] && g_NpadHandheldState[cur].buttons[buttonIndex])
{
return true;
}
}
else if (style.Test<nn::hid::NpadStyleJoyDual>())
{
if (!g_NpadJoyDualState[prev].buttons[buttonIndex] && g_NpadJoyDualState[cur].buttons[buttonIndex])
{
return true;
}
}
else if (style.Test<nn::hid::NpadStyleFullKey>())
{
if (!g_NpadFullKeyState[prev].buttons[buttonIndex] && g_NpadFullKeyState[cur].buttons[buttonIndex])
{
return true;
}
}
}
return false;
}
bool IsHeld(PadButton padButton) NN_NOEXCEPT
{
// debug pad
int buttonIndex = GetDebugPadButton(padButton);
if (g_DebugPadState[g_CurPadIndex].buttons[buttonIndex])
{
return true;
}
// npad
buttonIndex = GetNpadButton(padButton);
for (int i = 0; i < g_NumNpads; ++i)
{
nn::hid::NpadStyleSet style = nn::hid::GetNpadStyleSet(g_NpadIdTypes[i]);
if (style.Test<nn::hid::NpadStyleHandheld>())
{
if (g_NpadHandheldState[g_CurPadIndex].buttons[buttonIndex])
{
return true;
}
}
else if (style.Test<nn::hid::NpadStyleJoyDual>())
{
if (g_NpadJoyDualState[g_CurPadIndex].buttons[buttonIndex])
{
return true;
}
}
else if (style.Test<nn::hid::NpadStyleFullKey>())
{
if (g_NpadFullKeyState[g_CurPadIndex].buttons[buttonIndex])
{
return true;
}
}
}
return false;
}
nn::hid::AnalogStickState GetRightAnalogStick() NN_NOEXCEPT
{
if (g_DebugPadState[g_CurPadIndex].analogStickR.x != 0 || g_DebugPadState[g_CurPadIndex].analogStickR.y != 0)
{
return g_DebugPadState[g_CurPadIndex].analogStickR;
}
for (int i = 0; i < g_NumNpads; ++i)
{
nn::hid::NpadStyleSet style = nn::hid::GetNpadStyleSet(g_NpadIdTypes[i]);
if (style.Test<nn::hid::NpadStyleHandheld>())
{
return g_NpadHandheldState[g_CurPadIndex].analogStickR;
}
else if (style.Test<nn::hid::NpadStyleJoyDual>())
{
return g_NpadJoyDualState[g_CurPadIndex].analogStickR;
}
else if (style.Test<nn::hid::NpadStyleFullKey>())
{
return g_NpadFullKeyState[g_CurPadIndex].analogStickR;
}
}
nn::hid::AnalogStickState stick;
stick.x = 0;
stick.y = 0;
return stick;
}
nn::hid::AnalogStickState GetLeftAnalogStick() NN_NOEXCEPT
{
if (g_DebugPadState[g_CurPadIndex].analogStickL.x != 0 || g_DebugPadState[g_CurPadIndex].analogStickL.y != 0)
{
return g_DebugPadState[g_CurPadIndex].analogStickL;
}
for (int i = 0; i < g_NumNpads; ++i)
{
nn::hid::NpadStyleSet style = nn::hid::GetNpadStyleSet(g_NpadIdTypes[i]);
if (style.Test<nn::hid::NpadStyleHandheld>())
{
return g_NpadHandheldState[g_CurPadIndex].analogStickL;
}
else if (style.Test<nn::hid::NpadStyleJoyDual>())
{
return g_NpadJoyDualState[g_CurPadIndex].analogStickL;
}
else if (style.Test<nn::hid::NpadStyleFullKey>())
{
return g_NpadFullKeyState[g_CurPadIndex].analogStickL;
}
}
nn::hid::AnalogStickState stick;
stick.x = 0;
stick.y = 0;
return stick;
}
