void Font::Initialize(nn::mem::StandardAllocator* pAllocator, const void* fontData) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
NN_ASSERT(pAllocator != nullptr);
NN_ASSERT(fontData != nullptr);
m_pAllocator = pAllocator;
uint32_t fontEngineWorkBufSize = 512 * 1024;
m_FontEngineWorkBuf = m_pAllocator->Allocate(fontEngineWorkBufSize);
if (m_FontEngine.Initialize(m_FontEngineWorkBuf, fontEngineWorkBufSize) != nn::fontll::Success)
{
NN_ASSERT(false);
}
if (m_FontEngine.LoadFont(m_FontName, fontData, 0, nn::fontll::FontNameLengthMax) != nn::fontll::Success)
{
NN_ASSERT(false);
}
m_FontEngine.SetFont(m_FontName);
m_FontEngine.SetFlags(nn::fontll::ScalableFontEngine::Flags_NoEffect);
m_FontEngine.SetBoldWeight(100);
SetSize(32);
nn::fontll::Metrics metrics;
m_FontEngine.GetFontMetrics(&metrics);
m_IsInitialized = true;
}
}
void Font::InitializeFromBfttf(nn::mem::StandardAllocator* pAllocator, const void* bfttfFontData, size_t bfttfFontDataSize) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
void* decodedFontData = nn::fontll::ScalableFontEngineHelper::Decode(bfttfFontData, static_cast<uint32_t>(bfttfFontDataSize));
Initialize(pAllocator, decodedFontData);
}
}
void Font::InitializeFromNxSystemFont(nn::mem::StandardAllocator* pAllocator, nn::pl::SharedFontType fontType) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
while (nn::pl::GetSharedFontLoadState(fontType) == nn::pl::SharedFontLoadState_Loading)
{
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1));
}
Initialize(pAllocator, nn::pl::GetSharedFontAddress(fontType));
}
}
void Font::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_FontEngine.Finalize();
memset(m_FontName, 0, sizeof(m_FontName));
m_pAllocator->Free(m_FontEngineWorkBuf);
m_FontEngineWorkBuf = nullptr;
m_pAllocator = nullptr;
m_IsInitialized = false;
}
}
/**************************************************************************************************
*
* RenderStringToTexture methods
*
* Buffer pointed to by texture argument should be (textureWidth * textureHeight) bytes.
*
**************************************************************************************************/
const CursorPosition Font::RenderStringToTexture(void* texture, int textureWidth, int textureHeight, const char* string) NN_NOEXCEPT
{
return RenderStringToTexture(texture, textureWidth, textureHeight, string, 0, 0);
}
const CursorPosition Font::RenderStringToTexture(void* texture, int textureWidth, int textureHeight, const char* string, int startPosX, int startPosY) NN_NOEXCEPT
{
return RenderString(texture, textureWidth, textureHeight, static_cast<int>(m_FontSize * m_LineSpacing),
string, &m_FontEngine, startPosX, startPosY);
}
void Font::SetSize(int16_t fontSize) NN_NOEXCEPT
{
NN_ASSERT(fontSize > 0);
// nn::fontll's Fixed32 type is a 16.16 fixed point number, where the lower 16 bits are the fractional component.
m_FontSize = fontSize;
m_FontEngine.SetScale(m_FontSize << 16, 0, 0, m_FontSize << 16);
}
void Font::SetLineSpacing(float lineSpacing) NN_NOEXCEPT
{
NN_ASSERT(lineSpacing >= 1.0f);
m_LineSpacing = lineSpacing;
}

CursorPosition RenderString(void* texture, int textureWidth, int textureHeight, int lineHeight,
const char* string, nn::fontll::ScalableFontEngine* pFontEngine, int startPosX, int startPosY) NN_NOEXCEPT
{
uint8_t* output = static_cast<uint8_t*>(texture);
int x = startPosX;
int y = startPosY + lineHeight;
bool reachedEndWidth = false;
bool reachedEnd = false;
while (!reachedEnd)
{
char characterBuffer[4];
nn::util::CharacterEncodingResult result = nn::util::PickOutCharacterFromUtf8String(characterBuffer, &string);
if (result != nn::util::CharacterEncodingResult_Success)
{
break;
}
uint32_t character = 0;
nn::util::ConvertCharacterUtf8ToUtf32(&character, characterBuffer);
if (character == '\0')
{
break;
}
if (character == '\n')
{
x = 0;
y += lineHeight;
reachedEndWidth = false;
}
else
{
if (!reachedEndWidth)
{
nn::fontll::GlyphMap* pGlyphMap = pFontEngine->AcquireGlyphmap(character, nn::fontll::FormatGrayMap8);
int startX = x + pGlyphMap->loX;
int startY = y - pGlyphMap->hiY;
NN_ASSERT(startY >= 0);
if (startX < textureWidth && startY < textureHeight)
{
// copy character glyph over to our texture, line by line
for (int row = 0; (row < pGlyphMap->height && (startY + row) < textureHeight); ++row)
{
int destOffset = (startY + row) * textureWidth + startX;
int srcOffset = row * pGlyphMap->width;
int width = (pGlyphMap->width > (textureWidth - startX)) ? (textureWidth - startX) : pGlyphMap->width;
memcpy(output + destOffset, pGlyphMap->bits + srcOffset, width);
}
}
else
{
reachedEndWidth = (startX >= textureWidth);
reachedEnd = (pGlyphMap->height != 0 && startY >= textureHeight);
}
x += pGlyphMap->idX;
//remember to release GlyphMap from engine after using it
pFontEngine->ReleasesGlyph(pGlyphMap);
}
}
}
CursorPosition cursorPos;
cursorPos.SetX(x);
cursorPos.SetY(y - lineHeight);
cursorPos.SetReachedEnd(reachedEnd);
return cursorPos;
}
} // end namespace
