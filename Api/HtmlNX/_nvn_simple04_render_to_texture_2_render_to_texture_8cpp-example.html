void RenderToTexture::Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
//*****************************************************************************************************************
// from a rendering perspective, this demo is divided into two scene passes:
// 1) using SimplePrimitiveRenderer, render a scene to our own texture
// 2) take the texture, and render another scene with it, this time destined to be displayed to the device window
//*****************************************************************************************************************
// initialize shared memory pool to hold vertex data
size_t memPoolSize = 192 * 1024;
m_VertexMemPool.Initialize(m_pGraphicsSystem, nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED, memPoolSize, "RenderToTexture MemPool");
// initialize shared view transform
nn::util::Vector3f cameraPos(0, 0, 13);
nn::util::Vector3f targetPos(0, 0, 0);
nn::util::Vector3f up(0, 1, 0);
nn::util::MatrixLookAtRightHanded(&m_ViewMtx, cameraPos, targetPos, up);
// initialize shared render state
m_RenderState.Initialize();
m_RenderState.GetPolygonState()->SetFrontFace(nvn::FrontFace::CW);
m_RenderState.GetPolygonState()->SetCullFace(nvn::Face::BACK);
m_RenderState.GetPolygonState()->SetPolygonMode(nvn::PolygonMode::FILL);
m_RenderState.GetDepthStencilState()->SetDepthTestEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthWriteEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthFunc(nvn::DepthFunc::LESS);
// initialize pass-specific data
InitializeFirstPass();
InitializeSecondPass();
m_IsInitialized = true;
}
}
void RenderToTexture::DrawFrame() NN_NOEXCEPT
{
UpdateFirstPass();
DrawFirstPass();
UpdateSecondPass();
DrawSecondPass();
m_IsScreenResolutionChanged = false;
}
void RenderToTexture::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_Shader.Finalize();
m_pGraphicsSystem->GetAllocator()->Free(m_GlslcData);
m_VertexDataBuffer.Finalize();
m_IndexDataBuffer.Finalize();
m_UniformDataBuffer.Finalize();
m_TextureManager.Finalize();
m_Cone.Destroy();
m_Octahedron.Destroy();
m_PrimitiveRenderer.Finalize();
m_VertexMemPool.Finalize(); // This memory pool is used by m_PrimitiveRenderer, so it must be finalized after m_PrimitiveRenderer.
// Otherwise, NVN Debug Layer will emit "Unfinalized object inside memory pool" errors for all
// the nvn::Buffers, etc. that are created internally by SimplePrimitiveRenderer.
for (int i = 0; i < NumRenderPasses; ++i)
{
m_CommandBuffer[i].Finalize();
}
m_pGraphicsSystem = nullptr;
m_IsInitialized = false;
}
}

void RenderToTexture::InitializeFirstPass() NN_NOEXCEPT
{
// initialize SimplePrimitiveRenderer for rendering first pass to texture
m_PrimitiveRenderer.Initialize(m_pGraphicsSystem, &m_VertexMemPool);
m_PrimitiveRenderer.UpdateAmbientColor(Color(0x111111));
m_PrimitiveRenderer.UpdateLightSettings(0, PointLightSettings(Color(0xffffff), Color(0x000000), nn::util::Vector3f(0, 1, 0), 0.001f));
m_PrimitiveRenderer.UpdateLightSettings(1, PointLightSettings(Color(0x000000), Color(0xffffff), nn::util::Vector3f(5, 3, 3), 0.0005f));
// create a few shape objects
m_NumPlanes = 20;
m_NumOctahedronInstances = 5 * m_NumPlanes;
m_NumConeInstances = 4 * m_NumPlanes;
m_Octahedron.CreateOctahedron(&m_PrimitiveRenderer, m_NumOctahedronInstances);
m_Octahedron.SetViewTransform(&m_ViewMtx);
m_Cone.CreateCone(&m_PrimitiveRenderer, m_NumConeInstances);
m_Cone.SetViewTransform(&m_ViewMtx);
// set materials for the shapes
MaterialSettings material(Color(0x222222), Color(0x000000), Color(0xeeeeee), Color(0x000000), 1.0f);
for (size_t i = 0; i < m_Octahedron.GetNumParts() / 2; ++i)
{
material.SetDiffuse(Color(0xa1e79e));
m_Octahedron.SetMaterial(2 * i, material);
material.SetDiffuse(Color(0xbbe7e8));
m_Octahedron.SetMaterial(2 * i + 1, material);
}
material.SetDiffuse(Color(0xf900cd));
m_Cone.SetMaterial(0, material);
// prepare our render (i.e. color) and depth textures:
int width = m_pGraphicsSystem->GetRenderTargetWidth();
int height = m_pGraphicsSystem->GetRenderTargetHeight();
// render texture configuration
nvn::TextureBuilder renderTextureBuilder;
renderTextureBuilder.SetDefaults()
.SetDevice(m_pGraphicsSystem->GetDevice())
.SetSize2D(width, height)
.SetTarget(nvn::TextureTarget::TARGET_2D)
.SetFormat(nvn::Format::RGBA8);
size_t renderTextureSize = renderTextureBuilder.GetStorageSize();
size_t renderTextureAlignment = renderTextureBuilder.GetStorageAlignment();
// depth texture configuration
nvn::TextureBuilder depthTextureBuilder;
depthTextureBuilder.SetDefaults()
.SetSize2D(width, height)
.SetDevice(m_pGraphicsSystem->GetDevice())
.SetTarget(nvn::TextureTarget::TARGET_2D)
.SetFormat(nvn::Format::DEPTH16)
.SetFlags(nvn::TextureFlags::COMPRESSIBLE); // necessary for depth textures
size_t depthTextureSize = depthTextureBuilder.GetStorageSize();
size_t depthTextureAlignment = depthTextureBuilder.GetStorageAlignment();
// initialize our texture manager and allocate storage for our render/depth textures
size_t textureAreaSize = renderTextureSize + renderTextureAlignment + depthTextureSize + depthTextureAlignment;
m_TextureManager.Initialize(m_pGraphicsSystem,
nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED | nvn::MemoryPoolFlags::COMPRESSIBLE,
32, textureAreaSize, "RenderToTexture Texture Pool");
// finish process of configuring/initializing/registering textures
int renderTextureId = InitializeTexture(renderTextureBuilder);
int depthTextureId = InitializeTexture(depthTextureBuilder);
m_pRenderTexture = m_TextureManager.GetTexture(renderTextureId);
m_pDepthTexture = m_TextureManager.GetTexture(depthTextureId);
nvn::SamplerBuilder samplerBuilder;
samplerBuilder.SetDefaults()
.SetDevice(m_pGraphicsSystem->GetDevice())
.SetMinMagFilter(nvn::MinFilter::LINEAR, nvn::MagFilter::LINEAR);
int samplerId = m_TextureManager.AddSampler(samplerBuilder);
m_RenderTextureHandle = m_TextureManager.RegisterTextureHandle(renderTextureId, samplerId);
// initialize command buffer
// Since the command set payload for rendering this pass is larger than the prior samples, we increase
// the commandMemoryChunkSize. The application will run fine even without doing so, but doing so will have the
// effect of reducing the number of times the command buffer memory callback is called during command set recording.
m_CommandBuffer[0].Initialize(m_pGraphicsSystem, false, 16 * 1024);
GenerateFirstPassCommandBuffer();
}
void RenderToTexture::InitializeSecondPass() NN_NOEXCEPT
{
int64_t shaderBinFileSize = 0;
// read shader binary for second pass, create GraphicsShader object from it, and get uniform block reflection info
m_GlslcData = ReadFile(&shaderBinFileSize, "rom:/NvnSimple04.glslc", m_pGraphicsSystem->GetAllocator());
m_Shader.Initialize(m_pGraphicsSystem, static_cast<GLSLCoutput*>(m_GlslcData));
const UniformBlockInfo& uniformBlockInfo = m_Shader.LookupUniformBlockInfo("VertexShaderUniformBlock");
// allocate blocks of memory for vertex and uniform data
size_t attributeDataSize = NumVertices * sizeof(Vertex);
size_t indexDataSize = NumFaces * sizeof(g_VertexIndices);
size_t uniformAlignment = m_pGraphicsSystem->GetResourceInfo().uniformBufferAlignment;
m_VertexMemPool.AllocateBuffer(&m_VertexDataBuffer, attributeDataSize);
m_VertexMemPool.AllocateBuffer(&m_IndexDataBuffer, indexDataSize);
m_VertexMemPool.AllocateBuffer(&m_UniformDataBuffer, uniformBlockInfo.GetSize(), uniformAlignment);
// load vertex and index data into memory pool
Vertex* vertices = static_cast<Vertex*>(m_VertexDataBuffer.Map());
// we use a single vertex buffer with interleaved attributes
for (int i = 0; i < NumVertices; ++i)
{
Vertex* pVertex = vertices + i;
float x = g_VertexPositions[3 * i];
float y = g_VertexPositions[3 * i + 1];
float z = g_VertexPositions[3 * i + 2];
int uvIndex = i % NumFaces;
float u = g_VertexUvs[2 * uvIndex];
float v = g_VertexUvs[2 * uvIndex + 1];
pVertex->SetPos(x, y, z);
pVertex->SetUV(u, v);
}
UpdateTexcoords();
// indices
m_NumIndices = 6 * NumFaces;
uint32_t* indices = static_cast<uint32_t*>(m_IndexDataBuffer.Map());
for (uint32_t i = 0; i < m_NumIndices; ++i)
{
uint32_t delta = 4 * (i / 6);
uint32_t index = i % 6;
indices[i] = g_VertexIndices[index] + delta;
}
// compute and load uniform data for second pass into memory pool
char* baseUniformBlockAddr = static_cast<char*>(m_UniformDataBuffer.Map());
// load projection transform
char* projMtxAddr = baseUniformBlockAddr + uniformBlockInfo.LookupMemberInfo("projMtx").GetOffset();
nn::util::Matrix4x4fType projMtx;
nn::util::MatrixIdentity(&projMtx);
nn::util::MatrixPerspectiveFieldOfViewRightHanded(&projMtx, nn::util::FloatPi / 3.0f, 16.0f / 9.0f, 0.1f, 1000.0f);
LoadMatrix44(projMtxAddr, projMtx);
// compute model view transform location in memory pool
m_ModelViewMtxAddr = baseUniformBlockAddr + uniformBlockInfo.LookupMemberInfo("modelViewMtx").GetOffset();
// setup vertex attribute and stream state for second pass
const char* attributeNames[NumVertexAttributes];
// the ordering of the attribute names must logically match up with the attribute ordering in our vertex data structure (see above)
attributeNames[0] = "attr_position";
attributeNames[1] = "attr_uv";
SetupVertexAttribAndStreamStateInterleaved(attributeNames, NumVertexAttributes, m_Shader, m_VertexAttribStates, &m_VertexStreamState);
// uniform binding locations
m_VsUboLocation = uniformBlockInfo.GetBindingLocations(nvn::ShaderStage::VERTEX);
m_FsSamplerLocation = m_Shader.LookupUniformInfo("sampler").GetBindingLocations(nvn::ShaderStage::FRAGMENT);
// initialize command buffer
m_CommandBuffer[1].Initialize(m_pGraphicsSystem);
GenerateSecondPassCommandBuffer();
}
void RenderToTexture::UpdateFirstPass() NN_NOEXCEPT
{
// update model view transforms for the shape instances
int octaIndex = 0;
int coneIndex = 0;
float dx = 8.0f;
float dy = 4.4f;
float coneAngle = atanf(dy / dx);
nn::util::Matrix4x3fType transformMtx;
nn::util::MatrixIdentity(&transformMtx);
for (int row = -1; row <= 1; ++row)
{
for (int col = -1; col <= 1; ++col)
{
float z = 0.0f;
nn::util::MatrixIdentity(&transformMtx);
// cones
if (abs(row) == abs(col) && (row != 0 && col != 0))
{
nn::util::MatrixSetRotateXyz(&transformMtx, nn::util::Vector3f(m_ShapeAngle, 0, 0));
float theta = ((nn::util::FloatPi / 2) + (row * coneAngle)) * col;
nn::util::Matrix4x3fType tempMtx;
nn::util::MatrixIdentity(&tempMtx);
nn::util::MatrixSetRotateXyz(&tempMtx, nn::util::Vector3f(0, 0, theta));
nn::util::MatrixMultiply(&transformMtx, transformMtx, tempMtx);
for (int i = 0; i < m_NumPlanes; ++i)
{
nn::util::MatrixSetTranslate(&transformMtx, nn::util::Vector3f(col * dx, row * dy, z));
m_Cone.SetModelTransform(coneIndex++, transformMtx);
z -= 5.0f;
}
}
// octahedrons
else
{
nn::util::MatrixSetRotateXyz(&transformMtx, nn::util::Vector3f(0, m_ShapeAngle, 0));
for (int i = 0; i < m_NumPlanes; ++i)
{
nn::util::MatrixSetTranslate(&transformMtx, nn::util::Vector3f(col * dx, row * dy, z));
m_Octahedron.SetModelTransform(octaIndex++, transformMtx);
z -= 5.0f;
}
}
}
}
// update state for next frame
m_ShapeAngle += 0.01f;
if (m_ShapeAngle > (2 * nn::util::FloatPi))
{
m_ShapeAngle -= (2 * nn::util::FloatPi);
m_PullbackAnimation = !m_PullbackAnimation;
}
}
void RenderToTexture::GenerateFirstPassCommandBuffer() NN_NOEXCEPT
{
nvn::CommandBuffer* commandBuffer = m_CommandBuffer[0].StartRecording();
{
// first, we explicitly set our own render/depth texture here to render into
commandBuffer->SetRenderTargets(1, &m_pRenderTexture, nullptr, m_pDepthTexture, nullptr);
int width = m_pGraphicsSystem->GetScreenWidth();
int height = m_pGraphicsSystem->GetScreenHeight();
// we limit the scissor and viewport to a portion of the texture if screen resolution is less than 1080p.
// this will reduce GPU load under handheld mode, which runs at a lower GPU frequency.
commandBuffer->SetScissor(0, 0, width, height);
commandBuffer->SetViewport(0, 0, width, height);
Color clearColor(0x222222);
commandBuffer->ClearColor(0, clearColor.Get(), nvn::ClearColorMask::RGBA);
commandBuffer->ClearDepthStencil(1.0f, true, 0, 0);
m_RenderState.Bind(commandBuffer);
// draw our shapes:
m_PrimitiveRenderer.StartDrawing(commandBuffer);
// cones
m_Cone.StartDrawing(commandBuffer, 0);
for (int i = 0; i < m_NumConeInstances; ++i)
{
m_Cone.SetModelViewTransformAsCurrent(i);
m_Cone.Draw(commandBuffer);
}
// octahedrons
for (size_t p = 0; p < m_Octahedron.GetNumParts(); ++p)
{
m_Octahedron.StartDrawing(commandBuffer, p);
for (int i = 0; i < m_NumOctahedronInstances; ++i)
{
m_Octahedron.SetModelViewTransformAsCurrent(i);
m_Octahedron.Draw(commandBuffer);
}
}
}
m_CommandHandle[0] = m_CommandBuffer[0].EndRecording();
}
void RenderToTexture::DrawFirstPass() NN_NOEXCEPT
{
if (!m_IsPrebuiltCommandBufferUsed || m_IsScreenResolutionChanged)
{
GenerateFirstPassCommandBuffer();
}
m_pGraphicsSystem->GetQueue()->SubmitCommands(1, &m_CommandHandle[0]);
}
void RenderToTexture::UpdateSecondPass() NN_NOEXCEPT
{
// update modelView uniform
nn::util::Matrix4x3fType rotateMtx;
nn::util::Matrix4x3fType scaleMtx;
nn::util::Matrix4x3fType modelViewMtx;
nn::util::MatrixIdentity(&rotateMtx);
nn::util::MatrixSetRotateXyz(&rotateMtx, nn::util::Vector3f(0, m_BoxAngle, 0));
float scale = 3.705f;
nn::util::Vector3f scaleVec(16.0f / 9.0f * scale, scale, 16.0f / 9.0f * scale);
nn::util::MatrixIdentity(&scaleMtx);
nn::util::MatrixSetScale(&scaleMtx, scaleVec);
nn::util::MatrixMultiply(&modelViewMtx, scaleMtx, rotateMtx);
nn::util::MatrixSetTranslate(&modelViewMtx, nn::util::Vector3f(0, 0, m_BoxDepth));
nn::util::MatrixMultiply(&modelViewMtx, modelViewMtx, m_ViewMtx);
LoadMatrix43(m_ModelViewMtxAddr, modelViewMtx);
// update state for next frame
if (m_PullbackAnimation)
{
m_BoxDepth = -3.0f * (-cosf(m_ShapeAngle) + 1.0f);
m_BoxAngle = m_ShapeAngle / 2.0f;
}
else
{
m_BoxDepth = 0.0f;
m_BoxAngle = 0.0f;
}
}
void RenderToTexture::GenerateSecondPassCommandBuffer() NN_NOEXCEPT
{
nvn::CommandBuffer* pCommandBuffer = m_CommandBuffer[1].StartRecording();
{
// scissor/viewport
m_pGraphicsSystem->SetDefaultViewportAndScissor(pCommandBuffer);
// clear color and depth/stencil buffers
Color clearColor(0xc4f0ff);
pCommandBuffer->ClearColor(0, clearColor.Get(), nvn::ClearColorMask::RGBA);
pCommandBuffer->ClearDepthStencil(1.0f, true, 0, 0);
// render state
m_RenderState.Bind(pCommandBuffer);
// bind shader
m_Shader.Bind(pCommandBuffer);
// bind uniform buffer
BindUniformBuffer(nvn::ShaderStage::VERTEX, m_VsUboLocation, &m_UniformDataBuffer, pCommandBuffer);
// bind vertex buffer data
pCommandBuffer->BindVertexAttribState(NumVertexAttributes, m_VertexAttribStates);
pCommandBuffer->BindVertexStreamState(1, &m_VertexStreamState);
BindVertexBuffer(0, &m_VertexDataBuffer, pCommandBuffer);
// set texture and sampler pools
m_TextureManager.SetTextureAndSamplerPools(pCommandBuffer);
// bind texture
pCommandBuffer->BindTexture(nvn::ShaderStage::FRAGMENT, m_FsSamplerLocation, m_RenderTextureHandle);
// draw call
pCommandBuffer->DrawElements(nvn::DrawPrimitive::TRIANGLES, nvn::IndexType::UNSIGNED_INT, m_NumIndices, m_IndexDataBuffer.GetAddress());
}
m_CommandHandle[1] = m_CommandBuffer[1].EndRecording();
}
void RenderToTexture::DrawSecondPass() NN_NOEXCEPT
{
// we need to first reset the render target to the texture that the GraphicsSystem sends to the display
m_pGraphicsSystem->ResetRenderTarget();
if (!m_IsPrebuiltCommandBufferUsed || m_IsScreenResolutionChanged)
{
if (m_IsScreenResolutionChanged)
{
UpdateTexcoords();
}
GenerateSecondPassCommandBuffer();
}
m_pGraphicsSystem->GetQueue()->SubmitCommands(1, &m_CommandHandle[1]);
}
int RenderToTexture::InitializeTexture(nvn::TextureBuilder& textureBuilder) NN_NOEXCEPT
{
size_t size = textureBuilder.GetStorageSize();
size_t alignment = textureBuilder.GetStorageAlignment();
ptrdiff_t memPoolOffset = 0;
void* cpuAddr = m_TextureManager.AllocateTextureMemory(&memPoolOffset, size, alignment);
textureBuilder.SetPackagedTextureData(cpuAddr);
textureBuilder.SetStorage(m_TextureManager.GetMemoryPool()->GetMemoryPool(), memPoolOffset);
return m_TextureManager.AddTexture(textureBuilder);
}
void RenderToTexture::UpdateTexcoords() NN_NOEXCEPT
{
// This function update the texture coordinates of the second pass geometry that uses the
// first pass render texture. This can be used if the screen resolution has changed.
// Since we limit the viewport and scissor for the render texture based on the screen resolution
// as opposed to using different texture sizes for different screen size configurations,
// we also need to update the texture coordinates.
uint32_t screenWidth = m_pGraphicsSystem->GetScreenWidth();
uint32_t renderTextureWidth = m_pGraphicsSystem->GetRenderTargetWidth();
// note that we assume the aspect ratio is the same before/after change (hence we don't look at the height)
float uv = static_cast<float>(screenWidth) / static_cast<float>(renderTextureWidth);
if (m_pGraphicsSystem->GetScreenHeight() == 720)
{
uv = 0.666f; // seems to be a bit better in terms of the edges for 720p case
}
Vertex* vertices = static_cast<Vertex*>(m_VertexDataBuffer.Map());
for (int i = 0; i < NumVertices; ++i)
{
Vertex* pVertex = vertices + i;
if (pVertex->GetUv().GetX() != 0.0f)
{
pVertex->SetUV(uv, pVertex->GetUv().GetY());
}
if (pVertex->GetUv().GetY() != 0.0f)
{
pVertex->SetUV(pVertex->GetUv().GetX(), uv);
}
}
