// Ensure the ability to get the friend list.
void EnsureFriendListAvailable(const nn::account::Uid& user) NN_NOEXCEPT
{
NN_LOG("[EnsureFriendListAvailable]\n");
NN_LOG("nn::friends::EnsureFriendListAvailable() ...\n");
// While the nn::friends::GetFriendList() function references the cache in system storage,
// the cache may not yet exist immediately after the user account is linked to the Nintendo Account.
// You can use the nn::friends::EnsureFriendListAvailable() function to ensure that the cache exists in the system storage.
nn::friends::AsyncContext context;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::friends::EnsureFriendListAvailable(&context, user));
// There are two ways to wait on asynchronous processing.
// One method puts the event on standby, and the other method checks the completion flag for asynchronous processing.
// If the asynchronous processing is suspended, call nn::friends::AsyncContext::Cancel().
// Even after calling nn::friends::AsyncContext::Cancel(), you must still wait until asynchronous processing completes.
#if 1
nn::os::SystemEvent completionEvent;
NN_ABORT_UNLESS_RESULT_SUCCESS(context.GetSystemEvent(&completionEvent));
completionEvent.Wait();
#else
bool done = false;
while (context.HasDone(&done).IsSuccess() && !done)
{
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(10));
}
#endif
NN_LOG("nn::friends::EnsureFriendListAvailable() done!\n");
if (context.GetResult().IsFailure())
{
nn::err::ShowError(context.GetResult());
}
}
// Get the friend list.
void GetFriendList(const nn::account::Uid& user) NN_NOEXCEPT
{
NN_LOG("[GetFriendList]\n");
static nn::friends::Friend s_Friends[nn::friends::FriendCountMax] = {};
int count;
nn::friends::FriendFilter filter = {};
// Use the following filter to filter on friends playing the application specified with the same presence group ID.
// filter.presenceStatus = nn::friends::PresenceStatusFilter_OnlineOrOnlinePlay;
// filter.isSameAppPresenceOnly = true;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::friends::GetFriendList(&count, s_Friends, user, 0, NN_ARRAY_SIZE(s_Friends), filter));
auto getPresenceStatusString = [](nn::friends::PresenceStatus status) NN_NOEXCEPT -> const char*
{
switch (status)
{
case nn::friends::PresenceStatus_Offline:
return "Offline";
case nn::friends::PresenceStatus_Online:
return "Online";
case nn::friends::PresenceStatus_OnlinePlay:
return "OnlinePlay";
default:
return "";
}
};
NN_LOG("--------------------------------------------------------------------------------\n");
NN_LOG("GetFriendList: Count = %d\n", count);
NN_LOG("--------------------------------------------------------------------------------\n");
for (int i = 0; i < count; i++)
{
NN_LOG("Friend[%3d]\n", i);
NN_LOG(" - id = %016llx\n", s_Friends[i].GetAccountId().id);
NN_LOG(" - name = %s\n", s_Friends[i].GetNickname().name);
NN_LOG(" - status = %s\n", getPresenceStatusString(s_Friends[i].GetPresence().GetStatus()));
}
NN_LOG("--------------------------------------------------------------------------------\n");
}
// Get the list of friend profiles.
void GetFriendProfileList(const nn::account::Uid& user) NN_NOEXCEPT
{
NN_LOG("[GetFriendProfileList]\n");
static nn::account::NetworkServiceAccountId s_AccountIds[nn::friends::FriendCountMax] = {};
int total;
nn::friends::FriendFilter filter = {};
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::friends::GetFriendList(&total, s_AccountIds, user, 0, NN_ARRAY_SIZE(s_AccountIds), filter));
if (total == 0)
{
return;
}
static nn::friends::Profile s_Profiles[nn::friends::GetProfileCountMax] = {};
int offset = 0;
while (offset < total)
{
int count = std::min(total - offset, static_cast<int>(NN_ARRAY_SIZE(s_Profiles)));
NN_LOG("nn::friends::GetProfileList() ...\n");
// There are guidelines on the frequency at which nn::friends::GetProfileList() is called.
// Note: Avoid excessive calls.
nn::friends::AsyncContext context;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::friends::GetProfileList(&context, s_Profiles, user, &s_AccountIds[offset], count));
nn::os::SystemEvent completionEvent;
NN_ABORT_UNLESS_RESULT_SUCCESS(context.GetSystemEvent(&completionEvent));
completionEvent.Wait();
NN_LOG("nn::friends::GetProfileList() done!\n");
if (context.GetResult().IsFailure())
{
nn::err::ShowError(context.GetResult());
return;
}
for (int i = 0; i < count; i++)
{
nn::friends::Url imageUrl = {};
NN_ABORT_UNLESS_RESULT_SUCCESS(s_Profiles[i].GetProfileImageUrl(&imageUrl, nn::friends::ImageSize_64x64));
NN_LOG("FriendProfile[%3d]\n", offset + i);
NN_LOG(" - id = %016llx\n", s_Profiles[i].GetAccountId().id);
NN_LOG(" - name = %s\n", s_Profiles[i].GetNickname().name);
NN_LOG(" - imageUrl = %s\n", imageUrl.value);
}
offset += count;
}
}

extern "C" void nnMain()
{
NN_LOG("[sample] Start.\n");
nn::account::Initialize();
nn::nifm::Initialize();
nn::account::Uid user;
nn::Result result = nn::account::ShowUserSelector(&user);
if (nn::account::ResultCancelledByUser::Includes(result))
{
NN_LOG("[sample] Please select the user account.\n");
NN_LOG("[sample] End.\n");
return;
}
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
nn::account::UserHandle handle = {};
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::account::OpenUser(&handle, user));
NN_LOG("nn::nifm::SubmitNetworkRequestAndWait() ...\n");
// You must establish a network connection before calling functions that trigger communication.
nn::nifm::SubmitNetworkRequestAndWait();
NN_LOG("nn::nifm::SubmitNetworkRequestAndWait() done!\n");
if (!nn::nifm::IsNetworkAvailable())
{
NN_LOG("Network is not available.\n");
}
if (nn::nifm::IsNetworkAvailable())
{
EnsureFriendListAvailable(user);
}
else
{
NN_LOG("EnsureFriendListAvailable() skipped.\n");
}
GetFriendList(user);
if (nn::nifm::IsNetworkAvailable())
{
GetFriendProfileList(user);
}
else
{
NN_LOG("GetFriendProfileList() skipped.\n");
}
if (nn::nifm::IsNetworkAvailable())
{
EnsureBlockedUserListAvailable(user);
}
else
{
NN_LOG("EnsureBlockedUserListAvailable() skipped.\n");
}
GetBlockedUserList(user);
nn::account::CloseUser(handle);
NN_LOG("[sample] End.\n");
}
