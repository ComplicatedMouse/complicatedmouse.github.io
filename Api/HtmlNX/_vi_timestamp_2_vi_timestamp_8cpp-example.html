using FrameList = std::vector<nn::vi::FrameInfo>;
const int RenderCountMax = 300;
struct VsyncThreadArgs
{
nn::vi::Layer* pLayer;
nn::os::SystemEvent* pVsync;
};
std::uint8_t NN_OS_ALIGNAS_THREAD_STACK g_VsyncThreadStack[16384];
std::atomic<bool> g_IsRunning(true);
std::atomic<nn::vi::FrameInfo*> g_pLatestFrame(nullptr);
void VsyncThread(void* pArg) NN_NOEXCEPT
{
VsyncThreadArgs* pArgs = static_cast<VsyncThreadArgs*>(pArg);
FrameList frames;
frames.reserve(RenderCountMax);
// nn::vi keeps a short history of frames. The exact length may be queried like this:
const int HistoryLengthMax = nn::vi::ListFrameHistory(nullptr, 0, pArgs->pLayer);
auto pHistory = std::make_unique<nn::vi::FrameInfo[]>(HistoryLengthMax);
std::uint64_t vsyncCount = 0;
while( g_IsRunning.load(std::memory_order_acquire) )
{
pArgs->pVsync->Wait();
++vsyncCount;
// After vsync, the most recent enqueued frame's status may have changed.
int writeCount = nn::vi::ListFrameHistory(pHistory.get(), HistoryLengthMax, pArgs->pLayer);
for( int i = writeCount - 1; i >= 0; --i )
{
if( g_pLatestFrame != nullptr && pHistory[i].frameNumber <= g_pLatestFrame.load()->frameNumber )
{
continue;
}
switch( pHistory[i].status )
{
case nn::vi::FrameStatus_Enqueued:
NN_LOG("Vsync %llu: Frame %llu Enqueued\n", vsyncCount, pHistory[i].frameNumber);
frames.push_back(pHistory[i]);
break;
case nn::vi::FrameStatus_Presented:
{
std::uint64_t frameNumber = pHistory[i].frameNumber;
// This frame was presented, so try to update the the complete frame history that this sample tracks.
auto iterator = std::find_if(frames.rbegin(), frames.rend(), [frameNumber](const nn::vi::FrameInfo& info)
{
return info.frameNumber == frameNumber;
});
if( iterator != frames.rend() )
{
*iterator = pHistory[i];
}
else
{
frames.push_back(pHistory[i]);
iterator = frames.rbegin();
}
NN_LOG("Vsync %llu: Frame %llu Presented\n", vsyncCount, iterator->frameNumber);
NN_LOG(" From Enqueue: %lld us\n", (iterator->presentationTime - iterator->enqueueTime).ToTimeSpan().GetMicroSeconds());
if( g_pLatestFrame != nullptr )
{
NN_LOG(" From Last Frame: %lld us\n", (iterator->presentationTime - g_pLatestFrame.load(std::memory_order_relaxed)->presentationTime).ToTimeSpan().GetMicroSeconds());
}
// Update the rendering thread with the last frame that was presented.
g_pLatestFrame.store(&(*iterator), std::memory_order_release);
}
break;
default:
break;
}
}
}
}

{

NN_ASSERT_NOT_NULL(ptr);

}

{
NN_UNUSED(userPtr);

}

{
NN_UNUSED(userPtr);

NN_ASSERT_NOT_NULL(newPtr);

extern "C" void nnMain()
{
nv::SetGraphicsAllocator(NvAllocate, NvFree, NvReallocate, nullptr);
nv::SetGraphicsDevtoolsAllocator(NvAllocate, NvFree, NvReallocate, nullptr);
const std::size_t GraphicsMemorySize = 8 * 1024 * 1024;
MallocPtr graphicsMemory(aligned_alloc(4096, GraphicsMemorySize));
nv::InitializeGraphics(graphicsMemory.get(), GraphicsMemorySize);
NN_UTIL_SCOPE_EXIT
{
nv::FinalizeGraphics();
};
nn::vi::Initialize();
NN_UTIL_SCOPE_EXIT
{
nn::vi::Finalize();
};
nn::vi::Display* pDisplay;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::vi::OpenDefaultDisplay(&pDisplay));
nn::os::SystemEvent vsync;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::vi::GetDisplayVsyncEvent(vsync.GetBase(), pDisplay));
nn::vi::Layer* pLayer;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::vi::CreateLayer(&pLayer, pDisplay));
nn::gfx::Initialize();
NN_UTIL_SCOPE_EXIT
{
nn::gfx::Finalize();
};
GfxSystem gfx(pLayer);
VsyncThreadArgs args { pLayer, &vsync };
nn::os::ThreadType vsyncThread;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::CreateThread(&vsyncThread,
VsyncThread,
&args,
g_VsyncThreadStack,
sizeof(g_VsyncThreadStack),
nn::os::DefaultThreadPriority));
nn::os::SetThreadCoreMask(&vsyncThread, 2, 4);
nn::os::SetThreadCoreMask(nn::os::GetCurrentThread(), 0, 1);
// Synchronize vsync and rendering threads to start on a vsync.
vsync.Clear();
vsync.Wait();
nn::os::StartThread(&vsyncThread);
std::uint64_t frameNumber = 0;
nn::vi::FrameInfo* pOldLastFrame = nullptr;
for( int i = 0; i < RenderCountMax; ++i )
{
gfx.BeginDraw();
auto pTextWriter = gfx.GetTextWriter();
nn::util::Unorm8x4 color = { { 255, 255, 255, 255 } };
pTextWriter->SetTextColor(color);
pTextWriter->SetScale(1.75f, 1.75f);
pTextWriter->SetCursor(0.f, 0.f);
// Working one ahead of the last frame number.
pTextWriter->Print("Displaying Frame %lld\n\n", frameNumber + 1);
nn::vi::FrameInfo* pLastFrame = g_pLatestFrame.load(std::memory_order_acquire);
if( pLastFrame != nullptr )
{
pTextWriter->Print("Frame %lld Statistics:\n", pLastFrame->frameNumber);
pTextWriter->SetScale(1.25f, 1.25f);
const char* status;
switch( pLastFrame->status )
{
case nn::vi::FrameStatus_Enqueued:
status = "Enqueued";
break;
case nn::vi::FrameStatus_Presented:
status = "Presented";
break;
default:
status = "Unknown";
}
// Note that not all fields are valid unless the frame's status is nn::vi::FrameStatus_Presented.
pTextWriter->Print("\tStatus : %s\n", status);
pTextWriter->Print("\tFrom Enqueue : %lld us\n", (pLastFrame->presentationTime - pLastFrame->enqueueTime).ToTimeSpan().GetMicroSeconds());
if( pOldLastFrame != nullptr )
{
pTextWriter->Print("\tFrom Previous Frame : %lld us\n", (pLastFrame->presentationTime - pOldLastFrame->presentationTime).ToTimeSpan().GetMicroSeconds());
}
pOldLastFrame = pLastFrame;
}
gfx.EndDraw();
// After nn::gfx::Queue::Present is invoked, nn::vi will generate a frame number.
// This should be used to identify a specific frame in nn::vi::ListFrameHistory.
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::vi::GetLatestFrameNumber(&frameNumber, pLayer));
}
g_IsRunning.store(false, std::memory_order_release);
nn::os::WaitThread(&vsyncThread);
}
