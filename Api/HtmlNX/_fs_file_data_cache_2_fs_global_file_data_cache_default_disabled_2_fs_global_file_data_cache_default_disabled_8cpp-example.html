/*
* This is the definition for nninitStartup(), which initializes the memory management functionality of the application.
*
* This sample allocates all of the memory heaps available to the application to the default
* malloc() function provided by NintendoSDK. Initialize the application-specific memory management features as needed,
* such as enabling the virtual memory address management feature.
* For more information, see Features > Program Flow > Starting and Initializing Programs.
*/
// Global file data caching is not automatically enabled in the Windows environment.
// nninitStartup() is only defined when the application operating environment is the operating system of the NX development hardware.
#if defined(NN_BUILD_TARGET_PLATFORM_OS_NN)
extern "C" void nninitStartup()
{
// Gets the information for the memory available to the application.
nn::os::MemoryInfo memInfo;
nn::os::QueryMemoryInfo(&memInfo);
size_t size = static_cast<size_t>(memInfo.totalAvailableMemorySize) - memInfo.totalUsedMemorySize;
// Along with the preconditions for nn::os::SetMemoryHeapSize(), rounds down size as an integer multiplier of nn::os::MemoryBlockUnitSize.
size = nn::util::align_down(size, nn::os::MemoryBlockUnitSize);
// Sets the memory heap size.
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::SetMemoryHeapSize(size));
// From the memory heap, allocates the memory region to be used by malloc().
uintptr_t p;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::AllocateMemoryBlock(&p, size));
// Sets a memory region for malloc() use.
nn::init::InitializeAllocator(reinterpret_cast<void*>(p), size, true);
// Registers the memory allocator to use by implementation of the thread_local specifier.
nn::os::SetMemoryAllocatorForThreadLocal(
[](size_t size, size_t) { return std::malloc(size); },
[](void* ptr, size_t) { std::free(ptr); }
);
// The default nninitStartup() initializes the memory management functionality and then enables global file data caching.
// This sample does not use global file data caching, so it only initializes the memory management functionality.
}
#endif // defined(NN_BUILD_TARGET_PLATFORM_OS_NN)
/*
* The application entry point.
*/
extern "C" void nnMain()
{
// Before calling nnMain(), the nninitStartup() defined in this sample program is called.
// The nninitStartup() was rewritten, so global file data caching is not enabled.
// Uses the resource data (ROM) to perform a benchmark.
// Global file data caching is disabled.
RunRomAccessBenchmark();
}
