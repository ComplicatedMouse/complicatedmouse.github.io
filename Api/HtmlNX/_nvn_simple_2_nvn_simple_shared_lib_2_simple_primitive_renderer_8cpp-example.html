void SimplePrimitiveRenderer::Initialize(GraphicsSystem* pGraphicsSystem, GraphicsMemoryPool* pVertexDataMemPool, const char* shaderBinaryPath) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
m_pVertexMemoryPool = pVertexDataMemPool;
// create shader
int64_t shaderBinFileSize = 0;
m_GlslcData = ReadFile(&shaderBinFileSize, shaderBinaryPath, m_pGraphicsSystem->GetAllocator());
m_Shader.Initialize(m_pGraphicsSystem, static_cast<GLSLCoutput*>(m_GlslcData));
// initialize vertex shader global uniform data (projection matrix)
m_pGlobalVSUboInfo = &m_Shader.LookupUniformBlockInfo("GlobalVsUniformBlock");
nn::util::Matrix4x4fType projMtx;
nn::util::MatrixIdentity(&projMtx);
nn::util::MatrixPerspectiveFieldOfViewRightHanded(&projMtx, nn::util::FloatPi / 3.0f, 16.0f / 9.0f, 0.1f, 1000.0f);
m_UniformAlignment = m_pGraphicsSystem->GetResourceInfo().uniformBufferAlignment;
m_pVertexMemoryPool->AllocateBuffer(&m_GlobalVSUniformBuffer, m_pGlobalVSUboInfo->GetSize(), m_UniformAlignment);
m_ProjMtxAddr = static_cast<char*>(m_GlobalVSUniformBuffer.Map()) + m_pGlobalVSUboInfo->LookupMemberInfo("projection").GetOffset();
LoadMatrix44(m_ProjMtxAddr, projMtx);
// initialize fragment shader global uniform data (ambient color, light settings)
m_pGlobalFSUboInfo = &m_Shader.LookupUniformBlockInfo("GlobalFsUniformBlock");
m_pVertexMemoryPool->AllocateBuffer(&m_GlobalFSUniformBuffer, m_pGlobalFSUboInfo->GetSize(), m_UniformAlignment);
char* fsUboAddr = static_cast<char*>(m_GlobalFSUniformBuffer.Map());
m_pAmbientColor = reinterpret_cast<Color*>(fsUboAddr + m_pGlobalFSUboInfo->LookupMemberInfo("ambient").GetOffset());
m_pAmbientColor->Set(0xffffff);
for (int i = 0; i < NumLights; ++i)
{
char index[8];
snprintf(index, 8, "%d", i);
std::string uniformName("pointLights[");
uniformName.append(index).append("]");
size_t enabledOffset = m_pGlobalFSUboInfo->LookupMemberInfo((uniformName + ".enabled").c_str()).GetOffset();
m_PointLights[i].pEnabled = reinterpret_cast<uint32_t*>(fsUboAddr + enabledOffset);
size_t posOffset = m_pGlobalFSUboInfo->LookupMemberInfo((uniformName + ".pos").c_str()).GetOffset();
m_PointLights[i].pPos = reinterpret_cast<nn::util::Vector3f*>(fsUboAddr + posOffset);
size_t diffuseOffset = m_pGlobalFSUboInfo->LookupMemberInfo((uniformName + ".diffuse").c_str()).GetOffset();
m_PointLights[i].pDiffuse = reinterpret_cast<Color*>(fsUboAddr + diffuseOffset);
size_t specularOffset = m_pGlobalFSUboInfo->LookupMemberInfo((uniformName + ".specular").c_str()).GetOffset();
m_PointLights[i].pSpecular = reinterpret_cast<Color*>(fsUboAddr + specularOffset);
size_t attenuationOffset = m_pGlobalFSUboInfo->LookupMemberInfo((uniformName + ".attenuation").c_str()).GetOffset();
m_PointLights[i].pAttenuation = reinterpret_cast<float*>(fsUboAddr + attenuationOffset);
*(m_PointLights[i].pEnabled) = 0;
*(m_PointLights[i].pAttenuation) = 0;
m_PointLights[i].pPos->Set(0, 0, 0);
m_PointLights[i].pDiffuse->Set(0x000000);
m_PointLights[i].pSpecular->Set(0x000000);
}
// lookup information on per-primitive UBOs up front once, so that we can use that information later
m_pPrimitiveVSUboInfo = &m_Shader.LookupUniformBlockInfo("PrimitiveVsUniformBlock");
m_pModelViewInfo = &m_pPrimitiveVSUboInfo->LookupMemberInfo("modelView");
m_pModelViewInvTransposeInfo = &m_pPrimitiveVSUboInfo->LookupMemberInfo("modelViewInvTranspose");
m_pPrimitiveFSUboInfo = &m_Shader.LookupUniformBlockInfo("PrimitiveFsUniformBlock");
m_pMaterialAmbientInfo = &m_pPrimitiveFSUboInfo->LookupMemberInfo("material.ambient");
m_pMaterialDiffuseInfo = &m_pPrimitiveFSUboInfo->LookupMemberInfo("material.diffuse");
m_pMaterialSpecularInfo = &m_pPrimitiveFSUboInfo->LookupMemberInfo("material.specular");
m_pMaterialEmissionInfo = &m_pPrimitiveFSUboInfo->LookupMemberInfo("material.emission");
m_pMaterialSpecularExpInfo = &m_pPrimitiveFSUboInfo->LookupMemberInfo("material.specularExp");
// create VertexAttribState and VertexStreamState
const char* attributeNames[NumVertexAttributes] = { "attr_position", "attr_normal" };
SetupVertexAttribAndStreamStateInterleaved(attributeNames, NumVertexAttributes, m_Shader, m_VertexAttribStates, &m_VertexStreamState);
m_IsInitialized = true;
}
}
void SimplePrimitiveRenderer::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
for (PrimitiveData& primitiveData : m_Primitives)
{
primitiveData.GetVertexBuffer()->Finalize();
delete primitiveData.GetVertexBuffer();
}
for (MaterialData& mat : m_Materials)
{
mat.pUboBuffer->Finalize();
delete mat.pUboBuffer;
}
for (ModelViewTransformData& m : m_ModelViewTransforms)
{
m.pUboBuffer->Finalize();
delete m.pUboBuffer;
}
m_Primitives.clear();
m_Materials.clear();
m_ModelViewTransforms.clear();
m_GlobalVSUniformBuffer.Finalize();
m_GlobalFSUniformBuffer.Finalize();
m_pVertexMemoryPool->Finalize();
m_Shader.Finalize();
m_pGraphicsSystem->GetAllocator()->Free(m_GlslcData);
m_pGraphicsSystem = nullptr;
m_GlslcData = nullptr;
m_IsInitialized = false;
}
}

Material SimplePrimitiveRenderer::AddMaterial() NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
int id = static_cast<int>(m_Materials.size());
nvn::Buffer* uboBuf = m_pVertexMemoryPool->AllocateBuffer(m_pPrimitiveFSUboInfo->GetSize(), m_UniformAlignment);
char* uboBufAddr = static_cast<char*>(uboBuf->Map());
memset(uboBufAddr, 0, m_pPrimitiveFSUboInfo->GetSize());
MaterialData mat;
mat.pAmbient = reinterpret_cast<Color*>(uboBufAddr + m_pMaterialAmbientInfo->GetOffset());
mat.pDiffuse = reinterpret_cast<Color*>(uboBufAddr + m_pMaterialDiffuseInfo->GetOffset());
mat.pSpecular = reinterpret_cast<Color*>(uboBufAddr + m_pMaterialSpecularInfo->GetOffset());
mat.pEmission = reinterpret_cast<Color*>(uboBufAddr + m_pMaterialEmissionInfo->GetOffset());
mat.pSpecularExponent = reinterpret_cast<float*>(uboBufAddr + m_pMaterialSpecularExpInfo->GetOffset());
mat.pUboBuffer = uboBuf;
m_Materials.push_back(mat);
return Material(id);
}

void SimplePrimitiveRenderer::UpdateModelViewTransform(const ModelViewTransform& modelViewTransform, const nn::util::Matrix4x3fType& modelViewMtx) NN_NOEXCEPT
{

}
void SimplePrimitiveRenderer::UpdateModelViewTransform(const ModelViewTransform& modelViewTransform, const nn::util::Matrix4x4fType& modelViewMtx) NN_NOEXCEPT
{
// modelViewMtx data is copied into uniform memory
NN_ASSERT(m_IsInitialized);
NN_ASSERT(modelViewTransform >= 0 && modelViewTransform < static_cast<int>(m_ModelViewTransforms.size()));
ModelViewTransformData& modelViewTransformData = m_ModelViewTransforms[modelViewTransform];
LoadMatrix44(modelViewTransformData.pModelView, modelViewMtx);
nn::util::Matrix4x4fType inverseTranspose;
nn::util::Matrix4x4fType temp;
nn::util::MatrixInverse(&temp, modelViewMtx);
nn::util::MatrixTranspose(&inverseTranspose, temp);
LoadMatrix44(modelViewTransformData.pModelViewInvTranspose, inverseTranspose);
}
Primitive SimplePrimitiveRenderer::AddTriangle(const nn::util::Vector3f& p1, const nn::util::Vector3f& p2, const nn::util::Vector3f& p3) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
int id = static_cast<int>(m_Primitives.size());
size_t verticesSize = sizeof(Vertex) * 3;
size_t bufferSize = verticesSize + sizeof(uint32_t) * 3;
nvn::Buffer* pNewVertexBuffer = m_pVertexMemoryPool->AllocateBuffer(bufferSize);
char* vertexBufAddr = static_cast<char*>(pNewVertexBuffer->Map());
Vertex* vertices = reinterpret_cast<Vertex*>(vertexBufAddr);
uint32_t* indices = reinterpret_cast<uint32_t*>(vertexBufAddr + verticesSize);
nn::util::Vector3f normal;
ComputeNormal(&normal, p1, p2, p3);
vertices[0].SetPos(p1);
vertices[1].SetPos(p2);
vertices[2].SetPos(p3);
for (uint32_t i = 0; i < 3; ++i)
{
vertices[i].SetNormal(normal);
indices[i] = i;
}
PrimitiveData primitiveData;
primitiveData.SetVertexBuffer(pNewVertexBuffer);
primitiveData.SetIndicesCount(3);
primitiveData.SetBufferAddress(pNewVertexBuffer->GetAddress() + verticesSize);
m_Primitives.push_back(primitiveData);
return Primitive(id);
}
Primitive SimplePrimitiveRenderer::AddCone(float height, float radius, int numSlices) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT(height > 0.0f);
NN_ASSERT(radius > 0.0f);
NN_ASSERT(numSlices > 2);
int id = static_cast<int>(m_Primitives.size());
int numTopVertices = 2 * numSlices; // triangular top: two unique vertices per slice (we need multiple top point vertices for different normals)
int numBottomVertices = numSlices + 1; // circular bottom: one unique vertex per slice, +1 for radial center point
int numVertices = numTopVertices + numBottomVertices;
size_t verticesSize = sizeof(Vertex) * numVertices;
int numIndices = (2 * numSlices) * 3; // total number of triangles = 2 * numSlices (an extra set is needed for the base)
size_t bufferSize = verticesSize + (numIndices * sizeof(uint32_t));
nvn::Buffer* pNewVertexBuffer = m_pVertexMemoryPool->AllocateBuffer(bufferSize);
char* vertexBufAddr = static_cast<char*>(pNewVertexBuffer->Map());
Vertex* vertices = reinterpret_cast<Vertex*>(vertexBufAddr);
uint32_t* indices = reinterpret_cast<uint32_t*>(vertexBufAddr + verticesSize);
// go through each slice, and set vertex positions, normals, and indices
float sliceAngle = 2.0f * nn::util::FloatPi / numSlices;
float dy = radius * radius / height; // vertical displacement for normal vectors on cone sides (equation obtained via similar triangles)
for (int i = 0; i < numSlices; ++i)
{
float theta = sliceAngle * (i + 1);
float dx = radius * cosf(theta);
float dz = radius * sinf(theta);
// top vertices (cone surface)
vertices[2 * i + 0].SetPos( 0, height / 2, 0);
vertices[2 * i + 1].SetPos(dx, -height / 2, dz);
nn::util::Vector3f normal(dx, dy, dz);
nn::util::VectorNormalize(&normal, normal);
vertices[2 * i + 0].SetNormal(normal);
vertices[2 * i + 1].SetNormal(normal);
indices[3 * i + 0] = 2 * i + 1;
indices[3 * i + 1] = 2 * i + 0;
indices[3 * i + 2] = (i == 0) ? (2 * numSlices - 1) : (2 * i - 1);
// bottom vertices (base surface)
int dv = numTopVertices;
int di = numIndices / 2;
vertices[dv + i].SetPos(dx, -height / 2, dz);
vertices[dv + i].SetNormal(0, -1, 0);
indices[di + (3 * i + 0)] = dv + i;
indices[di + (3 * i + 1)] = (i == 0) ? (dv + numSlices - 1) : (dv + i - 1);
indices[di + (3 * i + 2)] = numVertices - 1;
}
// final vertex (center point on base of cone)
vertices[numVertices - 1].SetPos(0, -height / 2, 0);
vertices[numVertices - 1].SetNormal(0, -1, 0);
// finally, add the primitive
PrimitiveData primitiveData;
primitiveData.SetVertexBuffer(pNewVertexBuffer);
primitiveData.SetIndicesCount(numIndices);
primitiveData.SetBufferAddress(pNewVertexBuffer->GetAddress() + verticesSize);
m_Primitives.push_back(primitiveData);
return Primitive(id);
}

void SimplePrimitiveRenderer::StartDrawing(nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT(pCommandBuffer != nullptr);
m_Shader.Bind(pCommandBuffer);
pCommandBuffer->BindUniformBuffer(
nvn::ShaderStage::VERTEX,
m_pGlobalVSUboInfo->GetBindingLocations(nvn::ShaderStage::VERTEX),
m_GlobalVSUniformBuffer.GetAddress(),
m_GlobalVSUniformBuffer.GetSize()
);
pCommandBuffer->BindUniformBuffer(nvn::ShaderStage::FRAGMENT,
m_pGlobalFSUboInfo->GetBindingLocations(nvn::ShaderStage::FRAGMENT),
m_GlobalFSUniformBuffer.GetAddress(),
m_GlobalFSUniformBuffer.GetSize())
;
pCommandBuffer->BindVertexAttribState(NumVertexAttributes, m_VertexAttribStates);
pCommandBuffer->BindVertexStreamState(1, &m_VertexStreamState);
}

void SimplePrimitiveRenderer::Draw(const Primitive& primitive, nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
NN_ASSERT(pCommandBuffer != nullptr);
PrimitiveData& primitiveData = m_Primitives[primitive];
NN_ASSERT(primitiveData.GetModelViewTransform() >= 0 && primitiveData.GetModelViewTransform() < static_cast<int>(m_ModelViewTransforms.size()));
NN_ASSERT(primitiveData.GetMaterial() >= 0 && primitiveData.GetMaterial() < static_cast<int>(m_Materials.size()));
nvn::Buffer* pVsUboBuf = m_ModelViewTransforms[primitiveData.GetModelViewTransform()].pUboBuffer;
nvn::Buffer* pFsUboBuf = m_Materials[primitiveData.GetMaterial()].pUboBuffer;
pCommandBuffer->BindUniformBuffer(
nvn::ShaderStage::VERTEX,
m_pPrimitiveVSUboInfo->GetBindingLocations(nvn::ShaderStage::VERTEX),
pVsUboBuf->GetAddress(),
pVsUboBuf->GetSize()
);
pCommandBuffer->BindUniformBuffer(
nvn::ShaderStage::FRAGMENT,
m_pPrimitiveFSUboInfo->GetBindingLocations(nvn::ShaderStage::FRAGMENT),
pFsUboBuf->GetAddress(),
pFsUboBuf->GetSize()
);
// all of our shapes are composed to use the TRIANGLES primitive type
pCommandBuffer->DrawElements(nvn::DrawPrimitive::TRIANGLES, nvn::IndexType::UNSIGNED_INT, primitiveData.GetIndicesCount(), primitiveData.GetBufferAddress());
}
