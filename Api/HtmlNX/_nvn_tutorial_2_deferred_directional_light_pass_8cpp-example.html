/*
* DeferredDirectionalLightPass::CalculateMatrices
* -----------------------------------------------
* Creates the view and projection matrices used for
* rendering the shadow map.
*/
void DeferredDirectionalLightPass::CalculateShadowMatrices()
{
nn::util::Matrix4x3f temp;
nn::util::Vector3f target;
target.Set(m_LightPos.GetX() + m_LightDir.GetX(),
m_LightPos.GetY() + m_LightDir.GetY(),
m_LightPos.GetZ() + m_LightDir.GetZ());
/* View matrix */
nn::util::MatrixLookAtRightHanded(&temp, m_LightPos, target, nn::util::Vector3f(1.0f, 0.0f, 0.0f));
nn::util::MatrixConvert(&m_LightViewMatrix, temp);
/* Orthographics projection matrix */
nn::util::MatrixOrthographicOffCenterRightHanded(&m_LightProjMatrix,
-m_ShadowFrustumWidth / 2.0f,
m_ShadowFrustumWidth / 2.0f,
-m_ShadowFrustumHeight / 2.0f,
m_ShadowFrustumHeight / 2.0f,
m_ShadowNear,
m_ShadowFar);
}

/*
* DeferredDirectionalLightPass::UpdateUniforms
* --------------------------------------------
* Updates the lighting pass uniforms.
*/
void DeferredDirectionalLightPass::UpdateUniforms()
{
DeferredDirectionalLight::LightInfoUniformBlockData* uniformMapLightInfo = reinterpret_cast<DeferredDirectionalLight::LightInfoUniformBlockData*>(m_pLightUniformBlockLightInfo->GetMappedPointer());
uniformMapLightInfo->SetUniform_u_invScreenWidth(m_InvScreenWidth);
uniformMapLightInfo->SetUniform_u_invScreenHeight(m_InvScreenHeight);
uniformMapLightInfo->SetUniform_u_globalAmbient(m_GlobalAmbient);
float val[4];
nn::util::VectorStore(val, m_EyePos);
uniformMapLightInfo->SetUniform_u_eyePos(val);
nn::util::VectorStore(val, m_LightDir);
uniformMapLightInfo->SetUniform_u_lightDir(val);
nn::util::VectorStore(val, m_LightIntensity);
uniformMapLightInfo->SetUniform_u_lightIntensity(val);
uniformMapLightInfo->SetUniform_u_OutsideMapVal(m_OutsideShadowMapVal);
uniformMapLightInfo->SetUniform_u_invShadowMapWidth(1.0f / static_cast<float>(m_ShadowMapWidth));
uniformMapLightInfo->SetUniform_u_invShadowMapHeight(1.0f / static_cast<float>(m_ShadowMapHeight));
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, m_LightViewMatrix);
uniformMapLightInfo->SetUniform_u_lightViewMatrix(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_LightProjMatrix);
uniformMapLightInfo->SetUniform_u_lightProjMatrix(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_InvViewProj);
uniformMapLightInfo->SetUniform_u_invViewProj(*reinterpret_cast<float(*)[16]>(&temp));
DeferredDirectionalLight::TextureHandlesUniformBlockData* uniformMapTextureHandles = reinterpret_cast<DeferredDirectionalLight::TextureHandlesUniformBlockData*>(m_pLightUniformBlockTextureHandles->GetMappedPointer());
uniformMapTextureHandles->SetUniform_u_diffuseTex(m_DiffuseTex);
uniformMapTextureHandles->SetUniform_u_specularTex(m_SpecularTex);
uniformMapTextureHandles->SetUniform_u_normalTex(m_NormalTex);
uniformMapTextureHandles->SetUniform_u_shadowTex(m_ShadowTex);
uniformMapTextureHandles->SetUniform_u_depthTex(m_DepthTex);
}

/*
* DeferredDirectionalLightPass::Draw
* ----------------------------------
* Draws a screen space quad and calculates the lighting
* contribution of the direction light for each fragment.
*/
void DeferredDirectionalLightPass::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredDirectionalLight::LightInfoUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pLightUniformBlockLightInfo->GetCurrentBufferAddress(),
sizeof(DeferredDirectionalLight::LightInfoUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredDirectionalLight::TextureHandlesUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pLightUniformBlockTextureHandles->GetCurrentBufferAddress(),
sizeof(DeferredDirectionalLight::TextureHandlesUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
/*
* DeferredDirectionalLightPass::SetMatrices
* -----------------------------------------
* Calculates the inverse view*projection matrix used to
* reconstruct the world space position of a given fragment
* from the stored depth.
*/
void DeferredDirectionalLightPass::SetMatrices(const nn::util::Matrix4x4f& viewMtx, const nn::util::Matrix4x4f& projMtx)
{
nn::util::Matrix4x4f tempMat;
nn::util::MatrixMultiply(&tempMat, viewMtx, projMtx);
nn::util::MatrixInverse(&m_InvViewProj, tempMat);
}
