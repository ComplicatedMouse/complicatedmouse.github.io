void ControllerConnection()
{
var controllers = new Controller[4];
// Connect controllers.
controllers[0] = m_HidInputter.AddController(
new ControllerDeviceInfo(
ControllerDeviceType.JoyConLeft,
ControllerInterfaceType.Bluetooth,
Color.FromArgb(0, 0, 0),
Color.FromArgb(255, 0, 0)));
controllers[1] = m_HidInputter.AddController(
new ControllerDeviceInfo(
ControllerDeviceType.JoyConRight,
ControllerInterfaceType.Bluetooth,
Color.FromArgb(0, 0, 0),
Color.FromArgb(255, 0, 0)));
controllers[2] = m_HidInputter.AddController(
new ControllerDeviceInfo(
ControllerDeviceType.JoyConLeft,
ControllerInterfaceType.Rail,
Color.FromArgb(0, 0, 0),
Color.FromArgb(255, 0, 0)));
controllers[3] = m_HidInputter.AddController(
new ControllerDeviceInfo(
ControllerDeviceType.JoyConRight,
ControllerInterfaceType.Rail,
Color.FromArgb(0, 0, 0),
Color.FromArgb(255, 0, 0)));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(3));
// Disconnect controllers.
for (int i = 0; i < 4; ++i)
{
m_HidInputter.ReleaseController(controllers[i]);
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
// Add the Switch Pro Controller.
for (int i = 0; i < controllers.Length; ++i)
{
controllers[i] = m_HidInputter.AddController(
new ControllerDeviceInfo(
ControllerDeviceType.SwitchProController,
ControllerInterfaceType.Bluetooth,
Color.FromArgb(0, 0, 0),
Color.FromArgb(255, 0, 0)));
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(3));
// Disconnect controllers.
foreach (var controller in controllers)
{
m_HidInputter.ReleaseController(controller);
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
// Add the Joy-Con (R).
for (int i = 0; i < controllers.Length; ++i)
{
controllers[i] = m_HidInputter.AddController(
new ControllerDeviceInfo(
ControllerDeviceType.JoyConRight,
ControllerInterfaceType.Bluetooth,
Color.FromArgb(0, 0, 0),
Color.FromArgb(255, 0, 0)));
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(3));
// Disconnect controllers.
foreach (var controller in controllers)
{
m_HidInputter.ReleaseController(controller);
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
// Add the Joy-Con (L).
for (int i = 0; i < controllers.Length; ++i)
{
controllers[i] = m_HidInputter.AddController(
new ControllerDeviceInfo(
ControllerDeviceType.JoyConLeft,
ControllerInterfaceType.Bluetooth,
Color.FromArgb(0, 0, 0),
Color.FromArgb(255, 0, 0)));
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(3));
// Disconnect controllers.
foreach (var controller in controllers)
{
m_HidInputter.ReleaseController(controller);
}
}
void PushButton()
{
var controllers = new Controller[4];
for (int i = 0; i < controllers.Length; ++i)
{
controllers[i] = m_HidInputter.AddController();
}
Console.WriteLine("Push buttons");
controllers[0].PushButton(Button.A, TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
controllers[0].PushButton(Button.A, TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
controllers[1].PushButton(Button.B, TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
controllers[2].PushButton(Button.Plus, TimeSpan.FromSeconds(1));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
controllers[3].PushButton(Button.Minus, TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(4));
controllers[0].PushButton(Button.Home, TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(4));
controllers[0].PushButton(Button.Home, TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(4));
controllers[0].PushButton(Button.Capture, TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
Console.WriteLine("Push some buttons simultaneously");
controllers[0].PushMultiButton(
new Button[] { Button.A, Button.B, Button.X, Button.Y },
TimeSpan.FromMilliseconds(120));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
Console.WriteLine("Push some buttons sequentially");
controllers[0].PushMultiButtonSeq(
new Button[]{
Button.A, Button.B, Button.X, Button.Y,
Button.Left, Button.Right, Button.Up, Button.Down,
Button.ZL, Button.ZR, Button.L, Button.R,
Button.Plus, Button.Minus, Button.StickL, Button.StickR
},
TimeSpan.FromMilliseconds(120), TimeSpan.FromMilliseconds(90));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
Console.WriteLine("Push some controllers' button by press-release pattern");
foreach (var controller in controllers)
{
controller.PressButton(Button.A);
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(3));
foreach (var controller in controllers)
{
controller.ReleaseButton(Button.A);
}
foreach (var controller in controllers)
{
m_HidInputter.ReleaseController(controller);
}
}
void StickControl()
{
var controllers = new Controller[4];
for (int i = 0; i < controllers.Length; ++i)
{
controllers[i] = m_HidInputter.AddController();
}
Console.WriteLine("Tilt an analog stick");
controllers[0].TiltAnalogStick(
Stick.L, new StickStatus(180, 1.0f), TimeSpan.FromSeconds(1));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
controllers[0].TiltAnalogStick(
Stick.L, new StickStatus(-90, 0.8f), TimeSpan.FromSeconds(2));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
controllers[0].TiltAnalogStick(
Stick.R, new StickStatus(90, 1.0f), TimeSpan.FromSeconds(1));
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(1));
Console.WriteLine("Tilt an analog stick all around");
for (int degree = 0; degree < 360; degree += 5)
{
controllers[0].HoldAnalogStick(
Stick.R, new StickStatus(degree, 1.0f));
System.Threading.Thread.Sleep(TimeSpan.FromMilliseconds(10));
}
controllers[0].ReleaseAnalogStick(Stick.R);
for (int degree = 0; degree > -360; degree -= 5)
{
controllers[0].HoldAnalogStick(
Stick.L, new StickStatus(degree, 0.5f));
System.Threading.Thread.Sleep(TimeSpan.FromMilliseconds(10));
}
controllers[0].ReleaseAnalogStick(Stick.L);
Console.WriteLine("Tilt multi controllers' sticks all around");
for (int degree = 0; degree < 360; degree += 5)
{
foreach (var controller in controllers)
{
controller.HoldAnalogStick(
Stick.R, new StickStatus(degree, 1.0f));
controller.HoldAnalogStick(
Stick.L, new StickStatus(-degree, 1.0f));
}
System.Threading.Thread.Sleep(TimeSpan.FromMilliseconds(10));
}
foreach (var controller in controllers)
{
m_HidInputter.ReleaseController(controller);
}
}

static void Main(string[] args)
{
try
{
var targets = new List<Target>();
try
{
var allTargets = TargetManager.GetTargets();
foreach (var target in allTargets)
{
if (target.GetIsConnected())
{
targets.Add(target);
}
}
}
catch (TmException e)
{
Console.WriteLine(e.Message);
Console.WriteLine(e.StackTrace);
return;
}
if (targets.Count == 0)
{
Console.WriteLine("Connected target was not found.");
return;
}
Console.WriteLine("Please select target.");
int count = 0;
foreach (var target in targets)
{
Console.WriteLine($"{count++} : " + target.GetSerialNumber());
}
var input = Console.ReadLine();
int index;
if (!Int32.TryParse(input, out index))
{
Console.WriteLine("Invalid input.");
return;
}
if (index >= 0 && index < targets.Count)
{
var serial = targets[index].GetSerialNumber();
using (var hidInputer = HidInputter.CreateForDevkit(serial))
{
var program = new Program(hidInputer);
program.RunSamples();
}
}
else
{
Console.WriteLine("Invalid number was input.");
}
}
catch (Exception e)
{
Console.WriteLine(e.Message);
}
}
void RunSamples()
{
ControllerConnection();
PushButton();
StickControl();
TouchControl();
MultiTouchControl();
}
}
