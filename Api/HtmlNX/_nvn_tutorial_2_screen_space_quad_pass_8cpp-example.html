#include <nvntutorial/ScreenSpaceQuadPass.h>
#include <nvn/nvn.h>
#include <nvn/nvn_FuncPtrInline.h>
#include <nvntutorial/UniformBufferManager.h>
#include <nvntutorial/ManagedUniformBuffer.h>
#include <nvntutorial/ShaderHeaders/ScreenSpaceQuadDataHelper.h>
#include <nvntutorial/AssetFileDataHolder.h>
/*
* ScreenSpaceQuadPass Constructor
* -------------------------------
* Sets up default member data values.
*/
ScreenSpaceQuadPass::ScreenSpaceQuadPass() : m_pUniformBlockFS(nullptr),
m_InvWidth(0.0f),
m_InvHeight(0.0f)
{
m_Name = "ScreenSpaceQuadPass";
}
/*
* ScreenSpaceQuadPass Destructor
* ------------------------------
* Empty destructor.
*/
ScreenSpaceQuadPass::~ScreenSpaceQuadPass()
{
}
/*
* ScreenSpaceQuadPass::Init
* -------------------------
* Sets the asset data to use for drawing.
*/
void ScreenSpaceQuadPass::Init(AssetFileDataHolder* pData)
{
m_pAssetData = pData;
}
/*
* ScreenSpaceQuadPass::SetTextureHandle
* -------------------------------------
* Sets the handle of the texture to apply to the screen space quad.
*/
void ScreenSpaceQuadPass::SetTextureHandle(const NVNtextureHandle& handle)
{
m_Handle = handle;
}
/*
* ScreenSpaceQuadPass::SetInvScreenSize
* -------------------------------------
* Sets the inverse screen size.
*/
void ScreenSpaceQuadPass::SetInvScreenSize(float invWidth, float invHeight)
{
m_InvWidth = invWidth;
m_InvHeight = invHeight;
}
/*
* ScreenSpaceQuadPass::SetupUniforms
* ----------------------------------
* Sets up the uniform buffer.
*/
void ScreenSpaceQuadPass::SetupUniforms(UniformBufferManager* pUniformBufferManager)
{
m_pUniformBlockFS = pUniformBufferManager->CreateUniformBuffer(sizeof(ScreenSpaceQuad::BlockSamplerFSUniformBlockData));
}
/*
* ScreenSpaceQuadPass::UpdateUniforms
* -----------------------------------
* Updates the uniform buffer values.
*/
void ScreenSpaceQuadPass::UpdateUniforms()
{
ScreenSpaceQuad::BlockSamplerFSUniformBlockData* uniformMapFS = reinterpret_cast<ScreenSpaceQuad::BlockSamplerFSUniformBlockData*>(m_pUniformBlockFS->GetMappedPointer());
uniformMapFS->SetUniform_u_bindlessTex(m_Handle);
uniformMapFS->SetUniform_u_invScreenWidth(m_InvWidth);
uniformMapFS->SetUniform_u_invScreenHeight(m_InvHeight);
}
/*
* ScreenSpaceQuadPass::Draw
* -------------------------
* Draws the texture quad.
*/
void ScreenSpaceQuadPass::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
ScreenSpaceQuad::BlockSamplerFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pUniformBlockFS->GetCurrentBufferAddress(),
sizeof(ScreenSpaceQuad::BlockSamplerFSUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
