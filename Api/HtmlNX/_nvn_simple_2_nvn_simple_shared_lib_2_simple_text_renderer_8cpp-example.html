Initialize(pGraphicsSystem, DefaultTextureMemorySize, DefaultTextureDescriptorCount, DefaultUniformMemorySize, DefaultSimpleTextShaderBinaryPath);

void SimpleTextRenderer::Initialize(GraphicsSystem* pGraphicsSystem, size_t textureMemorySize, int textureDescriptorCount, size_t uniformMemorySize, const char* shaderBinaryPath) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
NN_ASSERT(textureMemorySize > 0);
NN_ASSERT(uniformMemorySize > 0);
m_pGraphicsSystem = pGraphicsSystem;
// read and initialize shader
int64_t shaderBinFileSize = 0;
m_GlslcData = ReadFile(&shaderBinFileSize, shaderBinaryPath, m_pGraphicsSystem->GetAllocator());
m_Shader.Initialize(m_pGraphicsSystem, static_cast<GLSLCoutput*>(m_GlslcData));
m_pUboInfo = &m_Shader.LookupUniformBlockInfo("TextRendererUniforms");
m_pDimensionsVarInfo = &m_pUboInfo->LookupMemberInfo("dimensions");
m_pOffsetVarInfo = &m_pUboInfo->LookupMemberInfo("offset");
m_pTextColorVarInfo = &m_pUboInfo->LookupMemberInfo("textColor");
m_SamplerLocation = m_Shader.LookupUniformInfo("sampler").GetBindingLocations(nvn::ShaderStage::FRAGMENT);
// initialize texture manager and memory pool for vertex/uniform data
m_TextureManager.Initialize(
m_pGraphicsSystem,
nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED,
textureDescriptorCount,
textureMemorySize,
"SimpleTextRenderer TexMemPool"
);
m_MemoryPool.Initialize(
m_pGraphicsSystem,
nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED,
uniformMemorySize,
"SimpleTextRenderer UniformMemPool"
);
// prepare shared vertex buffer
size_t vertexBufSize = sizeof(Vertex) * 4;
m_MemoryPool.AllocateBuffer(&m_VertexBuffer, vertexBufSize);
Vertex* vertices = static_cast<Vertex*>(m_VertexBuffer.Map());
vertices[0].SetPos(0, 0);
vertices[1].SetPos(1, 0);
vertices[2].SetPos(1, -1);
vertices[3].SetPos(0, -1);
vertices[0].SetUv(0, 0);
vertices[1].SetUv(1, 0);
vertices[2].SetUv(1, 1);
vertices[3].SetUv(0, 1);
// prepare shared index buffer
size_t indexBufSize = sizeof(uint32_t) * 6;
m_MemoryPool.AllocateBuffer(&m_IndexBuffer, indexBufSize);
uint32_t* indices = static_cast<uint32_t*>(m_IndexBuffer.Map());
indices[0] = 0;
indices[1] = 1;
indices[2] = 2;
indices[3] = 0;
indices[4] = 2;
indices[5] = 3;
// prepare buffer to hold all uniform data
size_t usedSize = nn::util::align_up(m_MemoryPool.GetUsedSize(), GraphicsMemoryPool::DefaultAlignment);
size_t uniformAlignment = m_pGraphicsSystem->GetResourceInfo().uniformBufferAlignment;
size_t remainingSize = uniformMemorySize - usedSize - uniformAlignment;
m_MemoryPool.AllocateBuffer(&m_UniformPool, remainingSize, uniformAlignment);
m_UboSize = nn::util::align_up(m_pUboInfo->GetSize(), uniformAlignment);
m_MaxUboCount = remainingSize / m_UboSize;
m_UboCount = 0;
// create VertexAttribState and VertexStreamState
const char* attributeNames[NumVertexAttributes] = { "attr_position", "attr_uv" };
SetupVertexAttribAndStreamStateInterleaved(attributeNames, NumVertexAttributes, m_Shader, m_VertexAttribStates, &m_VertexStreamState);
// initialize render state
m_RenderState.Initialize();
m_RenderState.GetColorState()->SetBlendEnable(0, true);
m_RenderState.GetBlendState()->SetBlendTarget(0);
m_RenderState.GetBlendState()->SetBlendEquation(nvn::BlendEquation::ADD, nvn::BlendEquation::ADD);
m_RenderState.GetBlendState()->SetBlendFunc(nvn::BlendFunc::SRC_ALPHA, nvn::BlendFunc::ONE_MINUS_SRC_ALPHA, nvn::BlendFunc::SRC_ALPHA, nvn::BlendFunc::ONE_MINUS_SRC_ALPHA);
m_RenderState.GetDepthStencilState()->SetDepthTestEnable(false);
m_RenderState.GetDepthStencilState()->SetDepthWriteEnable(false);
m_RenderState.GetPolygonState()->SetCullFace(nvn::Face::NONE);
m_IsInitialized = true;
}
}
void SimpleTextRenderer::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
for (TextAreaData& textArea: m_TextAreas)
{
textArea.GetTextBuffer()->Finalize();
}
m_TextAreas.clear();
m_TextureManager.Finalize();
m_VertexBuffer.Finalize();
m_IndexBuffer.Finalize();
m_UniformPool.Finalize();
m_MemoryPool.Finalize();
m_Shader.Finalize();
m_UboSize = 0;
m_UboCount = 0;
m_MaxUboCount = 0;
m_IsInitialized = false;
}
}

TextArea SimpleTextRenderer::CreateTextArea(int width, int height, Font* pFont, int16_t fontSize, float lineSpacing) NN_NOEXCEPT
{
NN_ASSERT(width > 0);
NN_ASSERT(height > 0);
width = static_cast<int>(width * m_ScalingFactor);
height = static_cast<int>(height * m_ScalingFactor);
size_t textBufferSize = width * height;
TextAreaData textAreaData;
textAreaData.SetAreaWidth(width);
textAreaData.SetAreaHeight(height);
textAreaData.SetFont(pFont);
textAreaData.SetFontSize(static_cast<int16_t>(fontSize * m_ScalingFactor));
textAreaData.SetLineSpacing(lineSpacing);
textAreaData.SetTextureDirty(true);
textAreaData.SetTextBuffer(m_TextureManager.GetMemoryPool()->AllocateBuffer(textBufferSize));
memset(textAreaData.GetTextBuffer()->Map(), 0, textBufferSize);
nvn::TextureBuilder textureBuilder;
textureBuilder.SetDefaults()
.SetDevice(m_pGraphicsSystem->GetDevice())
.SetSize2D(width, height)
.SetTarget(nvn::TextureTarget::TARGET_2D)
.SetFormat(nvn::Format::R8); // we use the R8 single-component texel format, which allows us to both save data,
// as well as utilize the intensity value for each texel as our alpha component
size_t textureSize = textureBuilder.GetStorageSize();
size_t textureAlignment = textureBuilder.GetStorageAlignment();
ptrdiff_t textureOffset = 0;
void* textureCpuAddr = m_TextureManager.AllocateTextureMemory(&textureOffset, textureSize, textureAlignment);
textureBuilder.SetPackagedTextureData(textureCpuAddr);
textureBuilder.SetStorage(m_TextureManager.GetMemoryPool()->GetMemoryPool(), textureOffset);
nvn::SamplerBuilder samplerBuilder;
samplerBuilder.SetDefaults()
.SetDevice(m_pGraphicsSystem->GetDevice())
.SetMinMagFilter(nvn::MinFilter::LINEAR, nvn::MagFilter::LINEAR);
int textureId = m_TextureManager.AddTexture(textureBuilder);
int samplerId = m_TextureManager.AddSampler(samplerBuilder);
textAreaData.SetTextureHandle(m_TextureManager.RegisterTextureHandle(textureId, samplerId));
textAreaData.SetTexture(m_TextureManager.GetTexture(textureId));
int id = static_cast<int>(m_TextAreas.size());
m_TextAreas.push_back(textAreaData);
return TextArea(id);
}
void SimpleTextRenderer::ClearAllTextAreas() NN_NOEXCEPT
{
m_TextAreas.clear();
m_TextureManager.Clear();
}
void SimpleTextRenderer::AppendText(const TextArea& textArea, const char* text) NN_NOEXCEPT
{
NN_ASSERT(textArea >= 0 && textArea < static_cast<int>(m_TextAreas.size()));
TextAreaData& textAreaData = m_TextAreas[textArea];
AppendTextString(textAreaData, text);
}

void SimpleTextRenderer::ClearText(const TextArea& textArea) NN_NOEXCEPT
{
NN_ASSERT(textArea >= 0 && textArea < static_cast<int>(m_TextAreas.size()));
TextAreaData& textAreaData = m_TextAreas[textArea];
memset(textAreaData.GetTextBuffer()->Map(), 0, textAreaData.GetTextBuffer()->GetSize());
textAreaData.GetCursor()->Clear();
textAreaData.SetTextureDirty(true);
}
void SimpleTextRenderer::StartDrawing(nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
#if defined (NN_BUILD_TARGET_PLATFORM_WIN)
m_TextureManager.GetMemoryPool()->Flush();
#endif
// First clear out our uniform buffer memory state. Uniform data is loaded on demand
// every time DrawTextArea is called subsequently.
m_UboCount = 0;
// next, add common gpu commands
m_RenderState.Bind(pCommandBuffer);
m_Shader.Bind(pCommandBuffer);
pCommandBuffer->BindVertexAttribState(NumVertexAttributes, m_VertexAttribStates);
pCommandBuffer->BindVertexStreamState(1, &m_VertexStreamState);
BindVertexBuffer(0, &m_VertexBuffer, pCommandBuffer);
}
void SimpleTextRenderer::DrawTextArea(const TextArea& textArea, float x, float y, const Color& textColor, nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
NN_ASSERT(textArea >= 0 && textArea < static_cast<int>(m_TextAreas.size()));
TextAreaData& textAreaData = m_TextAreas[textArea];
// first, update the texture if necessary
#if defined (NN_BUILD_TARGET_PLATFORM_NX)
if (textAreaData.IsTextureDirty()) // on Windows, for some reason the texture appears to be cleared out each frame, so we always upload the texture instead
#endif
{
UploadTexture(textAreaData, pCommandBuffer);
textAreaData.SetTextureDirty(false);
}
// next, load the TextArea state into the next uniform block, bind the UBO and texture, and add the draw call
nvn::BufferAddress uboAddr = LoadNextUniformBlockSlot(textAreaData, x, y, textColor);
int32_t bindingLocation = m_pUboInfo->GetBindingLocations(nvn::ShaderStage::VERTEX);
pCommandBuffer->BindUniformBuffer(nvn::ShaderStage::VERTEX, bindingLocation, uboAddr, m_pUboInfo->GetSize());
m_TextureManager.SetTextureAndSamplerPools(pCommandBuffer);
pCommandBuffer->BindTexture(nvn::ShaderStage::FRAGMENT, m_SamplerLocation, textAreaData.GetTextureHandle());
int numIndices = static_cast<int>(m_IndexBuffer.GetSize() / sizeof(uint32_t));
pCommandBuffer->DrawElements(nvn::DrawPrimitive::TRIANGLES, nvn::IndexType::UNSIGNED_INT, numIndices, m_IndexBuffer.GetAddress());
}
void SimpleTextRenderer::SetCanvasDimensions(float width, float height) NN_NOEXCEPT
{
// the canvas dimensions affect the positioning and relative size of drawn text areas,
// but has no impact on how much memory is used internally
NN_ASSERT(width > 0.0f);
NN_ASSERT(height > 0.0f);
m_CanvasWidth = width;
m_CanvasHeight = height;
}
void SimpleTextRenderer::SetScalingFactor(float scalingFactor) NN_NOEXCEPT
{
NN_ASSERT(scalingFactor > 0.0f);
// The scaling factor applies to the following:
// - x and y arguments to DrawTextArea.
// - width, height arguments to CreateTextArea, as well as the resulting fontSize.
// Therefore, the scaling factor affects how much texture memory is used, as well
// as the resolution/processing cost of the font rasterization.
// This scaling factor can be utilized for example if you wish to use the same
// coordinates and dimensions assuming a 720p canvas, yet render text at 1080p resolution,
// which in this case would be achieved by using a scalingFactor of 1080/720 = 1.5 and a
// 1080p canvas size.
m_ScalingFactor = scalingFactor;
}

nvn::BufferAddress SimpleTextRenderer::LoadNextUniformBlockSlot(TextAreaData& textAreaData, float x, float y, const Color& color) NN_NOEXCEPT
{
ptrdiff_t offset = m_UboCount * m_UboSize;
char* uboAddr = static_cast<char*>(m_UniformPool.Map()) + offset;
Vector2* pDimensions = reinterpret_cast<Vector2*>(uboAddr + m_pDimensionsVarInfo->GetOffset());
Vector2* pPosition = reinterpret_cast<Vector2*>(uboAddr + m_pOffsetVarInfo->GetOffset());
Color* pTextColor = reinterpret_cast<Color*>(uboAddr + m_pTextColorVarInfo->GetOffset());
// since range of screen coordinate space is from -1 to 1 in both x and y directions, we apply a scale of 2.0
// the scaling factor was already applied to the dimensions of the TextArea
float w = 2.0f * (textAreaData.GetAreaWidth() / m_CanvasWidth);
float h = 2.0f * (textAreaData.GetAreaHeight() / m_CanvasHeight);
// we must apply the scaling factor to the position coordinates
float offsetX = (2.0f * m_ScalingFactor * x / m_CanvasWidth) - 1.0f;
float offsetY = 1.0f - (2.0f * m_ScalingFactor * y / m_CanvasHeight);
pDimensions->Set(w, h);
pPosition->Set(offsetX, offsetY);
*pTextColor = color;
++m_UboCount;
return m_UniformPool.GetAddress() + offset;
}
void SimpleTextRenderer::UploadTexture(TextAreaData& textAreaData, nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
nvn::CopyRegion copyRegion;
copyRegion.depth = 1;
copyRegion.width = textAreaData.GetAreaWidth();
copyRegion.height = textAreaData.GetAreaHeight();
copyRegion.xoffset = 0;
copyRegion.yoffset = 0;
copyRegion.zoffset = 0;
// the nvn::CommandBuffer::CopyBufferToTexture command converts a linear source buffer into the tiled texture format
pCommandBuffer->CopyBufferToTexture(textAreaData.GetTextBuffer()->GetAddress(), textAreaData.GetTexture(), nullptr, &copyRegion, nvn::CopyFlags::NONE);
}
void SimpleTextRenderer::AppendTextString(TextAreaData& textAreaData, const char* text) NN_NOEXCEPT
{
// Note that we save and restore the font size/line spacing properties in case the user is
// using the font separately from the SimpleTextRenderer class, but if this cost is undesirable, it
// can be removed easily below.
int16_t prevSize = textAreaData.GetFont()->GetSize();
float prevLineSpacing = textAreaData.GetFont()->GetLineSpacing();
textAreaData.GetFont()->SetSize(textAreaData.GetFontSize());
textAreaData.GetFont()->SetLineSpacing(textAreaData.GetLineSpacing());
// write out the texels of the string
int x = textAreaData.GetCursor()->GetX();
int y = textAreaData.GetCursor()->GetY();
Font* pTextFont = textAreaData.GetFont();
NN_ASSERT_NOT_NULL(pTextFont);
CursorPosition textCursor = pTextFont->RenderStringToTexture(textAreaData.GetTextBuffer()->Map(), textAreaData.GetAreaWidth(), textAreaData.GetAreaHeight(), text, x, y);
textAreaData.SetCursor(textCursor);
textAreaData.SetTextureDirty(true);
// restore size/line spacing
pTextFont->SetSize(prevSize);
pTextFont->SetLineSpacing(prevLineSpacing);
}
