// Start the main loop.
void MainLoop() NN_NOEXCEPT
{
CONSOLE_LOG("* MainLoop() start.\n");
NN_ABORT_UNLESS_RESULT_SUCCESS(g_WebSocket.Initialize());
g_WebSocket.SetConnection(&g_Connection);
g_WebSocket.SetHandshakeCompletionEventHandler(HandshakeCompletionEventHandler, nullptr);
g_WebSocket.SetErrorEventHandler(ErrorEventHandler, nullptr);
// In the sample, these are set only when the subprotocol is chat.
// g_WebSocket.SetOpenEventHandler(OpenEventHandler, nullptr);
// g_WebSocket.SetMessageEventHandler(MessageEventHandler, nullptr);
// g_WebSocket.SetCloseEventHandler(CloseEventHandler, nullptr);
NN_UTIL_SCOPE_EXIT
{
g_WebSocket.Finalize();
CONSOLE_LOG("* MainLoop() end.\n");
};
nn::os::ThreadType thread;
bool isRunning = false;
bool isSceneChanged = true;
nn::hid::NpadButtonSet prevButtons = {};
while (NN_STATIC_CONDITION(true))
{
nn::hid::NpadButtonSet buttons = {};
gui::input::GetButtons(&buttons);
nn::hid::NpadButtonSet triggerButtons = ~prevButtons & buttons;
prevButtons = buttons;
if (!isRunning)
{
if (isSceneChanged)
{
SetInfoBar(u8" +: Exit ↑↓: Scroll A: Enter");
isSceneChanged = false;
}
if (triggerButtons.Test<nn::hid::NpadButton::A>())
{
g_CommunicationEndRequestEvent.Clear();
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::CreateThread(&thread, CommunicationThread, nullptr,
g_CommunicationThreadStack, sizeof (g_CommunicationThreadStack), nn::os::DefaultThreadPriority));
nn::os::StartThread(&thread);
isRunning = true;
isSceneChanged = true;
}
}
else
{
if (isSceneChanged)
{
SetInfoBar(u8" +: Exit ↑↓: Scroll A: Send X: Leave");
isSceneChanged = false;
}
if (triggerButtons.Test<nn::hid::NpadButton::A>())
{
SendChatMessage();
}
else if (triggerButtons.Test<nn::hid::NpadButton::X>() || nn::os::TryWaitThread(&thread))
{
g_CommunicationEndRequestEvent.Signal();
nn::os::DestroyThread(&thread);
isRunning = false;
isSceneChanged = true;
}
}
if (buttons.Test<nn::hid::NpadButton::Plus>())
{
g_CommunicationEndRequestEvent.Signal();
break;
}
else if (buttons.Test<nn::hid::NpadButton::Up>())
{
gui::console::Scroll(0, 1);
}
else if (buttons.Test<nn::hid::NpadButton::Down>())
{
gui::console::Scroll(0, -1);
}
gui::console::ProcessFrame();
}
if (isRunning)
{
nn::os::DestroyThread(&thread);
}
}
// Communication thread.
void CommunicationThread(void*) NN_NOEXCEPT
{
CONSOLE_LOG("* CommunicationThread() start.\n");
NN_UTIL_SCOPE_EXIT
{
CONSOLE_LOG("* CommunicationThread() end.\n");
};
#if !defined(USE_HTCS)
while (NN_STATIC_CONDITION(true))
{
nn::nifm::SubmitNetworkRequest();
while (nn::nifm::IsNetworkRequestOnHold())
{
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(10));
if (g_CommunicationEndRequestEvent.TryWait())
{
nn::nifm::CancelNetworkRequest();
return;
}
}
if (nn::nifm::IsNetworkAvailable())
{
break;
}
if (!nn::nifm::HandleNetworkRequestErrorResult())
{
CONSOLE_LOG("! Network connection could not be established.\n");
return;
}
}
NN_UTIL_SCOPE_EXIT
{
nn::nifm::CancelNetworkRequest();
};
#endif
// Make it possible to finish immediately if a cancel request is received during the connection process.
g_WebSocket.SetCancellationFunction(
[](void*) -> bool
{
return g_CommunicationEndRequestEvent.TryWait();
},
nullptr);
// Connect to the server.
// Block until the connection process is complete.
nn::Result result = g_WebSocket.Open(Uri, {"chat", "superchat"});
if (result.IsFailure())
{
if (!nn::websocket::ResultOperationCanceled::Includes(result))
{
CONSOLE_LOG("! nn::websocket::WebSocket::Open() failed. e = 0x%08X\n", result.GetInnerValueForDebug());
}
}
while (g_WebSocket.IsAvailable())
{
g_WebSocket.Perform();
if (g_CommunicationEndRequestEvent.TryWait())
{
// After adding the close request to the send queue, nn::websocket::WebSocket::Perform() must continue to be called until the request is actually received by the server.
// For this reason, never end this loop immediately.
g_WebSocket.Close();
}
// Sleep here to avoid a busy loop.
// The lower the nn::websocket::WebSocket::Perform() call frequency, the less responsive the communication process becomes.
// Adjust the call frequency based on the application's processing load and communication load.
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(10));
}
}

// Send a message.
void SendChatMessage() NN_NOEXCEPT
{
if (!g_WebSocket.IsAvailable())
{
CONSOLE_LOG("! WebSocket is not available.\n");
return;
}
std::unique_ptr<char[]> message(GetKeyboardInputWithHeader());
if (!message)
{
return;
}
nn::Result result = g_WebSocket.SendText(message.get());
if (result.IsFailure())
{
CONSOLE_LOG("! nn::websocket::WebSocket::SendText() failed. e = 0x%08X\n", result.GetInnerValueForDebug());
}
}
// Handle events when the handshake is completed.
bool HandshakeCompletionEventHandler(const nn::websocket::HeaderList& headers, const char* pSubProtocol, void*) NN_NOEXCEPT
{
NN_LOG("Response Headers:\n");
for (auto&& pHeader : headers)
{
// The header format is name: value.
// There is one space after the colon, regardless of the actual server response. (Ensured by the library.)
NN_LOG(" - %s\n", pHeader);
// Include the colon in the search when searching for headers.
// We recommend using case-insensitive header names. (See RFC7230#section-3.2.)
if (nn::util::Strnicmp(pHeader, "Set-Cookie:", sizeof ("Set-Cookie:") - 1) == 0)
{
NN_LOG(" - Cookie => '%s'\n", pHeader + sizeof ("Set-Cookie:"));
}
}
if (pSubProtocol && nn::util::Strncmp(pSubProtocol, "chat", sizeof ("chat")) == 0)
{
g_WebSocket.SetOpenEventHandler(OpenEventHandler, nullptr);
g_WebSocket.SetMessageEventHandler(MessageEventHandler, nullptr);
g_WebSocket.SetCloseEventHandler(CloseEventHandler, nullptr);
return true;
}
else
{
// If you do not care about the server response, you can return false to drop the connection.
return false;
}
}
// Handle server connection events.
void OpenEventHandler(void*) NN_NOEXCEPT
{
CONSOLE_LOG("[ You have entered the chat room. ]\n");
}
// Handle message receipt events.
void MessageEventHandler(void* pBuffer, size_t size, bool isText, void*) NN_NOEXCEPT
{
if (!isText)
{
return;
}
const size_t CommandSize = 4;
if (size < CommandSize)
{
return;
}
const char* pCommand = reinterpret_cast<const char*>(pBuffer);
const char* pMessage = pCommand + CommandSize;
size_t messsageSize = size - CommandSize;
if (nn::util::Strncmp(pCommand, "MSG:", CommandSize) == 0)
{
CONSOLE_LOG("> %.*s\n", static_cast<int>(messsageSize), pMessage);
}
else if (nn::util::Strncmp(pCommand, "ENT:", CommandSize) == 0)
{
CONSOLE_LOG("[ \"%.*s\" has entered the chat room. ]\n", static_cast<int>(messsageSize), pMessage);
}
else if (nn::util::Strncmp(pCommand, "LVE:", CommandSize) == 0)
{
CONSOLE_LOG("[ \"%.*s\" has left the chat room. ]\n", static_cast<int>(messsageSize), pMessage);
}
}
// Handle server disconnection events.
void CloseEventHandler(uint16_t code, const char* pReason, size_t length, void*) NN_NOEXCEPT
{
NN_UNUSED(code);
NN_UNUSED(pReason);
NN_UNUSED(length);
CONSOLE_LOG("[ You have left the chat room. ]\n");
}

// Get keyboard input. (With "MSG:" header.)
std::unique_ptr<char[]> GetKeyboardInputWithHeader() NN_NOEXCEPT
{
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
nn::swkbd::ShowKeyboardArg showArg = {};
size_t workBufferSize = nn::swkbd::GetRequiredWorkBufferSize(false);
std::unique_ptr<nn::Bit8[]> workBuffer(new nn::Bit8[workBufferSize + nn::os::MemoryPageSize]);
showArg.workBuf = reinterpret_cast<void*>(nn::util::align_up(reinterpret_cast<uintptr_t>(workBuffer.get()), nn::os::MemoryPageSize));
showArg.workBufSize = workBufferSize;
nn::swkbd::MakePreset(&showArg.keyboardConfig, nn::swkbd::Preset_Default);
showArg.keyboardConfig.keyboardMode = nn::swkbd::KeyboardMode_Full;
showArg.keyboardConfig.isUseUtf8 = true;
size_t stringBufferSize = nn::swkbd::GetRequiredStringBufferSize();
std::unique_ptr<nn::Bit8[]> stringBuffer(new nn::Bit8[stringBufferSize + nn::os::MemoryPageSize]);
nn::swkbd::String outputString = {};
outputString.ptr = reinterpret_cast<void*>(nn::util::align_up(reinterpret_cast<uintptr_t>(stringBuffer.get()), nn::os::MemoryPageSize));
outputString.bufSize = stringBufferSize;
nn::Result result = nn::swkbd::ShowKeyboard(&outputString, showArg);
if (nn::swkbd::ResultCanceled::Includes(result))
{
return nullptr;
}
if (result.IsFailure())
{
CONSOLE_LOG("! nn::swkbd::ShowKeyboard() failed. e = 0x%08X\n", result.GetInnerValueForDebug());
return nullptr;
}
std::unique_ptr<char[]> input(new char[sizeof ("MSG:") + stringBufferSize]);
nn::util::SNPrintf(input.get(), sizeof ("MSG:") + stringBufferSize, "MSG:%s", outputString.ptr);
#else
std::unique_ptr<char[]> input(new char[sizeof ("MSG:Hello")]);
// Return this hard-coded string in environments where no software keyboard is available.
std::memcpy(input.get(), "MSG:Hello", sizeof ("MSG:Hello"));
#endif
return input;
}
