/*
* DeferredPointLightObject::Shutdown
* ----------------------------------
* If the light is a shadow caster, the shadow map and
* memory pool are cleaned up.
*/
void DeferredPointLightObject::Shutdown()
{
if (m_ShadowCasting)
{
nvnTextureFinalize(&m_ShadowMap.m_Texture);
if (m_pShadowMapMemoryPool)
{
m_pShadowMapMemoryPool->Shutdown();
delete m_pShadowMapMemoryPool;
m_pShadowMapMemoryPool = NULL;
}
}
}
/*
* DeferredPointLightObject::SetPointLightState
* --------------------------------------------
* Sets the render state for the lighting pass.
*/
void DeferredPointLightObject::SetPointLightState()
{
/* Disable depth test/write so that objects with the point light radius get lit */
nvnDepthStencilStateSetStencilTestEnable(&m_DepthStencilState, NVN_TRUE);
nvnDepthStencilStateSetStencilFunc(&m_DepthStencilState, NVNface::NVN_FACE_FRONT_AND_BACK, NVNstencilFunc::NVN_STENCIL_FUNC_EQUAL);
nvnDepthStencilStateSetStencilOp(&m_DepthStencilState, NVNface::NVN_FACE_FRONT_AND_BACK, NVNstencilOp::NVN_STENCIL_OP_ZERO, NVNstencilOp::NVN_STENCIL_OP_ZERO, NVNstencilOp::NVN_STENCIL_OP_ZERO);
nvnDepthStencilStateSetDepthTestEnable(&m_DepthStencilState, NVN_FALSE);
nvnDepthStencilStateSetDepthWriteEnable(&m_DepthStencilState, NVN_FALSE);
/* Cull front faces so that the light works when the camera is inside of its radius */
nvnPolygonStateSetFrontFace(&m_PolygonState, NVNfrontFace::NVN_FRONT_FACE_CCW);
nvnPolygonStateSetCullFace(&m_PolygonState, NVNface::NVN_FACE_FRONT);
nvnPolygonStateSetPolygonMode(&m_PolygonState, NVNpolygonMode::NVN_POLYGON_MODE_FILL);
/* Set the blend state so that the lighting contribution is added to the final target */
nvnBlendStateSetBlendFunc(&m_BlendState, NVNblendFunc::NVN_BLEND_FUNC_ONE, NVNblendFunc::NVN_BLEND_FUNC_ONE, NVNblendFunc::NVN_BLEND_FUNC_ONE, NVNblendFunc::NVN_BLEND_FUNC_ZERO);
nvnBlendStateSetBlendEquation(&m_BlendState, NVNblendEquation::NVN_BLEND_EQUATION_ADD, NVNblendEquation::NVN_BLEND_EQUATION_ADD);
nvnColorStateSetBlendEnable(&m_ColorState, 0, NVN_TRUE);
}

* DeferredPointLightObject::SetPointLightState

void DeferredPointLightObject::SetDebugDrawState()

}

{
nvnDepthStencilStateSetStencilTestEnable(&m_DepthStencilState, NVN_TRUE);
nvnDepthStencilStateSetStencilFunc(&m_DepthStencilState, NVNface::NVN_FACE_FRONT_AND_BACK, NVNstencilFunc::NVN_STENCIL_FUNC_ALWAYS);
nvnDepthStencilStateSetDepthTestEnable(&m_DepthStencilState, NVN_TRUE);
nvnDepthStencilStateSetDepthWriteEnable(&m_DepthStencilState, NVN_FALSE);
nvnDepthStencilStateSetStencilOp(&m_DepthStencilState, NVNface::NVN_FACE_FRONT, NVNstencilOp::NVN_STENCIL_OP_KEEP, NVNstencilOp::NVN_STENCIL_OP_DECR_WRAP, NVNstencilOp::NVN_STENCIL_OP_KEEP);
nvnDepthStencilStateSetStencilOp(&m_DepthStencilState, NVNface::NVN_FACE_BACK, NVNstencilOp::NVN_STENCIL_OP_KEEP, NVNstencilOp::NVN_STENCIL_OP_INCR_WRAP, NVNstencilOp::NVN_STENCIL_OP_KEEP);
nvnPolygonStateSetCullFace(&m_PolygonState, NVNface::NVN_FACE_NONE);
nvnPolygonStateSetPolygonMode(&m_PolygonState, NVNpolygonMode::NVN_POLYGON_MODE_FILL);
}

/*
* DeferredPointLightObject::UpdateUniforms
* ----------------------------------------
* Updates the uniforms for the lighting pass.
*/
void DeferredPointLightObject::UpdateUniforms()
{
DeferredPointLight::LightInfoUniformBlockData* uniformMapLightInfo = reinterpret_cast<DeferredPointLight::LightInfoUniformBlockData*>(m_pLightUniformBlockLightInfo->GetMappedPointer());
uniformMapLightInfo->SetUniform_u_radius(m_Radius);
uniformMapLightInfo->SetUniform_u_invRadius(1.0f / m_Radius);
uniformMapLightInfo->SetUniform_u_invScreenHeight(m_InvScreenHeight);
uniformMapLightInfo->SetUniform_u_invScreenWidth(m_InvScreenWidth);
float val[4];
nn::util::VectorStore(val, m_EyePos);
uniformMapLightInfo->SetUniform_u_eyePos(val);
nn::util::VectorStore(val, m_LightPos);
uniformMapLightInfo->SetUniform_u_centerLightPos(val);
nn::util::VectorStore(val, m_Color * m_Intensity);
uniformMapLightInfo->SetUniform_u_lightIntensity(val);
PointLightDebug::ColorUniformBlockData* colorMap = reinterpret_cast<PointLightDebug::ColorUniformBlockData*>(m_pDebugColorBlock->GetMappedPointer());
nn::util::VectorStore(val, m_Color);
colorMap->SetUniform_u_color(val);
DeferredPointLight::BlockModelUniformBlockData* uniformMapBlockVS = reinterpret_cast<DeferredPointLight::BlockModelUniformBlockData*>(m_pBlockModel->GetMappedPointer());
nn::util::MatrixIdentity(&m_WorldMatrix);
nn::util::Matrix4x3fType scale;
nn::util::MatrixIdentity(&scale);
nn::util::Vector3fType scaleVector;
nn::util::VectorSet(&scaleVector, m_Radius, m_Radius, m_Radius);
nn::util::MatrixSetScale(&scale, scaleVector);
nn::util::Matrix4x3fType translate;
nn::util::MatrixIdentity(&translate);
nn::util::Vector3fType translateVector;
nn::util::VectorSet(&translateVector, m_LightPos.GetX(), m_LightPos.GetY(), m_LightPos.GetZ());
nn::util::MatrixSetTranslate(&translate, translateVector);
nn::util::Matrix4x3fType tempMat;
nn::util::MatrixMultiply(&tempMat, scale, translate);
nn::util::MatrixConvert(&m_WorldMatrix, tempMat);
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, m_WorldMatrix);
uniformMapBlockVS->SetUniform_u_modelMtx(*reinterpret_cast<float(*)[16]>(&temp));
if (m_ShadowCasting)
{
DeferredPointLight::ShadowMapInfoUniformBlockData* shadowBufferMap = reinterpret_cast<DeferredPointLight::ShadowMapInfoUniformBlockData*>(m_pShadowUniform->GetMappedPointer());
shadowBufferMap->SetUniform_u_shadowMap(m_ShadowMap.m_TextureHandle);
shadowBufferMap->SetUniform_u_poissonScaleY(m_PoissonRadius * m_InvShadowWidth);
shadowBufferMap->SetUniform_u_poissonScaleX(m_PoissonRadius * m_InvShadowHeight);
shadowBufferMap->SetUniform_u_shadowCullDistance(m_ShadowCullDistance);
}
}

/*
* DeferredPointLightObject::Draw
* ------------------------------
* Lighting pass for the point light. The light is drawn as
* a front face culled sphere. The resulting color is blended
* into the current render target.
*/
void DeferredPointLightObject::Draw(NVNcommandBuffer* pCommandBuffer) const
{
/* Bind the uniform buffers. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_VERTEX,
DeferredPointLight::BlockModelUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX),
m_pBlockModel->GetCurrentBufferAddress(),
sizeof(DeferredPointLight::BlockModelUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_VERTEX,
DeferredPointLight::BlockViewProjUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX),
g_pBlockViewProj->GetCurrentBufferAddress(),
sizeof(DeferredPointLight::BlockViewProjUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredPointLight::LightInfoUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pLightUniformBlockLightInfo->GetCurrentBufferAddress(),
sizeof(DeferredPointLight::LightInfoUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredPointLight::TextureHandlesUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
g_pLightUniformBlockTextureHandles->GetCurrentBufferAddress(),
sizeof(DeferredPointLight::TextureHandlesUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredPointLight::InvViewProjUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
g_pBlockInvViewProj->GetCurrentBufferAddress(),
sizeof(DeferredPointLight::InvViewProjUniformBlockData));
if (m_ShadowCasting)
{
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredPointLight::ShadowMapInfoUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pShadowUniform->GetCurrentBufferAddress(),
sizeof(DeferredPointLight::ShadowMapInfoUniformBlockData));
}
NVNcommandHandle drawHandle = m_LightGeometry->GetCommandHandle(m_pAssetData->GetProgramData()[0]->m_pProgramName);
nvnCommandBufferCallCommands(pCommandBuffer, 1, &drawHandle);
}

/*
* DeferredPointLightObject::SetMatrices
* -------------------------------------
* Sets up the view, projection, and inverse view*projection
* matrices and copies them to the appropriate uniform buffers.
* Because these values are constant for the whole scene, these
* uniform buffers are shared by all point lights and updated
* once per frame.
*/
void DeferredPointLightObject::SetMatrices(const nn::util::Matrix4x4fType& viewMtx, const nn::util::Matrix4x4fType& projMtx)
{
g_CameraMatrix = viewMtx;
g_ProjMatrix = projMtx;
DeferredPointLight::BlockViewProjUniformBlockData* uniformMap = reinterpret_cast<DeferredPointLight::BlockViewProjUniformBlockData*>(g_pBlockViewProj->GetMappedPointer());
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, g_CameraMatrix);
uniformMap->SetUniform_u_viewMtx(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, g_ProjMatrix);
uniformMap->SetUniform_u_projMtx(*reinterpret_cast<float(*)[16]>(&temp));
DeferredPointLight::InvViewProjUniformBlockData* invMap = reinterpret_cast<DeferredPointLight::InvViewProjUniformBlockData*>(g_pBlockInvViewProj->GetMappedPointer());
nn::util::Matrix4x4f tempMat, tempMat2;
nn::util::MatrixMultiply(&tempMat, g_CameraMatrix, g_ProjMatrix);
nn::util::MatrixInverse(&tempMat2, tempMat);
nn::util::MatrixStore(&temp, tempMat2);
invMap->SetUniform_u_invViewProj(*reinterpret_cast<float(*)[16]>(&temp));
}

/*

* ----------------------------------------
* Check whether to ignore an object for shadowing by name.
*/

{

/*
* DeferredPointLightObject::SetupShadows
* --------------------------------------
* If the light is a shadow caster, setup the shadow map memory
* pool, cubemap texture, and matrices.
*/
void DeferredPointLightObject::SetupShadows(bool shadowCasting,
uint32_t width,
uint32_t height,
float sampleRadius,
float cullDist,
NVNdevice* pDevice,
TextureIDManager* pTextureIDManager)
{
m_ShadowCasting = shadowCasting;
if (m_ShadowCasting)
{
m_ShadowMapWidth = width;
m_ShadowMapHeight = height;
NVNtextureBuilder textureBuilder;
nvnTextureBuilderSetDevice(&textureBuilder, pDevice);
nvnTextureBuilderSetDefaults(&textureBuilder);
NVNsamplerBuilder samplerBuilder;
nvnSamplerBuilderSetDevice(&samplerBuilder, pDevice);
nvnSamplerBuilderSetDefaults(&samplerBuilder);
/* Get the size of the cubemap */
size_t shadowTargetSize = TextureUtil::GetTextureSize(NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT,
m_ShadowMapWidth,
m_ShadowMapHeight,
NVN_TEXTURE_TARGET_CUBEMAP,
NVN_FORMAT_DEPTH32F,
textureBuilder);
/* Create a memory pool of the appropriate size */
m_pShadowMapMemoryPool = new MemoryPool();
m_pShadowMapMemoryPool->Init(NULL,
shadowTargetSize,
NVN_MEMORY_POOL_FLAGS_CPU_NO_ACCESS_BIT |
NVN_MEMORY_POOL_FLAGS_GPU_CACHED_BIT |
NVN_MEMORY_POOL_FLAGS_COMPRESSIBLE_BIT,
pDevice);
/* Create the texture */
TextureUtil::CreateRenderTarget(NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT,
m_ShadowMapWidth,
m_ShadowMapHeight,
NVN_TEXTURE_TARGET_CUBEMAP,
NVN_FORMAT_DEPTH32F,
m_pShadowMapMemoryPool,
0,
textureBuilder,
samplerBuilder,
pTextureIDManager,
pDevice,
&m_ShadowMap,
false);
m_InvShadowHeight = 1.0f / m_ShadowMapHeight;
m_InvShadowWidth = 1.0f / m_ShadowMapWidth;
m_PoissonRadius = sampleRadius;
m_ShadowCullDistance = cullDist;
/* Create a texture view for each face of the cubemap */
for (int i = 0; i < 6; i++)
{
nvnTextureViewSetDefaults(&m_ShadowViews[i]);
nvnTextureViewSetTarget(&m_ShadowViews[i], NVN_TEXTURE_TARGET_2D);
nvnTextureViewSetLayers(&m_ShadowViews[i], i, 1);
}
/* Perspective projection matrix */
nn::util::MatrixPerspectiveFieldOfViewRightHanded(&m_ShadowProjMatrix,
90.0f * DEG_TO_RAD, static_cast<float>(m_ShadowMapWidth) / static_cast<float>(m_ShadowMapHeight),
0.01f,
m_Radius);
nn::util::Vector3f lightPos = GetLightPos();
nn::util::Vector3f directions[6] = { nn::util::Vector3f(1.0f, 0.0f, 0.0f),
nn::util::Vector3f(-1.0f, 0.0f, 0.0f),
nn::util::Vector3f(0.0f, 1.0f, 0.0f),
nn::util::Vector3f(0.0f, -1.0f, 0.0f),
nn::util::Vector3f(0.0f, 0.0f, 1.0f),
nn::util::Vector3f(0.0f, 0.0f, -1.0f) };
nn::util::Vector3f up[6] = { nn::util::Vector3f(0.0f, -1.0f, 0.0f),
nn::util::Vector3f(0.0f, -1.0f, 0.0f),
nn::util::Vector3f(0.0f, 0.0f, 1.0f),
nn::util::Vector3f(0.0f, 0.0f, -1.0f),
nn::util::Vector3f(0.0f, -1.0f, 0.0f),
nn::util::Vector3f(0.0f, -1.0f, 0.0f) };
/* Create a view matrix for each face */
for (uint32_t mapIdx = 0; mapIdx < 6; ++mapIdx)
{
nn::util::Vector3f target = lightPos + directions[mapIdx];
nn::util::Matrix4x3f temp;
nn::util::MatrixLookAtRightHanded(&temp, lightPos, target, up[mapIdx]);
nn::util::MatrixConvert(&m_ShadowCameraMatrices[mapIdx], temp);
}
}
}
/*

/*
* DeferredPointLightObject::BindShadowMapUniforms
* -----------------------------------------------
* Binds the uniform buffer for shadow map rendering.
*/
void DeferredPointLightObject::BindShadowMapUniforms(NVNcommandBuffer* pCommandBuffer)
{
if (m_ShadowCasting)
{
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
PointShadowMap::LightInfoUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pObjectShadowMapUniform->GetCurrentBufferAddress(),
sizeof(PointShadowMap::LightInfoUniformBlockData));
}
}
/*
* DeferredPointLightObject::UpdateShadowMapUniforms
* -------------------------------------------------
* Updates the shadow map uniform buffer.
*/
void DeferredPointLightObject::UpdateShadowMapUniforms()
{
if (m_ShadowCasting)
{
PointShadowMap::LightInfoUniformBlockData* uniformMapLightInfo = reinterpret_cast<PointShadowMap::LightInfoUniformBlockData*>(m_pObjectShadowMapUniform->GetMappedPointer());
float val[4];
nn::util::VectorStore(val, m_LightPos);
uniformMapLightInfo->SetUniform_u_lightPos(val);
uniformMapLightInfo->SetUniform_u_radius(m_Radius);
}
}
