/**************************************************************************************************
*
* GraphicsSystem:
*
* A simple NVN-based rendering framework suitable for demo applications. Not meant for
* multi-instance usage. Conducts tasks that are necessary for rendering graphics using NVN for
* the user, including the following:
*
* - Initializes nvn::Device, nvn::Queue, and nvn::Window objects
* - Generates render textures and commands for setting render textures per frame
* - Conducts basic render loop processing
* - Creates shared memory pool for GraphicsShader (nvn::Program) instances
* - Creates shared memory pool for GraphicsCommandBuffer (nvn::CommandBuffer) instances
*
**************************************************************************************************/
class GraphicsSystem
{
public:
GraphicsSystem() NN_NOEXCEPT :
m_IsInitialized(false),
m_pAllocator(nullptr),
m_QueueMemory(nullptr),
m_CurRenderTargetIndex(0),
m_RenderTargetWidth(0),
m_RenderTargetHeight(0),
m_ScreenWidth(0),
m_ScreenHeight(0),

{}
