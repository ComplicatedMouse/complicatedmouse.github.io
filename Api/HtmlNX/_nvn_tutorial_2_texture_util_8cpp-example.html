/*
* GetTextureSize
* --------------
* Gets the size of a theoretical texture given the
* list of parameters.
*/
size_t GetTextureSize(int flags,
int width,
int height,
NVNtextureTarget target,
NVNformat format,
NVNtextureBuilder& textureBuilder,
int levelCount)
{
nvnTextureBuilderSetDefaults(&textureBuilder);
nvnTextureBuilderSetFlags(&textureBuilder, flags);
nvnTextureBuilderSetSize2D(&textureBuilder, width, height);
nvnTextureBuilderSetTarget(&textureBuilder, target);
nvnTextureBuilderSetFormat(&textureBuilder, format);
nvnTextureBuilderSetLevels(&textureBuilder, levelCount);
return nvnTextureBuilderGetStorageSize(&textureBuilder);
}
/*
* CreateTexture
* -------------
* Creates a texture with the given paramters, memory
* pool, and pool offset.
*/
void CreateTexture(int flags,
int width,
int height,
NVNtextureTarget target,
NVNformat format,
MemoryPool* pool,
ptrdiff_t poolOffset,
NVNtextureBuilder& textureBuilder,
NVNtexture* texture,
NVNtextureSwizzle r,
NVNtextureSwizzle g,
NVNtextureSwizzle b,
NVNtextureSwizzle a,
int levelCount)
{
nvnTextureBuilderSetDefaults(&textureBuilder);
nvnTextureBuilderSetFlags(&textureBuilder, flags);
nvnTextureBuilderSetSize2D(&textureBuilder, width, height);
nvnTextureBuilderSetTarget(&textureBuilder, target);
nvnTextureBuilderSetFormat(&textureBuilder, format);
nvnTextureBuilderSetSwizzle(&textureBuilder, r, g, b, a);
nvnTextureBuilderSetLevels(&textureBuilder, levelCount);
nvnTextureBuilderSetStorage(&textureBuilder, pool->GetMemoryPool(), poolOffset);
if (nvnTextureInitialize(texture, &textureBuilder) == false)
{
NN_ASSERT(0, "Texture Initialize Failed");
}
}
/*
* CreateRenderTarget
* ------------------
* Creates a texture, sampler, and texture handle.
*/
void CreateRenderTarget(int flags,
int width,
int height,
NVNtextureTarget target,
NVNformat format,
MemoryPool* pool,
ptrdiff_t poolOffset,
NVNtextureBuilder& textureBuilder,
NVNsamplerBuilder& samplerBuilder,
TextureIDManager* pTextureIDManager,
const NVNdevice* pDevice,
RenderTargetData* targetData,
bool shadow,
NVNtextureSwizzle r,
NVNtextureSwizzle g,
NVNtextureSwizzle b,
NVNtextureSwizzle a,
int levelCount)
{
CreateTexture(flags,
width,
height,
target,
format,
pool,
poolOffset,
textureBuilder,
&(targetData->m_Texture),
r,
g,
b,
a,
levelCount);
targetData->m_TextureID = pTextureIDManager->RegisterTexture(&targetData->m_Texture);
/* Setup the texture with a shadow sampler */
if (shadow)
{
nvnSamplerBuilderSetCompare(&samplerBuilder, NVN_COMPARE_MODE_COMPARE_R_TO_TEXTURE, NVN_COMPARE_FUNC_LESS);
}
nvnSamplerBuilderSetWrapMode(&samplerBuilder, NVN_WRAP_MODE_CLAMP_TO_EDGE, NVN_WRAP_MODE_CLAMP_TO_EDGE, NVN_WRAP_MODE_CLAMP_TO_EDGE);
NVNboolean success = nvnSamplerInitialize(&(targetData->m_Sampler), &samplerBuilder);
NN_ASSERT(success, "Failed to initialize sampler\n");
NN_UNUSED(success);
targetData->m_SamplerID = pTextureIDManager->RegisterSampler(&(targetData->m_Sampler));
targetData->m_TextureHandle = nvnDeviceGetTextureHandle(pDevice, targetData->m_TextureID, targetData->m_SamplerID);
nvnSamplerBuilderSetDefaults(&samplerBuilder);
}
}
