class GraphicsContext
{
public:
GraphicsContext() NN_NOEXCEPT :
m_IsInitialized(false) {}
void Initialize() NN_NOEXCEPT;

{
return m_IsInitialized;
}
nn::vi::NativeWindowHandle GetNativeWindowHandle() NN_NOEXCEPT
{
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
return m_NativeWindow;
#else
return reinterpret_cast<nn::vi::NativeWindowHandle>(m_hWnd);
#endif
}
private:
bool m_IsInitialized;
nn::vi::Display* m_pDisplay;
nn::vi::Layer* m_pLayer;
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
nn::vi::NativeWindowHandle m_NativeWindow;
void* m_GraphicsMemory;
#if defined(USE_NVNGD)
void* m_DevtoolsMemory;
nn::mem::StandardAllocator m_DevtoolsAllocator;
#endif
#else
HINSTANCE m_hInstance;
HWND m_hWnd;
#endif
};
void GraphicsContext::Initialize() NN_NOEXCEPT
{
if (!m_IsInitialized)
{
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
//----------------------------------------------------------------------------
// NX platform
//----------------------------------------------------------------------------
nv::SetGraphicsAllocator(NvAllocate, NvFree, NvReallocate, nullptr);
#if defined(USE_NVNGD)
// 128 MB is the bare minimum amount of memory for NVNGD.
// Even for a minimal application such as NvnSimple01ScreenSpaceTriangles, 128 MB will not
// be enough to use NVNGD's shader debugging capabilities (an invalid memory access will occur
// in the NVNGD runtime, causing the NX application to crash), so we provide 512 MB as recommended
// in the NVNGD documentation, which is enough to make shader debugging work in the aforementioned
// application.
const int graphicsDevToolsMemSize = 512 * 1024 * 1024;
m_DevtoolsMemory = malloc(graphicsDevToolsMemSize);
NN_ASSERT(m_DevtoolsMemory != nullptr);
m_DevtoolsAllocator.Initialize(m_DevtoolsMemory, graphicsDevToolsMemSize);
nv::SetGraphicsDevtoolsAllocator(NvDevtoolsAllocate, NvDevtoolsFree, NvDevtoolsReallocate, &m_DevtoolsAllocator);
#endif
const int graphicsSystemMemSize = 8 * 1024 * 1024;
m_GraphicsMemory = aligned_alloc(4096, graphicsSystemMemSize);
NN_ASSERT(m_GraphicsMemory != nullptr);
nv::InitializeGraphics(m_GraphicsMemory, graphicsSystemMemSize);
nn::vi::Initialize();
nn::Result result = nn::vi::OpenDefaultDisplay(&m_pDisplay);
NN_ASSERT(result.IsSuccess());
result = nn::vi::CreateLayer(&m_pLayer, m_pDisplay);
NN_ASSERT(result.IsSuccess());
result = nn::vi::GetNativeWindow(&m_NativeWindow, m_pLayer);
#else
//----------------------------------------------------------------------------
// Windows platform
//----------------------------------------------------------------------------
m_hInstance = GetModuleHandle(nullptr);
NN_ASSERT(m_hInstance);
nn::vi::Initialize();
nn::Result result = nn::vi::OpenDefaultDisplay(&m_pDisplay);
NN_ASSERT(result.IsSuccess());
result = nn::vi::CreateLayer(&m_pLayer, m_pDisplay);
NN_ASSERT(result.IsSuccess());
result = nn::vi::GetNativeWindow(reinterpret_cast<nn::vi::NativeWindowHandle*>(&m_hWnd), m_pLayer);
#if defined(USE_NVNGD)
// When using latest version of Windows 10 and latest NVN drives,
// use this to force application to load the tool's library.
// See more detials in NVN Graphics Debugger's documents.
const char* libraryName = "NV_TargetGLInterceptor.dll";
if (!LoadLibraryA(libraryName))
{
NN_LOG("Load NV_TargetGLInterceptor.dll failed.\n");
}
#endif
#endif
m_IsInitialized = true;
}
}

void InitializeGraphicsContext(nvn::DeviceGetProcAddressFunc* pNvnDeviceGetProcAddr, nn::vi::NativeWindowHandle* pNativeWindowHandle) NN_NOEXCEPT
{
NN_ASSERT(pNvnDeviceGetProcAddr != nullptr);
NN_ASSERT(pNativeWindowHandle != nullptr);
// First, initialize the graphics context (things like vi display, nv API on NX, and openGL context on Windows)
// if that has not been done yet. Note that there is no finalize equivalent to this code.
// For one, calling nv::InitializeGraphics multiple times across the application lifetime results in a crash as of
// NintendoSDK/NX Addon 4.4.0, even if nv::FinalizeGraphics is called in between.
// Explicit finalization of the graphics context is not necessary for smooth application shutdown either on NX or on Windows.
g_Context.Initialize();
// Next, query the NVN device entry point (i.e. get the pointer to the function for obtaining pointers
// to all the other NVN Device API functions), and pass back the native window handle to the caller.
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
//----------------------------------------------------------------------------
// NX platform
//----------------------------------------------------------------------------
*pNativeWindowHandle = g_Context.GetNativeWindowHandle();
#if !defined(USE_NVNGD) || (NN_SDK_VERSION_MAJOR >= 5)
*pNvnDeviceGetProcAddr = reinterpret_cast<nvn::DeviceGetProcAddressFunc>(nvnBootstrapLoader("nvnDeviceGetProcAddress"));
#else
*pNvnDeviceGetProcAddr = reinterpret_cast<nvn::DeviceGetProcAddressFunc>(nvnDeviceGetToolsProcAddress(nullptr, "nvnDeviceGetProcAddress"));
#endif
#else
//----------------------------------------------------------------------------
// Windows platform
//----------------------------------------------------------------------------
*pNativeWindowHandle = reinterpret_cast<nn::vi::NativeWindowHandle>(g_Context.GetNativeWindowHandle());
*pNvnDeviceGetProcAddr = reinterpret_cast<nvn::DeviceGetProcAddressFunc>(nvnBootstrapLoader("nvnDeviceGetProcAddress"));
#endif
}
void BootstrapNvnDevice(nvn::Device* pDevice, nvn::DeviceGetProcAddressFunc nvnDeviceGetProcAddr) NN_NOEXCEPT
{
NN_ASSERT(g_Context.IsInitialized());
// initialize the first few initial functions that are needed to create an NVN device
nvnLoadCPPProcs(nullptr, nvnDeviceGetProcAddr);
// initialize the NVN device object
uint32_t flags = 0;
#if defined(USE_NVNGD)
flags = NVN_DEVICE_FLAG_DEBUG_ENABLE_LEVEL_2_BIT | NVN_DEVICE_FLAG_DEBUG_SKIP_CALLS_ON_ERROR_BIT;
#endif
nvn::DeviceBuilder deviceBuilder;
deviceBuilder.SetDefaults();
deviceBuilder.SetFlags(flags);
if (!pDevice->Initialize(&deviceBuilder))
{
NN_ASSERT(0, "nvnDeviceInitialize");
}
// after creating the device, load the rest of the CPP API
nvnLoadCPPProcs(pDevice, nvnDeviceGetProcAddr);
// we install the debug callback if NVNGD is being used, but note that it can be used without NVNGD as well
if (flags & NVN_DEVICE_FLAG_DEBUG_ENABLE_LEVEL_2_BIT)
{
pDevice->InstallDebugCallback(reinterpret_cast<nvn::DebugCallbackFunc>(NvnDebugLayerCallback), nullptr, true);
}
// query the API versions and make sure they are as expected
int majorVersion = 0;
int minorVersion = 0;
pDevice->GetInteger(nvn::DeviceInfo::API_MAJOR_VERSION, &majorVersion);
pDevice->GetInteger(nvn::DeviceInfo::API_MINOR_VERSION, &minorVersion);
if (majorVersion != NVN_API_MAJOR_VERSION || minorVersion < NVN_API_MINOR_VERSION)
{
NN_ASSERT(0, "NVN SDK not supported by current driver.");
}
}
void NvnDebugLayerCallback(
nvn::DebugCallbackSource::Enum source,
nvn::DebugCallbackType::Enum type,
int id,
nvn::DebugCallbackSeverity::Enum severity,
const char * message,
void* userParam) NN_NOEXCEPT
{
NN_ASSERT(userParam == nullptr);
NN_LOG("NVN Debug Layer Callback:\n");
NN_LOG(" source: 0x%08x\n", source);
NN_LOG(" type: 0x%08x\n", type);
NN_LOG(" id: 0x%08x\n", id);
NN_LOG(" severity: 0x%08x\n", severity);
NN_LOG(" message: %s\n", message);
NN_UNUSED(source);
NN_UNUSED(type);
NN_UNUSED(id);
NN_UNUSED(severity);
NN_UNUSED(message);
NN_UNUSED(userParam);
}
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
void* NvAllocate(size_t size, size_t alignment, void* userPtr) NN_NOEXCEPT
{
NN_UNUSED(userPtr);
// According to specifications of aligned_alloc(), we need to coordinate the size parameter to become the integral multiple of alignment.
return aligned_alloc(alignment, nn::util::align_up(size, alignment));
}

{
NN_UNUSED(userPtr);
free(addr);
}
void* NvReallocate(void* addr, size_t newSize, void* userPtr) NN_NOEXCEPT
{
NN_UNUSED(userPtr);
return realloc(addr, newSize);
}

void NvDevtoolsFree(void* addr, void* userPtr) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(userPtr);

}
void* NvDevtoolsReallocate(void* addr, size_t newSize, void* userPtr) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(userPtr);
return reinterpret_cast<nn::mem::StandardAllocator*>(userPtr)->Reallocate(addr, newSize);
}
#endif // USE_NVNGD
#endif // NN_BUILD_TARGET_PLATFORM_NX
