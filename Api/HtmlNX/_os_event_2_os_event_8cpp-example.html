namespace {
const size_t ThreadStackSize = 8192; // Stack size of the event operation thread.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack1[ ThreadStackSize ]; // The first event operation thread stack.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack2[ ThreadStackSize ]; // The second event operation thread stack.
nn::os::EventType g_Event; // Operation target event.
} // Namespace.
//-----------------------------------------------------------------------------
//
// First thread (side waiting for the event).
//
void ThreadFunction1(void *arg)
{
NN_UNUSED(arg);
NN_LOG("This is Event sample code.\n");
nn::os::WaitEvent( &g_Event );
NN_LOG("First signal.\n");
nn::os::WaitEvent( &g_Event );
NN_LOG("Second signal.\n");
nn::os::WaitEvent( &g_Event );
NN_LOG("Third signal.\n");
nn::os::WaitEvent( &g_Event );
NN_LOG("Fourth signal.\n");
nn::os::WaitEvent( &g_Event );
NN_LOG("Fifth signal.\n");
nn::os::WaitEvent( &g_Event );
NN_LOG("Sixth signal.\n");
}
//
// Second thread (side giving notification of the event).
//
void ThreadFunction2(void *arg)
{
NN_UNUSED(arg);
nn::os::SleepThread( nn::TimeSpan::FromMilliSeconds(100) );
nn::os::SignalEvent( &g_Event );
nn::os::SleepThread( nn::TimeSpan::FromMilliSeconds(500) );
nn::os::SignalEvent( &g_Event );
nn::os::SleepThread( nn::TimeSpan::FromMilliSeconds(100) );
nn::os::SignalEvent( &g_Event );
nn::os::SleepThread( nn::TimeSpan::FromMilliSeconds(500) );
nn::os::SignalEvent( &g_Event );
nn::os::SleepThread( nn::TimeSpan::FromMilliSeconds(100) );
nn::os::SignalEvent( &g_Event );
nn::os::SleepThread( nn::TimeSpan::FromMilliSeconds(500) );
nn::os::SignalEvent( &g_Event );
}
//
// The main function.
//
extern "C" void nnMain()
{
nn::os::ThreadType thread1;
nn::os::ThreadType thread2;
nn::Result result;
// Initialize the event.
nn::os::InitializeEvent( &g_Event, false, nn::os::EventClearMode_AutoClear );
// Generate threads.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread1, ThreadFunction1, NULL, g_ThreadStack1, ThreadStackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread2, ThreadFunction2, NULL, g_ThreadStack2, ThreadStackSize, nn::os::DefaultThreadPriority ) );
// Start thread execution.
nn::os::StartThread( &thread1 );
nn::os::StartThread( &thread2 );
// Wait for the thread to end.
nn::os::WaitThread( &thread1 );
nn::os::WaitThread( &thread2 );
// Free threads.
nn::os::DestroyThread( &thread1 );
nn::os::DestroyThread( &thread2 );
// Finalize the event.
nn::os::FinalizeEvent( &g_Event );
}
