/*
* BoundingVolume Constructor
* --------------------------
* Sets up default member data values.
*/
BoundingVolume::BoundingVolume() : m_NumIndices(0),
m_VertexBufferSize(0),
m_pBlockViewProj(nullptr),
m_pBlockModel(nullptr),
m_pBlockColor(nullptr)
{
nn::util::MatrixIdentity(&m_WorldMatrix);
nn::util::MatrixIdentity(&m_CameraMatrix);
nn::util::MatrixIdentity(&m_ProjectionMatrix);
/* Enable depth testing but disable writing */
nvnDepthStencilStateSetDepthTestEnable(&m_DepthStencilState, NVN_TRUE);
nvnDepthStencilStateSetDepthWriteEnable(&m_DepthStencilState, NVN_FALSE);
}

BoundingVolume::~BoundingVolume()
{
}

/*
* BoundingVolume::Shutdown
* ------------------------
* Cleans up the vertex/index buffers and the memory pool
*/
void BoundingVolume::Shutdown()
{
nvnBufferFinalize(&m_VertexBuffer);
nvnBufferFinalize(&m_IndexBuffer);
m_VertexIndexBufferPool.Shutdown();
}
/*
* BoundingVolume::SetupVertexIndexBuffer
* --------------------------------------
* Sets up the vertex and index buffer.
*/
void BoundingVolume::SetupVertexIndexData(NVNdevice* pDevice,
const std::vector<nn::util::Vector3f>& vertexPositionData,
const std::vector<uint32_t>& indexData)
{
m_NumIndices = static_cast<int>(indexData.size());
m_VertexBufferSize = vertexPositionData.size() * sizeof(nn::util::Vector3f);
size_t indexBufferSize = indexData.size() * sizeof(uint32_t);
size_t vertexIndexMemoryPoolSize = m_VertexBufferSize + indexBufferSize;
/* Align the size of the memory pool */
if (vertexIndexMemoryPoolSize < g_MinimumPoolSize)
{
vertexIndexMemoryPoolSize = g_MinimumPoolSize;
}
else
{
vertexIndexMemoryPoolSize = static_cast<uint32_t>(Align(vertexIndexMemoryPoolSize, NVN_MEMORY_POOL_STORAGE_GRANULARITY));
}
/* Allocate the memory for the vertex/index buffer pool */
char* volumeDataBuffer = reinterpret_cast<char*>(AlignedAllocate(vertexIndexMemoryPoolSize, NVN_MEMORY_POOL_STORAGE_ALIGNMENT));
memcpy(volumeDataBuffer, vertexPositionData.data(), m_VertexBufferSize);
memcpy(volumeDataBuffer + m_VertexBufferSize, indexData.data(), indexBufferSize);
/* Wrap the memory pool around the allocated vertex/index memory */
m_VertexIndexBufferPool.Init(volumeDataBuffer,
vertexIndexMemoryPoolSize,
NVN_MEMORY_POOL_FLAGS_CPU_NO_ACCESS_BIT | NVN_MEMORY_POOL_FLAGS_GPU_CACHED_BIT,
pDevice);
NVNbufferBuilder bufferBuilder;
nvnBufferBuilderSetDevice(&bufferBuilder, pDevice);
nvnBufferBuilderSetDefaults(&bufferBuilder);
nvnBufferBuilderSetStorage(&bufferBuilder, m_VertexIndexBufferPool.GetMemoryPool(), 0, m_VertexBufferSize);
if (!nvnBufferInitialize(&m_VertexBuffer, &bufferBuilder))
{
NN_ASSERT(0, "Failed to initialize vertex buffer storage");
}
nvnBufferBuilderSetDefaults(&bufferBuilder);
nvnBufferBuilderSetStorage(&bufferBuilder, m_VertexIndexBufferPool.GetMemoryPool(), m_VertexBufferSize, indexBufferSize);
if (!nvnBufferInitialize(&m_IndexBuffer, &bufferBuilder))
{
NN_ASSERT(0, "Failed to initialize index buffer storage");
}
nvnVertexAttribStateSetDefaults(&m_VertexAttribStates[0]);
nvnVertexAttribStateSetStreamIndex(&m_VertexAttribStates[0], 0);
nvnVertexAttribStateSetFormat(&m_VertexAttribStates[0], NVN_FORMAT_RGB32F, 0);
nvnVertexStreamStateSetDefaults(&m_VertexStreamStates[0]);
nvnVertexStreamStateSetStride(&m_VertexStreamStates[0], sizeof(nn::util::Vector3f));
}

/*
* BoundingVolume::UpdateUniforms
* ------------------------------
* Updates the uniform buffers.
*/
void BoundingVolume::UpdateUniforms()
{
PointLightDebug::BlockViewProjUniformBlockData* uniformViewProjMap = reinterpret_cast<PointLightDebug::BlockViewProjUniformBlockData*>(m_pBlockViewProj->GetMappedPointer());
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, m_CameraMatrix);
uniformViewProjMap->SetUniform_u_viewMtx(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_ProjectionMatrix);
uniformViewProjMap->SetUniform_u_projMtx(*reinterpret_cast<float(*)[16]>(&temp));
PointLightDebug::BlockModelUniformBlockData* uniformModelMap = reinterpret_cast<PointLightDebug::BlockModelUniformBlockData*>(m_pBlockModel->GetMappedPointer());
nn::util::MatrixStore(&temp, m_WorldMatrix);
uniformModelMap->SetUniform_u_modelMtx(*reinterpret_cast<float(*)[16]>(&temp));
PointLightDebug::ColorUniformBlockData* uniformColorMap = reinterpret_cast<PointLightDebug::ColorUniformBlockData*>(m_pBlockColor->GetMappedPointer());
float val[4];
nn::util::VectorStore(val, m_Color);
uniformColorMap->SetUniform_u_color(val);
}
/*
* BoundingVolume::Draw
* --------------------
* Binds the vertex attributes, states and buffers of the volume then makes a draw call.
*/
void BoundingVolume::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Set vertex attributes and states. */
nvnCommandBufferBindVertexAttribState(pCommandBuffer, 1, &m_VertexAttribStates[0]);
nvnCommandBufferBindVertexStreamState(pCommandBuffer, 1, &m_VertexStreamStates[0]);
/* Bind the vertex buffer. */
nvnCommandBufferBindVertexBuffer(pCommandBuffer, 0, nvnBufferGetAddress(&m_VertexBuffer), m_VertexBufferSize);
/* Bind the uniform buffers. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_VERTEX,
PointLightDebug::BlockModelUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX),
m_pBlockModel->GetCurrentBufferAddress(),
sizeof(PointLightDebug::BlockModelUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_VERTEX,
PointLightDebug::BlockViewProjUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX),
m_pBlockViewProj->GetCurrentBufferAddress(),
sizeof(PointLightDebug::BlockViewProjUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
PointLightDebug::ColorUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockColor->GetCurrentBufferAddress(),
sizeof(PointLightDebug::ColorUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
nvnCommandBufferDrawElements(pCommandBuffer,
NVN_DRAW_PRIMITIVE_LINES,
NVN_INDEX_TYPE_UNSIGNED_INT,
m_NumIndices,
nvnBufferGetAddress(&m_IndexBuffer));
}
