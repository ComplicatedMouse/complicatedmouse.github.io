/*************************************************************************************************
*
* NvnSimpleTextureConverter.exe:
*
* A simple texture converter for NVN applications that tiles textures in an optimized format
* that is suitable for NX platform, built using the NVN Texture Packager.
*
* See TextureAsset.h for the format of the output texture asset.
*
* This tool can convert TGA and DDS file formats.
*
*************************************************************************************************/

void PrintUsage() NN_NOEXCEPT
{
printf("NvnSimpleTextureConverter: Converts a TGA/DDS image into an NVN hardware texture format.\n");
printf("Usage: NvnSimpleTextureConverter.exe <texture image path> -o <output image path>\n");
}
bool WriteToTextureAssetFile(const nvnTool::texpkg::NVNHWTexture& nvnHwTexture, const char* outputFilePath) NN_NOEXCEPT
{
// prepare TextureAsset data
size_t fileSize = TextureAsset::HeaderLength + static_cast<size_t>(nvnHwTexture.header.dataSize);
TextureAsset* pTextureAsset = static_cast<TextureAsset*>(malloc(fileSize));
memset(pTextureAsset, 0, TextureAsset::HeaderLength);
pTextureAsset->header.id = TextureAsset::HeaderId;
pTextureAsset->header.format = static_cast<nvn::Format::Enum>(nvnHwTexture.header.format);
pTextureAsset->header.target = static_cast<nvn::TextureTarget::Enum>(nvnHwTexture.header.target);
pTextureAsset->header.width = nvnHwTexture.header.width;
pTextureAsset->header.height = nvnHwTexture.header.height;
pTextureAsset->header.depth = nvnHwTexture.header.depth;
pTextureAsset->header.numMipMapLevels = nvnHwTexture.header.mipmapLevels;
pTextureAsset->header.texelsOffset = TextureAsset::HeaderLength;
pTextureAsset->header.texelsDataSize = static_cast<uint32_t>(nvnHwTexture.header.dataSize);
char* texels = reinterpret_cast<char*>(pTextureAsset) + pTextureAsset->header.texelsOffset;
memcpy(texels, nvnHwTexture.data, pTextureAsset->header.texelsDataSize);
// write TextureAsset to file
std::ofstream outputFile;
outputFile.open(outputFilePath, std::ofstream::out | std::ofstream::binary);
if (!outputFile.is_open())
{
printf("Could not open file %s for writing\n", outputFilePath);
return false;
}
// Note that we are writing out the byte stream as-is, so the endianness of the machine where
// this tool is being used must match the endianness of the target platform from which the output file
// will be loaded.
outputFile.write(reinterpret_cast<const char*>(pTextureAsset), fileSize);
outputFile.close();
free(pTextureAsset);
return true;
}

// main
//-------------------------------------------------------------------------------------------------
int main(int argc, char** argv)
{
if (argc != 4 || strcmp(argv[2], "-o") != 0)
{
printf("Unrecognized command arguments.\n");
PrintUsage();
return 1;
}
const char* inputFilePath = argv[1];
const char* outputFilePath = argv[3];
printf("NvnSimpleTextureConverter: converting %s into %s\n", inputFilePath, outputFilePath);
// load image library
nvnTool::texpkg::RawImageLib rawImageLib;
if (!LoadNVNImageLibrary(&rawImageLib, nvnImageDllName))
{
printf("LoadNvnImageLibrary failed.\n");
}
// load texture package library
nvnTool::texpkg::TexpkgLib texpkgLib;
if (!LoadTexpkgLibrary(&texpkgLib, nvnTexPkgDllName))
{
printf("NVN Texpkg Library failed to load\n");
return 1;
}
// create exporter to tile textures to hardware optimized format
nvnTool::texpkg::RawImageHardwareTextureExporter* pTextureExporter;
pTextureExporter = texpkgLib.createRawImageHardwareTextureExporter(&rawImageLib);
if (pTextureExporter == nullptr)
{
printf("Failed to create texture exporter\n");
return 1;
}
// create raw image and load image data into it
nvnTool::texpkg::RawImage* pRawImage = rawImageLib.createRawImage();
std::string filePath = inputFilePath;
{
// tga/dds formats are supported directly by TexturePackager library and its RawImageFormatManager class
std::wstring filename(filePath.begin(), filePath.end());
nvnTool::texpkg::RawImageFormatManager* pImageLoader;
pImageLoader = texpkgLib.createRawImageFormatManager(&rawImageLib, TEXPKG_NUM_DEFAULT_FORMAT_LOADERS,
nvnTool::texpkg::RawImageFormatDefaultHandlerLibraries);
if (pImageLoader == nullptr)
{
printf("Failed to create RawImageFormatManager\n");
return 1;
}
if (pImageLoader->Load(filename.c_str(), pRawImage) != TP_OK)
{
printf("Failed to load image: %s\n", inputFilePath);
PrintUsage();
return 1;
}
}
// convert raw image into hardware optimized format
nvnTool::texpkg::NVNHWTexture hwTexture;
if (pTextureExporter->Convert(pRawImage, &hwTexture) != TP_OK)
{
printf("Failed to convert image: %s\n", inputFilePath);
return 1;
}
// write hardware optimized texture to file as a TextureAsset
if (!WriteToTextureAssetFile(hwTexture, outputFilePath))
{
PrintUsage();
return 1;
}
return 0;
}
