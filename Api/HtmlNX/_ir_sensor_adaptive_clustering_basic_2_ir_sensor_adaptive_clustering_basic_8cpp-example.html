void CheckFirmwareUpdate(const nn::irsensor::IrCameraHandle& irCameraHandle)
{
const int FirmwareCheckTrialCountMax = 300;
auto counter = 0;
while (NN_STATIC_CONDITION(true))
{
auto isUpdateNeeded = false;
auto result = nn::irsensor::CheckFirmwareUpdateNecessity(&isUpdateNeeded, irCameraHandle);
if (result.IsSuccess())
{
// When successful, return a flag indicating whether a firmware update is required.
NN_LOG(" Controller firmware update checking succeeded.\n");
if (isUpdateNeeded)
{
// If a controller update is required, we recommend calling the update applet.
// The update applet updates all connected controllers in a block process, so be careful when calling the applet.
#if defined( NN_BUILD_CONFIG_OS_HORIZON )
// Call the forced update version of Controller Support without the selection UI.
// Call only once, regardless of success or failure.
nn::hid::ControllerFirmwareUpdateArg arg;
arg.enableForceUpdate = false;
nn::hid::ShowControllerFirmwareUpdate(arg);
#elif defined( NN_BUILD_CONFIG_OS_WIN )
NN_LOG("Calling ControllerFirmwareUpdate is not supported in the Windows environment.\n");
#else
#error "Unsupported operating system."
#endif
}
break;
}
else if (::nn::irsensor::ResultIrsensorUnavailable::Includes(result))
{
// Skip if the controller is not available for use.
NN_LOG(" Controller is not available.\n");
break;
}
else if (::nn::irsensor::ResultIrsensorFirmwareCheckIncompleted::Includes(result))
{
// Retry because FirmwareUpdate is being checked, or the applet is being called.
counter++;
if (counter > FirmwareCheckTrialCountMax)
{
// If the check continues for an extended time, we recommend timing out.
NN_LOG(" Firmware update checking timed out.\n");
break;
}
else
{
NN_LOG(" Firmware update checking is running.\n");
}
}
else
{
NN_ABORT(" Firmware update checking unexpected error\n");
}
::nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(15));
}
}
void PrintStatus(const nn::irsensor::IrCameraHandle& irCameraHandle, int64_t& latestSamplingNumber)
{
//Get the IR camera state.
auto irCameraStatus = nn::irsensor::GetIrCameraStatus(irCameraHandle);
//The IR camera is available.
if (irCameraStatus == nn::irsensor::IrCameraStatus_Available)
{
NN_LOG("The IR camera is available.\n");
}
//The controller is disconnected.
else if (irCameraStatus == nn::irsensor::IrCameraStatus_Unconnected)
{
NN_LOG("The IR camera is not connected.\n");
}
//The controller does not have an IR camera.
else if (irCameraStatus == nn::irsensor::IrCameraStatus_Unsupported)
{
NN_LOG("The IR camera is not supported.\n");
}
nn::irsensor::AdaptiveClusteringProcessorState state;
//Get the results from the adaptive clustering processor.
const auto DataCountMax = 1;
auto dataCount = 0;
auto result = nn::irsensor::GetAdaptiveClusteringProcessorStates(&state, &dataCount, DataCountMax, latestSamplingNumber, irCameraHandle);
if (nn::irsensor::ResultIrsensorDeviceNotReady::Includes(result))
{
// Data is still being prepared if a DeviceNotReady error is returned, so keep trying until proper data is obtained.
NN_LOG("Error: Data is not ready.\n");
return;
}
else if (nn::irsensor::ResultIrsensorUnsupported::Includes(result))
{
// If the controller does not have an IR Motion Camera,
// prompt the user to connect the appropriate controller.
NN_LOG("Error: Controller is unsupported.\n");
return;
}
else if (nn::irsensor::ResultIrsensorUnconnected::Includes(result))
{
// If a controller is not connected, we recommend prompting the user and retrying until a controller is connected.
NN_LOG("Error: Unconnected controller.\n");
return;
}
else if (nn::irsensor::ResultIrsensorDeviceError::Includes(result))
{
// If a DeviceError occurs, the IR Motion Camera cannot be used, and the processor stops.
// If this occurs, the detailed error contents are displayed in the error viewer or the controller update is displayed,
// and the user can attempt to resolve the issue.
NN_LOG("Error: DeviceError occured.\n");
isFinished = true;
return;
}
else if (nn::irsensor::ResultIrsensorDeviceResourceNotAvailable::Includes(result))
{
// If there is a device resource violation, we recommend that you either notify the user that the device
// cannot be used due to the number of connected wireless controllers, or suggest how to improve the situation.
NN_LOG("Error: DeviceResource is not available.\n");
return;
}
//Sampling number.
NN_LOG("SamplingNumber : %lld\n", state.samplingNumber);
latestSamplingNumber = state.samplingNumber;
//The time that has elapsed since the process started.
NN_LOG("TimeStamp: %lld us\n", state.timeStamp.GetMicroSeconds());
//Number of objects detected.
NN_LOG("ObjectCount: %d\n", state.objectCount);
//Object detection precision.
NN_LOG("AccuracyLevel: %s\n", AccuracyLevelText[state.accuracyLevel]);
//Environment noise state.
NN_LOG("AmbientNoiseLevel: %s\n", AmbientNoiseLevelText[state.ambientNoiseLevel]);
for (auto i = 0; i < state.objectCount; ++i)
{
//State of missing objects.
auto isIncomplete = state.objects[i].isIncomplete;
//Object centroid.
auto centroidX = state.objects[i].centroid.x;
auto centroidY = state.objects[i].centroid.y;
//Object area.
auto area = state.objects[i].area;
NN_LOG("Object(%d)\n", i);
NN_LOG(" IsIncomplete: %s\n", isIncomplete ? "True" : "False");
NN_LOG(" Centroid: (%3.1f, %3.1f)\n", centroidX, centroidY);
NN_LOG(" Area: %f\n", area);
}
NN_LOG("\n");
}
void Update()
{
for(auto i = 0; i < NpadIdCountMax; i++)
{
auto style = nn::hid::GetNpadStyleSet(npadIds[i]);
if (style.Test<nn::hid::NpadStyleJoyRight>())
{
nn::hid::NpadJoyRightState stateJoyRight;
nn::hid::GetNpadState(&stateJoyRight, npadIds[i]);
// End when the + Button is pressed.
if (stateJoyRight.buttons.Test<nn::hid::NpadButton::Plus>())
{
isFinished = true;
}
//If a button is being pressed, get and output the state and processing result of the IR Motion Camera.
else if (stateJoyRight.buttons.IsAnyOn())
{
NN_LOG("The right controller's buttons are pressed.\n");
PrintStatus(irCameraHandles[i], g_LatestSamplingNumber[i]);
}
}
if (style.Test<nn::hid::NpadStyleJoyLeft>())
{
nn::hid::NpadJoyLeftState stateJoyLeft;
nn::hid::GetNpadState(&stateJoyLeft, npadIds[i]);
// - End when a button is pressed.
if (stateJoyLeft.buttons.Test<nn::hid::NpadButton::Minus>())
{
isFinished = true;
}
//If a button is being pressed, get and output the state and processing result of the IR Motion Camera.
else if (stateJoyLeft.buttons.IsAnyOn())
{
NN_LOG("The left controller's buttons are pressed.\n");
PrintStatus(irCameraHandles[i], g_LatestSamplingNumber[i]);
}
}
if (style.Test<nn::hid::NpadStyleHandheld>())
{
nn::hid::NpadHandheldState stateHandheld;
nn::hid::GetNpadState(&stateHandheld, npadIds[i]);
// End when the + Button is pressed.
if (stateHandheld.buttons.Test<nn::hid::NpadButton::Plus>())
{
isFinished = true;
}
//If a button is being pressed, get and output the state and processing result of the IR Motion Camera.
else if (stateHandheld.buttons.IsAnyOn())
{
NN_LOG("The handheld controller's buttons are pressed.\n");
PrintStatus(irCameraHandles[i], g_LatestSamplingNumber[i]);
}
}
if (style.Test<nn::hid::NpadStyleFullKey>())
{
nn::hid::NpadFullKeyState stateFullKey;
nn::hid::GetNpadState(&stateFullKey, npadIds[i]);
// End when the + Button is pressed.
if (stateFullKey.buttons.Test<nn::hid::NpadButton::Plus>())
{
isFinished = true;
}
//If a button is being pressed, get and output the state and processing result of the IR Motion Camera.
else if (stateFullKey.buttons.IsAnyOn())
{
NN_LOG("The FullKey controller's buttons are pressed.\n");
PrintStatus(irCameraHandles[i], g_LatestSamplingNumber[i]);
}
}
}
}

extern "C" void nnMain()
{
NN_LOG("IR Sensor Adaptive Clustering Basic Sample(Npad) Start.\n");
nn::hid::InitializeNpad();
//Set the Npad play style.
nn::hid::SetSupportedNpadStyleSet(
::nn::hid::NpadStyleJoyRight::Mask
| ::nn::hid::NpadStyleJoyLeft::Mask
| ::nn::hid::NpadStyleHandheld::Mask
| ::nn::hid::NpadStyleFullKey::Mask
);
//Set which NpadId to use.
nn::hid::SetSupportedNpadIdType(npadIds, sizeof(npadIds) / sizeof(npadIds[0]));
for (auto i = 0; i < NpadIdCountMax; ++i)
{
// Assign one each.
nn::hid::SetNpadJoyAssignmentModeSingle(npadIds[i]);
//Get the IR camera handle.
irCameraHandles[i] = nn::irsensor::GetIrCameraHandle(npadIds[i]);
//Initialize the IR camera.
nn::irsensor::Initialize(irCameraHandles[i]);
NN_LOG("NpadPlayerNumber(%d)\n",npadIds[i]);
// Check FirmwareUpdate.
CheckFirmwareUpdate(irCameraHandles[i]);
}
NN_LOG("When you press any button, you can get IR sensor state.\n");
NN_LOG("Press the + Button or - Button to shut down this application.\n");
for (auto i = 0; i < NpadIdCountMax; ++i)
{
nn::irsensor::AdaptiveClusteringProcessorConfig config;
nn::irsensor::GetAdaptiveClusteringProcessorDefaultConfig(&config);
//Starting the adaptive clustering processor.
nn::irsensor::RunAdaptiveClusteringProcessor(irCameraHandles[i], config);
}
while(!isFinished)
{
Update();
::nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(15));
}
for (auto i = 0; i < NpadIdCountMax; ++i)
{
//Stopping the adaptive clustering processor.
nn::irsensor::StopImageProcessor(irCameraHandles[i]);
//Finalize the IR camera.
nn::irsensor::Finalize(irCameraHandles[i]);
}
NN_LOG("IR Sensor Adaptive Clustering Basic Sample Done.\n");
}
