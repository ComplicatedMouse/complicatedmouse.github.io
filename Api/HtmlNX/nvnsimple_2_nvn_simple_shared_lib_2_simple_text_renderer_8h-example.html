/**************************************************************************************************
*
* SimpleTextRenderer:
*
* A simple class that can be used to display text using Fonts to the screen.
*
* Client programs will create text areas based on a canvas size and Font, rasterize text into
* those areas through AppendText* methods, and then render the text areas through DrawTextArea.
*
* Printf-style formatted text is supported.
*
* Depends on SimpleTextRenderer.vs and SimpleTextRenderer.fs.
*
**************************************************************************************************/
class SimpleTextRenderer
{
public:
SimpleTextRenderer() NN_NOEXCEPT :
m_IsInitialized(false),
m_pGraphicsSystem(nullptr),
m_GlslcData(nullptr),
m_CanvasWidth(1280.0f),
m_CanvasHeight(720.0f),
m_UboSize(0),
m_UboCount(0),
m_MaxUboCount(0),
m_SamplerLocation(0),
m_ScalingFactor(1.0f) {}
void Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT;
void Initialize(GraphicsSystem* pGraphicsSystem, size_t textureMemorySize, int textureDescriptorCount, size_t uniformMemorySize, const char* shaderBinaryPath) NN_NOEXCEPT;
void Finalize() NN_NOEXCEPT;
TextArea CreateTextArea(int width, int height, Font* pFont) NN_NOEXCEPT;
TextArea CreateTextArea(int width, int height, Font* pFont, int16_t fontSize, float lineSpacing) NN_NOEXCEPT;
void ClearAllTextAreas() NN_NOEXCEPT;
void AppendText(const TextArea& textArea, const char* text) NN_NOEXCEPT;
void AppendTextPrintf(const TextArea& textArea, const char * format, ...) NN_NOEXCEPT;
void ClearText(const TextArea& textArea) NN_NOEXCEPT;
void StartDrawing(nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT;
void DrawTextArea(const TextArea& textArea, float x, float y, const Color& textColor, nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT;
void SetCanvasDimensions(float width, float height) NN_NOEXCEPT;
void SetScalingFactor(float scalingFactor) NN_NOEXCEPT;
float GetCanvasWidth() const NN_NOEXCEPT
{
return m_CanvasWidth;
}
float GetCanvasHeight() const NN_NOEXCEPT
{
return m_CanvasHeight;
}
float GetScalingFactor() const NN_NOEXCEPT

}
