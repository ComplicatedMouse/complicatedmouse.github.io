#pragma once
#include <vector>
#include <nn/mem.h>
#include <nvn/nvn_Cpp.h>
#include <nvn/nvn_CppMethods.h>
//-------------------------------------------------------------------------------------------------
// Forward declarations
//-------------------------------------------------------------------------------------------------
class GraphicsSystem;
class GraphicsMemoryPool;
/**************************************************************************************************
*
* GraphicsCommandBuffer:
*
* A lightweight wrapper for an nvn::CommandBuffer. Manages the details of allocating and adding
* command/control memory on demand through NVN's command buffer memory callback mechanism.
*
**************************************************************************************************/
class GraphicsCommandBuffer
{
public:
GraphicsCommandBuffer() NN_NOEXCEPT :
m_IsInitialized(false),
m_IsManualResetMode(false),
m_pGraphicsSystem(nullptr),
m_pMemoryPool(nullptr),
m_CommandMemChunkSize(0),
m_ControlMemChunkSize(0) {}
void Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT;
void Initialize(GraphicsSystem* pGraphicsSystem, bool manualResetMode) NN_NOEXCEPT;
void Initialize(GraphicsSystem* pGraphicsSystem, bool manualResetMode, size_t commandMemChunkSize) NN_NOEXCEPT;
void Finalize() NN_NOEXCEPT;
void Reset() NN_NOEXCEPT;
nvn::CommandBuffer* StartRecording() NN_NOEXCEPT;
nvn::CommandHandle EndRecording() NN_NOEXCEPT;
void EndRecordingAndExecute() NN_NOEXCEPT;
void PrintCommandBufferUsage() NN_NOEXCEPT;
nvn::CommandBuffer* GetCommandBuffer() NN_NOEXCEPT
{
return &m_CommandBuffer;
}
bool IsInitialized() const NN_NOEXCEPT
{
return m_IsInitialized;
}
friend void NVNAPIENTRY CommandBufferMemoryCallback(nvn::CommandBuffer* pCmdBuf, nvn::CommandBufferMemoryEvent::Enum event, size_t minSize, void* callbackData) NN_NOEXCEPT;
static const size_t DefaultCommandMemChunkSize = 1024;
private:
ptrdiff_t AllocateNewCommandMemChunk() NN_NOEXCEPT;
void* AllocateNewControlMemChunk() NN_NOEXCEPT;
bool AddNextCommandMemChunk() NN_NOEXCEPT;
bool AddNextControlMemChunk() NN_NOEXCEPT;
void MemoryCallbackHandler(nvn::CommandBufferMemoryEvent::Enum event, size_t minSize) NN_NOEXCEPT;
bool m_IsInitialized;
bool m_IsManualResetMode;
GraphicsSystem* m_pGraphicsSystem;
GraphicsMemoryPool* m_pMemoryPool;
nvn::CommandBuffer m_CommandBuffer;
size_t m_NumCommandMemChunksAdded;
size_t m_CommandMemChunkSize;
std::vector<ptrdiff_t> m_CommandMemChunkOffsets;
size_t m_NumControlMemChunksAdded;
size_t m_ControlMemChunkSize;
std::vector<void*> m_ControlMemChunks;
};
