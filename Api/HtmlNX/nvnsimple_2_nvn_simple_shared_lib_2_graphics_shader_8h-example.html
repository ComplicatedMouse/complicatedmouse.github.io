/**************************************************************************************************
*
* GraphicsShader:
*
* A lightweight wrapper for an nvn::Program object. Manages the details of nvn::Program
* creation based on shader program data compiled by GLSLC. Also provides reflection information
* for vertex attributes, uniform blocks, and inidividual uniforms (see Lookup* methods).
*
* See GraphicsShader::Initialize for implementation details on how nvn::Program is set up,
* and how reflection information is obtained and recorded.
*
* See NvnSimpleShaderCompiler project for details on how GLSLCoutput, which is used as the input for
* creating an instance of this class, is generated.
*
**************************************************************************************************/
class GraphicsShader
{
public:
GraphicsShader() NN_NOEXCEPT :
m_IsInitialized(false),
m_pGraphicsSystem(nullptr),
m_Program(),
m_NumShaders(0),
m_ShaderStageBits(0),
m_VertexAttributeMap(),
m_UniformBlockMap(),
m_UniformMap(),
m_RecommendedScratchMemSize(0)
{}
void Initialize(GraphicsSystem* pGraphicsSystem, const GLSLCoutput* pGlslcData) NN_NOEXCEPT;
void Finalize() NN_NOEXCEPT;
void Bind(nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT;
const VertexAttributeInfo& LookupVertexAttributeInfo(const char* attributeName) const NN_NOEXCEPT;
const UniformBlockInfo& LookupUniformBlockInfo(const char* uniformBlockName) const NN_NOEXCEPT;
const UniformInfo& LookupUniformInfo(const char* uniformName) const NN_NOEXCEPT;
size_t GetNumVertexAttributes() const NN_NOEXCEPT;
size_t GetNumUniformBlocks() const NN_NOEXCEPT;
size_t GetNumUniforms() const NN_NOEXCEPT;
size_t GetRecommendedScratchMemorySize() const NN_NOEXCEPT;
nvn::Program* GetProgram() NN_NOEXCEPT
{
return &m_Program;
}
nvn::ShaderStageBits GetShaderStageBits() const NN_NOEXCEPT
{
return m_ShaderStageBits;
}
private:
bool m_IsInitialized;
GraphicsSystem* m_pGraphicsSystem;
nvn::Program m_Program;
size_t m_NumShaders;
nvn::ShaderStageBits m_ShaderStageBits;
std::map<std::string, VertexAttributeInfo> m_VertexAttributeMap;
std::map<std::string, UniformBlockInfo> m_UniformBlockMap;
std::map<std::string, UniformInfo> m_UniformMap;
size_t m_RecommendedScratchMemSize;
};
