/*
* ScalableAOPass::UpdateUniforms
* ------------------------------
* Updates the uniforms.
*/
void ScalableAOPass::UpdateUniforms()
{
ScalableAO::BlockSamplerFSUniformBlockData* blockSamplerFS = reinterpret_cast<ScalableAO::BlockSamplerFSUniformBlockData*>(m_pBlockSamplerFS->GetMappedPointer());
blockSamplerFS->SetUniform_u_ViewSpaceDepthTex(m_ViewSpaceDepthTexHandle);
blockSamplerFS->SetUniform_u_Radius(m_Radius);
blockSamplerFS->SetUniform_u_IntensityDivR6(m_Intensity / std::powf(m_Radius, 6));
blockSamplerFS->SetUniform_u_Bias(m_Bias);
blockSamplerFS->SetUniform_u_ProjectionScale(m_ProjectionScale);
blockSamplerFS->SetUniform_u_PowerExponent(m_PowerExponent);
float temp[4];
nn::util::Float4x4 mat;
nn::util::MatrixStore(&mat, m_ProjMatrix);
temp[0] = -2.0f / (m_Width * mat.m[0][0]);
temp[1] = -2.0f / (m_Height * mat.m[1][1]);
temp[2] = (1.0f - mat.m[0][2]) / mat.m[0][0];
temp[3] = (1.0f + mat.m[1][2]) / mat.m[1][1];
blockSamplerFS->SetUniform_u_ProjectionInfo(temp);
}
/*
* ScalableAOPass::Draw
* --------------------
* Draws the SAO.
*/
void ScalableAOPass::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
ScalableAO::BlockSamplerFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockSamplerFS->GetCurrentBufferAddress(),
sizeof(ScalableAO::BlockSamplerFSUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
