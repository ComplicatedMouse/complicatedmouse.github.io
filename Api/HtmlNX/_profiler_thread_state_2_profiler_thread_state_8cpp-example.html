namespace
{
const size_t ThreadStackSize = 8192; // stack size of the thread
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack1[ThreadStackSize]; // first thread stack
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack2[ThreadStackSize]; // second thread stack
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack3[ThreadStackSize]; // third thread stack
int g_SharedResult = 0; // shared resource for the mutex
nn::os::MutexType g_Mutex; // operation target mutex
bool volatile g_ShouldExit = false; // flag to tell a thread to exit
}
int64_t CalculateFibonacci(int n)
{
int64_t retVal = n;
if (n > 1)
{
retVal = CalculateFibonacci(n - 1) + CalculateFibonacci(n - 2);
}
return retVal;
}
// Function run by each thread which will all attempt to access a shared resource
void ThreadSample(void *arg)
{
int64_t resultFromCalculations = 0;
int fibonacciSeed = 0;
while (!g_ShouldExit)
{
fibonacciSeed = (rand() % 5) + 15;
resultFromCalculations = CalculateFibonacci(fibonacciSeed);
if (resultFromCalculations != 0)
{
nn::os::LockMutex(&g_Mutex);
// Calculation that uses the shared resource
g_SharedResult = g_SharedResult % resultFromCalculations;
nn::os::UnlockMutex(&g_Mutex);
}
}
}
extern "C" void nnMain()
{
nn::os::ThreadType thread1;
nn::os::ThreadType thread2;
nn::os::ThreadType thread3;
nn::Result result;
srand(0);
// Initialize the mutex
nn::os::InitializeMutex(&g_Mutex, false, 0);
// Initialize the profiler
char* buffer = new char[nn::profiler::MinimumBufferSize];
result = nn::profiler::Initialize(buffer, nn::profiler::MinimumBufferSize);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Start thread 1 on core 0
result = nn::os::CreateThread(&thread1, ThreadSample, NULL, g_ThreadStack1, ThreadStackSize, nn::os::DefaultThreadPriority, 0);
NN_ASSERT(result.IsSuccess(), "Cannot create thread 1.");
nn::os::SetThreadName(&thread1, "Sample Thread 1");
// Start thread 2 on core 1
result = nn::os::CreateThread(&thread2, ThreadSample, NULL, g_ThreadStack2, ThreadStackSize, nn::os::DefaultThreadPriority, 1);
NN_ASSERT(result.IsSuccess(), "Cannot create thread 2.");
nn::os::SetThreadName(&thread2, "Sample Thread 2");
// Start thread 3 on core 2
result = nn::os::CreateThread(&thread3, ThreadSample, NULL, g_ThreadStack3, ThreadStackSize, nn::os::DefaultThreadPriority, 2);
NN_ASSERT(result.IsSuccess(), "Cannot create thread 3.");
nn::os::SetThreadName(&thread3, "Sample Thread 3");
// Start thread execution
nn::os::StartThread(&thread1);
nn::os::StartThread(&thread2);
nn::os::StartThread(&thread3);
const int totalFrameCount = 120 * 60;
for (int frame = 0; frame < totalFrameCount; ++frame)
{
auto startTime = nn::os::GetSystemTick();
// Record the top of our main loop. This is used as the framerate in the GUI.
nn::profiler::RecordHeartbeat(nn::profiler::Heartbeats_Main);
// Determine how long we have spent on this frame so far.
// Set the wait time so that we attempt to maintain a consistent framerate.
auto stopTime = nn::os::GetSystemTick();
int elapsedMicroSeconds = (stopTime - startTime).ToTimeSpan().GetMicroSeconds();
int waitTime = std::max(0, 16667 - elapsedMicroSeconds);
nn::os::SleepThread(nn::TimeSpan::FromMicroSeconds(waitTime));
}
// Each thread should now finish running
g_ShouldExit = true;
// Wait for the thread to be terminated
nn::os::WaitThread(&thread1);
nn::os::WaitThread(&thread2);
nn::os::WaitThread(&thread3);
// Destroy threads
nn::os::DestroyThread(&thread1);
nn::os::DestroyThread(&thread2);
nn::os::DestroyThread(&thread3);
// Finalize the profiler
result = nn::profiler::Finalize();
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
delete[] buffer;
// Finalize the mutex
nn::os::FinalizeMutex(&g_Mutex);
}
