/*
* ScreenSpaceQuadDepthPass Constructor
* ------------------------------------
* Sets up default member data values.
*/
ScreenSpaceQuadDepthPass::ScreenSpaceQuadDepthPass() : m_Near(0.0f),
m_Far(1.0f)
{
m_Name = "ScreenSpaceQuadDepthPass";
}
/*
* ScreenSpaceQuadDepthPass Destructor
* -----------------------------------
* Empty destructor.
*/
ScreenSpaceQuadDepthPass::~ScreenSpaceQuadDepthPass()
{
}
/*
* ScreenSpaceQuadDepthPass::SetNearFar
* ------------------------------------
* Sets the near and far plane values used to visualize
* the depth buffer.
*/
void ScreenSpaceQuadDepthPass::SetNearFar(float near, float far)
{
m_Near = near;
m_Far = far;
}
/*
* ScreenSpaceQuadDepthPass::SetupUniforms
* ---------------------------------------
* Sets up the uniform buffer.
*/
void ScreenSpaceQuadDepthPass::SetupUniforms(UniformBufferManager* pUniformBufferManager)
{
m_pUniformBlockFS = pUniformBufferManager->CreateUniformBuffer(sizeof(ScreenSpaceQuadDepth::BlockSamplerFSUniformBlockData));
}
/*
* ScreenSpaceQuadDepthPass::UpdateUniforms
* ----------------------------------------
* Updates the uniform buffer.
*/
void ScreenSpaceQuadDepthPass::UpdateUniforms()
{
ScreenSpaceQuadDepth::BlockSamplerFSUniformBlockData* uniformMapFS = reinterpret_cast<ScreenSpaceQuadDepth::BlockSamplerFSUniformBlockData*>(m_pUniformBlockFS->GetMappedPointer());
uniformMapFS->SetUniform_u_bindlessTex(m_Handle);
uniformMapFS->SetUniform_u_invScreenWidth(m_InvWidth);
uniformMapFS->SetUniform_u_invScreenHeight(m_InvHeight);
uniformMapFS->SetUniform_u_far(m_Far);
uniformMapFS->SetUniform_u_near(m_Near);
}
/*
* ScreenSpaceQuadDepthPass::Draw
* ------------------------------
* Draws a screen space quad that visualizes the values of the given
* depth texture.
*/
void ScreenSpaceQuadDepthPass::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
ScreenSpaceQuadDepth::BlockSamplerFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pUniformBlockFS->GetCurrentBufferAddress(),
sizeof(ScreenSpaceQuadDepth::BlockSamplerFSUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
