/*
* FrustumCullingPass::Shutdown
* ----------------------------
* Cleans up all dynamically allocated memory and NVN objects.
*/
void FrustumCullingPass::Shutdown()
{
if (m_pViewFrustumDebug)
{
m_pViewFrustumDebug->Shutdown();
AlignedDeallocate(m_pViewFrustumDebug);
m_pViewFrustumDebug = nullptr;
}
for (uint32_t i = 0; i < m_BoundingBoxes.size(); ++i)
{
if (m_BoundingBoxes[i])
{
m_BoundingBoxes[i]->Shutdown();
AlignedDeallocate(m_BoundingBoxes[i]);
}
}
m_BoundingBoxes.clear();
if (m_pInstancesBuffer)
{
nvnBufferFinalize(m_pInstancesBuffer);
delete m_pInstancesBuffer;
m_pInstancesBuffer = nullptr;
}
if (m_pCullingDebugBuffer)
{
nvnBufferFinalize(m_pCullingDebugBuffer);
delete m_pCullingDebugBuffer;
m_pCullingDebugBuffer = nullptr;
}
if (m_pInstancesPool)
{
m_pInstancesPool->Shutdown();
delete m_pInstancesPool;
m_pInstancesPool = nullptr;
}
if (m_pCullingDebugPool)
{
m_pCullingDebugPool->Shutdown();
delete m_pCullingDebugPool;
m_pCullingDebugPool = nullptr;
}
}
/*
* FrustumCullingPass::SetupInstanceData
* -------------------------------------
* Creates a storage buffer that will contain the AABBs and
* the model matrix for each object's mesh. The order of this
* buffer is the same as the indirect data buffer.
*/
void FrustumCullingPass::SetupInstanceData(NVNdevice* pDevice, std::map<std::string, PBRObject::ModelCulled>& models, std::vector<PBRObject::ObjectCulled>& sceneObjects)
{
std::vector<InstanceData> instances;
for (uint32_t i = 0; i < sceneObjects.size(); ++i)
{
sceneObjects[i].SetInstanceDataBufferOffset(static_cast<uint32_t>(instances.size() * sizeof(InstanceData)));
std::vector<PBRModelData::LodInfo> lodInfo = sceneObjects[i].GetLodInfo();
if (lodInfo.size() > 0)
{
for (uint32_t j = 0; j < lodInfo.size(); ++j)
{
std::string modelName = lodInfo[j].m_Name;
if (modelName == "")
continue;
const std::vector<PBRObject::Mesh*>& meshes = models.at(modelName).GetMeshes();
for (uint32_t k = 0; k < meshes.size(); ++k)
{
InstanceData instance;
const std::vector<float>& aabb = dynamic_cast<PBRObject::MeshCulled*>(meshes[k])->GetBoundingBox();
instance.aabbMin.Set(aabb[0], aabb[1], aabb[2], 1.0f);
instance.aabbMax.Set(aabb[3], aabb[4], aabb[5], 1.0f);
instance.modelMatrix = sceneObjects[i].GetWorldMatrix();
instances.push_back(instance);
}
}
}
else
{
std::string modelName = sceneObjects[i].GetModel();
const std::vector<PBRObject::Mesh*>& meshes = models.at(modelName).GetMeshes();
for (uint32_t j = 0; j < meshes.size(); ++j)
{
InstanceData instance;
const std::vector<float>& aabb = dynamic_cast<PBRObject::MeshCulled*>(meshes[j])->GetBoundingBox();
instance.aabbMin.Set(aabb[0], aabb[1], aabb[2], 1.0f);
instance.aabbMax.Set(aabb[3], aabb[4], aabb[5], 1.0f);
instance.modelMatrix = sceneObjects[i].GetWorldMatrix();
instances.push_back(instance);
}
}
}
m_InstancesBufferSize = instances.size() * sizeof(InstanceData);
/* The SSBO must have a properly aligned memory block to use. */
void* instancesData = AlignedAllocate(m_InstancesBufferSize, sizeof(nn::util::Vector4f));
memcpy(instancesData, instances.data(), m_InstancesBufferSize);
m_pInstancesPool = new MemoryPool;
m_pInstancesPool->Init(instancesData,
Align(m_InstancesBufferSize, sizeof(nn::util::Vector4f)),
NVN_MEMORY_POOL_FLAGS_CPU_UNCACHED_BIT |
NVN_MEMORY_POOL_FLAGS_GPU_CACHED_BIT,
pDevice);
m_pInstancesBuffer = new NVNbuffer;
NVNbufferBuilder bufferBuilder;
nvnBufferBuilderSetDevice(&bufferBuilder, pDevice);
nvnBufferBuilderSetDefaults(&bufferBuilder);
nvnBufferBuilderSetStorage(&bufferBuilder, m_pInstancesPool->GetMemoryPool(), 0, Align(m_InstancesBufferSize, sizeof(nn::util::Vector4f)));
if (!nvnBufferInitialize(m_pInstancesBuffer, &bufferBuilder))
{
NN_ASSERT(0, "Failed to initialize uniform block's buffer");
}
}

/*
* FrustumCullingPass::SetupDebugDraw
* ----------------------------------
* Sets up the bounding boxes for each mesh
* and the camera frustum.
*/
void FrustumCullingPass::SetupDebugDraw(NVNdevice* pDevice,
UniformBufferManager* pUniformBufferManager,
AssetFileDataHolder* pViewFrustumDebugDraw,
AssetFileDataHolder* pObjectDebugDraw,
std::map<std::string, PBRObject::ModelCulled>& models,
std::vector<PBRObject::ObjectCulled>& sceneObjects,
PBRObject::ModelCulled* pDebugCamera)
{
static std::vector<uint32_t> indexData = { 0,2, 2,3, 3,1, 1,0,
0,4, 1,5, 2,6, 3,7,
4,6, 6,7, 7,5, 5,4 };
/* Setup camera frustum shape */
std::vector<nn::util::Vector3f> vertexPositionData;
vertexPositionData.resize(8);
/* near face */
vertexPositionData[0].Set(1.0f, 1.0f, -1.0f);
vertexPositionData[1].Set(-1.0f, 1.0f, -1.0f);
vertexPositionData[2].Set(1.0f, -1.0f, -1.0f);
vertexPositionData[3].Set(-1.0f, -1.0f, -1.0f);
/* far face */
vertexPositionData[4].Set(1.0f, 1.0f, 1.0f);
vertexPositionData[5].Set(-1.0f, 1.0f, 1.0f);
vertexPositionData[6].Set(1.0f, -1.0f, 1.0f);
vertexPositionData[7].Set(-1.0f, -1.0f, 1.0f);
void* place = AlignedAllocate(sizeof(ViewFrustum), NN_ALIGNOF(ViewFrustum));
m_pViewFrustumDebug = new(place) ViewFrustum;
m_pViewFrustumDebug->Init(pViewFrustumDebugDraw);
m_pViewFrustumDebug->SetupVertexIndexData(pDevice, vertexPositionData, indexData);
m_pViewFrustumDebug->SetupUniforms(pUniformBufferManager);
/*
* Draw bounding box around base model and it's all meshes
* as the LODs models will have roughly the same bounding box.
*/
for (uint32_t i = 0; i < sceneObjects.size(); ++i)
{
const std::vector<PBRObject::Mesh* >& meshes = models.at(sceneObjects[i].GetModel()).GetMeshes();
for (uint32_t j = 0; j < meshes.size(); ++j)
{
const std::vector<float>& aabb = dynamic_cast<PBRObject::MeshCulled*>(meshes[j])->GetBoundingBox();
/* near face */
vertexPositionData[0].Set(aabb[3], aabb[4], aabb[2]);
vertexPositionData[1].Set(aabb[0], aabb[4], aabb[2]);
vertexPositionData[2].Set(aabb[3], aabb[1], aabb[2]);
vertexPositionData[3].Set(aabb[0], aabb[1], aabb[2]);
/* far face */
vertexPositionData[4].Set(aabb[3], aabb[4], aabb[5]);
vertexPositionData[5].Set(aabb[0], aabb[4], aabb[5]);
vertexPositionData[6].Set(aabb[3], aabb[1], aabb[5]);
vertexPositionData[7].Set(aabb[0], aabb[1], aabb[5]);
place = AlignedAllocate(sizeof(BoundingVolume), NN_ALIGNOF(BoundingVolume));
BoundingVolume* boundingBox = new(place) BoundingVolume;
boundingBox->Init(pObjectDebugDraw);
boundingBox->SetupVertexIndexData(pDevice, vertexPositionData, indexData);
boundingBox->SetupUniforms(pUniformBufferManager);
m_BoundingBoxes.push_back(boundingBox);
}
}
size_t instancesSize = m_InstancesBufferSize / sizeof(InstanceData);
m_CullingDebugBufferSize = instancesSize * sizeof(nn::util::Vector4f);
void* cullingDebug = AlignedAllocate(m_CullingDebugBufferSize, sizeof(nn::util::Vector4f));
memset(cullingDebug, 0, m_CullingDebugBufferSize);
m_pCullingDebugPool = new MemoryPool;
m_pCullingDebugPool->Init(cullingDebug,
Align(m_CullingDebugBufferSize, sizeof(nn::util::Vector4f)),
NVN_MEMORY_POOL_FLAGS_CPU_UNCACHED_BIT |
NVN_MEMORY_POOL_FLAGS_GPU_CACHED_BIT,
pDevice);
m_pCullingDebugBuffer = new NVNbuffer;
NVNbufferBuilder bufferBuilder;
nvnBufferBuilderSetDevice(&bufferBuilder, pDevice);
nvnBufferBuilderSetDefaults(&bufferBuilder);
nvnBufferBuilderSetStorage(&bufferBuilder, m_pCullingDebugPool->GetMemoryPool(), 0, Align(m_CullingDebugBufferSize, sizeof(nn::util::Vector4f)));
if (!nvnBufferInitialize(m_pCullingDebugBuffer, &bufferBuilder))
{
NN_ASSERT(0, "Failed to initialize uniform block's buffer");
}
m_pDebugDrawShader = pObjectDebugDraw;
m_pDebugCamera = pDebugCamera;
m_pBlockViewProjCamera = pUniformBufferManager->CreateUniformBuffer(sizeof(PointLightDebug::BlockViewProjUniformBlockData));
m_pBlockModelCamera = pUniformBufferManager->CreateUniformBuffer(sizeof(PointLightDebug::BlockModelUniformBlockData));
m_pBlockColorCamera = pUniformBufferManager->CreateUniformBuffer(sizeof(PointLightDebug::ColorUniformBlockData));
nvnPolygonStateSetDefaults(&m_DebugPolygonState);
nvnPolygonStateSetPolygonMode(&m_DebugPolygonState, NVN_POLYGON_MODE_LINE);
}

/*
* FrustumCullingPass::UpdateUniforms
* ----------------------------------
* Updates view and projection matrices needed by compute shader
* to transform AABBs.
*/
void FrustumCullingPass::UpdateUniforms()
{
FrustumCulling::ViewProjUniformBlockData* viewProj = reinterpret_cast<FrustumCulling::ViewProjUniformBlockData*>(m_pViewProj->GetMappedPointer());
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, m_CullingCameraInfo.m_Matrix);
viewProj->SetUniform_u_View(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_ProjectionMatrix);
viewProj->SetUniform_u_Projection(*reinterpret_cast<float(*)[16]>(&temp));
}
/*
* FrustumCullingPass::UpdateInstancesBuffer
* -----------------------------------------
* Updates the instance buffer with the models matrices
* and AABBs.
*/
void FrustumCullingPass::UpdateInstancesBuffer(std::map<std::string, PBRObject::ModelCulled>& models, std::vector<PBRObject::ObjectCulled>& sceneObjects)
{
InstanceData* instances = reinterpret_cast<InstanceData*>(nvnBufferMap(m_pInstancesBuffer));
nvnBufferInvalidateMappedRange(m_pInstancesBuffer, 0, m_InstancesBufferSize);
for (uint32_t i = 0; i < sceneObjects.size(); ++i)
{
uint32_t instanceIdx = sceneObjects[i].GetInstanceDataBufferOffset() / sizeof(InstanceData);
std::vector<PBRModelData::LodInfo> lodInfo = sceneObjects[i].GetLodInfo();
if (lodInfo.size() > 0)
{
for (uint32_t j = 0; j < lodInfo.size(); ++j)
{
std::string modelName = lodInfo[j].m_Name;
if (modelName == "")
continue;
const std::vector<PBRObject::Mesh*>& meshes = models.at(modelName).GetMeshes();
for (uint32_t k = 0; k < meshes.size(); ++k)
{
const std::vector<float>& aabb = dynamic_cast<PBRObject::MeshCulled*>(meshes[k])->GetBoundingBox();
instances[instanceIdx + j + k].aabbMin.Set(aabb[0], aabb[1], aabb[2], 1.0f);
instances[instanceIdx + j + k].aabbMax.Set(aabb[3], aabb[4], aabb[5], 1.0f);
instances[instanceIdx + j + k].modelMatrix = sceneObjects[i].GetWorldMatrix();
}
}
}
else
{
std::string modelName = sceneObjects[i].GetModel();
const std::vector<PBRObject::Mesh*>& meshes = models.at(modelName).GetMeshes();
for (uint32_t j = 0; j < meshes.size(); ++j)
{
const std::vector<float>& aabb = dynamic_cast<PBRObject::MeshCulled*>(meshes[j])->GetBoundingBox();
instances[instanceIdx + j].aabbMin.Set(aabb[0], aabb[1], aabb[2], 1.0f);
instances[instanceIdx + j].aabbMax.Set(aabb[3], aabb[4], aabb[5], 1.0f);
instances[instanceIdx + j].modelMatrix = sceneObjects[i].GetWorldMatrix();
}
}
}
}
/*
* FrustumCullingPass::UpdateDebugDraw
* -----------------------------------
* Updates MVP matrices of the bounding boxes and
* the camera frustum shape.
*/
void FrustumCullingPass::UpdateDebugDraw(std::map<std::string, PBRObject::ModelCulled>& models, std::vector<PBRObject::ObjectCulled>& sceneObjects)
{
/* Update the camera view frustum */
m_pViewFrustumDebug->SetCameraMatrix(m_CameraMatrix);
m_pViewFrustumDebug->SetCullingCameraMatrix(m_CullingCameraInfo.m_Matrix);
m_pViewFrustumDebug->SetProjectionMatrix(m_ProjectionMatrix);
m_pViewFrustumDebug->SetColor(nn::util::Vector4f(1.0f, 0.0f, 1.0f, 1.0f));
m_pViewFrustumDebug->UpdateUniforms();
/* Get the color values from compute shader culling debug buffer */
nn::util::Vector4f* bbColors = reinterpret_cast<nn::util::Vector4f*>(nvnBufferMap(m_pCullingDebugBuffer));
nvnBufferInvalidateMappedRange(m_pCullingDebugBuffer, 0, m_CullingDebugBufferSize);
uint32_t bbIdx = 0;
for (uint32_t i = 0; i < sceneObjects.size(); ++i)
{
uint32_t instanceIdx = sceneObjects[i].GetInstanceDataBufferOffset() / sizeof(InstanceData);
std::vector<PBRObject::Mesh* > meshes = models.at(sceneObjects[i].GetModel()).GetMeshes();
for (uint32_t j = 0; j < meshes.size(); ++j)
{
m_BoundingBoxes[bbIdx]->SetWorldMatrix(sceneObjects[i].GetWorldMatrix());
m_BoundingBoxes[bbIdx]->SetCameraMatrix(m_CameraMatrix);
m_BoundingBoxes[bbIdx]->SetProjectionMatrix(m_ProjectionMatrix);
m_BoundingBoxes[bbIdx]->SetColor(bbColors[instanceIdx + j]);
m_BoundingBoxes[bbIdx]->UpdateUniforms();
++bbIdx;
}
}
PointLightDebug::BlockViewProjUniformBlockData* uniformViewProjMap = reinterpret_cast<PointLightDebug::BlockViewProjUniformBlockData*>(m_pBlockViewProjCamera->GetMappedPointer());
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, m_CameraMatrix);
uniformViewProjMap->SetUniform_u_viewMtx(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_ProjectionMatrix);
uniformViewProjMap->SetUniform_u_projMtx(*reinterpret_cast<float(*)[16]>(&temp));
PointLightDebug::BlockModelUniformBlockData* uniformModelMap = reinterpret_cast<PointLightDebug::BlockModelUniformBlockData*>(m_pBlockModelCamera->GetMappedPointer());
nn::util::Matrix4x3fType tempCameraScale;
nn::util::MatrixIdentity(&tempCameraScale);
nn::util::MatrixSetScale(&tempCameraScale, nn::util::Vector3f(0.25f, 0.25f, 0.25f));
nn::util::Matrix4x3fType tempCameraRotateCamera;
nn::util::MatrixIdentity(&tempCameraRotateCamera);
nn::util::MatrixCameraRotateZxy(&tempCameraRotateCamera,
nn::util::Vector3f(0.0f, 0.0f, 0.0f),
nn::util::Vector3f(m_CullingCameraInfo.m_XRotation,
m_CullingCameraInfo.m_YRotation,
0.0f));
nn::util::Matrix4x3fType tempCameraRotate;
nn::util::MatrixTranspose(&tempCameraRotate, tempCameraRotateCamera);
nn::util::Matrix4x3fType tempCameraTranslate;
nn::util::MatrixIdentity(&tempCameraTranslate);
nn::util::MatrixSetTranslate(&tempCameraTranslate, m_CullingCameraInfo.m_Position - m_CullingCameraInfo.m_ViewVector);
nn::util::Matrix4x3fType tempMat1;
nn::util::Matrix4x3fType tempMat2;
nn::util::MatrixMultiply(&tempMat1, tempCameraScale, tempCameraRotate);
nn::util::MatrixMultiply(&tempMat2, tempMat1, tempCameraTranslate);
nn::util::Matrix4x4fType tempCameraModelMat4x4;
nn::util::MatrixConvert(&tempCameraModelMat4x4, tempMat2);
nn::util::MatrixStore(&temp, tempCameraModelMat4x4);
uniformModelMap->SetUniform_u_modelMtx(*reinterpret_cast<float(*)[16]>(&temp));
PointLightDebug::ColorUniformBlockData* uniformColorMap = reinterpret_cast<PointLightDebug::ColorUniformBlockData*>(m_pBlockColorCamera->GetMappedPointer());
float val[4] = { 1.0f, 0.0f, 1.0f, 1.0f };
uniformColorMap->SetUniform_u_color(val);
}
/*
* FrustumCullingPass::DispatchCompute
* -----------------------------------
* Dispatches the compute shader.
*/
void FrustumCullingPass::DispatchCompute(NVNcommandBuffer* pCommandBuffer, NVNbuffer* pDrawIndirectBuffer, const size_t& drawIndirectBufferSize)
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_COMPUTE,
FrustumCulling::ViewProjUniformBlockData::GetBinding(NVN_SHADER_STAGE_COMPUTE),
m_pViewProj->GetCurrentBufferAddress(),
sizeof(FrustumCulling::ViewProjUniformBlockData));
/* Bind the storage buffer. */
nvnCommandBufferBindStorageBuffer(pCommandBuffer,
NVN_SHADER_STAGE_COMPUTE,
FrustumCulling::InstancesDataStorageBlockData::GetBinding(NVN_SHADER_STAGE_COMPUTE),
nvnBufferGetAddress(m_pInstancesBuffer),
m_InstancesBufferSize);
nvnCommandBufferBindStorageBuffer(pCommandBuffer,
NVN_SHADER_STAGE_COMPUTE,
FrustumCulling::IndirectDrawsStorageBlockData::GetBinding(NVN_SHADER_STAGE_COMPUTE),
nvnBufferGetAddress(pDrawIndirectBuffer),
drawIndirectBufferSize);
nvnCommandBufferBindStorageBuffer(pCommandBuffer,
NVN_SHADER_STAGE_COMPUTE,
FrustumCulling::CullingDebugStorageBlockData::GetBinding(NVN_SHADER_STAGE_COMPUTE),
nvnBufferGetAddress(m_pCullingDebugBuffer),
m_CullingDebugBufferSize);
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
size_t instancesSize = m_InstancesBufferSize / sizeof(InstanceData);
int groupsX = static_cast<int>(instancesSize / g_NumWorkGroups + ((instancesSize % g_NumWorkGroups > 0) ? 1 : 0));
nvnCommandBufferDispatchCompute(pCommandBuffer, groupsX, 1, 1);
}
/*
* FrustumCullingPass::DrawFrustumView
* -----------------------------------
* Draws the camera frustum.
*/
void FrustumCullingPass::DrawFrustumView(NVNcommandBuffer* pCommandBuffer)
{
m_pViewFrustumDebug->BindState(pCommandBuffer);
m_pViewFrustumDebug->Draw(pCommandBuffer);
std::vector<NVNProgramData*>& programData = m_pDebugDrawShader->GetProgramData();
nvnCommandBufferBindPolygonState(pCommandBuffer, &m_DebugPolygonState);
/* Bind the uniform buffers. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_VERTEX,
PointLightDebug::BlockModelUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX),
m_pBlockModelCamera->GetCurrentBufferAddress(),
sizeof(PointLightDebug::BlockModelUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_VERTEX,
PointLightDebug::BlockViewProjUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX),
m_pBlockViewProjCamera->GetCurrentBufferAddress(),
sizeof(PointLightDebug::BlockViewProjUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
PointLightDebug::ColorUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockColorCamera->GetCurrentBufferAddress(),
sizeof(PointLightDebug::ColorUniformBlockData));
NVNcommandHandle drawHandle = m_pDebugCamera->GetCommandHandle(programData[0]->m_pProgramName);
nvnCommandBufferCallCommands(pCommandBuffer, 1, &drawHandle);
}
