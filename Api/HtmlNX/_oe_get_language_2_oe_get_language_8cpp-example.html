namespace
{
// Store the configured language and the string to display in a map.
const std::map<std::string, std::string> g_LanguageMap = {
{ "en-US", "American English" },
{ "en-GB", "British English" },
{ "ja", "Japanese" },
{ "fr", "French" },
{ "de", "German" },
{ "es-419", "LatinAmerican Spanish" },
{ "es", "Spanish" },
{ "it", "Italian" },
{ "nl", "Dutch" },
{ "fr-CA", "Canadian French" },
{ "pt", "Portuguese" },
{ "ru", "Russian" },
{ "zh-Hans", "Simplified Chinese" },
{ "zh-Hant", "Traditional Chinese" },
{ "ko", "Korean" },
};
} // Namespace.
extern "C" void nnMain()
{
NN_LOG("\n\nOeGetLanguage\n");
// Get the language.
const auto code = nn::oe::GetDesiredLanguage();
const auto it = g_LanguageMap.find(code.string);
if (it != g_LanguageMap.end())
{
// Output the language to the log if the language is supported.
NN_LOG("%s\n\n\n", it->second.c_str());
}
else
{
// Output an error if the language is not supported.
NN_LOG("It is a language not supported. \n\n\n");
}
}
