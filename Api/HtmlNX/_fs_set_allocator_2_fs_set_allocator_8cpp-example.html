#include <cstdlib>

#include <nn/fs.h>
#include <nn/nn_Assert.h>
#include <nn/nn_Abort.h>
#include <nn/nn_Log.h>
namespace {
void* Allocate(size_t size)
{
auto p = std::malloc(size);
NN_LOG("Allocated: 0x%p (%4zu bytes)\n", p, size);
return p;
}
void Deallocate(void* p, size_t size)
{
NN_UNUSED(size);
std::free(p);
NN_LOG("Deallocated: 0x%p (%4zu bytes)\n", p, size);
}
} // Namespace.
extern "C" void nnMain()
{
nn::Result result;
const int FileSize = 1024;
// Sets the memory allocator to use in the FS library.
// You must set an allocator that returns 8-byte aligned addresses.
// The configured allocator is used when mounting or opening files.
NN_LOG("Set the allocator.\n");
nn::fs::SetAllocator(Allocate, Deallocate);
NN_LOG("Mount the host root.\n");
result = nn::fs::MountHostRoot();
if( nn::fs::ResultTargetNotFound::Includes(result) )
{
// The host PC was not recognized.
// Correctly connect to the host PC.
NN_ASSERT(false, "Target not found.\n");
return;
}
// Any failures not listed are implementation failures. Always abort when they occur.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
NN_LOG("Create C:/Windows/Temp/FsSmplFile/.\n");
result = nn::fs::CreateFile("C:/Windows/Temp/FsSmplFile", FileSize);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// Create a directory.
NN_ASSERT(false, "Parent directory not found.");
return;
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file already exists.
// This error handling is not needed when the target file exists and that is not an issue.
// Note that there is no assurance that the file size is equal to FileSize when you are not using error handling.
// Delete the target file and re-create it when necessary.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient available space.
// Increase the available space on the host.
NN_ABORT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Open the file.\n");
nn::fs::FileHandle fileHandle;
result = nn::fs::OpenFile(&fileHandle, "C:/Windows/Temp/FsSmplFile", nn::fs::OpenMode_Read);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when you are only opening existing files.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open, using OpenMode_Write.
// This error handling is not necessary when there is no possibility of the file being open already.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Close the file.\n");
nn::fs::CloseFile(fileHandle);
NN_LOG("Delete the file.\n");
result = nn::fs::DeleteFile("C:/Windows/Temp/FsSmplFile");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when the target file exists without fail.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Unmount the host root.\n");
nn::fs::UnmountHostRoot();
return;
}
