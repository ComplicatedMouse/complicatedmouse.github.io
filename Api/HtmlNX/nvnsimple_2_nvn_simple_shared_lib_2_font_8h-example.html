/**************************************************************************************************
* CursorPosition:
* Represents a cursor location that can be used to continue to render subsequent text after
* calling Font::RenderStringToTexture.
**************************************************************************************************/
class CursorPosition
{
public:
CursorPosition() NN_NOEXCEPT :
m_X(0), m_Y(0), m_IsReachedEnd(false) {}
void Clear() NN_NOEXCEPT
{
m_X = 0;
m_Y = 0;
m_IsReachedEnd = false;
}
const int GetX() const NN_NOEXCEPT
{
return m_X;
}
const int GetY() const NN_NOEXCEPT
{
return m_Y;
}
const bool IsReachedEnd() const NN_NOEXCEPT
{
return m_IsReachedEnd;
}
void SetX(const int x) NN_NOEXCEPT
{
this->m_X = x;
}
void SetY(const int y) NN_NOEXCEPT
{
this->m_Y = y;
}
void SetReachedEnd(bool reachedEnd) NN_NOEXCEPT
{
m_IsReachedEnd = reachedEnd;
}
private:
int m_X;
int m_Y;
bool m_IsReachedEnd;
};
/**************************************************************************************************
*
* Font:
*
* A font class that supports all the font asset types that nn::fontll::ScalableFontEngine
* supports, namely TTF, OTF, and BFTTF (which includes the NX system fonts).
*
* The RenderStringToTexture methods allow the user to rasterize text into a linear buffer,
* based on adjustable size and line spacing properties. To use these methods, input strings
* should be encoded by UTF-8, and will be converted to UTF-32 for glyphmap generation.
*
**************************************************************************************************/
class Font
{
public:
Font() NN_NOEXCEPT :
m_IsInitialized(false),
m_FontEngineWorkBuf(nullptr),
m_FontSize(0),
m_LineSpacing(1.2f),
m_pAllocator(nullptr) {}
void Initialize(nn::mem::StandardAllocator* pAllocator, const void* fontData) NN_NOEXCEPT; // for OTF/TTF
void InitializeFromBfttf(nn::mem::StandardAllocator* pAllocator, const void* bfttfFontData, size_t bfttfFontDataSize) NN_NOEXCEPT; // for BFTTF
void InitializeFromNxSystemFont(nn::mem::StandardAllocator* allocator, nn::pl::SharedFontType fontType) NN_NOEXCEPT; // for NX system font
void Finalize() NN_NOEXCEPT;
void SetSize(int16_t fontSize) NN_NOEXCEPT;
int16_t GetSize() const NN_NOEXCEPT
{
return m_FontSize;
}
void SetLineSpacing(float lineSpacing) NN_NOEXCEPT;
float GetLineSpacing() NN_NOEXCEPT
{
return m_LineSpacing;
}
const CursorPosition RenderStringToTexture(void* texture, int textureWidth, int textureHeight, const char* string) NN_NOEXCEPT;
const CursorPosition RenderStringToTexture(void* texture, int textureWidth, int textureHeight, const char* string, int startPosX, int startPosY) NN_NOEXCEPT;
nn::fontll::ScalableFontEngine* GetFontEngine() NN_NOEXCEPT
{
return &m_FontEngine;
}
private:
bool m_IsInitialized;
nn::fontll::ScalableFontEngine m_FontEngine;
void* m_FontEngineWorkBuf;
char m_FontName[nn::fontll::FontNameLengthMax];
int16_t m_FontSize;
float m_LineSpacing;
nn::mem::StandardAllocator* m_pAllocator;
};
