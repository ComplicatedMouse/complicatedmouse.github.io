}

ScalableAOBlurPass::~ScalableAOBlurPass()

}
/*
* ScalableAOBlurPass::Init
* ------------------------
* Sets the asset data to use for drawing.
*/
void ScalableAOBlurPass::Init(AssetFileDataHolder* pData)
{
m_pAssetData = pData;
}
/*
* ScalableAOBlurPass::SetupUniforms
* ---------------------------------
* Sets up the uniform buffers.
*/
void ScalableAOBlurPass::SetupUniforms(UniformBufferManager* pUniformBufferManager)
{
m_pBlockSamplerFSH = pUniformBufferManager->CreateUniformBuffer(sizeof(ScalableAOBlur::BlockSamplerFSUniformBlockData));
m_pBlockSamplerFSV = pUniformBufferManager->CreateUniformBuffer(sizeof(ScalableAOBlur::BlockSamplerFSUniformBlockData));
}
/*
* ScalableAOBlurPass::SetTextureHandles
* ------------------------------------
* Sets the texture handles for the texture that will be blurred.
*/
void ScalableAOBlurPass::SetTextureHandles(const NVNtextureHandle& textureHandleH, const NVNtextureHandle& textureHandleV)
{
m_TextureHandleH = textureHandleH;
m_TextureHandleV = textureHandleV;
}
/*
* ScalableAOBlurPass::SetRenderTargets
* ------------------------------------
* Sets up the render targets.
*/
void ScalableAOBlurPass::SetRenderTargets(NVNtexture* renderTargetH, NVNtexture* renderTargetV)
{
m_RenderTargetH = renderTargetH;
m_RenderTargetV = renderTargetV;
}
/*
* ScalableAOBlurPass::UpdateUniforms
* ----------------------------------
* Updates the uniform buffers.
*/
void ScalableAOBlurPass::UpdateUniforms()
{
ScalableAOBlur::BlockSamplerFSUniformBlockData* blockSamplerFSH = reinterpret_cast<ScalableAOBlur::BlockSamplerFSUniformBlockData*>(m_pBlockSamplerFSH->GetMappedPointer());
blockSamplerFSH->SetUniform_u_TextureHandle(m_TextureHandleH);
blockSamplerFSH->SetUniform_u_Axis(0);
ScalableAOBlur::BlockSamplerFSUniformBlockData* blockSamplerFSV = reinterpret_cast<ScalableAOBlur::BlockSamplerFSUniformBlockData*>(m_pBlockSamplerFSV->GetMappedPointer());
blockSamplerFSV->SetUniform_u_TextureHandle(m_TextureHandleV);
blockSamplerFSV->SetUniform_u_Axis(1);
}
/*
* ScalableAOBlurPass::Draw
* ------------------------
* Draws a screen space quad that converts dapth values into a view space z buffer.
*/
void ScalableAOBlurPass::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
NVNtexture* targetsH[] = { m_RenderTargetH };
NVNtexture* targetsV[] = { m_RenderTargetV };
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Horizontal pass */
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
ScalableAOBlur::BlockSamplerFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockSamplerFSH->GetCurrentBufferAddress(),
sizeof(ScalableAOBlur::BlockSamplerFSUniformBlockData));
/* Bind the render target. */
nvnCommandBufferSetRenderTargets(pCommandBuffer, 1, targetsH, NULL, NULL, NULL);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
nvnCommandBufferBarrier(pCommandBuffer, NVNbarrierBits::NVN_BARRIER_ORDER_FRAGMENTS_BIT);
/* Vertical pass */
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
ScalableAOBlur::BlockSamplerFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockSamplerFSV->GetCurrentBufferAddress(),
sizeof(ScalableAOBlur::BlockSamplerFSUniformBlockData));
/* Bind the render target. */
nvnCommandBufferSetRenderTargets(pCommandBuffer, 1, targetsV, NULL, NULL, NULL);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
