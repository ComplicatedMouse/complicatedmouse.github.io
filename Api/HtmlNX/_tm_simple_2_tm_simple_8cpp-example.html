DWORD WINAPI ProgramExitMonitorThread(LPVOID arg)
{
HANDLE hExitEvent = CreateEvent(NULL, false, false, L"TmProgramExitEvent");
_ASSERT(hExitEvent != NULL);
nn::tm::Error errorExitEvent = nn::tm::RegisterProgramExitEvent(hExitEvent);
if (errorExitEvent != nn::tm::Error_Ok)
{
nn::tm::ErrorText errorText;
nn::tm::GetErrorTextEn(&errorText, errorExitEvent);
printf("RegisterProgramExitEvent failed because %s(%d).\n", errorText.text, errorExitEvent);
return errorExitEvent;
}
while (true)
{
WaitForSingleObject(hExitEvent, INFINITE);
printf("ProgramExitEvent Signaled\n");
}
return 0;
}

DWORD WINAPI DisplayHtcsPortmapping(LPVOID arg)
{
HANDLE hHtcsUpdateEvent = CreateEvent(NULL, false, false, L"TmHtcsUpdateEvent");
_ASSERT(hHtcsUpdateEvent != NULL);
nn::tm::Error errorUpdateEvent = nn::tm::RegisterHtcsPortMappingUpdateEvent(hHtcsUpdateEvent);
if (errorUpdateEvent != nn::tm::Error_Ok)
{
nn::tm::ErrorText errorText;
nn::tm::GetErrorTextEn(&errorText, errorUpdateEvent);
printf("RegisterHtcsPortMappingUpdateEvent failed because %s(%d).\n", errorText.text, errorUpdateEvent);
return errorUpdateEvent;
}
nn::tm::HtcsAddress addresses[64];
int32_t size;
while (true)
{
WaitForSingleObject(hHtcsUpdateEvent, INFINITE);
nn::tm::Error errorGetHtcsPoints = nn::tm::GetHtcsEntryEndPoints(addresses, &size, sizeof(addresses) / sizeof(addresses[0]), nullptr, nullptr);
if (errorGetHtcsPoints != nn::tm::Error_Ok)
{
nn::tm::ErrorText errorText;
nn::tm::GetErrorTextEn(&errorText, errorGetHtcsPoints);
printf("GetHtcsEntryEndPoints failed because %s(%d).\n", errorText.text, errorGetHtcsPoints);
continue;
}
std::string out;
out += "---- htcs ports updated ----\n";
for (int i = 0; i < size; i++)
{
out += std::to_string(i);
out += ", ";
out += addresses[i].ipAndPortAddress;
out += "\n";
}
out += "---- htcs port list end ----\n";
printf(out.c_str());
}
return 0;
}

printf("\tAdd target with IP address\n");
printf("other) ");
printf("\tReturn to menu\n");
int selectedTarget = GetInputNumber();
if (0 <= selectedTarget && selectedTarget < targetCount)
{
nn::tm::TargetHandle targetHandle;
RETURN_IF_FAILURE(nn::tm::AddTarget(&targetHandle, pDiscoveryHandles[selectedTarget]));
printf("Added.\n");
}
if (selectedTarget == targetCount)
{
AddTargetWithIp();
}
}

{

{
RETURN_IF_FAILURE(nn::tm::GetTargetProgramState(&programState, target));
Sleep(100);
}
ShowProgramName(target);

while (1)
{

break;
default:
TerminateSampleProgram(target);
return;
}
}

void TargetSetting() {
int32_t targetCount;
RETURN_IF_FAILURE(nn::tm::GetTargetCount(&targetCount));
if (targetCount == 0)
{
printf("No target is found. Skip TargetSetting\n");
return;
}
std::unique_ptr<nn::tm::TargetHandle[]> pTargetHandles(new nn::tm::TargetHandle[targetCount]);
RETURN_IF_FAILURE(nn::tm::GetTargets(pTargetHandles.get(), &targetCount, targetCount));
printf("Please input option number of target to operate:\n");
ShowRegisteredTargetsWithoutDetails(pTargetHandles.get(), targetCount);
printf("other) ");
printf("\tReturn to menu\n");
int selectedTarget = GetInputNumber();
if (selectedTarget == targetCount)
{
return;
}
if (0 <= selectedTarget && selectedTarget < targetCount)
{
nn::tm::TargetHandle target = pTargetHandles[selectedTarget];
while (1) {
ShowTargetInformation(target);
ShowSettingMenu(target);
SettingCommand selectedCommand = static_cast<SettingCommand>(GetInputNumber());
if (selectedCommand < 0 || selectedCommand >= SettingCommand_Return)
{
break;
}
SettingCommandProcessor(selectedCommand, target);
}
}
}
void IncompleteDiscoveryScan()
{

PrintDiscoveryState(discoveryState);
}
