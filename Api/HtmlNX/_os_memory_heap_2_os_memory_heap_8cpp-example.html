#include <stdint.h>
#include <nn/nn_Macro.h>
#include <nn/nn_Abort.h>
#include <nn/nn_Log.h>
#include <nn/os.h>
//-----------------------------------------------------------------------------
const size_t HeapSize = 32u * 1024 * 1024;
//-----------------------------------------------------------------------------
// nninitStartup() is invoked before calling nnMain().
//
extern "C" void nninitStartup()
{
// Allocate the memory heap.
auto result = nn::os::SetMemoryHeapSize( HeapSize );
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
}
//-----------------------------------------------------------------------------
// Sample Program
//
void MemorySample()
{
// Repeatedly acquire memory while increasing the size.
size_t size = nn::os::MemoryBlockUnitSize;
while (size <= HeapSize)
{
uintptr_t address;
auto result = nn::os::AllocateMemoryBlock( &address, size );
if (!result.IsSuccess())
{
// Exit the loop if memory acquisition fails.
break;
}
NN_LOG("Allocated memory block address: 0x%p\n", address);
NN_LOG("Allocated memory block size: 0x%zx\n", size);
NN_LOG("\n");
nn::os::FreeMemoryBlock( address, size );
size = size * 2;
}
}
//-----------------------------------------------------------------------------
// The main function.
//
extern "C" void nnMain()
{
// Generate memory heap information.
NN_LOG("Size of memory heap: 0x%zx\n", HeapSize);
NN_LOG("\n");
// Issue the sample program function.
MemorySample();
NN_LOG("End\n");
}
