#include <nn/nn_Log.h>
#include <nn/nn_Assert.h>
#include <nn/TargetConfigs/build_Base.h>
#include <nn/init.h>
#include <nn/hid.h>
#include <nn/gll.h>
#if defined(NN_BUILD_CONFIG_OS_HORIZON)
#include <nn/oe.h>
#endif
#include "GraphicsHelper.h"
#include "Renderer.h"
extern "C" void nnMain()
{
nn::hid::InitializeTouchScreen();
nn::hid::TouchScreenState< 1 > touchScreenState = {};
GraphicsHelper graphicsHelper;
graphicsHelper.Initialize();
nngllResult gllResult = nngllInitializeGl();
NN_UNUSED( gllResult );
NN_ASSERT_EQUAL( gllResult, nngllResult_Succeeded );
NN_LOG( "GL_VERSION: %s\n", ::glGetString( GL_VERSION ) );
NN_LOG( "GL_SHADING_LANGUAGE_VERSION: %s\n", glGetString( GL_SHADING_LANGUAGE_VERSION ) );
NN_LOG( "\nSupported extensions:\n" );
// Check whether the function pointer is null to determine whether that feature is loaded.
if( glGetStringi != nullptr )
{
// GL 3.0 or later.
int extensionCount = 0;
glGetIntegerv( GL_NUM_EXTENSIONS, &extensionCount );
for( int idxExtension = 0; idxExtension < extensionCount; ++idxExtension )
{
auto pExtensionName = reinterpret_cast< const char* >( ::glGetStringi( GL_EXTENSIONS, idxExtension ) );
NN_LOG( "%s\n", pExtensionName );
}
}
else
{
// Version earlier than GL 3.0.
for( auto pExtensionNames = reinterpret_cast<const char*>( ::glGetString( GL_EXTENSIONS ) );
*pExtensionNames != '\0'; ++pExtensionNames )
{
NN_LOG( "%c", *pExtensionNames == ' ' ? '\n' : *pExtensionNames );
}
}
// Pass an extension name to NN_GLL_IS_SUPPORTED() to determine whether that extension is supported.
if( NN_GLL_IS_SUPPORTED( GL_EXT_depth_bounds_test ) )
{
// GL_EXT_depth_bounds_test is supported.
glEnable( GL_DEPTH_BOUNDS_TEST_EXT );
glDepthBoundsEXT( 0.0f, 0.5f );
}
Renderer renderer;
renderer.Initialize();
#if defined(NN_BUILD_CONFIG_OS_HORIZON)
nn::os::SystemEvent resolutionChangeEvent;
nn::oe::GetDefaultDisplayResolutionChangeEvent(&resolutionChangeEvent);
int width;
int height;
// Check initial resolution.
nn::oe::GetDefaultDisplayResolution(&width, &height);
graphicsHelper.Resize(width, height);
#endif
for( int frame = 0; touchScreenState.count < 1; ++frame )
{
#if defined(NN_BUILD_CONFIG_OS_HORIZON)
if (resolutionChangeEvent.TryWait())
{
// Rendering will match the display output resolution.
// Note the display subsystem can upscale and downscale the output buffers
// (in most cases) when the buffer dimensions and display output resolution
// do not match.
nn::oe::GetDefaultDisplayResolution(&width, &height);
graphicsHelper.Resize(width, height);
}
#endif
renderer.Render( frame );
glFinish();
graphicsHelper.SwapBuffers();
nn::hid::GetTouchScreenStates( &touchScreenState, 1 );
}
renderer.Finalize();
graphicsHelper.Finalize();
}
