// For searching for tag names in this application.
const std::string tagsToSearchByApplication[] = {"application_id_of_related_item", "application_id_of_related_item_2"};
constexpr auto TagsToSearchByApplicationCountMax = sizeof(tagsToSearchByApplication) / sizeof(*tagsToSearchByApplication);
}
RelatedTitleBundleScene::RelatedTitleBundleScene(SceneData* pSceneDataInfo) NN_NOEXCEPT
: SceneBase(pSceneDataInfo)
, m_RelatedTitleBundleStatus(RelatedTitleBundleStatus_Initialize)
, m_RelatedTitleCount(0)
, m_DetailWindow(DetailWindow(700, 5, DetailWindow::WindowStyle::WindowStyle_RelatedTitle))
{
NN_ASSERT_NOT_NULL(pSceneDataInfo);
pSceneData->SetLoadCompleted(false);
}
RelatedTitleBundleScene::~RelatedTitleBundleScene() NN_NOEXCEPT
{
}
// The list to display during the selection process.
void RelatedTitleBundleScene::DrawSelectRelatedTitleBundleCallBack(int listIndex, SceneBase* pScene, int x, int y, bool isSelected) NN_NOEXCEPT
{
NN_UNUSED(isSelected);
NN_ASSERT_NOT_NULL(pScene);
RelatedTitleBundleScene* pRelatedTitleBundleScene = dynamic_cast<RelatedTitleBundleScene*>(pScene);
const auto contentIndex = pRelatedTitleBundleScene->m_DisplayListIndexs[listIndex];
const auto& titleItem = pRelatedTitleBundleScene->m_RelatedTitleItem[contentIndex.titleIndex].titleRequest.GetResponse().GetTitleItems()[contentIndex.listIndex];
const ContentPricesResponse& contentPriceResponse = pRelatedTitleBundleScene->m_RelatedTitleItem[contentIndex.titleIndex].pricesRequest.GetResponse();
const ContentPrice& contentPrice = contentPriceResponse.GetContentPrices()[contentIndex.priceIndex];
ImGui::SetCursorPos(ImVec2(x, y));
ImGui::SetWindowFontScale(1.0f);
ImGui::Text("%s", titleItem.GetName().c_str());
// Display more details about the selected item.
if (isSelected == true)
{
// Display titles, bundles, and so on.
pRelatedTitleBundleScene->m_DetailWindow.DrawContentType(titleItem.GetContentType());
// Item Name / Price / Description / Exceptions Display
pRelatedTitleBundleScene->m_DetailWindow.DrawDetailInfomation(
titleItem.GetName(),
contentPriceResponse.GetIncTaxMessage(),
titleItem.GetDescription(),
titleItem.GetDisclaimer(),
pRelatedTitleBundleScene->m_RelatedTitleItem[contentIndex.titleIndex].titleRequest.GetResponse().GetLang(),
contentPrice.GetPrice()
);
}
}
// Screen rendering.
void RelatedTitleBundleScene::Draw() NN_NOEXCEPT
{
switch (m_RelatedTitleBundleStatus)
{
case RelatedTitleBundleStatus_None:
break;
case RelatedTitleBundleStatus_Initialize:
case RelatedTitleBundleStatus_ResponseWait:
case RelatedTitleBundleStatus_ReadTitle:
case RelatedTitleBundleStatus_ReadTitleResponseWait:
case RelatedTitleBundleStatus_ContentPricesInitialize:
case RelatedTitleBundleStatus_ContentPricesResponseWait:
ImGui::SetWindowFontScale(1.8f);
ImGui::SetCursorPos(ImVec2(550, 675));
ImGui::TextColored(White, "Now Loading...");
break;
case RelatedTitleBundleStatus_Start:
{
// Display the selection screen.
m_Select.DrawSelectionList(DrawSelectRelatedTitleBundleCallBack, this);
// Note: There is no way to ensure that the language of the response is one of the system languages.
// Do not treat this condition as an error. Continue to display product information as is.
nn::settings::LanguageCode lang = nn::oe::GetDesiredLanguage();
if (::nn::util::Strncmp(lang.string, m_RelatedTitleItem[0].titleRequest.GetResponse().GetLang().c_str(), sizeof(lang.string)) != 0)
{
// Notation displays (translation):
// Product information cannot be displayed using the languages configured for this application. Display using an available language.
ImGui::SetWindowFontScale(1.1f);
ImGui::SetCursorPos(ImVec2(10, 570));
ImGui::TextColored(Red, "Product information cannot be displayed in this language currently set for this application. The supported language will be displayed.");
}
// Displays sale status.
ImGui::SetWindowFontScale(1.0f);
ImGui::SetCursorPos(ImVec2(50, 50));
ImGui::TextColored(White, "Selected sales status : %s", pSceneData->GetSalesStatusStirng().c_str());
// Description of operations.
ImGui::SetWindowFontScale(2.0f);
ImGui::SetCursorPos(ImVec2(10, 600));
ImGui::TextColored(White, "Select an item by up or down button and Push A button to retrieve each item.\nPush B button to move to the selected scene for sales status.");
// Display the frame for the Details window.
m_DetailWindow.DrawFrame("Related title / bundle detail infomation.");
break;
}
default:
NN_LOG("Status that does not exist.\n");
break;
}
}
// Input management processing.
void RelatedTitleBundleScene::Control(const nns::hid::GamePad& gamePad) NN_NOEXCEPT
{
switch (m_RelatedTitleBundleStatus)
{
case RelatedTitleBundleStatus_Initialize:
{
// Check for the maximum number of application tags specified.
NN_ASSERT(TagsToSearchByApplicationCountMax <= RelatedTitleBundleScene::RelatedTitleMax);
// Request to get one's own TitleItem.
if (false == m_RequestTitleItem.SetupRequest(pSceneData->GetUserContext()->id, "{applicationId}"))
{
// Invalid server response error.
NN_LOG("Invalid server response.\n");
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
else
{
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_ResponseWait;
break;
}
break;
}
case RelatedTitleBundleStatus_ResponseWait:
if (!m_RequestTitleItem.IsCompleted())
{
if (gamePad.HasAnyButtonsUp(nns::hid::Button::B::Mask) == true)
{
// Canceling.
m_RequestTitleItem.Cancel();
break;
}
}
else
{
try
{
if (m_RequestTitleItem.WaitFinish(pSceneData->GetUserContext()->userHandle) == false)
{
//If an error or cancellation occurs.
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
else
{
if (m_RequestTitleItem.ConvertResponse() == false)
{
// Display an error when the parse fails.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
else
{
// Display information about the different, related title from the read data.
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_ReadTitle;
break;
}
}
}
catch (std::bad_alloc& e)
{
// Processes cancellation when memory allocation fails.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
catch (std::exception& e)
{
// Although only bad_alloc is expected in the sample, this processing is used when other exceptions are thrown.
// Process cancellation.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
}
break;
// Read a different, related title.
case RelatedTitleBundleStatus_ReadTitle:
{
// Search for the application_id_of_related_item and application_id_of_related_item_2 in the tag,
// and gets the ApplicationId information of other related titles that are displayed.
if (m_RelatedTitleCount >= TagsToSearchByApplicationCountMax)
{
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_InitializeSelection;
break;
}
std::string tagData("");
for (const auto& relatedTitleBundle : m_RequestTitleItem.GetResponse().GetTitleItems())
{
for (const auto& tag : relatedTitleBundle.GetTagCollection().GetItemTags())
{
if (tag.GetKey().compare(tagsToSearchByApplication[m_RelatedTitleCount]) == 0)
{
tagData = tag.GetValue();
break;
}
}
}
// Extract applicationId. (Get the 16 digits from the front of a 16-digit hexadecimal number.)
std::cmatch applicationId;
if (std::regex_search(tagData.c_str(), applicationId, std::regex("[0-9a-fA-F]{16}")) == false)
{
if (m_RelatedTitleCount == 0)
{
// The application ID format does not match or does not exist.
NN_LOG("ApplicationId format mismatch.\n");
// The error when required data does not exist.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_ThereIsNoRequiredData);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
{
// If there are no ApplicationId entries for other related titles after the first, proceed to selection initialization.
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_InitializeSelection;
break;
}
}
auto& item = m_RelatedTitleItem[m_RelatedTitleCount];
item.applicationID = applicationId.str();
// Request to get the TitleItem of a different, related title.
const auto isOnsaleOnly = pSceneData->GetSalesStatus() != SceneData::SalesStatus::SalesStatus_All;
const auto isSuccess = item.titleRequest.SetupRequest(pSceneData->GetUserContext()->id, item.applicationID, isOnsaleOnly);
if (isSuccess == false)
{
// Invalid server response error.
NN_LOG("Invalid server response.\n");
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
else
{
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_ReadTitleResponseWait;
break;
}
break;
}
case RelatedTitleBundleStatus_ReadTitleResponseWait:
{
auto& item = m_RelatedTitleItem[m_RelatedTitleCount];
if (!item.titleRequest.IsCompleted())
{
if (gamePad.HasAnyButtonsUp(nns::hid::Button::B::Mask) == true)
{
// Canceling.
item.titleRequest.Cancel();
break;
}
}
else
{
try
{
if (item.titleRequest.WaitFinish(pSceneData->GetUserContext()->userHandle) == false)
{
//If an error or cancellation occurs.
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
else
{
if (item.titleRequest.ConvertResponse() == false)
{
// Display an error when the parse fails.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
else
{
// After finishing reading information for a different, related title, proceed to reading price information for other related titles.
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_ContentPricesInitialize;
break;
}
}
}
catch (std::bad_alloc& e)
{
// Processes cancellation when memory allocation fails.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
catch (std::exception& e)
{
// Although only bad_alloc is expected in the sample, this processing is used when other exceptions are thrown.
// Process cancellation.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
}
break;
}
// Reading the price information for a different, related title.
case RelatedTitleBundleStatus_ContentPricesInitialize:
{
auto& item = m_RelatedTitleItem[m_RelatedTitleCount];
if (item.titleRequest.GetResponse().GetTitleItems().size() == 0)
{
// If there are no related titles, transition to the process for initializing the selection.
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_InitializeSelection;
break;
}
// Create a list of nsUid values from the data for all of the different, related titles.
nn::ec::NsUid nsUidList[SceneData::NsUidListMax];
int i;
for (i = 0; i < item.titleRequest.GetResponse().GetTitleItems().size() && i < SceneData::NsUidListMax; i++)
{
nsUidList[i].value = item.titleRequest.GetResponse().GetTitleItems()[i].GetNsUid().value;
}
const auto isSuccess = item.pricesRequest.SetupRequest(pSceneData->GetUserContext()->id, nsUidList, i, item.applicationID);
if (isSuccess == false)
{
// Return to the previous screen when an error occurs.
NN_LOG("Request or initialization failed.\n");
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_ContentPricesResponseWait;
}
break;
}
case RelatedTitleBundleStatus_ContentPricesResponseWait:
{
auto& item = m_RelatedTitleItem[m_RelatedTitleCount];
if (!item.pricesRequest.IsCompleted())
{
if (gamePad.HasAnyButtonsUp(nns::hid::Button::B::Mask) == true)
{
// Canceling.
item.pricesRequest.Cancel();
break;
}
}
else
{
try
{
if (item.pricesRequest.WaitFinish(pSceneData->GetUserContext()->userHandle) == false)
{
//If an error or cancellation occurs.
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
else
{
if (item.pricesRequest.ConvertResponse() == false)
{
// Display an error when the parse fails.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
else
{
// Confirm that length matches the number of items.
if (item.pricesRequest.GetResponse().GetLength() != item.pricesRequest.GetResponse().GetContentPrices().size())
{
// length does not match the number of items.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_ItemCountDifferent);
SceneBase::SetStatus(DisplayStatus_Error);
NN_LOG("Content Prices length Error.\n");
break;
}
m_RelatedTitleCount++;
// After finishing reading information for the different, related title, proceed to reading information for the next different, related title.
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_ReadTitle;
break;
}
}
}
catch (std::bad_alloc& e)
{
// Processes cancellation when memory allocation fails.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
catch (std::exception& e)
{
// Although only bad_alloc is expected in the sample, this processing is used when other exceptions are thrown.
// Process cancellation.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
break;
}
}
break;
}
case RelatedTitleBundleStatus_InitializeSelection:
{
// Store the array of indexes of the different, related titles to display on the screen.
for (int titleIndex = 0; titleIndex < m_RelatedTitleCount; titleIndex++)
{
auto& item = m_RelatedTitleItem[titleIndex];
const auto relatedTitleResponse = item.titleRequest.GetResponse();
const auto pricesResponse = item.pricesRequest.GetResponse();
for (int i = 0; i < relatedTitleResponse.GetTitleItems().size(); i++)
{
nn::ec::NsUid itemNsUid;
itemNsUid.value = relatedTitleResponse.GetTitleItems()[i].GetNsUid().value;
// Search ContentPrices for items with the same NsUid.
for (int j = 0; j < pricesResponse.GetContentPrices().size(); j++)
{
if ((itemNsUid.value == pricesResponse.GetContentPrices()[j].GetNsUid().value)
&& (pricesResponse.GetContentPrices()[j].GetPrice().GetSalesStatus().compare("not_onsale") != 0))
{
// Store the Content array indexes.
m_DisplayListIndexs.push_back(SelectionListInformation{ titleIndex, i, j });
break;
}
}
}
}
// The initial settings for the choices.
int listMax = static_cast<int>(m_DisplayListIndexs.size());
if (0 == listMax)
{
// Display an error if there are no items to display.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_NoneItem);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
int row = 10; // Number of lines
int offsetX = 600; // The amount to shift the selection horizontally.
int offsetY = 50; // The amount to shift the selection vertically.
int positionX = 10; // The coordinate for the left position of the selection.
int positionY = 100; // The coordinate for the upper position of the selection.
float cursorSize = 1.5; // The size of the cursor.
int cursorOffsetX = 0; // Fine tuning the x-position of the cursor.
int cursorOffsetY = -5; // Fine tuning the y-position of the cursor.
m_Select.SetInitialValue(
listMax
, row
, offsetX
, offsetY
, positionX
, positionY
, pSceneData->GetSelectedPositionRelatedTitleBundle()
, cursorSize
, cursorOffsetX
, cursorOffsetY
);
// Processing for informing the main screen that loading completed and deleting the previous screen.
pSceneData->SetLoadCompleted(true);
m_RelatedTitleBundleStatus = RelatedTitleBundleStatus_Start;
break;
}
case RelatedTitleBundleStatus_Start:
int selected;
Select::SelectStatus selectStatus;
// Execute the selection process.
m_Select.InputSelectionList(&selectStatus, &selected, gamePad);
pSceneData->SetSelectedPositionRelatedTitleBundle(selected);
if (selectStatus == Select::SelectStatus_Cancel)
{
// If the process has been canceled.
SceneBase::SetStatus(DisplayStatus_RelatedTitleSalesStatusSelection);
break;
}
if (selectStatus == Select::SelectStatus_Decision)
{
// If A is pressed.
nn::ec::NsUid nsUid;
const auto contentIndex = m_DisplayListIndexs[selected];
nsUid.value = m_RelatedTitleItem[contentIndex.titleIndex].titleRequest.GetResponse().GetTitleItems()[contentIndex.listIndex].GetNsUid().value;
pSceneData->SetNsUid(nsUid);
SceneBase::SetStatus(DisplayStatus_ShowShopProductDetails);
break;
}
break;
default:
break;
}
