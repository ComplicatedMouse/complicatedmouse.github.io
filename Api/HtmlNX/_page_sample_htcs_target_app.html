This sample program shows how to communicate with the host PC.

Overview

This documentation describes how to use the htcs library for communication with the host PC.

For more information, see htcs Library.

File Structure

This sample program is located in Samples/Sources/Applications/Htcs/ .

Directory Overview
HtcsHostApp A sample application that runs on the host PC.
HtcsTargetApp A sample application that runs on the target. This section describes these applications.

Required Environment

Requires the installation of Target Manager.

How to Use

Nothing specific.

Important Information

This sample program does not display anything on the screen. The execution results are output to a log.

Execution Procedure

If Target Manager is not running, start it before executing this sample program. After that, build and run HtcsHostApp and HtcsTargetApp.

109 void WaitHostCompleted(int sock)

111  nn::htcs::FdSet fd;
112  nn::htcs::FdSetZero(&fd);
113  nn::htcs::FdSetSet(sock, &fd);

115  NN_ABORT_UNLESS(nn::htcs::Select(sock + 1, &fd, nullptr, nullptr, nullptr) > 0, "Server: Select failed (%d)\n", nn::htcs::GetLastError());
116  if (nn::htcs::FdSetIsSet(sock, &fd))
117  {
118  char buf;

127 // The sent data is sent back to the peer as is.
128 void EchoBack(int sock)
129 {
130  char buf[128] = {};
131  nn::htcs::ssize_t receivedBytes = nn::htcs::Recv(sock, buf, sizeof(buf), 0);
132  if (receivedBytes < 0)
133  {
134  NN_LOG("Receive error. %d\n", nn::htcs::GetLastError());
135  return;
136  }
137  else if (receivedBytes == 0)
138  {
139  NN_LOG("Gracefully closed.\n");
140  return;
141  }
142 
143  NN_LOG("Received \"%s\".\n", buf);
144 
145  nn::htcs::ssize_t sentBytes = nn::htcs::Send(sock, buf, receivedBytes, 0);
146  if (sentBytes < 0)
147  {
148  NN_LOG("Send error. %d\n", nn::htcs::GetLastError());
149  }
150  NN_ABORT_UNLESS(nn::htcs::Send(sock, "\n", 1, 0) != -1, "Server: Send failed (%d)\n", nn::htcs::GetLastError());
151 
152  WaitHostCompleted(sock);
153 }

155 void NonBlockingEchoBack(int sock)
156 {
157  nn::htcs::ssize_t receivedBytes = 0;
158 
159  char buf[128] = {};
160 
161  nn::htcs::FdSet fd;
162  nn::htcs::TimeVal time;

164  while (true)

166  time.tv_sec = 0;
167  time.tv_usec = 0;

169  nn::htcs::FdSetSet(sock, &fd);

171  int result = nn::htcs::Select(sock + 1, &fd, nullptr, nullptr, &time);
172  if (result > 0)
173  {
174  break;
175  }
176  else if (result < 0)

178  NN_LOG("Select error (%d)\n", nn::htcs::GetLastError());
179  return;
180  }

182  nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1000));
183  }
184  if (nn::htcs::FdSetIsSet(sock, &fd))
185  {
186  receivedBytes = nn::htcs::Recv(sock, buf, sizeof(buf) - 1, 0);
187  if (receivedBytes < 0)
188  {
189  NN_LOG("Receive error (%d)\n", nn::htcs::GetLastError());
190  return;
191  }
192  else if (receivedBytes == 0)
193  {
194  NN_LOG("Gracefully closed.\n");
195  return;
196  }
197  }
198  else
199  {
200  NN_ABORT();
201  }
202 
203  NN_LOG("Received \"%s\".\n", buf);
204 
205  nn::htcs::ssize_t sentBytes = 0;
206  time.tv_sec = 0;
207  time.tv_usec = 10;
208  nn::htcs::FdSetZero(&fd);
209  nn::htcs::FdSetSet(sock, &fd);
210  NN_ABORT_UNLESS(nn::htcs::Select(sock + 1, nullptr, &fd, nullptr, &time) != -1, "Server: Select failed (%d)\n", nn::htcs::GetLastError());
211  if (nn::htcs::FdSetIsSet(sock, &fd))
212  {
213  sentBytes = nn::htcs::Send(sock, buf, receivedBytes, 0);
214  if (sentBytes < 0)
215  {
216  NN_LOG("Send error (%d)\n", nn::htcs::GetLastError());

219  }
220  else

222  NN_ABORT();

228 void ServerSample()
229 {
230  // Register the service.
231  nn::htcs::SockAddrHtcs serviceAddr;
232  serviceAddr.family = nn::htcs::HTCS_AF_HTCS;
233  serviceAddr.peerName = nn::htcs::GetPeerNameAny();
234  std::strcpy(serviceAddr.portName.name, "ServerInTarget");
235 
236  int serviceSocket = nn::htcs::Socket();
237  NN_ABORT_UNLESS(nn::htcs::Bind(serviceSocket, &serviceAddr) != -1, "Server: Bind failed (%d)\n", nn::htcs::GetLastError());
238  NN_ABORT_UNLESS(nn::htcs::Listen(serviceSocket, 1) != -1, "Server: Listen failed (%d)\n", nn::htcs::GetLastError());
239 
240  // Wait for the connection.
241  NN_LOG("Waiting for the connection from the host.\n");
242  int sock = nn::htcs::Accept(serviceSocket, nullptr);
243  NN_ABORT_UNLESS(sock >= 0, "Server: Accept failed. Error(%d).\n", nn::htcs::GetLastError());
244  NN_LOG("Accepted the client on the host.\n");
245 
246  // Echo back test.
247  EchoBack(sock);
248 
249  // Disconnect.
250  nn::htcs::Close(sock);
251  nn::htcs::Close(serviceSocket);
252  NN_LOG("Closed.\n");
253 }

255 void ClientSample()
256 {
257  // Create the server address information for the host you want to connect to.
258  nn::htcs::SockAddrHtcs addr;
259  addr.family = nn::htcs::HTCS_AF_HTCS;
260  addr.peerName = nn::htcs::GetDefaultHostName();
261  std::strcpy(addr.portName.name, "ServerInHost");
262 
263  // Try to connect until a connection is made.
264  int sock = nn::htcs::Socket();
265  while (NN_STATIC_CONDITION(true))
266  {
267  if (nn::htcs::Connect(sock, &addr) == 0)
268  {
269  NN_LOG("Connected to the server on the host.\n");
270  break;
271  }
272  nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1000));
273  }
274 
275  // Echo back test.
276  EchoBack(sock);
277 
278  // Disconnect.
279  nn::htcs::Close(sock);
280  NN_LOG("Closed.\n");
281 }

283 void NonBlockingServerSample()
284 {
285  // Register the service.
286  nn::htcs::SockAddrHtcs address;
287  address.family = nn::htcs::HTCS_AF_HTCS;
288  address.peerName = nn::htcs::GetPeerNameAny();
289  std::strcpy(address.portName.name, "ServerInTarget");
290 
291  nn::htcs::FdSet fd;
292  nn::htcs::FdSetZero(&fd);
293 
294  int listenSocket = nn::htcs::Socket();
295  NN_ABORT_UNLESS(nn::htcs::Fcntl(listenSocket, nn::htcs::HTCS_F_SETFL, nn::htcs::HTCS_O_NONBLOCK) != -1, "Server: Fcntl failed (%d)\n", nn::htcs::GetLastError());
296  NN_ABORT_UNLESS(nn::htcs::Bind(listenSocket, &address) != -1, "Server: Bind failed (%d)\n", nn::htcs::GetLastError());
297  NN_ABORT_UNLESS(nn::htcs::Listen(listenSocket, 1) != -1, "Server: Listen failed (%d)\n", nn::htcs::GetLastError());
298 
299  NN_LOG("Waiting for the connection from the host.\n");
300  nn::htcs::FdSetSet(listenSocket, &fd);
301  NN_ABORT_UNLESS(nn::htcs::Select(listenSocket + 1, &fd, nullptr, nullptr, nullptr) != -1, "Server: Select failed (%d)\n", nn::htcs::GetLastError());
302  if (nn::htcs::FdSetIsSet(listenSocket, &fd))
303  {
304  int socket = nn::htcs::Accept(listenSocket, nullptr);
305  if (socket < 0)
306  {
307  NN_LOG("L%d LastError = %d\n", __LINE__, nn::htcs::GetLastError());
308  }
309 
310  NonBlockingEchoBack(socket);
311 
312  nn::htcs::Close(socket);
313  }
314  nn::htcs::Close(listenSocket);
315  NN_LOG("Closed.\n");
316 }

318 void NonBlockingClientSample()
319 {
320  // Create the server address information for the host you want to connect to.
321  nn::htcs::SockAddrHtcs addr;
322  addr.family = nn::htcs::HTCS_AF_HTCS;
323  addr.peerName = nn::htcs::GetDefaultHostName();
324  std::strcpy(addr.portName.name, "ServerInHost");
325 
326  int sock = nn::htcs::Socket();
327  nn::htcs::Fcntl(sock, nn::htcs::HTCS_F_SETFL, nn::htcs::HTCS_O_NONBLOCK);
328  while (true)
329  {
330  NN_ABORT_UNLESS(nn::htcs::Connect(sock, &addr) < 0, "Server: Connect unexpected success\n");
331  if (nn::htcs::GetLastError() == nn::htcs::HTCS_EINPROGRESS
332  || nn::htcs::GetLastError() == nn::htcs::HTCS_EISCONN)
333  {
334  break;
335  }
336  else if (nn::htcs::GetLastError() == nn::htcs::HTCS_EADDRNOTAVAIL
337  || nn::htcs::GetLastError() == nn::htcs::HTCS_EAGAIN)
338  {
339  NN_LOG("Waiting for the host address to become valid.\n");
340  nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1000));
341  continue;
342  }
343  else
344  {
345  NN_LOG("Unexpected connection error (%d)\n", nn::htcs::GetLastError());
346  return;
347  }
348  }
349 
350  // Echo back test.
351  NonBlockingEchoBack(sock);
352 
353  // Disconnect.
354  nn::htcs::Close(sock);
355  NN_LOG("Closed.\n");
356 }

Sample Program Description

This sample program has the following flow.

  • Run the HTSC communication sample with the target acting as the server.
  • Run the HTSC communication sample with the target acting as the client.

Acting separately as the server and the client, return data sent from the host back to the host unmodified.
