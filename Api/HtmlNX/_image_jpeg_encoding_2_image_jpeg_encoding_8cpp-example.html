static const size_t BytesPerPixel = sizeof(nn::Bit32); // Defines the number of bytes per pixel.
struct MetaData
{
// 0th TIFF IFD category metadata.
nn::image::ExifOrientation orientation; // Image direction.
const char *software; // Software name.
size_t softwareSize; // Number of bytes of the software name including the terminating character.
const char *dateTime; // Imaging date and time (20 bytes including the main single character).
// 0th Exif IFD category metadata.
const void *makerNote; // Manufacturer's notes.
size_t makerNoteSize; // Manufacturer's note byte count.
nn::image::Dimension effectiveDim; // Effective image size.
const char *uniqueId; // Image unique ID (33 bytes including the terminating character).
};
void PrintMetaData(const MetaData &metaData)
{
const char *orientationStr;
switch (metaData.orientation)
{
case nn::image::ExifOrientation_Normal:
orientationStr = "Upright.";
break;
case nn::image::ExifOrientation_FlipHorizontal:
orientationStr = "Inverted in the horizontal direction.";
break;
case nn::image::ExifOrientation_Rotate180:
orientationStr = "Rotated 180 degrees clockwise.";
break;
case nn::image::ExifOrientation_FlipVertical:
orientationStr = "Inverted in the vertical direction.";
break;
case nn::image::ExifOrientation_FlipTopRightToLeftBottom:
orientationStr = "Inverted around an angled line from the right down so that the right end of the top side goes to the bottom end of the left side.";
break;
case nn::image::ExifOrientation_Rotate270:
orientationStr = "Rotated 270 degrees clockwise.";
break;
case nn::image::ExifOrientation_FlipTopLeftToRightBottom:
orientationStr = "Inverted around an angled line from the left down so that the left end of the top side goes to the bottom end of the right side.";
break;
case nn::image::ExifOrientation_Rotate90:
orientationStr = "Rotated 90 degrees clockwise.";
break;
default:
NN_UNEXPECTED_DEFAULT;
}
LogIfNotNull(" - Image direction.", orientationStr);
LogIfNotNull(" - Software name.", metaData.software);
LogIfNotNull(" - Imaging date and time.", metaData.dateTime);
if (metaData.makerNote != nullptr)
{
NN_LOG(" - Manufacturer's note: %zu bytes.\n", metaData.makerNoteSize);
}
}
void GetDownsampledPixelData(
Buffer *pOutPixelData,
const nn::image::Dimension pixelDim,
const Buffer &sourceData,
const nn::image::Dimension sourceDim)
{
// Downsamples the pixel data. (The GPU is supposed to do this, but does not, so this does the same thing using the CPU.)
uint8_t *sourceBuf = reinterpret_cast<uint8_t*>(sourceData.GetDataPtr());
size_t sourceWidthAligned = sourceData.GetDataSize() / sourceDim.height / BytesPerPixel;
uint8_t *pixelBuf = reinterpret_cast<uint8_t*>(pOutPixelData->GetDataPtr());
size_t dstWidthAligned = pOutPixelData->GetDataSize() / pixelDim.height / BytesPerPixel;
int scale = sourceDim.width / pixelDim.width;
for (int y = 0; y < pixelDim.height; y++)
{
for (int x = 0; x < pixelDim.width; x++)
{
uint8_t *source = sourceBuf + (y * scale * sourceWidthAligned + x * scale) * BytesPerPixel;
uint8_t *pixel = pixelBuf + (y * dstWidthAligned + x) * BytesPerPixel;
// [Input Example]
// Created by changing the channel to explicitly indicate that it is a thumbnail.
pixel[0] = source[2]; // R <- B
pixel[1] = source[0]; // G <- R
pixel[2] = source[1]; // B <- G
pixel[3] = source[3]; // A
}
}
}
bool TryEncodeWithGivenQuality(
size_t *pOutActualSize,
Buffer *pOutJpeg,
nn::image::JpegEncoder *pEncoder,
const int quality)
{
// Encoding quality settings.
pEncoder->SetQuality(quality);
// Analyze pixel data.
nn::image::JpegStatus jpegStatus = pEncoder->Analyze();
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok);
// Encoding process.
jpegStatus = pEncoder->Encode(
pOutActualSize,
pOutJpeg->GetDataPtr(), pOutJpeg->GetDataSize());
switch (jpegStatus)
{
case nn::image::JpegStatus_ShortOutput:
return false;
default:
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok); // Unknown error.
}
return true;
}
bool EncodeThumbnailData(
size_t *pOutActualSize,
Buffer *pOutJpeg,
const Buffer &sourceData,
const nn::image::Dimension &sourceDim,
const int scale,
const int alignmentSize)
{
// Calculate the size of the thumbnail image.
nn::image::Dimension dim = {sourceDim.width / scale, sourceDim.height / scale};
nn::image::Dimension dimAligned = {
dim.width % alignmentSize == 0? dim.width: dim.width - dim.width % alignmentSize + alignmentSize,
dim.height};
// Pixel data for the thumbnail. This work buffer can be deallocated if the thumbnail encoding has finished.
Buffer pixelData(dimAligned.width * dimAligned.height * BytesPerPixel);
// Reduce pixel data.
GetDownsampledPixelData(&pixelData, dim, sourceData, sourceDim);
// Encoder instance.
nn::image::JpegEncoder encoder(nn::util::pmr::NewDeleteResource());
// Sets pixel data.
encoder.SetPixelData(pixelData.GetDataPtr(), dimAligned, alignmentSize);
// For example, this process can be retried while manipulating the image quality parameters until it fits in the buffer.
static const int LowestQuality = 1;
// [Input Example]
// Set the image quality. (Starting from 100, the sampling ratio is fixed by default to 4:2:0.)
int lastQuality = 100;
while (!TryEncodeWithGivenQuality(pOutActualSize, pOutJpeg, &encoder, lastQuality))
{
if (lastQuality <= LowestQuality)
{
// An error occurs if the quality cannot be lowered any further.
NN_LOG("The buffer where the JPEG data is written is insufficient. Cancels the thumbnail image creation.");
return false;
}
// Lowers the quality and retries with the lower limit of LowestQuality (1).
lastQuality = std::max(lastQuality - 7, LowestQuality);
NN_LOG("The buffer where the JPEG data is written is insufficient. Lower the quality to %d and try again.", lastQuality);
}
// pixelData can be deallocated after this point.
NN_LOG(" - Thumbnail: width %zu px, height %zu px, image quality %d, %zu bytes\n", dim.width, dim.height, lastQuality, *pOutActualSize);
return true;
}
void SetupExifBuilder(
nn::image::ExifBuilder *pOutExifBuilder,
Buffer *pThumbnail,
const Buffer &sourceData,
const nn::image::Dimension &sourceDim)
{
NN_LOG("Set the Exif information.");
// [Input Example]
// Set metadata. Because only a pointer is set without copying character or binary strings, make sure that you do not deallocate the wrong region.
static const char software[] = "NintendoSDK Sample Program for JPEG library";
static char dateTime[20] = "2015:08:07 09:00:00";
static const nn::Bit8 makerNote[] = {0xBE, 0xEF};
static const char uniqueId[] = "0123456789ABCDEF0123456789abcdef";
MetaData metaData = {};
metaData.orientation = nn::image::ExifOrientation_FlipVertical; // Vertical inversion.
metaData.software = software; // Imaging software name.
metaData.softwareSize = sizeof(software);
metaData.dateTime = dateTime; // - Capture date and time.
metaData.makerNote = makerNote; // Manufacturer's notes.
metaData.makerNoteSize = sizeof(makerNote);
metaData.uniqueId = uniqueId; // Unique ID.
PrintMetaData(metaData);
pOutExifBuilder->SetOrientation(metaData.orientation);
pOutExifBuilder->SetSoftware(metaData.software, metaData.softwareSize);
pOutExifBuilder->SetDateTime(metaData.dateTime, 20);
pOutExifBuilder->SetMakerNote(metaData.makerNote, metaData.makerNoteSize);
pOutExifBuilder->SetUniqueId(metaData.uniqueId, 33);
// [Input Example]
// Create the thumbnail image (no alignment (1)).
size_t thumbSize;
// The application matches the size to display as a thumbnail. (For example, a length of 120 pixels.)
int scale = std::max(1, std::max(sourceDim.width / 120, sourceDim.height / 120));
if (!EncodeThumbnailData(&thumbSize, pThumbnail, sourceData, sourceDim, scale, 1))
{
NN_LOG("The thumbnail image could not be created. Omitted.");
}
else
{
pOutExifBuilder->SetThumbnail(pThumbnail->GetDataPtr(), thumbSize);
}
// Analysis of thumbnail pixel data.
nn::image::JpegStatus exifStatus = pOutExifBuilder->Analyze();
switch (exifStatus)
{
case nn::image::JpegStatus_OutputOverabounds:
// In this case, ExifBuilder.IsBuildable() is false.
NN_LOG("Due to an excess of metadata being set, the Exif information size exceeds the maximum (65,527 bytes).");
return;
default:
NN_ASSERT(exifStatus == nn::image::JpegStatus_Ok); // Unknown error code.
}
}
void EncodePixelData(
size_t *pOutActualSize,
Buffer *pOutJpeg,
const Buffer &pixelData,
const nn::image::Dimension dim,
const int alignmentSize)
{
/* ----------------------------------------------------------------------------------------
Exif information build phase.
(Skip this phase if you are not using either metadata or thumbnails.)
*/
Buffer thumbJpeg(32 * 1024); // Reserve enough memory for the thumbnail. This buffer cannot be deallocated until the JPEG encoding of the source data is finished.
Buffer exifWorkBuf(nn::image::ExifBuilder::GetWorkBufferSize()); // This buffer cannot be deallocated until the JPEG encoding of the source data is finished.
nn::image::ExifBuilder builder(exifWorkBuf.GetDataPtr(), exifWorkBuf.GetDataSize());
// ExifBuilder is configured with the appropriate metadata.
SetupExifBuilder(&builder, &thumbJpeg, pixelData, dim);
/* ----------------------------------------------------------------------------------------
Encoding Setting Phase
*/
// Encoder instance.
nn::image::JpegEncoder encoder(nn::util::pmr::NewDeleteResource());
// Set the pixel data.
encoder.SetPixelData(pixelData.GetDataPtr(), dim, alignmentSize);
// [Input Example]
// Set with regard to image quality. (Maximum image quality 100, sampling ratio 4:4:4.)
NN_LOG("Encoding settings.\n");
NN_LOG(" - Picture quality parameter: 100\n");
NN_LOG(" - YCbCr samplet ratio: 4:4:4\n");
encoder.SetQuality(100);
encoder.SetSamplingRatio(nn::image::JpegSamplingRatio_444);
/* ----------------------------------------------------------------------------------------
Pixel Data Analysis Phase
*/
// Analyze the pixel data.
nn::image::JpegStatus jpegStatus = encoder.Analyze();
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok); // Currently, only OK is returned.
/* ----------------------------------------------------------------------------------------
JPEG Encoding Phase
*/
NN_LOG("Ecodes as JPEG data.");
// Encoding process.
if (builder.IsBuildable())
{
// If an ExifBuilder with IsBuildable() set to true is available, encode by setting Exif information.
jpegStatus = encoder.Encode(
pOutActualSize,
pOutJpeg->GetDataPtr(), pOutJpeg->GetDataSize(),
&builder); // Set Exif information.
}
else
{
// IsBuildable() is false if an error occurs during the Exif information build phase.
// In this case, encodes without setting Exif information.
jpegStatus = encoder.Encode(
pOutActualSize,
pOutJpeg->GetDataPtr(), pOutJpeg->GetDataSize());
}
// Deallocates the work memory for thumbnails and ExifBuilder and the work memory for JpegEncoder.
switch (jpegStatus)
{
case nn::image::JpegStatus_ShortOutput:
NN_LOG("The buffer where the JPEG data is written is insufficient.");
return;
default:
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok); // Unknown error.
NN_LOG("Output JPEG data: Width %zu px, Height %zu px, %zu bytes\n", dim.width, dim.height, *pOutActualSize);
}
}
}
extern "C" void nnMain()
{
// Initialize the file system.
nn::fs::SetAllocator(Allocate, Deallocate);
// Mount the file system on the PC.
// Note: Set an appropriate directory to mount as the host.
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::MountHost("Asset", "C:\\JpegInputDir"));
// For example, when encoding a texture, require the width to be 4-byte aligned.
static const int AlignmentSize = 4; // Specify the alignment.
static const nn::image::Dimension dim = {639, 480}; // Set to an inconvenient 639 to generate padding.
static const nn::image::Dimension dimAligned = {
(dim.width % AlignmentSize == 0)? dim.width: dim.width - dim.width % AlignmentSize + AlignmentSize,
dim.height}; // Aligned size.
// This buffer can be deallocated if JPEG encoding has finished.
Buffer pixelData(dimAligned.width * dimAligned.height * BytesPerPixel);
// Initialize the color buffer.
uint8_t *pixelBuf = reinterpret_cast<uint8_t*>(pixelData.GetDataPtr());
for (int y = 0; y < dim.height; y++)
{
const uint8_t R = static_cast<uint8_t>(((float)y / dim.height * 0xFF) + 0.5f);
for (int x = 0; x < dim.width; x++)
{
const uint8_t G = static_cast<uint8_t>(((float)x / dim.width) * 0xFF + 0.5f);
const uint8_t B = static_cast<uint8_t>(
(std::sqrt(x * x + y * y) / sqrt(dim.width * dim.width + dim.height * dim.height))
* 0xFF + 0.5f);
// Arranged by R, G, B, and A regardless of endianness.
uint8_t *pixel = pixelBuf + (x + y * dimAligned.width) * BytesPerPixel;
pixel[0] = R;
pixel[1] = G;
pixel[2] = B;
pixel[3] = 0xFF; // A
}
}
// JPEG encoding process.
size_t actualSize;
Buffer jpegData(1024 * 1024); // Reserves a suitable 1 MiB for JPEG data.
EncodePixelData(&actualSize, &jpegData, pixelData, dim, AlignmentSize);
pixelData.Release(); // Explicitly release for sampling.
FileData::Save("Asset:/output.jpg", jpegData.GetDataPtr(), actualSize);
jpegData.Release(); // Explicitly release for sampling.
// FS finalization process.
nn::fs::Unmount("Asset");
return;
}
