// The animator generation processing can be customized to generate a proprietary animator.
class MyLayout : public nn::ui2d::Layout
{
public:
//---------------------------------------------------------------------------
// Search for the animator using the segment tag name.
nn::ui2d::Animator* FindAnimatorByTagName(const char* pName)
{
auto it_end = GetAnimTransformList().end();
for (auto it = GetAnimTransformList().begin(); it != it_end; ++it)
{
MyGroupAnimator* pAnimator = nn::ui2d::DynamicCast<MyGroupAnimator*>(&*it);
if (pAnimator != nullptr && pAnimator->GetTagName() != nullptr)
{
if (nn::ui2d::detail::EqualsName<nn::ui2d::AnimTagNameStrMax>(pAnimator->GetTagName(), pName))
{
return pAnimator;
}
}
}
return nullptr;
}
protected:
//------------------------------------------------------------------------------
// Override the processing to replace GroupAnimator, created at times such as when creating buttons, with MyGroupAnimator.
virtual nn::ui2d::GroupAnimator* DoCreateAndSetupGroupAnimator_(nn::gfx::Device* pDevice, const char* pTagName, const nn::ui2d::AnimResource& animResource, bool enabled) NN_OVERRIDE
{
// Generate MyGroupAnimator.
MyGroupAnimator* pAnimator = CreateAnimTransform<MyGroupAnimator>(pDevice, animResource);
pAnimator->Setup(animResource, GetGroupContainer(), 0, enabled);
// Remember the segment tag name as an additional Setup.
pAnimator->SetTagName(pTagName);
return pAnimator;
}
//------------------------------------------------------------------------------
// Override the processing for replacing GroupAnimator, created at such times as when creating buttons, with MyGroupAnimator.
virtual nn::ui2d::GroupArrayAnimator* DoCreateAndSetupGroupArrayAnimator_(nn::gfx::Device* pDevice, const char* pTagName, const nn::ui2d::AnimResource& animResource, bool enabled) NN_OVERRIDE
{
NN_UNUSED(pTagName);
const nn::ui2d::ResAnimationBlock *const pAnimBlock = animResource.GetResourceBlock();
if (!pAnimBlock)
{
return NULL;
}
// Break up and use CreateAnimTransform to reduce the new count.
void* pPtr = Layout::AllocateMemory(sizeof(MyGroupArrayAnimator) + animResource.GetGroupCount() * sizeof(nn::ui2d::Group*));
if (pPtr == NULL) {
return NULL;
}
MyGroupArrayAnimator* pAnimator = new(pPtr) MyGroupArrayAnimator();
pAnimator->SetResource(pDevice, GetResourceAccessor(), pAnimBlock);
this->GetAnimTransformList().push_back(*pAnimator);
pAnimator->Setup(animResource, GetGroupContainer(), nn::util::BytePtr(pPtr, sizeof(MyGroupArrayAnimator)).Get<nn::ui2d::Group*>(), enabled);
pAnimator->SetTagName(pTagName);
return pAnimator;
}

//------------------------------------------------------------------------------
// Initialize all segment tag animations linked to the loaded layout.
virtual void CreateAnimators_(nn::gfx::Device* pDevice, nn::ui2d::Layout* pLayout) NN_OVERRIDE
{
if (GetPreviewMode() == PreviewMode_Animation || GetLayout() == nullptr)
{
return;
}
// Make a list of the segment tags from the layout, and create all animations in the non-playback state.
MyLayout* pMyLayout = nn::ui2d::DynamicCast<MyLayout*>(pLayout);
const int animTagCount = pLayout->AcquireAnimTagNameCount();
for (int i = 0; i < animTagCount; i++)
{
const char* pAnimTagName = pLayout->AcquireAnimTagNameByIndex(i);
if (pAnimTagName != nullptr)
{
if (pMyLayout->FindAnimatorByTagName(pAnimTagName) == nullptr)
{
pMyLayout->CreateGroupAnimatorAuto(pDevice, pAnimTagName, false);
}
}
}
}

}

// File Loading Function Using nn::fs
//------------------------------------------------------------------------------
bool FileLoad_(
void** pOutBuffer,
size_t* pOutBufferSize,
const char* filePath,
SimpleAllocator* pAllocator,
int alignment)
{
char filePathName[nn::fs::EntryNameLengthMax];
memcpy(reinterpret_cast<void*>(&filePathName), filePath, strlen(filePath) + 1);
nn::fs::FileHandle fsHandle;
nn::Result ret = nn::fs::OpenFile(&fsHandle, static_cast<char*>(filePathName), static_cast<nn::fs::OpenMode>(nn::fs::OpenMode_Read));
if (ret.IsFailure())
{
return false;
}
int64_t fsSize;
nn::fs::GetFileSize(&fsSize, fsHandle);
*pOutBufferSize = static_cast<size_t>(fsSize);
*pOutBuffer = pAllocator->Alloc(*pOutBufferSize + 4, alignment);
if (!(*pOutBuffer))
{
nn::fs::CloseFile(fsHandle);
return false;
}
memset(*pOutBuffer, 0, *pOutBufferSize + 4);
nn::fs::ReadFile(fsHandle, 0, *pOutBuffer, *pOutBufferSize);
nn::fs::CloseFile(fsHandle);
return true;
}

// Update the input device state for viewer update purposes.
static nn::ui2d::viewer::Viewer::InputDeviceState CalcForInputForViewer_()
{
// Update the input device.
{
g_ControllerManager.Update();
}
// Set the values in a structure for transmitting the input state to the viewer.
// Because this sample does not use pad operations, pad-related updates are omitted.
nn::ui2d::viewer::Viewer::InputDeviceState inputDeviceState;
memset(&inputDeviceState, 0, sizeof(nn::ui2d::viewer::Viewer::InputDeviceState));
#if defined(NN_BUILD_CONFIG_OS_WIN32)
inputDeviceState.isMousePointerOn = true;
inputDeviceState.pointerX = static_cast<float>(g_UserInputs.mouseState.x);
inputDeviceState.pointerY = static_cast<float>(g_UserInputs.mouseState.y);
inputDeviceState.isTrigMouseLButton = g_UserInputs.mouseStateOld.buttons.IsAllOff() && g_UserInputs.mouseState.buttons.IsAnyOn();
inputDeviceState.isReleaseMouseLButton = g_UserInputs.mouseStateOld.buttons.IsAnyOn() && g_UserInputs.mouseState.buttons.IsAllOff();
#else
inputDeviceState.isMousePointerOn = false;
nns::hid::TouchScreen* pTouchScreen = reinterpret_cast<nns::hid::TouchScreen*>(g_ControllerManager.GetController(nns::hid::ControllerId_TouchScreen, 0));
const std::vector<nns::hid::TouchScreen::TouchState>& states = pTouchScreen->GetTouchStates();
for (size_t i = 0; i < states.size(); i++)
{
// If there were multiple touches, only process the first one.
const nns::hid::TouchScreen::TouchState& state = states[i];
inputDeviceState.pointerX = state.position.x * (g_GfxFramework.GetDisplayWidth() / 1280.0f);
inputDeviceState.pointerY = state.position.y * (g_GfxFramework.GetDisplayHeight() / 720.0f);
inputDeviceState.isMousePointerOn = true;
inputDeviceState.isTrigMouseLButton = state.IsBegan();
inputDeviceState.isReleaseMouseLButton = state.IsEnded();
break;
}
#endif // defined(NN_BUILD_CONFIG_OS_WIN32)
return inputDeviceState;
}

g_FrameIndex += 1;
}

// This is the provisional game background rendering.
// Draw the grid and cube using PrimitiveRenderer.
void DrawDummyGameScene_(nn::gfx::CommandBuffer& commandBuffer)
{
// Configure the view and projection.
const float radius = 20.f;
const float x = radius * sin(g_FrameIndex / 500.f);
const float z = radius * cos(g_FrameIndex / 500.f);
g_pPrimitiveRenderer->SetDefaultParameters();
nn::util::Matrix4x3fType viewMatrix;
nn::util::Matrix4x4fType projectionMatrix;
nn::util::Matrix4x3f modelMatrix;
nn::util::MatrixIdentity(&modelMatrix);
nn::util::Vector3f translate;
nn::util::VectorSet(&translate, 0.f, 0.f, 1.f);
nn::util::MatrixSetAxisW(&modelMatrix, translate);
const nn::util::Uint8x4 white = { { 255, 255, 255, 255 } };
const nn::util::Uint8x4 red = { { 255, 0, 0, 255 } };
const nn::util::Uint8x4 green = { { 0, 255, 0, 255 } };
const nn::util::Uint8x4 blue = { { 0, 0, 255, 255 } };
// viewMatrix
const nn::util::Vector3fType camPos = { x, 10.f, z };
const nn::util::Vector3fType camTarget = { 0.f, 0.f, 0.f };
const nn::util::Vector3fType camUp = { 0.f, 1.f, 0.f };
nn::util::MatrixLookAtRightHanded(&viewMatrix, camPos, camTarget, camUp);
g_pPrimitiveRenderer->SetViewMatrix(&viewMatrix);
// projectionMatrix
const float fovy = nn::util::FloatPi / 3.0f;
const float aspect = static_cast<float>(g_GfxFramework.GetDisplayWidth()) / static_cast<float>(g_GfxFramework.GetDisplayHeight());
nn::util::MatrixPerspectiveFieldOfViewRightHanded(&projectionMatrix, fovy, aspect, 0.1f, 1000.f);

// Depth enable.

// Build an independent HUD viewer feature for the animator class.
// Display the internal state and perform the playback operation.
void AddOneAnimationNodeToHud_(nn::ui2d::viewer::DebugHudService& hudService, nn::ui2d::Animator* pAnimator)
{
hudService.StartLayout(pAnimator->GetTagName());

// Express the playback state simply as a character.

hudService.GenerateLabel(playingState);
// Play button.
if (hudService.GenerateButton(pAnimator->IsPlaying() ? "Play" : "Stop"))
{
if (pAnimator->IsPlaying())
{

pAnimator->SetEnabled(false);
}
else
{
pAnimator->SetEnabled(true);
pAnimator->PlayAuto(1.0f);
}
}
// Slider
int iframe = static_cast<int>(pAnimator->GetFrame());
const char* pLable = pAnimator->GetTagName() == nullptr ? " " : pAnimator->GetTagName();
if (hudService.GenerateSlider(pLable, &iframe, 0, static_cast<int>(pAnimator->GetFrameMax())))
{
pAnimator->SetEnabled(true);
pAnimator->SetFrame(static_cast<float>(iframe));
}
}
hudService.EndLayout();
}
//---------------------------------------------------------------------------
// Build an independent HUD viewer feature for the layout.
void AddLayoutNodesToHudRecursively_(nn::ui2d::viewer::DebugHudService& hudService, nn::ui2d::Layout* pLayout)
{
// Do not perform node creation if there are no subordinate animators or parts.
if (pLayout->GetAnimTransformList().empty() && pLayout->GetPartsPaneList().empty())
{
return;
}
// Start the node corresponding to the layout.
if (pLayout->IsPartsLayout())
{
char nodeName[64];
nn::util::SNPrintf(nodeName, 64, "%s[%s]", pLayout->GetRootPane()->GetName(), pLayout->GetName());
hudService.StartNode(nodeName);
}
else
{
hudService.StartNode(pLayout->GetName());
}
// Hierarchy below the layout.
{
// Create the animator node.
for (auto it = pLayout->GetAnimTransformList().begin(), it_end = pLayout->GetAnimTransformList().end(); it != it_end; ++it)
{
auto* pAnimator = nn::ui2d::DynamicCast<nn::ui2d::Animator*>(&(*it));
if (pAnimator != nullptr)
{
AddOneAnimationNodeToHud_(hudService, pAnimator);
}
}
// Perform recursive processing for parts panes.
for (nn::ui2d::PartsPaneList::iterator it = pLayout->GetPartsPaneList().begin(), it_end = pLayout->GetPartsPaneList().end(); it != it_end; ++it)
{
AddLayoutNodesToHudRecursively_(hudService, static_cast<nn::ui2d::Layout*>((*it).GetLayout()));
}
}
// End the node corresponding to the layout.
hudService.EndNode();
}

void GfxFrameworkMakeCommandCallback_(nns::gfx::GraphicsFramework* pGraphicsFramework, int bufferIndex, void* pUserData)
{
NN_UNUSED(pUserData);
const int cmdBufferIndex = bufferIndex;
pGraphicsFramework->BeginFrame(cmdBufferIndex);
nn::gfx::ColorTargetView* pTargetView = pGraphicsFramework->GetColorTargetView();
nn::gfx::DepthStencilView* pDepthStencilView = pGraphicsFramework->GetDepthStencilView();

// Clear the framebuffer.
{

g_LayoutViewer.GetBackgroundColor(bgColor);
pCmdBuffer->ClearColor(pTargetView, bgColor[0], bgColor[1], bgColor[2], bgColor[3], NULL);
pCmdBuffer->ClearDepthStencil(pDepthStencilView, 1.0f, 0, nn::gfx::DepthStencilClearMode_DepthStencil, NULL);
}
// Set the render state.
pCmdBuffer->SetRenderTargets(1, &pTargetView, pDepthStencilView);
// Configure the viewport scissor settings.
pCmdBuffer->SetViewportScissorState(pGraphicsFramework->GetViewportScissorState());
// Blend mode.
pCmdBuffer->SetBlendState(pGraphicsFramework->GetBlendState(nns::gfx::GraphicsFramework::BlendStateType_Alpha));
}
g_pPrimitiveRenderer->Update(cmdBufferIndex);
if (g_LayoutViewer.IsBackgroundImageEnabled())
{
// Render the viewer background image.
const float width = g_LayoutViewer.GetBackgroundImageSize().width / g_GfxFramework.GetDisplayWidth() * 2.0f;
const float height = g_LayoutViewer.GetBackgroundImageSize().height / g_GfxFramework.GetDisplayHeight() * 2.0f;
nn::util::Vector3fType center = { -width * 0.5f, -height * 0.5f, 0.0f };
nn::util::Vector3fType size = { width, height, 0.0f };
const nn::util::Uint8x4 white = { { 255, 255, 255, 255 } };
g_pPrimitiveRenderer->SetDefaultParameters();
g_pPrimitiveRenderer->SetColor(white);
g_pPrimitiveRenderer->DrawQuad(
pCmdBuffer,
center,
size,
g_LayoutViewer.GetBackgroundImage(),
g_GraphicsResource.GetSamplerDescriptorSlot(nn::ui2d::TexWrap_Clamp, nn::ui2d::TexWrap_Clamp, nn::ui2d::TexFilter_Linear, nn::ui2d::TexFilter_Linear));
pCmdBuffer->FlushMemory(nn::gfx::GpuAccess_ColorBuffer);
pCmdBuffer->InvalidateMemory(nn::gfx::GpuAccess_Texture);
}
else
{
// Render the provisional game background.
{
DrawDummyGameScene_(*pCmdBuffer);
pCmdBuffer->FlushMemory(nn::gfx::GpuAccess_ColorBuffer);
pCmdBuffer->InvalidateMemory(nn::gfx::GpuAccess_Texture);
}
}
// Free the lifespan texture per frame.
g_DynamicTextureManager.FreePerFrameDynamicTexture();
// Render the layout viewer.
{
// Depth disabled.
g_pPrimitiveRenderer->SetDepthStencilState(pCmdBuffer, nns::gfx::PrimitiveRenderer::DepthStencilType::DepthStencilType_DepthNoWriteTest);
g_LayoutViewer.DrawCaptureTexture(pGraphicsFramework->GetDevice(), *pCmdBuffer);
// Set the render state.
pCmdBuffer->SetRenderTargets(1, &pTargetView, pDepthStencilView);
pCmdBuffer->SetViewportScissorState(pGraphicsFramework->GetViewportScissorState());
pCmdBuffer->SetBlendState(pGraphicsFramework->GetBlendState(nns::gfx::GraphicsFramework::BlendStateType_Alpha));
g_LayoutViewer.DrawLayout(*pCmdBuffer);
g_LayoutViewer.DrawSystem(*pCmdBuffer);
}

{

info.SetDefault();
info.SetGpuAccessFlags(gpuAccessFlags);
// Set an appropriate size for the size.
// Adjust the size according to the size required by the data to display.
const size_t constantBufferSize = 1024 * 1024 * 2;
const size_t bufferAlignment = nn::gfx::Buffer::GetBufferAlignment(pDevice, info);
const size_t constantBufferTotalSize = nn::util::align_up(constantBufferSize, bufferAlignment);
nn::gfx::MemoryPool* pMemoryPool = nullptr;

{
nn::gfx::MemoryPoolInfo memPoolInfo;
memPoolInfo.SetDefault();
memPoolInfo.SetMemoryPoolProperty(nn::gfx::MemoryPoolProperty_CpuUncached | nn::gfx::MemoryPoolProperty_GpuCached);
const size_t memoryPoolSize = nn::util::align_up(constantBufferTotalSize, nn::gfx::MemoryPool::GetPoolMemorySizeGranularity(pDevice, memPoolInfo));
const size_t MemoryPoolAlignment = nn::gfx::MemoryPool::GetPoolMemoryAlignment(pDevice, memPoolInfo);
const size_t ui2dDrawMemoryPoolsize = memoryPoolSize * nn::ui2d::ConstantBufferCount;
g_offsetToUi2dDrawMemoryPool = g_GfxFramework.AllocatePoolMemory(memoryPoolType, ui2dDrawMemoryPoolsize, MemoryPoolAlignment);
NN_ASSERT(g_offsetToUi2dDrawMemoryPool != nn::gfx::util::MemoryPoolAllocator::InvalidOffset, "Can't allocate the memory pool[size=%d] for ui2d-constantBuffer. Too large data to preview !", ui2dDrawMemoryPoolsize);
}
pMemoryPool = g_GfxFramework.GetMemoryPool(memoryPoolType);
}
{
nn::font::GpuBuffer::InitializeArg arg;
arg.SetGpuAccessFlag(gpuAccessFlags);
arg.SetBufferSize(constantBufferTotalSize);
arg.SetBufferCount(nn::ui2d::ConstantBufferCount);
arg.SetMemoryPool(pMemoryPool);
arg.SetMemoryPoolOffset(g_offsetToUi2dDrawMemoryPool);
arg.SetAllocator(AllocFunction_, NULL);
arg.SetAllocateSyncFlag(false);
g_Ui2dConstantBuffer.Initialize(pDevice, arg);
}
