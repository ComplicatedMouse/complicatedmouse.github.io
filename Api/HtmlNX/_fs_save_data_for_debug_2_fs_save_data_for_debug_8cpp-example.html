#include <cstdlib>
#include <cstring>
#include <nn/fs.h>
#include <nn/nn_Assert.h>
#include <nn/nn_Abort.h>
#include <nn/nn_Log.h>
#include <nn/fs/fs_Debug.h>
namespace {
struct SaveDataStructure
{
int value;
};
} // Namespace.
extern "C" void nnMain()
{
nn::Result result;
// Mount the save data as "save."
NN_LOG("Mount save data as \"save\".\n");
result = nn::fs::MountSaveDataForDebug("save");
// Always abort when a failure occurs.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
const size_t FileSize = sizeof(SaveDataStructure);
// Check the existence of the file.
{
nn::fs::DirectoryEntryType directoryEntryType;
result = nn::fs::GetEntryType(&directoryEntryType, "save:/file");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// Create the file if it does not already exist.
result = nn::fs::CreateFile("save:/file", FileSize);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// This error handling is not necessary when parent directories exist without fail.
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file already exists.
// This error handling is not needed when the file already exists and that is not an issue.
// Note that there is no assurance that the file size is equal to FileSize when you are not using error handling.
// Delete the target file and re-create it when necessary.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient data save region for the save data.
// Increase the data save region.
NN_ABORT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// Write initial data.
NN_LOG("Create the initial data as save:/file.\n");
nn::fs::FileHandle fileHandle;
result = nn::fs::OpenFile(&fileHandle, "save:/file", nn::fs::OpenMode_Write);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when you are only opening existing files.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open already.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
SaveDataStructure data = {0};
result = nn::fs::WriteFile(fileHandle, 0, &data, sizeof(data), nn::fs::WriteOption::MakeValue(nn::fs::WriteOptionFlag_Flush));
if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient data save region for the save data.
// This error handling is not necessary when you would not specify OpenMode_AllowAppend when opening files.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
nn::fs::CloseFile(fileHandle);
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
}
// Load and update data.
{
SaveDataStructure data;
nn::fs::FileHandle fileHandle;
result = nn::fs::OpenFile(&fileHandle, "save:/file", nn::fs::OpenMode_Read | nn::fs::OpenMode_Write);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when you are only opening existing files.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open already.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// Read data.
size_t readSize = 0;
NN_LOG("Read save:/file.\n");
// No error handling is needed because an abort occurs within the library when loading fails.
(void)nn::fs::ReadFile(&readSize, fileHandle, 0, &data, sizeof(data));
NN_ASSERT_EQUAL(readSize, sizeof(data));
NN_LOG("data.value: 0x%x\n", data.value);
// Updates the data.
NN_LOG("Increment data.value: 0x%x -> 0x%x\n", data.value, data.value + 1);
data.value++;
// Write updated data.
NN_LOG("Write save:/file.\n");
result = nn::fs::WriteFile(fileHandle, 0, &data, sizeof(data), nn::fs::WriteOption::MakeValue(nn::fs::WriteOptionFlag_Flush));
if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient data save region for the save data.
// This error handling is not necessary when you would not specify OpenMode_AllowAppend when opening files.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Close save:/file.\n");
nn::fs::CloseFile(fileHandle);
// Commit the updated save data.
// No error handling is needed because an abort occurs within the library when a commit fails.
NN_LOG("Commit \"save\".\n");
(void)nn::fs::Commit("save");
}
// Unmount.
NN_LOG("Unmount \"save\".\n\n");
nn::fs::Unmount("save");
return;
}
