#include <gtest/gtest.h>
void InitialzeCurrentDirectory()
{
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
// Get the command-line argument.
int argc = nn::os::GetHostArgc();
char** argv = nn::os::GetHostArgv();
char currentDirectory[nn::fs::EntryNameLengthMax] = {};
for (int i = 0; i < argc; ++i) {
if (strcmp("--current_directory", argv[i]) == 0) {
if (i < argc - 1)
nn::util::Strlcpy(currentDirectory, argv[i + 1], nn::fs::EntryNameLengthMax);
}
}
// Abort if the current directory is not set with the options.
if (nn::util::Strncmp(currentDirectory, "\0", nn::fs::EntryNameLengthMax) == 0)
{
NN_ABORT("Please set current directory using --current_directory option\n");
}
// Initialization that should be performed before GoogleTest static objects are initialized.
// Set the current directory for GoogleTest.
chdir(currentDirectory);
#endif
}
// Startup Functions
extern "C" void nninitStartup()
{
// Change the size of the memory heap region.
const size_t MemoryHeapSize = 2ull * 1024 * 1024;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::SetMemoryHeapSize(MemoryHeapSize));
// Get the memory block from the memory heap.
uintptr_t address;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::AllocateMemoryBlock(&address, MemoryHeapSize));
nn::init::InitializeAllocator(reinterpret_cast<void*>(address), MemoryHeapSize);
InitialzeCurrentDirectory();
}
// First test
TEST(AssertTest, SuccessTest)
{
// Test for whether ASSERT succeeds.
ASSERT_EQ(10, 10);
}
// Second test
TEST(AssertTest, FailTest)
{
// Test for whether ASSERT fails.
ASSERT_EQ(10, 20);
}
extern "C" void nnMain()
{
// Get the command-line argument.
int argc = nn::os::GetHostArgc();
char** argv = nn::os::GetHostArgv();
// Mount the file system on the host PC.
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::MountHostRoot());
// Initialize GoogleTest.
testing::InitGoogleTest(&argc, argv);
// Run the defined TEST.
// Return 0 if all tests are successful, and 1 otherwise.
int testResult = RUN_ALL_TESTS();
// Unmount the file system mounted with MountHostRoot().
nn::fs::UnmountHostRoot();
std::quick_exit(testResult);
}
