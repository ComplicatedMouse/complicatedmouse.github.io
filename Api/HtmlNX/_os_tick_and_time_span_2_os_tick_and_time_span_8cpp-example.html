#include <stdint.h>
#include <nn/nn_Macro.h>
#include <nn/nn_Log.h>
#include <nn/nn_TimeSpan.h>
#include <nn/os.h>
//-----------------------------------------------------------------------------
//
// Sample Program
//
void SampleCodeFunction()
{
// Get and display the tick frequency.
int64_t freq = nn::os::GetSystemTickFrequency();
NN_LOG("Tick Frequency = %lld\n\n", freq);
// Get and display the tick value for one second.
// At the same time, also display the value from converting to the TimeSpan format and then converting to milliseconds.
NN_LOG(" Tick Count TimeSpan\n");
NN_LOG("------------------------------------------\n");
for (int i=0; i<10; ++i)
{
nn::os::Tick tick = nn::os::GetSystemTick();
nn::TimeSpan timeSpan = nn::os::ConvertToTimeSpan( tick );
NN_LOG("%d: %16lld %16lld(msec)\n", i, tick.GetInt64Value(), timeSpan.GetMilliSeconds());
// Wait one second.
nn::os::SleepThread( nn::TimeSpan::FromMilliSeconds(1000) );
}
}
//
// The main function.
//
extern "C" void nnMain()
{
// Issue sample code function.
SampleCodeFunction();
}
