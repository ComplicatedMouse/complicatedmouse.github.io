#if defined(NN_BUILD_TARGET_PLATFORM_NX)
// Handles system messages.
void MessageHandlerThreadFunction(void*) NN_NOEXCEPT
{
while (true)
{
nn::oe::Message message = nn::oe::PopNotificationMessage();
nn::oe::PerformanceMode perfMode;
switch (message)
{
case nn::oe::MessageExitRequest:
g_ShouldExit = true;
return; // Time to exit the function, so stop processing messages and exit this thread.
case nn::oe::MessageOperationModeChanged:
NN_LOG("** Operation mode changed to ");
g_OperationMode = nn::oe::GetOperationMode();
g_IsOperationModeChanged = true;
if (g_OperationMode == nn::oe::OperationMode_Console)
{
NN_LOG("Console\n");
}
else if (g_OperationMode == nn::oe::OperationMode_Handheld)
{
NN_LOG("Handheld\n");
}
break;
case nn::oe::MessagePerformanceModeChanged:
NN_LOG("** Performance mode changed to ");
// we don't actually use the performance mode to change graphics system parameters, but it could
// be used for similar purposes as operation mode as above, so we leave this in for reference.
// performance mode and operation mode are not necessarily one-to-one; console mode w/ low battery
// will yield the Normal performance mode.
perfMode = nn::oe::GetPerformanceMode();
if (perfMode == nn::oe::PerformanceMode_Boost)
{
NN_LOG("Boost\n");
}
else if (perfMode == nn::oe::PerformanceMode_Normal)
{
NN_LOG("Normal\n");
}
break;
default:
break;
}
}
}
#endif

void UpdateCommandBufferReuseStatus(SimpleTextRenderer* pTextRenderer, const TextArea& textArea, SceneBaseClass* pScene) NN_NOEXCEPT
{
bool isUsingPrebuiltCb = pScene->IsPrebuiltCommandBufferUsed();
pTextRenderer->ClearText(textArea);
pTextRenderer->AppendTextPrintf(textArea, u8"Command buffer reuse : %s", (isUsingPrebuiltCb) ? u8"ON" : u8"OFF");
}
}

void SimpleFrameworkInitialize(GraphicsSystem* pGraphicsSystem, const GraphicsSystemParams& graphicsSystemParams) NN_NOEXCEPT
{
if (!g_IsInitialized)
{
NN_ASSERT(pGraphicsSystem != nullptr);
g_pGraphicsSystem = pGraphicsSystem;
nn::os::MemoryInfo memoryInfo;
nn::os::QueryMemoryInfo(&memoryInfo);
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
NN_LOG("------------------------------------------------------------------------------\n");
PrintMemoryLine("Total available memory: %12zu bytes (%8.2f MB)\n", memoryInfo.totalAvailableMemorySize);
PrintMemoryLine("Program size (code + data): %12zu bytes (%8.2f MB)\n", memoryInfo.programSize);
PrintMemoryLine("Initial allocator (malloc/new) has available: %12zu bytes (%8.2f MB)\n", nn::init::GetAllocator()->GetTotalFreeSize());
NN_LOG("------------------------------------------------------------------------------\n\n");
#endif
size_t appAllocatorSize = 67108864llu; // 64 MB
g_MemoryBlock = malloc(appAllocatorSize);
NN_ASSERT(g_MemoryBlock != nullptr);
g_Allocator.Initialize(g_MemoryBlock, appAllocatorSize);
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
PrintMemoryLine("Created application allocator of size %zu bytes (%.2f MB)\n", appAllocatorSize);
PrintMemoryLine("%zu bytes (%.2f MB) remaining in initial allocator (malloc/new)\n", nn::init::GetAllocator()->GetTotalFreeSize());
#endif
size_t romCacheBufferSize = 0;
nn::fs::QueryMountRomCacheSize(&romCacheBufferSize);
g_RomCacheBuffer = g_Allocator.Allocate(romCacheBufferSize);
if (nn::fs::MountRom("rom", g_RomCacheBuffer, romCacheBufferSize).IsFailure())
{
NN_ABORT("Failed to mount rom fs.\n");
}
InitPad();
pGraphicsSystem->Initialize(&g_Allocator, graphicsSystemParams);
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
nn::oe::SetPerformanceModeChangedNotificationEnabled(true);
nn::oe::SetOperationModeChangedNotificationEnabled(true);
size_t stackSize = 8192;
g_MessageHandlerThreadStack = g_Allocator.Allocate(stackSize, nn::os::ThreadStackAlignment);
nn::os::CreateThread(
&g_MessageHandlerThread,
MessageHandlerThreadFunction,
nullptr,
g_MessageHandlerThreadStack,
stackSize,
nn::os::DefaultThreadPriority
);
nn::os::StartThread(&g_MessageHandlerThread);
nn::oe::EnterExitRequestHandlingSection();
#endif
#if defined(USE_NXCP)
size_t profilerBufferSize = nn::profiler::MinimumBufferSize;
g_ProfilerBuffer = malloc(profilerBufferSize);
NN_ASSERT(g_ProfilerBuffer != nullptr);
nn::Result result = nn::profiler::Initialize(g_ProfilerBuffer, profilerBufferSize);
NN_ASSERT(result.IsSuccess());
#endif
g_IsInitialized = true;
}
}
void SimpleFrameworkFinalize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT
{
if (g_IsInitialized)
{
NN_ASSERT(pGraphicsSystem != nullptr);
#if defined(USE_NXCP)
nn::profiler::Finalize();
g_Allocator.Free(g_ProfilerBuffer);
#endif
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
nn::oe::LeaveExitRequestHandlingSection();
nn::os::WaitThread(&g_MessageHandlerThread);
nn::os::DestroyThread(&g_MessageHandlerThread);
g_Allocator.Free(g_MessageHandlerThreadStack);
#endif
pGraphicsSystem->Finalize();
nn::fs::Unmount("rom");
g_Allocator.Free(g_RomCacheBuffer);
g_Allocator.Finalize();
std::free(g_MemoryBlock);
g_IsInitialized = false;
}
}
bool SimpleFrameworkUpdate() NN_NOEXCEPT
{
#if defined(USE_NXCP)
nn::profiler::RecordHeartbeat(nn::profiler::Heartbeats_Main);
#endif
UpdatePad();
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
if (g_IsOperationModeChanged)
{
int width = g_pGraphicsSystem->GetRenderTargetWidth();
int height = g_pGraphicsSystem->GetRenderTargetHeight();
if (width == 1920 && height == 1080)
{
int newWidth = width;
int newHeight = height;
if (g_OperationMode == nn::oe::OperationMode_Handheld)
{
newWidth = 1280;
newHeight = 720;
}
g_pGraphicsSystem->SetScreenResolution(newWidth, newHeight);
}
g_IsOperationModeChanged = false;
}
#endif
return g_ShouldExit;
}
void SimpleFrameworkMain(SceneBaseClass* pScene, const GraphicsSystemParams& graphicsSystemParams) NN_NOEXCEPT
{
GraphicsSystem graphicsSystem;
SimpleFrameworkInitialize(&graphicsSystem, graphicsSystemParams);
pScene->Initialize(&graphicsSystem);
while (!SimpleFrameworkUpdate())
{
if (graphicsSystem.ScreenResolutionWasChanged())
{
pScene->NotifyScreenResolutionChange();
}
graphicsSystem.DrawBegin();
pScene->DrawFrame();
graphicsSystem.DrawEnd();
}
pScene->Finalize();
SimpleFrameworkFinalize(&graphicsSystem);
}
void SimpleFrameworkMainWithDebugInfo(SceneBaseClass* pScene, const GraphicsSystemParams& graphicsSystemParams, const char* sampleName, const Color& textColor) NN_NOEXCEPT
{
GraphicsSystem graphicsSystem;
SimpleFrameworkInitialize(&graphicsSystem, graphicsSystemParams);
pScene->Initialize(&graphicsSystem);
Font font;
font.InitializeFromNxSystemFont(graphicsSystem.GetAllocator(), nn::pl::SharedFontType_Standard);
SimpleTextRenderer textRenderer;
textRenderer.Initialize(&graphicsSystem);
textRenderer.SetScalingFactor(1.5f); // 1080 / 720
textRenderer.SetCanvasDimensions(graphicsSystemParams.GetRenderTargetWidth() * 1.0f, graphicsSystemParams.GetRenderTargetHeight() * 1.0f);
TextArea controlsTextArea = textRenderer.CreateTextArea(1200, 50, &font, 20, font.GetLineSpacing());
TextArea infoTextArea = textRenderer.CreateTextArea(1200, 50, &font, 20, font.GetLineSpacing());
TextArea sampleNameTextArea = textRenderer.CreateTextArea(500, 50, &font, 20, 1.5f);
TextArea commandBufferReuseTextArea = textRenderer.CreateTextArea(500, 50, &font, 20, 1.5f);
textRenderer.AppendText(controlsTextArea, u8"| A: toggle command buffer reuse on/off | B: show/hide info |");
textRenderer.AppendTextPrintf(sampleNameTextArea, u8"Sample : %s", sampleName);
UpdateCommandBufferReuseStatus(&textRenderer, commandBufferReuseTextArea, pScene);
GraphicsCommandBuffer commandBuffer;
commandBuffer.Initialize(&graphicsSystem);
nn::os::Tick frameStartTick(0);
const int TimeInfoUpdateInterval = 20;
int numFramesWithoutTimeUpdate = 0;
double frameTimeTotal = 0.0;
double drawCpuTimeTotal = 0.0;
bool isFirstFrame = true;
bool showInfoEnabled = true;
//main loop
while (!SimpleFrameworkUpdate())
{
if (isFirstFrame)
{
frameStartTick = GetTick();
isFirstFrame = false;
}
else
{
double frameTime = GetTimeDeltaAsMilliseconds(frameStartTick);
frameStartTick = GetTick();
frameTimeTotal += frameTime;
}
if (IsTriggered(PadButton_A))
{
pScene->TogglePrebuiltCommandBufferUse();
UpdateCommandBufferReuseStatus(&textRenderer, commandBufferReuseTextArea, pScene);
}
if (IsTriggered(PadButton_B))
{
showInfoEnabled = !showInfoEnabled;
}
if (graphicsSystem.ScreenResolutionWasChanged())
{
pScene->NotifyScreenResolutionChange();
}
graphicsSystem.DrawBegin();
nn::os::Tick beforeDrawTick = GetTick();
pScene->DrawFrame();
drawCpuTimeTotal += GetTimeDeltaAsMicroseconds(beforeDrawTick);
++numFramesWithoutTimeUpdate;
if (numFramesWithoutTimeUpdate == TimeInfoUpdateInterval)
{
double frameTimeAvg = frameTimeTotal / numFramesWithoutTimeUpdate;
double drawCpuTimeAvg = drawCpuTimeTotal / numFramesWithoutTimeUpdate;
textRenderer.ClearText(infoTextArea);
textRenderer.AppendTextPrintf(infoTextArea, u8"| Frame time: %.2f ms | draw CPU time: %.2f µs |", frameTimeAvg, drawCpuTimeAvg);
frameTimeTotal = 0.0;
drawCpuTimeTotal = 0.0;
numFramesWithoutTimeUpdate = 0;
}
if (showInfoEnabled)
{
nvn::CommandBuffer* pCommandBuffer = commandBuffer.StartRecording();
{
textRenderer.StartDrawing(pCommandBuffer);
textRenderer.DrawTextArea(sampleNameTextArea, 20, 15, textColor, pCommandBuffer);
textRenderer.DrawTextArea(commandBufferReuseTextArea, 20, 50, textColor, pCommandBuffer);
textRenderer.DrawTextArea(controlsTextArea, 20, 640, textColor, pCommandBuffer);
textRenderer.DrawTextArea(infoTextArea, 20, 675, textColor, pCommandBuffer);
}
commandBuffer.EndRecordingAndExecute();
}
graphicsSystem.DrawEnd();
}
pScene->Finalize();
SimpleFrameworkFinalize(&graphicsSystem);
}
