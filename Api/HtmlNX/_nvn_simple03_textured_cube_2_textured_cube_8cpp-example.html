void TexturedCube::Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
int64_t shaderBinFileSize = 0;
// read shader binary, create GraphicsShader object from it, and get uniform block reflection info
m_GlslcData = ReadFile(&shaderBinFileSize, "rom:/NvnSimple03.glslc", m_pGraphicsSystem->GetAllocator());
m_Shader.Initialize(m_pGraphicsSystem, static_cast<GLSLCoutput*>(m_GlslcData));
m_pUniformBlockInfo = &m_Shader.LookupUniformBlockInfo("VertexShaderUniformBlock");
// create memory pool for vertex data
size_t attributeDataSize = sizeof(g_VertexPositions) + sizeof(g_VertexUvs);
size_t indexDataSize = sizeof(g_VertexIndices) * 6;
size_t uniformAlignment = m_pGraphicsSystem->GetResourceInfo().uniformBufferAlignment;
size_t alignedUniformBlockSize = nn::util::align_up(m_pUniformBlockInfo->GetSize(), uniformAlignment);
size_t memPoolSize = attributeDataSize + indexDataSize + alignedUniformBlockSize + 1024; // add in extra 1 KB to hold linear texture below
m_VertexAndUniformDataMemPool.Initialize(m_pGraphicsSystem, nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED, memPoolSize, "TexturedCube MemPool");
// allocate blocks of memory for vertex and uniform data
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_VertexDataBuffer, attributeDataSize);
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_IndexDataBuffer, indexDataSize);
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_UniformDataBuffer, indexDataSize, uniformAlignment);
// load vertex and index data into memory pool
const int numFaces = 6;
const int numVertices = 4 * numFaces;
Vertex* vertices = static_cast<Vertex*>(m_VertexDataBuffer.Map());
// we use a single vertex buffer with interleaved attributes
for (int i = 0; i < numVertices; ++i)
{
Vertex* pVertex = vertices + i;
float x = g_VertexPositions[3 * i];
float y = g_VertexPositions[3 * i + 1];
float z = g_VertexPositions[3 * i + 2];
float u = g_VertexUvs[2 * i];
float v = g_VertexUvs[2 * i + 1];
pVertex->m_Pos.Set(x, y, z);
pVertex->m_Uv.Set(u, v);
}
// indices
m_NumIndices = 6 * numFaces;
uint32_t* indices = static_cast<uint32_t*>(m_IndexDataBuffer.Map());
for (uint32_t i = 0; i < m_NumIndices; ++i)
{
uint32_t delta = 4 * (i / 6);
uint32_t index = i % 6;
indices[i] = g_VertexIndices[index] + delta;
}
// compute and load uniform data into memory pool
char* baseUniformBlockAddr = static_cast<char*>(m_UniformDataBuffer.Map());
// projection transform
const UniformBlockMemberInfo& projMtxInfo = m_pUniformBlockInfo->LookupMemberInfo("projMtx");
char* projMtxAddr = baseUniformBlockAddr + projMtxInfo.GetOffset();
nn::util::Matrix4x4fType projMtx;
nn::util::Float4x4 mtxStorage;
nn::util::MatrixIdentity(&projMtx);
nn::util::MatrixPerspectiveFieldOfViewRightHanded(&projMtx, nn::util::FloatPi / 3.0f, 16.0f / 9.0f, 0.1f, 1000.0f);
nn::util::MatrixStore(&mtxStorage, projMtx);
memcpy(projMtxAddr, &mtxStorage, projMtxInfo.GetSize());
// The modelView transform will update per frame, so we don't actually load anything here.
// Instead, we compute the view transform up front and set the address of the modelView uniform for later computation.
const UniformBlockMemberInfo& modelViewMtxInfo = m_pUniformBlockInfo->LookupMemberInfo("modelViewMtx");
m_ModelViewMtxAddr = baseUniformBlockAddr + modelViewMtxInfo.GetOffset();
m_Theta = 0.0f;
m_IsRotateHorizontal = true;
nn::util::Vector3f cameraPos(0, 2, 4);
nn::util::Vector3f targetPos(0, 0, 0);
nn::util::Vector3f up(0, 1, 0);
nn::util::MatrixLookAtRightHanded(&m_ViewMtx, cameraPos, targetPos, up);
// set up NVN VertexStreamState and VertexAttribState objects based on vertex attribute names and shader
// the ordering of the attribute names must logically match up with the attribute ordering in our vertex data structure (see above)
const char* attributeNames[NumVertexAttributes] = { "attr_position", "attr_uv" };
// we use an interleaved vertex buffer layout for this demo
SetupVertexAttribAndStreamStateInterleaved(attributeNames, NumVertexAttributes, m_Shader, m_VertexAttribStates, &m_VertexStreamState);
// load texture files
const int NumTextureAssets = 4;
const char* textureFiles[NumTextureAssets] = { "rom:/CubeTexture.tex", "rom:/BrickTexture.tex", "rom:/DdsTexture.tex", "rom:/DdsTextureIndexed.tex" };
TextureAsset* textureAssets[NumTextureAssets];
int totalSize = 0; // calculate total size required to store the textures to be loaded
for (int i = 0; i < NumTextureAssets; ++i)
{
textureAssets[i] = ReadTexture(textureFiles[i], m_pGraphicsSystem->GetAllocator());
totalSize += textureAssets[i]->header.texelsDataSize;
}
// create texture manager instance
m_TextureManager.Initialize(m_pGraphicsSystem, nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED,
32, totalSize, "TexturedCube Texture Pool");
// initialize nvn::Textures from texture asset files
int textureIds[NumTextureAssets];
for (int i = 0; i < NumTextureAssets; ++i)
{
textureIds[i] = m_TextureManager.AddTexture(textureAssets[i]);
}
// inititalize linear nvn::Texture
m_LinearTexWidth = 16;
m_LinearTexHeight = 8;
nvn::TextureBuilder textureBuilder;
textureBuilder.SetDefaults()
.SetDevice(m_pGraphicsSystem->GetDevice())
.SetFlags(nvn::TextureFlags::LINEAR)
.SetTarget(nvn::TextureTarget::TARGET_2D)
.SetFormat(nvn::Format::RGBA8)
.SetSize2D(m_LinearTexWidth, m_LinearTexHeight);
// stride must be set when building linear textures w/ nvn
size_t linearTexStride = m_LinearTexWidth * 4; // 4 bytes per pixel
NN_ASSERT(linearTexStride % m_pGraphicsSystem->GetResourceInfo().linearTextureStrideAlignment == 0);
textureBuilder.SetStride(linearTexStride);
// We place the linear texture in a separate memory pool from our tiled textures. Instead, we place it in m_VertexAndUniformDataMemPool.
// One reason: we want to update the linear texture per-frame, but on Windows, CPU_UNCACHED is not supported for memory pools
// with non-linear textures in them, and furthermore flushing such memory pools causes texture corruption, so using the same
// memory pool is not an option here.
size_t linearTexSize = textureBuilder.GetStorageSize();
size_t linearTexAlign = textureBuilder.GetStorageAlignment();
ptrdiff_t linearTexStartOffset = m_VertexAndUniformDataMemPool.Allocate(linearTexSize, linearTexAlign);
m_LinearTexels = reinterpret_cast<uint32_t*>(m_VertexAndUniformDataMemPool.GetBaseCpuAddress() + linearTexStartOffset);
textureBuilder.SetStorage(m_VertexAndUniformDataMemPool.GetMemoryPool(), linearTexStartOffset);
// make a checkerboard pattern for the linear texture
for (int row = 0; row < m_LinearTexHeight; ++row)
{
for (int col = 0; col < m_LinearTexWidth; ++col)
{
uint32_t color = 0xffffffff;
if ((row % 2) == (col % 2))
{
color = 0xff000000;
}
m_LinearTexels[row * m_LinearTexWidth + col] = color;
}
}
int linearTexId = m_TextureManager.AddTexture(textureBuilder);
// initialize nvn::Samplers
const int NUM_SAMPLERS = 2;
int samplerIds[NUM_SAMPLERS];
nvn::SamplerBuilder samplerBuilder;
samplerBuilder.SetDefaults();
samplerBuilder.SetDevice(m_pGraphicsSystem->GetDevice());
samplerBuilder.SetMinMagFilter(nvn::MinFilter::LINEAR, nvn::MagFilter::LINEAR);
samplerIds[0] = m_TextureManager.AddSampler(samplerBuilder);
samplerBuilder.SetMinMagFilter(nvn::MinFilter::NEAREST, nvn::MagFilter::NEAREST);
samplerIds[1] = m_TextureManager.AddSampler(samplerBuilder);
// generate texture handles
m_TextureHandles[0] = m_TextureManager.RegisterTextureHandle(textureIds[0], samplerIds[0]);
m_TextureHandles[1] = m_TextureManager.RegisterTextureHandle(textureIds[1], samplerIds[1]);
m_TextureHandles[2] = m_TextureManager.RegisterTextureHandle(textureIds[1], samplerIds[0]);
m_TextureHandles[3] = m_TextureManager.RegisterTextureHandle(textureIds[2], samplerIds[0]);
m_TextureHandles[4] = m_TextureManager.RegisterTextureHandle(textureIds[3], samplerIds[1]);
m_TextureHandles[LinearTextureIndex] = m_TextureManager.RegisterTextureHandle(linearTexId, samplerIds[1]);
// free memory for texture files since texel data was loaded into TextureManager MemoryPool
for (int i = 0; i < NumTextureAssets; ++i)
{
m_pGraphicsSystem->GetAllocator()->Free(textureAssets[i]);
}
// get shader binding location for sampler uniform
m_SamplerLocation = m_Shader.LookupUniformInfo("sampler").GetBindingLocations(nvn::ShaderStage::FRAGMENT);
// initialize render state
m_RenderState.Initialize();
m_RenderState.GetColorState()->SetBlendEnable(0, true);
m_RenderState.GetBlendState()->SetBlendTarget(0);
m_RenderState.GetBlendState()->SetBlendEquation(nvn::BlendEquation::ADD, nvn::BlendEquation::ADD);
m_RenderState.GetBlendState()->SetBlendFunc(nvn::BlendFunc::SRC_ALPHA, nvn::BlendFunc::ONE_MINUS_SRC_ALPHA, nvn::BlendFunc::SRC_ALPHA, nvn::BlendFunc::ONE_MINUS_SRC_ALPHA);
m_RenderState.GetPolygonState()->SetFrontFace(nvn::FrontFace::CW);
m_RenderState.GetPolygonState()->SetCullFace(nvn::Face::BACK);
m_RenderState.GetPolygonState()->SetPolygonMode(nvn::PolygonMode::FILL);
m_RenderState.GetDepthStencilState()->SetDepthTestEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthWriteEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthFunc(nvn::DepthFunc::LESS);
// build our command buffer to hold GPU commands to draw our triangles up front once
m_CommandBuffer.Initialize(m_pGraphicsSystem);
GenerateCommandBuffer();
m_CurTextureHandleIndex = 0;
m_IsInitialized = true;
}
}//NOLINT(impl/function_size)
void TexturedCube::DrawFrame() NN_NOEXCEPT
{
// update modelView uniform
nn::util::Matrix4x3fType rotateMtx;
nn::util::Matrix4x3fType modelViewMtx;
nn::util::Vector3f rotate(0, 0, 0);
if (m_IsRotateHorizontal)
{
rotate.SetY(m_Theta);
}
else
{
rotate.SetX(m_Theta);
}
nn::util::MatrixIdentity(&rotateMtx);
nn::util::MatrixSetRotateXyz(&rotateMtx, rotate);
nn::util::MatrixMultiply(&modelViewMtx, rotateMtx, m_ViewMtx);
LoadMatrix43(m_ModelViewMtxAddr, modelViewMtx);
// check to see if the texture handle is to be updated
bool updateTextureHandle = false;
if (IsTriggered(PadButton_X))
{
m_CurTextureHandleIndex = (m_CurTextureHandleIndex + 1) % NumTextureHandles;
updateTextureHandle = true;
}
// generate the command buffer if necessary and execute it
if (!m_IsPrebuiltCommandBufferUsed || updateTextureHandle || m_IsScreenResolutionChanged)
{
GenerateCommandBuffer();
}
m_pGraphicsSystem->GetQueue()->SubmitCommands(1, &m_CommandHandle);
// update per frame state
m_Theta += 0.01f;
if (m_Theta > 2 * nn::util::FloatPi)
{
m_Theta -= (2 * nn::util::FloatPi);
m_IsRotateHorizontal = !m_IsRotateHorizontal;
}
// with our linear texture, we update the texels on the CPU
// given that our texture memory pool is CPU_UNCACHED, the updated texture will be reflected to the GPU
if (m_CurTextureHandleIndex == LinearTextureIndex)
{
float ramp = sinf(m_Theta / 2.0f);
for (int i = 0; i < m_LinearTexWidth * m_LinearTexHeight; ++i)
{
uint32_t texel = m_LinearTexels[i];
// update the white squares to ramp the red component from 0.0 to 1.0
if (texel != 0xff000000)
{
uint8_t red = static_cast<uint8_t>(0xff * ramp);
texel = (texel & 0xffffff00) | red;
m_LinearTexels[i] = texel;
}
}
}
}
void TexturedCube::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_TextureManager.Finalize();
m_CommandBuffer.Finalize();
m_VertexDataBuffer.Finalize();
m_IndexDataBuffer.Finalize();
m_UniformDataBuffer.Finalize();
m_VertexAndUniformDataMemPool.Finalize();
m_Shader.Finalize();
m_pGraphicsSystem->GetAllocator()->Free(m_GlslcData);
m_GlslcData = nullptr;
m_pGraphicsSystem = nullptr;
m_CommandHandle = 0;
m_ModelViewMtxAddr = nullptr;
m_Theta = 0.0f;
m_IsRotateHorizontal = true;
m_IsInitialized = false;
}
}

void TexturedCube::GenerateCommandBuffer() NN_NOEXCEPT
{
nvn::CommandBuffer* pCommandBuffer = m_CommandBuffer.StartRecording();
{
// scissor/viewport
m_pGraphicsSystem->SetDefaultViewportAndScissor(pCommandBuffer);
// clear color and depth/stencil buffers
static float color[4] = { 0.2f, 0.2f, 0.2f, 1.0f };
pCommandBuffer->ClearColor(0, color, nvn::ClearColorMask::RGBA);
pCommandBuffer->ClearDepthStencil(1.0f, true, 0, 0);
// render state
m_RenderState.Bind(pCommandBuffer);
// bind shader
m_Shader.Bind(pCommandBuffer);
// bind uniform buffer
int32_t bindingLocation = m_pUniformBlockInfo->GetBindingLocations(nvn::ShaderStage::VERTEX);
BindUniformBuffer(nvn::ShaderStage::VERTEX, bindingLocation, &m_UniformDataBuffer, pCommandBuffer);
// bind vertex buffer data
pCommandBuffer->BindVertexAttribState(NumVertexAttributes, m_VertexAttribStates);
pCommandBuffer->BindVertexStreamState(1, &m_VertexStreamState);
BindVertexBuffer(0, &m_VertexDataBuffer, pCommandBuffer);
// set texture and sampler pools
m_TextureManager.SetTextureAndSamplerPools(pCommandBuffer);
// bind texture
pCommandBuffer->BindTexture(nvn::ShaderStage::FRAGMENT, m_SamplerLocation, m_TextureHandles[m_CurTextureHandleIndex]);
// draw call
pCommandBuffer->DrawElements(nvn::DrawPrimitive::TRIANGLES, nvn::IndexType::UNSIGNED_INT, m_NumIndices, m_IndexDataBuffer.GetAddress());
}
m_CommandHandle = m_CommandBuffer.EndRecording();
}
