#include <cinttypes>
#include <cstdlib>
#include <cstring>
#include <nn/fs.h>
#include <nn/nn_Assert.h>
#include <nn/nn_Log.h>
#include <nn/nn_Abort.h>
extern "C" void nnMain()
{
nn::Result result;
// Mount the file system on the host PC.
// "host" is specified for the mount name and "C:/Windows/Temp/" for the root path of the file system.
NN_LOG("Mount C:/Windows/Temp to host\n");
result = nn::fs::MountHost("host", "C:/Windows/Temp");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// Cannot find the target path on the host PC.
// Make sure that the C:/Windows/Temp/ path exists on the host PC.
NN_ASSERT(false, "Target directory not found.\n");
return;
}
else if( nn::fs::ResultTargetNotFound::Includes(result) )
{
// The host PC was not recognized.
// Correctly connect to the host PC.
NN_ASSERT(false, "Host PC not found.\n");
return;
}
// Any failures not listed are implementation failures. Always abort when they occur.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
static const size_t FileSize = 512;
// Create directories and files.
{
NN_LOG("Create\thost:/FsSmplDir\n");
result = nn::fs::CreateDirectory("host:/FsSmplDir");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// This error handling is not necessary when parent directories exist without fail.
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target directory already exists.
// Add processing for purposes such as deleting the target directory and recreating it, or ignoring that issue to continue.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient storage region.
// Increase the storage region availability on the host.
NN_ASSERT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Create\thost:/FsSmplDir/dir1\n");
result = nn::fs::CreateDirectory("host:/FsSmplDir/dir1");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// This error handling is not necessary when parent directories exist without fail.
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target directory already exists.
// Add processing for purposes such as deleting the target directory and recreating it, or ignoring that issue to continue.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient storage region.
// Increase the storage region availability on the host.
NN_ASSERT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Create\thost:/FsSmplDir/file1\n");
result = nn::fs::CreateFile("host:/FsSmplDir/file1", FileSize);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// This error handling is not necessary when parent directories exist without fail.
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file already exists.
// This error handling is not needed when the file already exists and that is not an issue.
// Note that there is no assurance that the file size is equal to FileSize when you are not using error handling.
// Delete the target file and re-create it when necessary.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient storage region.
// Increase the storage region availability on the host.
NN_ASSERT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
}
// Manipulate a directory.
{
NN_LOG("Show entries in host:/FsSmplDir\n");
nn::fs::DirectoryHandle directoryHandle;
result = nn::fs::OpenDirectory(&directoryHandle, "host:/FsSmplDir", nn::fs::OpenDirectoryMode_All);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target directory cannot be found.
// This error handling is not necessary when the target directory exists without fail.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
nn::fs::DirectoryEntry directoryEntry[2];
int64_t readNum = 0;
// No error handling is needed because an abort occurs within the library when loading fails.
(void)nn::fs::ReadDirectory(&readNum, directoryEntry, directoryHandle, 2);
for (int i = 0; i < readNum; i++)
{
NN_LOG("%10s <type = %4s, size = %" PRId64 ">\n", directoryEntry[i].name,
directoryEntry[i].directoryEntryType == nn::fs::DirectoryEntryType_Directory ? "Dir" : "File",
directoryEntry[i].fileSize);
}
nn::fs::CloseDirectory(directoryHandle);
}
// Manipulate a file.
{
nn::fs::FileHandle fileHandle;
result = nn::fs::OpenFile(&fileHandle, "host:/FsSmplDir/file1", nn::fs::OpenMode_Read | nn::fs::OpenMode_Write | nn::fs::OpenMode_AllowAppend);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when the target file exists without fail.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open already.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
static const size_t BufferSize = 1024;
static char writeBuffer[BufferSize];
for (int i = 0; i < BufferSize; i++)
{
writeBuffer[i] = static_cast<char>(i);
}
NN_LOG("Write\thost:/FsSmplDir/file1\n");
result = nn::fs::WriteFile(fileHandle, 0, writeBuffer, BufferSize, nn::fs::WriteOption());
if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient storage region.
// Increase the storage region availability on the host.
NN_ASSERT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Flush\thost:/FsSmplDir/file1\n");
// No error handling is needed because an abort occurs within the library when flushing fails.
(void)nn::fs::FlushFile(fileHandle);
int64_t fileSize = 0;
NN_LOG("Read\thost:/FsSmplDir/file1\n");
// No error handling is needed because an abort occurs within the library when failing to get the size.
(void)nn::fs::GetFileSize(&fileSize, fileHandle);
NN_ASSERT_EQUAL(fileSize, 1024);
static char readBuffer[BufferSize];
// No error handling is needed because an abort occurs within the library when loading fails.
(void)nn::fs::ReadFile(fileHandle, 0, readBuffer, static_cast<size_t>(fileSize));
NN_ASSERT(std::memcmp(writeBuffer, readBuffer, BufferSize) == 0);
nn::fs::CloseFile(fileHandle);
}
// Delete directories and files.
result = nn::fs::DeleteDirectoryRecursively("host:/FsSmplDir");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target directory cannot be found.
// This error handling is not necessary when the target directory exists without fail.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// Unmount.
NN_LOG("Unmount host\n\n");
nn::fs::Unmount("host");
//-----------------------------------------------------------------------------
// You can specify an absolute path to the file system's root path on the host computer.
// The path that is the starting point can be configured using the project properties and TargetManager.
// For more information, see NintendoSDK Documents.
NN_LOG("MountHost with a relative path.\n");
result = nn::fs::MountHost("host", "./");
if( nn::fs::ResultTargetNotFound::Includes(result) )
{
// The host PC was not recognized.
// Correctly connect to the host PC.
NN_ASSERT(false, "Target not found.\n");
return;
}
// Any failures not listed are implementation failures. Always abort when they occur.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
NN_LOG("Create\thost:/FsSmplFile.\n");
result = nn::fs::CreateFile("host:/FsSmplFile", FileSize);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// This error handling is not necessary when parent directories exist without fail.
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file already exists.
// This error handling is not needed when the file already exists and that is not an issue.
// Note that there is no assurance that the file size is equal to FileSize when you are not using error handling.
// Delete the target file and re-create it when necessary.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient storage region.
// Increase the storage region availability on the host.
NN_ASSERT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Delete\thost:/FsSmplFile.\n");
result = nn::fs::DeleteFile("host:/FsSmplFile");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when the target file exists without fail.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// Unmount.
NN_LOG("Unmount host\n\n");
nn::fs::Unmount("host");
//-----------------------------------------------------------------------------
// Use MountHostRoot when performing operations under the same path name as that of the host PC.
NN_LOG("Mount Host Root\n");
result = nn::fs::MountHostRoot();
if( nn::fs::ResultTargetNotFound::Includes(result) )
{
// The host PC was not recognized.
// Correctly connect to the host PC.
NN_ASSERT(false, "Target not found.\n");
return;
}
// Any failures not listed are implementation failures. Always abort when they occur.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
NN_LOG("Create\tC:/Windows/Temp/FsSmplFile\n");
result = nn::fs::CreateFile("C:/Windows/Temp/FsSmplFile", FileSize);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// This error handling is not necessary when parent directories exist without fail.
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file already exists.
// This error handling is not needed when the file already exists and that is not an issue.
// Note that there is no assurance that the file size is equal to FileSize when you are not using error handling.
// Delete the target file and re-create it when necessary.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient storage region.
// Increase the storage region availability on the host.
NN_ASSERT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
NN_LOG("Delete\tC:/Windows/Temp/FsSmplFile\n");
result = nn::fs::DeleteFile("C:/Windows/Temp/FsSmplFile");
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when the target file exists without fail.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open.

// No further error handling is necessary because

NN_LOG("Unmount Host Root");
nn::fs::UnmountHostRoot();
