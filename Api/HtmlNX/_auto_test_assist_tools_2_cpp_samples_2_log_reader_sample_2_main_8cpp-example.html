#include <cstdio>
#include <memory>
#include <vector>
#include <Windows.h>
#include "LogReaderWrapper.h"
#include "TmUtilityWrapper.h"
using namespace autoTestAssistTool;
void PrintLogLine()
{
constexpr int logBufferSize = 1024;
char logBuffer[logBufferSize];
size_t logSize = 0;
logReader::LogReaderResult result;
result = logReader::GetLogLineSize(&logSize);
if (result != logReader::LogReaderResult_Success)
{
printf("GetLogLineSize was failed\n");
return;
}
if (logSize <= logBufferSize)
{
result = logReader::GetLogLine(logBuffer, logBufferSize);
if (result == logReader::LogReaderResult_Success)
{
printf("%s\n", logBuffer);
}
else
{
printf("GetLogLine was failed\n");
}
}
else
{
std::unique_ptr<char> largeLogBuffer(new char[logSize]);
result = logReader::GetLogLine(largeLogBuffer.get(), logSize);
if (result == logReader::LogReaderResult_Success)
{
printf("%s\n", largeLogBuffer.get());
}
else
{
printf("GetLogLine was failed\n");
}
}
}
void LogSample()
{
printf("-----------------------------\n");
printf("Print log\n");
logReader::LogReaderResult result;
for (int i = 0; i < 10; ++i)
{
result = logReader::WaitForNextLine(1000);
if (result == logReader::LogReaderResult_Success)
{
logReader::MoveToNextLine();
PrintLogLine();
}
else
{
printf("New log was not found\n");
}
}
Sleep(5000);
printf("-----------------------------\n");
printf("Print log after waiting for outputting specified text\n");
// Set the string to use for regular expression matching.
constexpr char keyString[] = u8".*";
logReader::MoveToNewestLine();
logReader::WaitUntilMatched(keyString, 10 * 1000);
if (result == logReader::LogReaderResult_Success)
{
PrintLogLine();
}
else
{
printf("Matching timeout\n");
}
printf("----------------------------\n");
printf("Move to 10 sec before\n");
logReader::MoveToFormerLineByTime(1000 * 10);
PrintLogLine();
printf("-----------------------------\n");
printf("Move to newest log\n");
logReader::MoveToNewestLine();
PrintLogLine();
printf("-----------------------------\n");
printf("Search log before\n");
logReader::SearchFormerLog(keyString);
PrintLogLine();
}
void RunSamples()
{
LogSample();
}
int main()
{
SetConsoleOutputCP(CP_UTF8);
int targetCount = 0;
tmUtility::GetConnectedTargetCount(&targetCount);
if (targetCount < 1)
{
printf("Connected target was not found.\n");
return 0;
}
std::vector<SerialNumberString> serialList;
if (targetCount > 0)
{
SerialNumberString* serialListBuffer = new SerialNumberString[targetCount];
for (int i = 0; i < targetCount; ++i)
{
serialListBuffer[i].value[0] = '\0';
}
tmUtility::GetConnectedTargetSerialNumbers(serialListBuffer, targetCount);
for (int i = 0; i < targetCount; ++i)
{
if (strlen(serialListBuffer[i].value) > 0)
{
serialList.push_back(serialListBuffer[i]);
}
}
delete[] serialListBuffer;
}
printf("Please select target.\n");
for (int i = 0; i < serialList.size(); ++i)
{
printf("[%d] : %s\n", i, serialList.at(i).value);
}
int selection = -1;
scanf_s("%d", &selection);
if (selection < 0 || selection >= serialList.size())
{
printf("Invalid input.");
return 0;
}
logReader::LogReaderResult result =
logReader::StartLogStoring(serialList.at(selection), 1000);
if (result != logReader::LogReaderResult_Success)
{
printf("StartLogStoring was failed.\n");
return 0;
}
RunSamples();
logReader::StopLogStoring();
return 0;
}
