struct ShaderGlslcInfo

nvn::Format ToVertexAttribNvnFormat(GLSLCpiqTypeEnum glslcVertexAttribType) NN_NOEXCEPT
{
nvn::Format format = nvn::Format::NONE;
// only has entries for the ones we use
switch (glslcVertexAttribType)
{
case GLSLC_PIQ_TYPE_FLOAT_VEC2: format = nvn::Format::RG32F; break;
case GLSLC_PIQ_TYPE_FLOAT_VEC3: format = nvn::Format::RGB32F; break;
case GLSLC_PIQ_TYPE_FLOAT_VEC4: format = nvn::Format::RGBA32F; break;
default:
NN_ASSERT(0, "unsupported GLSLCpiqTypeEnum (%d) encountered.\n", glslcVertexAttribType);
}
return format;
}
ptrdiff_t GetAttributeWidth(GLSLCpiqTypeEnum glslcVertexAttribType) NN_NOEXCEPT
{
ptrdiff_t width = 0;
// not comprehensive - only has entries for the ones used in the NvnSimple programs
switch (glslcVertexAttribType)
{
case GLSLC_PIQ_TYPE_BOOL: width = 4; break;
case GLSLC_PIQ_TYPE_FLOAT: width = 4; break;
case GLSLC_PIQ_TYPE_FLOAT_VEC2: width = 8; break;
case GLSLC_PIQ_TYPE_FLOAT_VEC3: width = 12; break;
case GLSLC_PIQ_TYPE_FLOAT_VEC4: width = 16; break;
case GLSLC_PIQ_TYPE_MAT4: width = 64; break;
case GLSLC_PIQ_TYPE_INT: width = 4; break;
case GLSLC_PIQ_TYPE_UINT64: width = 8; break;
case GLSLC_PIQ_TYPE_SAMPLER: width = 8; break;
case GLSLC_PIQ_TYPE_TEXTURE_2D: width = 8; break;
default:
NN_ASSERT(0, "unsupported GLSLCpiqTypeEnum (%d) encountered.\n", glslcVertexAttribType);
}
return width;
}
nvn::ShaderStageBits ConvertToShaderStageBits(NVNshaderStage shaderStage) NN_NOEXCEPT
{
nvn::ShaderStageBits stageBits = 0;
switch (shaderStage)
{
case NVN_SHADER_STAGE_VERTEX: stageBits = nvn::ShaderStageBits::VERTEX; break;
case NVN_SHADER_STAGE_FRAGMENT: stageBits = nvn::ShaderStageBits::FRAGMENT; break;
case NVN_SHADER_STAGE_GEOMETRY: stageBits = nvn::ShaderStageBits::GEOMETRY; break;
case NVN_SHADER_STAGE_TESS_CONTROL: stageBits = nvn::ShaderStageBits::TESS_CONTROL; break;
case NVN_SHADER_STAGE_TESS_EVALUATION: stageBits = nvn::ShaderStageBits::TESS_EVALUATION; break;
case NVN_SHADER_STAGE_COMPUTE: stageBits = nvn::ShaderStageBits::COMPUTE; break;
default:
NN_ASSERT(0, "unknown NVNshaderStage (%d) encountered.\n", shaderStage);
}
return stageBits;
}

void GraphicsShader::Initialize(GraphicsSystem* pGraphicsSystem, const GLSLCoutput* pGlslcData) NN_NOEXCEPT
{
// [pGlslcData] - contains the shader binary code and reflection information (as generated by GLSLC)
// to initialize a GraphicsShader instance with.
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
size_t memPoolChunkSize = 0;
int shaderAlignment = 0;
int shaderSections = pGlslcData->numSections;
std::vector<ShaderGlslcInfo> shaderGlslcInfos;
m_pGraphicsSystem->GetDevice()->GetInteger(nvn::DeviceInfo::UNIFORM_BUFFER_ALIGNMENT, &shaderAlignment);
m_RecommendedScratchMemSize = 0;
// go through all of the glslc data, section by section
for (int i = 0; i < shaderSections; ++i)
{
GLSLCsectionTypeEnum type = pGlslcData->headers[i].genericHeader.common.type;
// found shader code, so record the necessary data to later initialize NVN shader data structures
if (type == GLSLC_SECTION_TYPE_GPU_CODE)
{
const GLSLCgpuCodeHeader* gpuSection = &(pGlslcData->headers[i].gpuCodeHeader);
const char* baseAddr = reinterpret_cast<const char*>(pGlslcData) + gpuSection->common.dataOffset;
ShaderGlslcInfo info;
info.control = baseAddr + gpuSection->controlOffset;
info.controlSize = gpuSection->controlSize;
info.data = baseAddr + gpuSection->dataOffset;
info.dataSize = gpuSection->dataSize;
info.stage = gpuSection->stage;
info.dataMemPoolOffset = memPoolChunkSize;
if (gpuSection->scratchMemBytesRecommended > m_RecommendedScratchMemSize)
{
m_RecommendedScratchMemSize = gpuSection->scratchMemBytesRecommended;
}
shaderGlslcInfos.push_back(info);
memPoolChunkSize += nn::util::align_up(info.dataSize, shaderAlignment);
m_ShaderStageBits |= ConvertToShaderStageBits(info.stage);
}
// found reflection information about the shader, so record that data
else if (type == GLSLC_SECTION_TYPE_REFLECTION)
{
const char* baseAddr = reinterpret_cast<const char*>(pGlslcData) + pGlslcData->headers[i].genericHeader.common.dataOffset;
const GLSLCprogramReflectionHeader* pReflectionHeader = &(pGlslcData->headers[i].programReflectionHeader);
const GLSLCuniformBlockInfo* pUniformBlock = reinterpret_cast<const GLSLCuniformBlockInfo*>(baseAddr + pReflectionHeader->uniformBlockOffset);
const GLSLCuniformInfo* pUniform = reinterpret_cast<const GLSLCuniformInfo*>(baseAddr + pReflectionHeader->uniformOffset);
const GLSLCprogramInputInfo* pProgramInput = reinterpret_cast<const GLSLCprogramInputInfo*>(baseAddr + pReflectionHeader->programInputsOffset);
const char* stringPool = baseAddr + pReflectionHeader->stringPoolOffset;
int uniformBlockIndex = 0;
std::vector<UniformBlockInfo*> uniformBlockList; // temporary vector to aid index-based lookup of uniform block for uniforms
// type 1: uniform blocks
for (uint32_t j = 0; j < pReflectionHeader->numUniformBlocks; ++j)
{
const char* name = stringPool + pUniformBlock->nameInfo.nameOffset;
UniformBlockInfo& blockInfo = m_UniformBlockMap[name];
blockInfo.m_BlockIndex = uniformBlockIndex;
blockInfo.m_Size = pUniformBlock->size;
for (int k = 0; k < 6; ++k)
{
blockInfo.m_BindingLocations[k] = pUniformBlock->bindings[k];
}
uniformBlockList.push_back(&blockInfo);
++pUniformBlock;
++uniformBlockIndex;
}
// type 2: individual uniforms
for (uint32_t j = 0; j < pReflectionHeader->numUniforms; ++j)
{
const char* name = stringPool + pUniform->nameInfo.nameOffset;
// type 2a: uniforms inside a uniform block
if (pUniform->isInUBO)
{
NN_ASSERT(pUniform->blockNdx >= 0);
UniformBlockMemberInfo& uniformInfo = uniformBlockList[pUniform->blockNdx]->m_UniformMap[name];
uniformInfo.SetSize(GetAttributeWidth(pUniform->type));
uniformInfo.SetParentUniformBlockIndex(pUniform->blockNdx);
uniformInfo.SetOffset(pUniform->blockOffset);
}
// type 2b: uniforms outside of a uniform block
// Note that NVN only allows sampler, image, and texture types for such uniforms that are outside of a uniform block.
// GLSLC will generate compiler errors for any other uniform types that come outside of a uniform block.
else
{
UniformInfo& uniformInfo = m_UniformMap[name];
uniformInfo.SetGLSLCpiqType(pUniform->type);
for (int k = 0; k < 6; ++k)
{
uniformInfo.SetBindingLocations(static_cast<nvn::ShaderStage::Enum>(k), pUniform->bindings[k]);
}
}
++pUniform;
}
// type 3: vertex attributes
for (uint32_t j = 0; j < pReflectionHeader->numProgramInputs; ++j)
{
const char* name = stringPool + pProgramInput->nameInfo.nameOffset;
VertexAttributeInfo& vertexAttribute = m_VertexAttributeMap[name];
vertexAttribute.SetLocationInShader(pProgramInput->location);
vertexAttribute.SetFormat(ToVertexAttribNvnFormat(pProgramInput->type));
vertexAttribute.SetElementWidth(GetAttributeWidth(pProgramInput->type));
++pProgramInput;
}
} // reflection handling end
} // end for loop
m_NumShaders = shaderGlslcInfos.size();
// allocate a slab of memory for our shader program data from our shared memory pool for shaders
GraphicsMemoryPool* pMemPool = m_pGraphicsSystem->GetMemoryPoolForShaders();
ptrdiff_t memPoolChunkBaseOffset = pMemPool->Allocate(memPoolChunkSize);
// last 1 KB of memory pool for shader must be left empty, so we check that here
NN_ASSERT(memPoolChunkBaseOffset + memPoolChunkSize <= pMemPool->GetPoolSize() - 1024);
// copy shader program data into our slab of memory in the memory pool and populate NVN ShaderData to pass into NVN program object
nvn::ShaderData* shaderData = new nvn::ShaderData[m_NumShaders];
uint8_t* shaderDataBaseCpuAddr = pMemPool->GetBaseCpuAddress() + memPoolChunkBaseOffset;
nvn::BufferAddress shaderDataBaseGpuAddr = pMemPool->GetBaseGpuAddress() + memPoolChunkBaseOffset;
for (size_t i = 0; i < m_NumShaders; ++i)
{
ShaderGlslcInfo* pInfo = &shaderGlslcInfos[i];
memcpy(shaderDataBaseCpuAddr + pInfo->dataMemPoolOffset, pInfo->data, pInfo->dataSize);
shaderData[i].data = shaderDataBaseGpuAddr + pInfo->dataMemPoolOffset;
shaderData[i].control = pInfo->control;
}
// finally, create NVN program object
if (!m_Program.Initialize(m_pGraphicsSystem->GetDevice()))
{
NN_ASSERT(0, "nvn::Program::Initialize failed");
}
if (!m_Program.SetShaders(static_cast<int>(m_NumShaders), shaderData))
{
NN_ASSERT(0, "nvn::Program::SetShaders failed");
}
delete[] shaderData;
m_IsInitialized = true;
}
}//NOLINT(impl/function_size)
void GraphicsShader::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_Program.Finalize();
m_NumShaders = 0;
m_pGraphicsSystem = nullptr;
m_IsInitialized = false;
}
}

const UniformBlockMemberInfo& UniformBlockInfo::LookupMemberInfo(const char* uniformName) const NN_NOEXCEPT
{
std::map<std::string, UniformBlockMemberInfo>::const_iterator iter;
iter = m_UniformMap.find(uniformName);
if (iter == m_UniformMap.end())
{
NN_ASSERT(0, "LookupMemberInfo: Uniform \"%s\" does not exist in map.", uniformName);
}
return iter->second;
}
