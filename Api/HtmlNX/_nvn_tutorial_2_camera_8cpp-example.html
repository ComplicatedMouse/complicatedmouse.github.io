#include <nvntutorial/TutorialUtil.h>
#if NN_BUILD_TARGET_PLATFORM_OS_WIN
#include <nvntutorial/InputManager.Windows.h>
#else
#include <nvntutorial/InputManager.Nx.h>
#endif
/*
* Camera Constructor
* ------------------
* Sets up default mamber data values.
*/
Camera::Camera() : m_XRot(0.0f),
m_YRot(0.0f),
m_RotSensitivity(0.0f),
m_PosSensitivity(0.0f),
m_pInputManager(nullptr)
{
m_Position.Zero();
m_View.Zero();
m_Up.Zero();
m_Right.Zero();
nn::util::MatrixIdentity(&m_CameraMatrix);
}
/*
* Camera Destructor
* -----------------
* Empty destructor.
*/
Camera::~Camera()
{
}
/*
* Camera::SetCameraState
* ----------------------
* Overwrite the camera state with a new position and
* orientation.
*/
void Camera::SetCameraState(float xRot, float yRot, const nn::util::Vector3f& pos)
{
m_XRot = xRot;
m_YRot = yRot;
m_Position = pos;
UpdateRotation();
nn::util::MatrixCameraRotateZxy(&m_CameraMatrix,
m_Position,
nn::util::Vector3f(GetXRotationAsRadian(),
GetYRotationAsRadian(),
0.0f));
}
/*
* Camera::Init
* ------------
* Setup the camera input.
*/
void Camera::Init(InputManager* pInputManager, float posSensitivity, float rotSensitivity)
{
m_PosSensitivity = posSensitivity;
m_RotSensitivity = rotSensitivity;
m_pInputManager = pInputManager;
}
/*
* Camera::Update
* --------------
* Update the camera's position/orientation based on
* user input.
*/
void Camera::Update()
{
#if NN_BUILD_TARGET_PLATFORM_OS_WIN
int rotY = 0;
int rotX = 0;
InputManagerWin* pWinInput = dynamic_cast<InputManagerWin*>(m_pInputManager);
if (pWinInput->GetMouseButtonState(InputManagerWin::MouseButton_Left) == InputManager::ButtonState_Pressed)
{
pWinInput->GetMouseDeltas(rotY, rotX);
rotX *= -1;
}
#else
float rotY = 0;
float rotX = 0;
dynamic_cast<InputManagerNX*>(m_pInputManager)->GetRightStickState(rotY, rotX);
#endif
m_XRot += static_cast<float>(rotX) * m_RotSensitivity;
m_YRot -= static_cast<float>(rotY) * m_RotSensitivity;
/* We don't want to rotate up more than 90 degrees, so we cap it. */
if (m_XRot > 90.0f)
{
m_XRot = 90.0f;
}
/* We don't want to rotate down more than 90 degrees, so we cap it. */
if (m_XRot < -90.0f)
{
m_XRot = -90.0f;
}
UpdateRotation();
UpdateTranslation();
nn::util::MatrixCameraRotateZxy(&m_CameraMatrix,
m_Position,
nn::util::Vector3f(GetXRotationAsRadian(),
GetYRotationAsRadian(),
0.0f));
}
/*
* Camera::UpdateRotation
* ----------------------
* Update the camera's orientation.
*/
void Camera::UpdateRotation()
{
nn::util::Matrix4x3f camMat43;
nn::util::MatrixCameraRotateZxy(&camMat43,
nn::util::Vector3f(0.0f, 0.0f, 0.0f),
nn::util::Vector3f(GetXRotationAsRadian(),
GetYRotationAsRadian(),
0.0f));
nn::util::Vector3f dir;
nn::util::Vector3f up;
nn::util::Vector3f right;
camMat43.Inverse();
nn::util::VectorTransform(&dir, nn::util::Vector3f(0.0f, 0.0f, -1.0f), camMat43);
nn::util::VectorNormalize(&m_View, dir);
nn::util::VectorTransform(&up, nn::util::Vector3f(0.0f, 1.0f, 0.0f), camMat43);
nn::util::VectorNormalize(&m_Up, up);
nn::util::VectorCross(&right, dir, up);
nn::util::VectorNormalize(&m_Right, right);
}
/*
* Camera::UpdateTranslation
* -------------------------
* Update the camera's position.
*/
void Camera::UpdateTranslation()
{
#if NN_BUILD_TARGET_PLATFORM_OS_WIN
InputManagerWin* pWinInput = dynamic_cast<InputManagerWin*>(m_pInputManager);
float mod = 1.0f;
if (pWinInput->GetKeyboardButtonState(CHAR_LEFT_SHIFT) == InputManager::ButtonState_Pressed ||
pWinInput->GetKeyboardButtonState(CHAR_RIGHT_SHIFT) == InputManager::ButtonState_Pressed)
{
mod = 0.25f;
}
if (pWinInput->GetKeyboardButtonState('w') == InputManager::ButtonState_Pressed)
{
m_Position += m_View * m_PosSensitivity * mod;
}
if (pWinInput->GetKeyboardButtonState('s') == InputManager::ButtonState_Pressed)
{
m_Position -= m_View * m_PosSensitivity * mod;
}
if (pWinInput->GetKeyboardButtonState('d') == InputManager::ButtonState_Pressed)
{
m_Position += m_Right * m_PosSensitivity * mod;
}
if (pWinInput->GetKeyboardButtonState('a') == InputManager::ButtonState_Pressed)
{
m_Position -= m_Right * m_PosSensitivity * mod;
}
if (pWinInput->GetKeyboardButtonState('e') == InputManager::ButtonState_Pressed)
{
m_Position += m_Up * m_PosSensitivity * mod;
}
if (pWinInput->GetKeyboardButtonState('q') == InputManager::ButtonState_Pressed)
{
m_Position -= m_Up * m_PosSensitivity * mod;
}
#else
float leftX = 0;
float leftY = 0;
InputManagerNX* pNXInput = dynamic_cast<InputManagerNX*>(m_pInputManager);
pNXInput->GetLeftStickState(leftX, leftY);
m_Position += leftY * m_View * m_PosSensitivity;
m_Position += leftX * m_Right * m_PosSensitivity;
if (pNXInput->GetGamepadButtonState(InputManagerNX::GamepadButton_ZL) == InputManager::ButtonState_Pressed)
{
m_Position -= m_Up * m_PosSensitivity;
}
if (pNXInput->GetGamepadButtonState(InputManagerNX::GamepadButton_ZR) == InputManager::ButtonState_Pressed)
{
m_Position += m_Up * m_PosSensitivity;
}
#endif
}
