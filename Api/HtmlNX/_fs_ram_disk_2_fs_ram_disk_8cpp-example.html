#include <cstdlib>
#include <cstring>
#include <nn/fs.h>
#include <nn/nn_Assert.h>
#include <nn/nn_Abort.h>
#include <nn/nn_Log.h>
#include <nn/fs/fs_Debug.h>
extern "C" void nnMain()
{
nn::Result result;
// Allocates 32 MB of memory for RamDisk.
// Consider the amount of memory to allocate based on the needs of your application.
const size_t SpaceSize = 32 * 1024 * 1024;
// Calculate the size of the buffer required for RamDisk and allocate it.
size_t bufferSize;
bufferSize = nn::fs::GetRamDiskBufferSize(SpaceSize);
void* ramDiskBuffer = std::malloc(bufferSize);
NN_ABORT_UNLESS_NOT_NULL(ramDiskBuffer);
// Mount RamDisk with the name 'ram.'
NN_LOG("Mount RamDisk as \"ram\"\n");
NN_LOG("SpaceSize : %zu\n", SpaceSize);
NN_LOG("BufferSize : %zu\n", bufferSize);
result = nn::fs::MountRamDisk("ram", ramDiskBuffer, bufferSize);
// Always abort when a failure occurs.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Ensure that the file system is always empty immediately after calling nn::fs::MountRamDisk().
const size_t FileSize = 1024;
const size_t WriteBufferSize = 1024;
const size_t ReadBufferSize = 1024;
static char writeBuffer[WriteBufferSize];
static char readBuffer[ReadBufferSize];
// Create files and write data.
{
result = nn::fs::CreateFile("ram:/file", FileSize);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// There are no directories within the path.
// This error handling is not necessary when parent directories exist without fail.
}
else if( nn::fs::ResultPathAlreadyExists::Includes(result)
|| nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file already exists.
// This error handling is not needed when the file already exists and that is not an issue.
// Note that there is no assurance that the file size is equal to FileSize when you are not using error handling.
// Delete the target file and re-create it when necessary.
}
else if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient data save area.
// Increase the data save region.
NN_ABORT("Usable space not enough.\n");
return;
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// Write initial data.
NN_LOG("Create ram:/file\n");
nn::fs::FileHandle fileHandle;
result = nn::fs::OpenFile(&fileHandle, "ram:/file", nn::fs::OpenMode_Write);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when you are only opening existing files.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open already.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
for (int i = 0; i < WriteBufferSize; i++)
{
writeBuffer[i] = static_cast<char>(i);
}
NN_LOG("Write ram:/file\n");
result = nn::fs::WriteFile(fileHandle, 0, &writeBuffer, sizeof(writeBuffer), nn::fs::WriteOption::MakeValue(nn::fs::WriteOptionFlag_Flush));
if( nn::fs::ResultUsableSpaceNotEnough::Includes(result) )
{
// Insufficient data save area.
// This error handling is not necessary when you would not specify OpenMode_AllowAppend when opening files.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// There is no need to call nn::fs::Commit() because RamDisk does not support journaling features.
nn::fs::CloseFile(fileHandle);
}
// Read data from the created files.
{
nn::fs::FileHandle fileHandle;
result = nn::fs::OpenFile(&fileHandle, "ram:/file", nn::fs::OpenMode_Read);
if( nn::fs::ResultPathNotFound::Includes(result) )
{
// The target file cannot be found.
// This error handling is not necessary when you are only opening existing files.
}
else if( nn::fs::ResultTargetLocked::Includes(result) )
{
// The target file is already open.
// This error handling is not necessary when there is no possibility of the file being open already.
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// Read data.
size_t readSize = 0;
NN_LOG("Read ram:/file\n");
// No error handling is needed because an abort occurs within the library when loading fails.
(void)nn::fs::ReadFile(&readSize, fileHandle, 0, &readBuffer, sizeof(readBuffer));
NN_ASSERT_EQUAL(readSize, sizeof(readBuffer));
NN_ASSERT(std::memcmp(writeBuffer, readBuffer, ReadBufferSize) == 0);
nn::fs::CloseFile(fileHandle);
}
// Unmount.
NN_LOG("Unmount \"ram\"\n\n");
nn::fs::Unmount("ram");
return;
}
