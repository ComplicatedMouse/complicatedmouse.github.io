namespace {
void StartDemo() noexcept
{
// ##################################################################################
// Application initialization sequence.
// ##################################################################################
auto pGraphicsSystem = std::make_unique<GraphicsSystem>();
pGraphicsSystem->SetApplicationHeap(&GetStandardAllocator());
pGraphicsSystem->Initialize();
InitializeTimeDemo();
// Initialize SaveDataAccessor.
auto pSaveDataAccessor = std::make_unique<SaveDataAccessor>();
// Pad demo initialization.
constexpr char NpadProgramName[] = " Npad";
auto pNpadDemo = std::make_shared<NpadDemo>();
pNpadDemo->SetName(NpadProgramName);
pNpadDemo->SetProperty(PluginProperty::Drawable);
GetPluginManager().AddBeginPlugin(pNpadDemo);
// Graphics demo initialization.
constexpr char GfxProgramName[] = "PrimitiveRenderer";
auto pGfxDemo = std::make_shared<GfxDemo>();
pGfxDemo->SetName(GfxProgramName);
GetPluginManager().AddBeginPlugin(pGfxDemo);
pGraphicsSystem->InitializeDebugFont();
// Scene plug-in initialization.
GetPluginManager().InitializeScenePlugin(pGraphicsSystem.get(),pSaveDataAccessor.get());
// NPad plug-in initialization.
GetPluginManager().InitializeNpadPlugin();
// ##################################################################################
// Main loop.
// ##################################################################################
auto isFirstDrawing = true;
while( NN_STATIC_CONDITION(true) )
{
int moveCount = pNpadDemo->GetSceneChangeCount();
GetPluginManager().ChangeScene(moveCount);
GetPluginManager().Procedure();
pGraphicsSystem->BeginDraw();
pGfxDemo->Draw();
GetPluginManager().DrawScene();
pGraphicsSystem->EndDraw();
if ( isFirstDrawing )
{
// Display the log when Demo1 occurred normally.
NN_LOG("Demo1 Initialize Success\n");
isFirstDrawing = false;
}
constexpr auto FrameRate = 60;
pGraphicsSystem->Synchronize(nn::TimeSpan::FromNanoSeconds(1000 * 1000 * 1000 / FrameRate));
// Determine whether there has been an exit notification.
if (pNpadDemo->EndTrigger())
{
break;
}
}
// ##################################################################################
// Finalize the Application
// ##################################################################################
// When nn::applet::IsExpectedToCloseApplication() == true, you automatically
// lose the ability to call sleep-related callbacks.
// NPad plug-in finalization.
GetPluginManager().FinalizeNpadPlugin();
// Scene plug-in finalization.
GetPluginManager().FinalizeScenePlugin();
pGraphicsSystem->FinalizeDebugFont();
FinalizeTimeDemo();
pGraphicsSystem->Finalize();
}
}

// Main Function
// The main function.

extern "C" void nnMain() noexcept
{
size_t cacheSize = 0;
nn::Result result;
constexpr size_t MemorySize = 100 * 1024 * 1024;
alignas(4096) static nn::Bit8 s_Memory[MemorySize];
GetStandardAllocator().Initialize(
s_Memory, MemorySize
);
// ##################################################################################
// Memory-related initialization.
// ##################################################################################
result = nn::fs::QueryMountRomCacheSize(&cacheSize);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
auto mountRomCacheBuffer = std::make_unique<unsigned char[]>(cacheSize);
NN_ASSERT_NOT_NULL(mountRomCacheBuffer.get());
result = nn::fs::MountRom("Contents", mountRomCacheBuffer.get(), cacheSize);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
GetPluginManager().InitializePlatformPlugin();
nn::os::InitializeEvent(&GetGlobalEvent(), false, nn::os::EventClearMode_ManualClear);
// Start the demo.
StartDemo();
nn::fs::Unmount("Contents");
GetPluginManager().FinalizePlatformPlugin();
GetStandardAllocator().Finalize();
}
