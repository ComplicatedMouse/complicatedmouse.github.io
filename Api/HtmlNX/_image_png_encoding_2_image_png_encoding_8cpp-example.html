bool EncodePngData(
size_t *pOutDataActualSize,
Buffer *pOutPngDataBuffer,
void* pixelData,
size_t pixelDataSize,
nn::image::PixelFormat pixelFormat,
nn::image::Dimension dimension,
size_t stride) NN_NOEXCEPT
{
// Initialize the PNG encoder.
nn::image::PngEncoder encoder;
encoder.Initialize(Allocate, nullptr, Free, nullptr);
// Register pixel data.
encoder.SetPixelData(pixelData, pixelDataSize, pixelFormat, dimension, stride);
// Allocate the output buffer.
pOutPngDataBuffer->Initialize(encoder.GetPngDataSizeMax());
// Encode PNG data.
nn::image::PngStatus status = encoder.Encode(pOutDataActualSize, pOutPngDataBuffer->GetDataPtr(), pOutPngDataBuffer->GetDataSize());
switch (status)
{
case nn::image::PngStatus_OutOfMemory:
NN_LOG("Insufficient memory during encoding.\n");
return false;
case nn::image::PngStatus_InternalError:
NN_LOG("Internal error during encoding.\n");
return false;
case nn::image::PngStatus_ShortOfOutputBuffer:
NN_LOG("Insufficient output buffer.\n");
return false;
default:
NN_ASSERT(status == nn::image::PngStatus_Ok);
}
return true;
}
extern "C" void nnMain()
{
// Initialize the file system.
// Get the output path of the program.
const size_t assetRootLength = 512;
char assetRoot[assetRootLength];
nn::diag::GetModulePath(assetRoot, assetRootLength, reinterpret_cast<uintptr_t>(&nnMain));
// Delete the \ImagePngEncoder.* part.
char* path = NULL;
path = std::strstr(assetRoot, "ImagePngEncoding.");
NN_ASSERT_NOT_NULL(path);
*path = '\0';
// Save the generated PNG file in the application's output directory.
nn::Result result = nn::fs::MountHost("Outputs", assetRoot);
NN_ASSERT(result.IsSuccess());
NN_UNUSED(result);
// Build the pixel data.
int pixelDataWidth = 1280;
int pixelDataHeight = 720;
size_t pixelDepth = 4;
size_t rowStride = pixelDataWidth * pixelDepth;
size_t pixelDataSize = pixelDataHeight * rowStride;
void* pixelData = malloc(pixelDataSize);
uint8_t* pixels = reinterpret_cast<uint8_t*>(pixelData);
float increamentX = 1.0f / pixelDataWidth;
float increamentY = 1.0f / pixelDataHeight;
for (int y = 0; y < pixelDataHeight; y++)
{
const uint8_t R = static_cast<uint8_t>(y * increamentY * 0xFF);
for (int x = 0; x < pixelDataWidth; x++)
{
const uint8_t A = ((x / 64) % 2 == 0 && (y / 64) % 2 == 1) || ((x / 64) % 2 == 1 && (y / 64) % 2 == 0) ?
static_cast<uint8_t>(y * increamentY * 0xFF) : 255;
const uint8_t G = static_cast<uint8_t>(x * increamentX * 0xFF);
const uint8_t B = 0xFF - static_cast<uint8_t>(x * increamentX * 0xFF);
uint8_t* pPixel = pixels + (pixelDataWidth * y + x) * pixelDepth;
pPixel[0] = R;
pPixel[1] = G;
pPixel[2] = B;
pPixel[3] = A;
}
}
// Encode the pixel data.
Buffer encodedPngData;
size_t encodedPngDataSize = 0;
nn::image::Dimension dimension = { pixelDataWidth , pixelDataHeight };
if(EncodePngData(&encodedPngDataSize, &encodedPngData, pixelData, pixelDataSize, nn::image::PixelFormat_Rgba32, dimension, rowStride) != true)
{
NN_LOG("Failed to encode pixel data.\n");
return;
}
// Save the encoded PNG data.
FileData::Save("Outputs:/output.png", encodedPngData.GetDataPtr(), encodedPngDataSize);
NN_LOG("Encoding succeeded. Saved the \"output.png\" PNG file to \n %s.\n", assetRoot);
// Finalize the file system.
nn::fs::Unmount("Outputs");
return;
}
