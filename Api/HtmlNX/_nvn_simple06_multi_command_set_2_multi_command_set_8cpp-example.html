// Public API

void MultiCommandSet::Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
// initialize memory pool to hold vertex data
size_t memPoolSize = 32 * 1024;
m_VertexMemPool.Initialize(m_pGraphicsSystem, nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED, memPoolSize, "MultiCommandSet MemPool");
// initialize SimplePrimitiveRenderer
m_PrimitiveRenderer.Initialize(m_pGraphicsSystem, &m_VertexMemPool);
m_PrimitiveRenderer.UpdateAmbientColor(Color(0x111111));
m_PrimitiveRenderer.UpdateLightSettings(0, PointLightSettings(Color(0xffffff), Color(0x000000), nn::util::Vector3f(0, 1, 0), 0.001f));
m_PrimitiveRenderer.UpdateLightSettings(1, PointLightSettings(Color(0x000000), Color(0x333333), nn::util::Vector3f(5, 3, 3), 0.0005f));
nn::util::Matrix4x4fType projMtx;
nn::util::MatrixIdentity(&projMtx);
nn::util::MatrixOrthographicOffCenterRightHanded(&projMtx, -9.6f, 9.6f, -5.4f, 5.4f, -1000.0f, 1000.0f);
m_PrimitiveRenderer.UpdateProjectionTransform(projMtx);
// initialize view transform
nn::util::Vector3f cameraPos(0, 0, 15);
nn::util::Vector3f targetPos(0, 0, 0);
nn::util::Vector3f up(0, 1, 0);
nn::util::MatrixLookAtRightHanded(&m_ViewMtx, cameraPos, targetPos, up);
// create our shape to draw, and initialize different materials and modelviews per instance to draw
m_Cone = m_PrimitiveRenderer.AddCone(2, 1.0f, 100);
MaterialSettings materialSettings(Color(0x222222), Color(0x000000), Color(0xeeeeee), Color(0x000000), 1.0f);
Color diffuseColors[NumConeInstances] =
{
Color(0x19d7e8), Color(0x71e5ef), Color(0xcbf4f8),
Color(0xcfe707), Color(0xe6f18e), Color(0xf7f9e5),
Color(0xe8378b), Color(0xe17ead), Color(0xe8ccd9)
};
for (int i = 0; i < NumConeInstances; ++i)
{
materialSettings.SetDiffuse(diffuseColors[i]);
m_Materials[i] = m_PrimitiveRenderer.AddMaterial(materialSettings);
m_ModelViews[i] = m_PrimitiveRenderer.AddModelViewTransform();
}
float dx = 5.0f;
float dy = 3.0f;
nn::util::Matrix4x3fType translationMtx;
nn::util::Matrix4x3fType modelViewMtx;
int mvIndex = 0;
for (int row = -1; row <= 1; ++row)
{
for (int col = -1; col <= 1; ++col)
{
nn::util::MatrixIdentity(&translationMtx);
nn::util::MatrixSetTranslate(&translationMtx, nn::util::Vector3f(col * dx, row * dy, 0.0f));
nn::util::MatrixIdentity(&modelViewMtx);
nn::util::MatrixMultiply(&modelViewMtx, translationMtx, m_ViewMtx);
m_PrimitiveRenderer.UpdateModelViewTransform(m_ModelViews[mvIndex++], modelViewMtx);
}
}
// initialize render state
m_RenderState.Initialize();
m_RenderState.GetPolygonState()->SetFrontFace(nvn::FrontFace::CW);
m_RenderState.GetPolygonState()->SetCullFace(nvn::Face::BACK);
m_RenderState.GetPolygonState()->SetPolygonMode(nvn::PolygonMode::FILL);
m_RenderState.GetDepthStencilState()->SetDepthTestEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthWriteEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthFunc(nvn::DepthFunc::LESS);
// initialize command buffer w/ manualResetMode so we can explicitly control when command buffer memory is reset
m_CommandBuffer.Initialize(m_pGraphicsSystem, true);
GenerateCommandSets();
m_IsInitialized = true;
}
}
void MultiCommandSet::DrawFrame() NN_NOEXCEPT
{
if (!m_IsPrebuiltCommandBufferUsed || m_IsScreenResolutionChanged)
{
GenerateCommandSets();
}
m_pGraphicsSystem->GetQueue()->SubmitCommands(1, &m_RenderContextCommandHandle);
m_pGraphicsSystem->GetQueue()->SubmitCommands(NumConeInstances, m_CommandHandles);
m_IsScreenResolutionChanged = false;
}
void MultiCommandSet::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_PrimitiveRenderer.Finalize();
m_VertexMemPool.Finalize();
m_CommandBuffer.Finalize();
m_pGraphicsSystem = nullptr;
m_IsInitialized = false;
}
}
//-------------------------------------------------------------------------------------------------
// Private methods
//-------------------------------------------------------------------------------------------------
void MultiCommandSet::GenerateCommandSets() NN_NOEXCEPT
{
// reset command memory here since we are using the manual reset mode
m_CommandBuffer.Reset();
// first command set - common initial rendering commands for the scene
{
nvn::CommandBuffer* pCommandBuffer = m_CommandBuffer.StartRecording();
m_pGraphicsSystem->SetDefaultViewportAndScissor(pCommandBuffer);
Color clearColor(0x222222);
pCommandBuffer->ClearColor(0, clearColor.Get(), nvn::ClearColorMask::RGBA);
pCommandBuffer->ClearDepthStencil(1.0f, true, 0, 0);
m_RenderState.Bind(pCommandBuffer);
m_PrimitiveRenderer.StartDrawing(pCommandBuffer);
m_PrimitiveRenderer.BindPrimitive(m_Cone, pCommandBuffer);
}
m_RenderContextCommandHandle = m_CommandBuffer.EndRecording();
// additional command sets - one per each shape instance that is drawn
for (int i = 0; i < NumConeInstances; ++i)
{
nvn::CommandBuffer* pCommandBuffer = m_CommandBuffer.StartRecording();
{
m_PrimitiveRenderer.SetBoundPrimitiveMaterial(m_Materials[i]);
m_PrimitiveRenderer.SetBoundPrimitiveModelView(m_ModelViews[i]);
m_PrimitiveRenderer.DrawBoundPrimitive(pCommandBuffer);
}
m_CommandHandles[i] = m_CommandBuffer.EndRecording();
}
}

{

}
