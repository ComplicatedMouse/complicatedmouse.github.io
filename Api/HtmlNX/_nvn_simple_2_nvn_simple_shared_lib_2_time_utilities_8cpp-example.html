#include "TimeUtilities.h"
//-------------------------------------------------------------------------------------------------
// Public API
//-------------------------------------------------------------------------------------------------
nn::os::Tick GetTick() NN_NOEXCEPT
{
return nn::os::GetSystemTick();
}
nn::TimeSpan GetTimeDelta(nn::os::Tick prevTick) NN_NOEXCEPT
{
return (nn::os::GetSystemTick() - prevTick).ToTimeSpan();
}
double GetTimeDeltaAsMilliseconds(nn::os::Tick prevTick) NN_NOEXCEPT
{
nn::TimeSpan timeSpan = GetTimeDelta(prevTick);
return static_cast<double>(timeSpan.GetNanoSeconds()) / 1000000.0;
}
double GetTimeDeltaAsMicroseconds(nn::os::Tick prevTick) NN_NOEXCEPT
{
nn::TimeSpan timeSpan = GetTimeDelta(prevTick);
return static_cast<double>(timeSpan.GetNanoSeconds()) / 1000.0;
}
double GetTimeDeltaAsNanoseconds(nn::os::Tick prevTick) NN_NOEXCEPT
{
nn::TimeSpan timeSpan = GetTimeDelta(prevTick);
return static_cast<double>(timeSpan.GetNanoSeconds());
}
