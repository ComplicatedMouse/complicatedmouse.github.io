#include <cstdlib>
#include <cstring>
#include <new>
#include <nn/fs.h>
#include <nn/nn_Assert.h>
#include <nn/nn_Log.h>
#include <nn/nn_Abort.h>
extern "C" void nnMain()
{
nn::Result result;
size_t cacheSize = 0;
char* cacheBuffer = nullptr;
// Mounts the file system.
// Mounting requires a cache buffer.
{
NN_LOG("Mount Rom\n");
// Gets the buffer size needed for the file system metadata cache.
// No error handling is required because the library cancels the operation if it fails to get any data.
(void)nn::fs::QueryMountRomCacheSize(&cacheSize);
// Allocates the cache buffer.
cacheBuffer = new(std::nothrow) char[cacheSize];
if (cacheBuffer == nullptr)
{
NN_ASSERT(false, "Cache buffer is null.\n");
return;
}
// Mounts the file system.
// Do not release the cache buffer until you unmount.
result = nn::fs::MountRom("rom", cacheBuffer, cacheSize);
// Always abort when a failure occurs.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
}
// Loads the resource data.
{
const char* FilePath = "rom:/test/testdata";
nn::fs::FileHandle file;
result = nn::fs::OpenFile(&file, FilePath, nn::fs::OpenMode_Read);
if (nn::fs::ResultPathNotFound::Includes(result))
{
// The target file cannot be found.
// This error handling is not necessary when you are only opening existing files.
NN_ASSERT(false, "File not found.\n");
return;
}
// The ResultTargetLocked handling is not necessary because you cannot open using OpenMode_Write when mounting using MountRom.
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
char buffer[1024];
size_t readSize = 0;
// Load the resource data to buffer.
// No error handling is needed. An end occurs within the library when loading fails.
NN_LOG("Read\t%s.\n", FilePath);
(void)nn::fs::ReadFile(&readSize, file, 0, buffer, sizeof(buffer));
NN_LOG("\tsize: %zu\n", readSize);
NN_LOG("\tdata: ");
for (int i = 0; i < static_cast<int>(readSize); ++i)
{
NN_LOG("%02x ", buffer[i]);
}
NN_LOG("\n");
nn::fs::CloseFile(file);
}
// Unmounts the file system.
{
NN_LOG("Unmount Rom\n\n");
nn::fs::Unmount("rom");
// Releases the cache buffer.
delete[] cacheBuffer;
cacheBuffer = nullptr;
}
return;
}
