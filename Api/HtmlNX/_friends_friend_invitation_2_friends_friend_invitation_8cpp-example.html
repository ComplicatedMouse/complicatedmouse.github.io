#define SAMPLE_LOG(...) console::Printf(SampleConsoleIndex, __VA_ARGS__)
void Initialize() NN_NOEXCEPT;
void Finalize() NN_NOEXCEPT;

extern "C" void nnMain()
{
Initialize();
SAMPLE_LOG("[sample] Start.\n");
console::Printf(SampleUsageIndex, u8"A: Send friend invitation\n");
console::Printf(SampleUsageIndex, u8"X: Exit\n");
console::Printf(SampleUsageIndex, u8"↑: Scroll (Up)\n");
console::Printf(SampleUsageIndex, u8"↓: Scroll (Down)\n");
nn::account::UserHandle handle = {};
nn::account::Uid openedUser = nn::account::InvalidUid;
nn::account::Uid invitedUser = nn::account::InvalidUid;
nn::Bit8 parameter[nn::friends::FriendInvitationApplicationParameterSizeMax] = {};
size_t size = 0;
nn::os::SystemEventType* pEvent = nn::friends::GetFriendInvitationNotificationEvent();
// Use the following method to start applications that support the friend notification feature from a friend invitation, or to check for friend invitation notifications.
//
// - Use the nn::friends::GetFriendInvitationNotificationEvent() function to view the obtained event signaled state.
// - Use the nn::friends::TryPopFriendInvitationNotificationInfo() function to attempt to get parameters.
//
// If an application is started from a friend invitation, the event obtained by nn::friends::GetFriendInvitationNotificationEvent() is in the signaled state from the beginning.
//
// To emulate starting an application from a friend invitation, use the 'friends set-friend-invitation' command.
// ====================================================================================================
// RunOnTarget $NINTENDO_SDK_ROOT/TargetTools/NX-NXFP2-a64/DevMenuCommand/Release/DevMenuCommand.nsp \
// -- friends set-friend-invitation --user 0 --file D:/parameter.dat
// ====================================================================================================
// To cancel emulation of starting an application from a friend invite, either restart the system or use the 'friends clear-friend-invitation' command.
// ====================================================================================================
// RunOnTarget $NINTENDO_SDK_ROOT/TargetTools/NX-NXFP2-a64/DevMenuCommand/Release/DevMenuCommand.nsp \
// -- friends clear-friend-invitation
// ====================================================================================================
if (nn::os::TryWaitSystemEvent(pEvent))
{
SAMPLE_LOG("(!) FriendInvitationNotificationEvent was signaled. (Application launched from friend invitation)\n");
if (nn::friends::TryPopFriendInvitationNotificationInfo(&invitedUser, &size, parameter, sizeof (parameter)))
{
// When the Console account selection at startup is set to None in Authoring Editor,
// check whether the application was started from a friend invitation. If it was, then it is clear
// that user performed the operation, so open the account of the invitee.
//
// When Console account selection at startup is set to Required or RequiredWithNetworkServiceAccountAvailable,
// the invited user account is the same as the user account selected at startup. (Note 1)
//
// This sample sets Console account selection at startup to None.
//
// Note 1:
// This is not always the case when using DevMenuCommand for emulation.
// For example, if you set the parameters with 'friends set-friend-invitation --user 1' and then select the 0th user account, they do not match.
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::account::OpenUser(&handle, invitedUser));
openedUser = invitedUser;
DumpFriendInvitationApplicationParameter(parameter, size);
}
}
if (openedUser == nn::account::InvalidUid)
{
// Selects a user account because the application was not started from a friend invitation.
nn::Result result = nn::account::ShowUserSelector(&openedUser);
if (nn::account::ResultCancelledByUser::Includes(result))
{
SAMPLE_LOG("nn::account::ShowUserSelector() is canceled by user.\n");
return;
}
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::account::OpenUser(&handle, openedUser));
}
NN_UTIL_SCOPE_EXIT
{
nn::account::CloseUser(handle);
};
// Friend invitation notifications may occur if anything other than OnRestart is set for Distribute Runtime Parameters in Authoring Editor.
// By using the nn::friends::GetFriendInvitationNotificationEvent() function to monitor obtained events, you can detect notifications in real time.
//
// Use the 'friends push-friend-invitation' command to emulate friend invitation notifications.
// The '--suppress-auto-kill' option must be added to RunOnTarget to prevent the application from closing automatically.
// ====================================================================================================
// RunOnTarget $NINTENDO_SDK_ROOT/TargetTools/NX-NXFP2-a64/DevMenuCommand/Release/DevMenuCommand.nsp \
// --suppress-auto-kill -- friends push-friend-invitation --user 0 --file D:/parameter.dat
// ====================================================================================================
while (NN_STATIC_CONDITION(true))
{
nn::hid::NpadButtonSet holdButtons = {};
nn::hid::NpadButtonSet triggerButtons = {};
GetButtons(&holdButtons, &triggerButtons);
if (nn::os::TryWaitSystemEvent(pEvent))
{
SAMPLE_LOG("(!) FriendInvitationNotificationEvent was signaled.\n");
if (nn::friends::TryPopFriendInvitationNotificationInfo(&invitedUser, &size, parameter, sizeof (parameter)))
{
if (invitedUser == openedUser)
{
DumpFriendInvitationApplicationParameter(parameter, size);
}
else
{
// A friend invitation notification was sent to a user account that is not in the Open state, so it is necessary to switch user accounts.
//
// If it is difficult to implement user account switching, set Distribute Runtime Parameters to AlwaysIfUserStateMatched.
// This causes friend invite notifications to only be sent to Open user accounts. (Note 2)
//
// Note 2:
// This is not always the case when using DevMenuCommand for emulation.
// Friend invitation notifications are sent regardless of the Distribute Runtime Parameters settings and user account Open state.
nn::account::CloseUser(handle);
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::account::OpenUser(&handle, invitedUser));
SAMPLE_LOG("(!) User account was switched.\n");
openedUser = invitedUser;
DumpFriendInvitationApplicationParameter(parameter, size);
}
}
}
else if (triggerButtons.Test<nn::hid::NpadButton::A>())
{
// This game mode description is displayed on the details screen for the friend invitation for the user (friend) the notification is sent to.
// Enter information such as "which game mode you are inviting to" and "match conditions."
static nn::friends::FriendInvitationGameModeDescription s_Description = {};
// Strings freely entered by the user may not be included in the description.
// For example, when playing in a stage created by the user, you can enter the automatically generated stage ID, but not the stage name described arbitrarily by the user.
const char* StageId = "XXX-YYY-ZZZ";
// The description can only be set in US English and Japanese in this sample.
// You cannot include line breaks or tab characters in the description.
// We recommend including descriptions in the application in the supported languages set in Authoring Editor.
// For the languages for which you are not adding a description, specify an empty string. (The variable declarations in this sample initialize that string as '= {};'.
nn::util::SNPrintf(s_Description.enUs, s_Description.TextSize,
u8"Let's play game! / GAME MODE: Cooperative play / STAGE ID: %s", StageId);
nn::util::SNPrintf(s_Description.ja, s_Description.TextSize,
u8"一緒に遊びましょう！ / ゲームモード: 協力プレイ / ステージ ID: %s", StageId);
// Application-specific parameters.
static char s_Parameter[nn::friends::FriendInvitationApplicationParameterSizeMax] = {};
// This sample describes parameters in JSON format.
// The application can send any binary data (not in text format).
int length = nn::util::SNPrintf(s_Parameter, sizeof (s_Parameter), "{\"session_id\":12345,\"stage_id\":\"%s\"}", StageId);
nn::Result result = nn::friends::StartFriendInvitation(openedUser, nn::friends::FriendInvitationInviteeCountMax,
s_Description, s_Parameter, static_cast<size_t>(length));
if (!(result.IsSuccess() || nn::friends::ResultAppletCanceled::Includes(result)))
{
SAMPLE_LOG("nn::friends::StartFriendInvitation() failed. e=%08X\n", result.GetInnerValueForDebug());
}
}
else if (holdButtons.Test<nn::hid::NpadButton::Up>())
{
console::Scroll(SampleConsoleIndex, 1);
}
else if (holdButtons.Test<nn::hid::NpadButton::Down>())
{
console::Scroll(SampleConsoleIndex, -1);
}
else if (triggerButtons.Test<nn::hid::NpadButton::X>())
{
break;
}
const int HeartbeatCount = 3 * 60;
static int s_Counter = 0;
// Generates the log at a frequency of once per 180 count (approximately 3 seconds) to check the application behavior.
if (s_Counter++ % HeartbeatCount == 0)
{
SAMPLE_LOG("%c\n", "-/|\\"[(s_Counter / HeartbeatCount) % 4]);
}
console::ProcessFrame();
}
SAMPLE_LOG("[sample] End.\n");
Finalize();
} // NOLINT(impl/function_size)
void Initialize() NN_NOEXCEPT
{
console::Initialize();
console::ConsoleCreateParam cp0 = console::GetDefaultConsoleCreateParam();
cp0.viewWidth = 300;
cp0.viewHeight = 320;
cp0.isTeeEnabled = false;
console::CreateConsole(SampleUsageIndex, cp0);
console::ConsoleCreateParam cp1 = console::GetDefaultConsoleCreateParam();
cp1.bufferHeight = 10000;
cp1.viewX = cp0.viewX + cp0.viewWidth + 10;
cp1.viewWidth = console::DisplayWidth - cp1.viewX;
cp1.isFixedWidth = true;
console::CreateConsole(SampleConsoleIndex, cp1);
nn::hid::InitializeNpad();
nn::hid::SetSupportedNpadStyleSet(nn::hid::NpadStyleFullKey::Mask | nn::hid::NpadStyleHandheld::Mask);
nn::hid::SetSupportedNpadIdType(SupportedNpadIds, NN_ARRAY_SIZE(SupportedNpadIds));
nn::account::Initialize();
}

console::Finalize();
}
void DumpFriendInvitationApplicationParameter(const void* pParameter, size_t size) NN_NOEXCEPT
{
SAMPLE_LOG("---------------------------------------------------------------------------\n");
SAMPLE_LOG("Size = %zu\n", size);
SAMPLE_LOG("---------------------------------------------------------------------------\n");
SAMPLE_LOG("ADDRESS 00 01 02 03 04 05 06 07 08 09 0A 0B 0C 0D 0E 0F 0123456789ABCDEF\n");
SAMPLE_LOG("---------------------------------------------------------------------------\n");
const nn::Bit8* p = reinterpret_cast<const nn::Bit8*>(pParameter);
for (size_t i = 0; i < size; i += 16)
{
char line[80];
int offset = nn::util::SNPrintf(line, sizeof (line), "%08X ", static_cast<int>(i));
size_t count = std::min(size - i, static_cast<size_t>(16));
for (size_t c = 0; c < 16; c++)
{
if (c < count)
{
offset += nn::util::SNPrintf(&line[offset], sizeof (line) - offset, "%02X ", p[i + c]);
}
else
{
offset += nn::util::SNPrintf(&line[offset], sizeof (line) - offset, " ");
}
}
offset += nn::util::SNPrintf(&line[offset], sizeof (line) - offset, " ");
for (size_t c = 0; c < 16; c++)
{
char ch = ' ';
if (c < count)
{
if (std::isprint(p[i + c]))
{
ch = static_cast<char>(p[i + c]);
}
else
{
ch = '.';
}
}
offset += nn::util::SNPrintf(&line[offset], sizeof (line) - offset, "%c", ch);
}
SAMPLE_LOG("%s\n", line);
}
SAMPLE_LOG("---------------------------------------------------------------------------\n");
}
void GetButtons(nn::hid::NpadButtonSet* pHoldButtons, nn::hid::NpadButtonSet* pTriggerButtons) NN_NOEXCEPT
{
static nn::hid::NpadButtonSet s_PrevButtons = {};
nn::hid::NpadButtonSet buttons = {};
nn::hid::NpadStyleSet style1 = nn::hid::GetNpadStyleSet(nn::hid::NpadId::No1);
nn::hid::NpadStyleSet style2 = nn::hid::GetNpadStyleSet(nn::hid::NpadId::Handheld);
if (style1.Test<nn::hid::NpadStyleFullKey>())
{
nn::hid::NpadFullKeyState state;
nn::hid::GetNpadState(&state, nn::hid::NpadId::No1);
buttons = state.buttons;
}
else if (style2.Test<nn::hid::NpadStyleHandheld>())
{
nn::hid::NpadHandheldState state;
nn::hid::GetNpadState(&state, nn::hid::NpadId::Handheld);
buttons = state.buttons;
}
nn::hid::NpadButtonSet trigger = (~s_PrevButtons) & buttons;
s_PrevButtons = buttons;
*pHoldButtons = buttons;
*pTriggerButtons = trigger;
}
