}
void* ReadFile(int64_t* pOutSize, const char* path, nn::mem::StandardAllocator* pAllocator, size_t align) NN_NOEXCEPT
{
nn::fs::FileHandle fileHandle;
nn::Result result = nn::fs::OpenFile(&fileHandle, path, nn::fs::OpenMode_Read);
if (result.IsFailure())
{
NN_ABORT("OpenFile failed.\n");
}
int64_t fileSize = 0;
nn::fs::GetFileSize(&fileSize, fileHandle);
void* buf = pAllocator->Allocate(static_cast<size_t>(fileSize), align);
result = nn::fs::ReadFile(fileHandle, 0, buf, static_cast<size_t>(fileSize));
if (result.IsFailure())
{
NN_ABORT("ReadFile failed.\n");
}
nn::fs::CloseFile(fileHandle);
*pOutSize = fileSize;
return buf;
}
TextureAsset* ReadTexture(const char* path, nn::mem::StandardAllocator* pAllocator) NN_NOEXCEPT
{
return ReadTexture(path, pAllocator, DeafultAlignment);
}
TextureAsset* ReadTexture(const char* path, nn::mem::StandardAllocator* pAllocator, size_t align) NN_NOEXCEPT
{
int64_t textureAssetFileSize = 0;
TextureAsset* pTextureAsset = static_cast<TextureAsset*>(ReadFile(&textureAssetFileSize, path, pAllocator, align));
NN_ASSERT(pTextureAsset->header.id == TextureAsset::HeaderId);
NN_ASSERT(static_cast<size_t>(textureAssetFileSize) == (TextureAsset::HeaderLength + pTextureAsset->header.texelsDataSize));
return pTextureAsset;
}
