namespace {
const size_t ThreadStackSize = 8192; // Stack size of the barrier operation thread.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack1[ ThreadStackSize ]; // The first thread stack.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack2[ ThreadStackSize ]; // The second thread stack.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack3[ ThreadStackSize ]; // The third thread stack.
nn::os::BarrierType g_Barrier; // Operation target barrier.
} // Namespace.
//-----------------------------------------------------------------------------
//
// Random wait time.
//
void WaitRandom(const char* name)
{
for (int i = 0; i < 3; ++i)
{
// Use the Tick value in place of a random number.
int64_t tick = nn::os::GetSystemTick().GetInt64Value();
auto waitTime = nn::TimeSpan::FromMilliSeconds( ((tick & 0xf) + 1) * 10 );
NN_LOG("Wait %lld milliseconds.\n", name, waitTime.GetMilliSeconds());
nn::os::SleepThread( waitTime );
}
}
//
// The first barrier wait thread.
//
void ThreadFunction1(void *arg)
{
const char* name = reinterpret_cast<const char*>( arg );
NN_LOG("Start\n");
WaitRandom( name );
// Barrier synchronization wait.
NN_LOG("Wait barrier synchronization.\n");
nn::os::AwaitBarrier( &g_Barrier );
NN_LOG("Resume after barrier wait.\n");
}
//
// The second barrier wait thread.
//
void ThreadFunction2(void *arg)
{
const char* name = reinterpret_cast<const char*>( arg );
NN_LOG("Start\n");
WaitRandom( name );
// Barrier synchronization wait.
NN_LOG("Wait barrier synchronization.\n");
nn::os::AwaitBarrier( &g_Barrier );
NN_LOG("Resume after barrier wait.\n");
}
//
// The third barrier wait thread.
//
void ThreadFunction3(void *arg)
{
const char* name = reinterpret_cast<const char*>( arg );
NN_LOG("Start\n");
WaitRandom( name );
// Barrier synchronization wait.
NN_LOG("Wait barrier synchronization.\n");
nn::os::AwaitBarrier( &g_Barrier );
NN_LOG("Resume after barrier wait.\n");
}
//
// The main function.
//
extern "C" void nnMain()
{
nn::os::ThreadType thread1;
nn::os::ThreadType thread2;
nn::os::ThreadType thread3;
nn::Result result;
// Initialize the barrier.
nn::os::InitializeBarrier( &g_Barrier, 3 );
// Generate threads.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread1, ThreadFunction1, const_cast<char*>("Thread1"), g_ThreadStack1, ThreadStackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread2, ThreadFunction2, const_cast<char*>("Thread2"), g_ThreadStack2, ThreadStackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread3, ThreadFunction3, const_cast<char*>("Thread3"), g_ThreadStack3, ThreadStackSize, nn::os::DefaultThreadPriority ) );
// Start thread execution.
nn::os::StartThread( &thread1 );
nn::os::StartThread( &thread2 );
nn::os::StartThread( &thread3 );
// Wait for the thread to end.
nn::os::WaitThread( &thread1 );
nn::os::WaitThread( &thread2 );
nn::os::WaitThread( &thread3 );
// Free threads.
nn::os::DestroyThread( &thread1 );
nn::os::DestroyThread( &thread2 );
nn::os::DestroyThread( &thread3 );
// Finalize the barrier.
nn::os::FinalizeBarrier( &g_Barrier );
// Verify that all threads have terminated.
NN_LOG("Ok.\n");
}
