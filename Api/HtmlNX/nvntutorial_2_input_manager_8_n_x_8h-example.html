#pragma once
#include <nn/hid.h>
#include <nn/hid/hid_Npad.h>
#include <nn/hid/hid_NpadJoy.h>
#include <nn/hid/hid_DebugPad.h>
#include <unordered_map>
#include <nvntutorial/InputManager.h>
#define NPAD 0 // NOLINT(preprocessor/const)
#define DEBUG_PAD 1 // NOLINT(preprocessor/const)
#define PAD_TYPE DEBUG_PAD // NOLINT(preprocessor/const)
class InputManagerNX : public InputManager
{
public:
InputManagerNX();
~InputManagerNX();
virtual void Init();
virtual void Update();
enum GamepadButton
{
GamepadButton_A,
GamepadButton_B,
GamepadButton_X,
GamepadButton_Y,
#if (PAD_TYPE == NPAD)
GamepadButton_StickL,
GamepadButton_StickR,
#endif
GamepadButton_L,
GamepadButton_R,
GamepadButton_ZL,
GamepadButton_ZR,
GamepadButton_Plus,
GamepadButton_Minus,
GamepadButton_Left,
GamepadButton_Up,
GamepadButton_Right,
GamepadButton_Down,
GamePadButton_NUM
};
ButtonState GetGamepadButtonState(GamepadButton button);
void GetLeftStickState(float& leftX, float& leftY);
void GetRightStickState(float& rightX, float& rightY);
private:
void RemapSticks(int inX, int inY, float& outX, float& outY);
ButtonState m_CurrentGamepadButtonState[GamePadButton_NUM];
#if (PAD_TYPE == NPAD)
nn::hid::NpadIdType m_NpadIds[5] = { nn::hid::NpadId::No1,
nn::hid::NpadId::No2,
nn::hid::NpadId::No3,
nn::hid::NpadId::No4,
nn::hid::NpadId::Handheld };
static const int g_NpadIdCountMax = sizeof(m_NpadIds) / sizeof(nn::hid::NpadIdType);
nn::hid::NpadHandheldState m_OldNpadHandheldState;
nn::hid::NpadHandheldState m_CurrentNpadHandheldState;
nn::hid::NpadFullKeyState m_OldNpadFullKeyState[g_NpadIdCountMax];
nn::hid::NpadFullKeyState m_CurrentNpadFullKeyState[g_NpadIdCountMax];
nn::hid::NpadJoyDualState m_OldNpadJoyDualState[g_NpadIdCountMax * 2];
nn::hid::NpadJoyDualState m_CurrentNpadJoyDualState[g_NpadIdCountMax * 2];
#elif (PAD_TYPE == DEBUG_PAD)
nn::hid::DebugPadState m_OldDebugPadState;
nn::hid::DebugPadState m_CurrentDebugPadState;
#else
#error
#endif
static const float g_AnalogStickDeadZoneFloat;
static const float g_AnalogStickDeadZoneFloatScale;
float m_LeftStickX;
float m_LeftStickY;
float m_RightStickX;
float m_RightStickY;
std::unordered_map<GamepadButton, int> m_GamepadBitflagMapping;
};
