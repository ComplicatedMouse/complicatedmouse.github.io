#include <stdint.h>
#include <nn/nn_Macro.h>
#include <nn/nn_Log.h>
#include <nn/nn_Abort.h>
#include <nn/os.h>
//-----------------------------------------------------------------------------
namespace {
const size_t ThreadStackSize = 8192; // Stack size of the thread operation thread.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack1[ ThreadStackSize ]; // The first thread stack.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack2[ ThreadStackSize ]; // The second thread stack.
nn::os::ThreadType g_Thread1;
nn::os::ThreadType g_Thread2;
} // Namespace.
//-----------------------------------------------------------------------------
//
// Display thread information.
//
void PrintThreadInfo(const char* name, nn::os::ThreadType* thread)
{
NN_LOG("[%s]\n", name);
const char* threadName = nn::os::GetThreadNamePointer( thread );
NN_LOG("Name: %s.\n", threadName);
NN_LOG("Priority: %d\n", nn::os::GetThreadPriority( thread ));
NN_LOG("Core number: %d\n", nn::os::GetCurrentCoreNumber());
NN_LOG("\n");
}
//
// First thread.
//
void ThreadFunction1(void *arg)
{
NN_UNUSED(arg);
PrintThreadInfo( "thread1", &g_Thread1 );
}
//
// Second thread.
//
void ThreadFunction2(void *arg)
{
NN_UNUSED(arg);
// Wait for thread 1 to terminate.
nn::os::WaitThread( &g_Thread1 );
PrintThreadInfo( "thread2", &g_Thread2 );
// Change the name of thread 2.
NN_LOG("Change the name of thread 2.\n");
nn::os::SetThreadName( &g_Thread2, "Thread2_NewName" );
// Change the priority of thread 2.
NN_LOG("Change the priority of thread 2 to 5.\n");
nn::os::ChangeThreadPriority( &g_Thread2, 5 );
NN_LOG("\n");
PrintThreadInfo( "thread2", &g_Thread2 );
}
//
// The main function.
//
extern "C" void nnMain()
{
nn::Result result;
// Generate threads.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &g_Thread1, ThreadFunction1, NULL, g_ThreadStack1, ThreadStackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &g_Thread2, ThreadFunction2, NULL, g_ThreadStack2, ThreadStackSize, nn::os::DefaultThreadPriority ) );
// Start thread execution.
nn::os::StartThread( &g_Thread1 );
nn::os::StartThread( &g_Thread2 );
// Wait for the thread to end.
nn::os::WaitThread( &g_Thread1 );
nn::os::WaitThread( &g_Thread2 );
// Free threads.
nn::os::DestroyThread( &g_Thread1 );
nn::os::DestroyThread( &g_Thread2 );
}
