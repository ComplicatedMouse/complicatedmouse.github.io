#include <nn/nn_Common.h>
#include <nn/nn_Log.h>
#include <nn/oe.h>
//-----------------------------------------------------------------------------
//
// Main function.
//
extern "C" void nnMain()
{
// Initialize the application operating environment library.
nn::oe::Initialize();
// Enable notifications of changes in operating mode.
nn::oe::SetOperationModeChangedNotificationEnabled(true);
// Enable notifications of changes in performance mode.
nn::oe::SetPerformanceModeChangedNotificationEnabled(true);
// Display the current operation mode and performance mode.
NN_LOG("The current operation mode is %s mode.\n",
nn::oe::GetOperationMode() == nn::oe::OperationMode_Handheld ? "Handheld" : "Console");
NN_LOG("The current performance mode is %s mode.\n",
nn::oe::GetPerformanceMode() == nn::oe::PerformanceMode_Normal ? "Normal" : "Boost");
// Continue to receive notification messages.
for (;;)
{
auto message = nn::oe::PopNotificationMessage();
switch (message)
{
// The operation mode has been changed.
case nn::oe::MessageOperationModeChanged :
NN_LOG("The operation mode has been changed. The current operation mode is %s mode.\n",
nn::oe::GetOperationMode() == nn::oe::OperationMode_Handheld ? "Handheld" : "Console");
break;
// The performance mode has been changed.
case nn::oe::MessagePerformanceModeChanged :
NN_LOG("The performance mode has been changed. The current performance mode is %s mode.\n",
nn::oe::GetPerformanceMode() == nn::oe::PerformanceMode_Normal ? "Normal" : "Boost");
break;
default:
// Do nothing.
break;
}
}
}
