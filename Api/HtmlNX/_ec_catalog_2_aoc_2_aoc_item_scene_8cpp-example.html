void AocItemScene::SetAocInstalled()
{
// Get the AocIndex that is already installed.
const int installAocCount = nn::aoc::CountAddOnContent();
std::unique_ptr<nn::aoc::AddOnContentIndex[]> installAocBuffer(new nn::aoc::AddOnContentIndex[installAocCount]);
nn::aoc::ListAddOnContent(installAocBuffer.get(), 0, installAocCount);
// Store the downloadable content installation status.
for (auto aocItem : m_RequestAocItems.GetResponse().GetAocItems())
{
const std::vector<int> aocIndexes = aocItem.GetAocIndexes();
if (installAocCount == 0 || aocIndexes.size() == 0)
{
m_IsAocItemsInstalled.push_back(false);
continue;
}
bool isAocInstalled = true;
for (auto aocIndex : aocIndexes)
{
int i;
for (i = 0; i < installAocCount; i++)
{
if (aocIndex == installAocBuffer[i])
{
break;
}
}
// The process ends immediately if no downloadable content is found.
if (i == installAocCount)
{
isAocInstalled = false;
break;
}
}
m_IsAocItemsInstalled.push_back(isAocInstalled);
}
}
// Display the list to show during selection and details about downloadable content items.
void AocItemScene::DrawSelectListCallBack(int listIndex, SceneBase* pScene, int x, int y, bool isSelected) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pScene);
AocItemScene* pAocItemScene = dynamic_cast<AocItemScene*>(pScene);
const int aocItemIndex = pAocItemScene->m_DisplayList[listIndex];
const int contentPriceIndex = pAocItemScene->m_ContentPricesIndexs[aocItemIndex];
const AocItem& aocItem = pAocItemScene->m_RequestAocItems.GetResponse().GetAocItems()[aocItemIndex];
const ContentPrice& contentPrice = pAocItemScene->m_RequestContentPrices.GetResponse().GetContentPrices()[contentPriceIndex];
const bool isAocInstalled = pAocItemScene->m_IsAocItemsInstalled[aocItemIndex];
ImGui::SetWindowFontScale(1.0f);
ImGui::SetCursorPos(ImVec2(x, y));
ImGui::Text("%s", aocItem.GetName().c_str());
// Display more details about the selected item.
if (isSelected == true)
{
// Display downloaded content, bundles, and so on.
pAocItemScene->m_DetailWindow.DrawContentType(aocItem.GetContentType());
// Item Name / Price / Description / Exceptions Display
pAocItemScene->m_DetailWindow.DrawDetailInfomation(
aocItem.GetName(),
pAocItemScene->m_RequestContentPrices.GetResponse().GetIncTaxMessage(),
aocItem.GetDescription(),
aocItem.GetDisclaimer(),
pAocItemScene->m_RequestAocItems.GetResponse().GetLang(),
contentPrice.GetPrice()
);
// Display installation information only for downloadable content.
if (aocItem.GetContentType() == "aoc")
{
pAocItemScene->m_DetailWindow.DrawInstallationText(isAocInstalled);
}
}
}
// Screen rendering.
void AocItemScene::Draw() NN_NOEXCEPT
{
switch (m_AocItemStatus)
{
case AocItemStatus_None:
break;
case AocItemStatus_AocItemsInitialize:
case AocItemStatus_AocItemsResponseWait:
case AocItemStatus_ContentPricesInitialize:
case AocItemStatus_ContentPricesResponseWait:
ImGui::SetWindowFontScale(1.8f);
ImGui::SetCursorPos(ImVec2(550, 675));
ImGui::TextColored(White, "Now Loading...");
break;
case AocItemStatus_AocItemListStart:
{
// Display the selection screen.
m_Select.DrawSelectionList(DrawSelectListCallBack, this);
// Note: There is no way to ensure that the language of the response is one of the system languages.
// Do not treat this condition as an error. Continue to display product information as is.
nn::settings::LanguageCode lang = nn::oe::GetDesiredLanguage();
if (::nn::util::Strncmp(lang.string, m_RequestAocItems.GetResponse().GetLang().c_str(), sizeof(lang.string)) != 0)
{
// Notation displays (translation):
// Product information cannot be displayed using the languages configured for this application. Display using an available language.
ImGui::SetWindowFontScale(1.1f);
ImGui::SetCursorPos(ImVec2(10, 570));
ImGui::TextColored(Red, "Product information cannot be displayed in this language currently set for this application. The supported language will be displayed.");
}
// Displays sale status.
ImGui::SetWindowFontScale(1.0f);
ImGui::SetCursorPos(ImVec2(50, 50));
ImGui::TextColored(White, "Selected sales status : %s", pSceneData->GetSalesStatusStirng().c_str());
// Description of operations.
ImGui::SetWindowFontScale(1.5f);
ImGui::SetCursorPos(ImVec2(10, 600));
ImGui::TextColored(White, "Select an item by up or down or left or right button and Push A button to jump to the item in eshop.\nPush B button to move to the selected scene for sales status.");
// Display the frame for the Details window.
m_DetailWindow.DrawFrame("AOC item detail infomation.");
break;
}
default:
NN_LOG("Status that does not exist.\n");
break;
}
}
// Input management processing.
void AocItemScene::Control(const nns::hid::GamePad& gamePad) NN_NOEXCEPT
{
// B Button for canceling loading.
bool isCancelButtonPushed = gamePad.HasAnyButtonsUp(nns::hid::Button::B::Mask);
switch (m_AocItemStatus)
{
case AocItemStatus_AocItemsInitialize:
{
const auto isOnsaleOnly = pSceneData->GetSalesStatus() != SceneData::SalesStatus::SalesStatus_All;
const auto isSuccess = m_RequestAocItems.SetupRequest(pSceneData->GetUserContext()->id, isOnsaleOnly);
if (isSuccess == false)
{
// Return to the previous screen when an error occurs.
NN_LOG("Request or initialization failed.\n");
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
m_AocItemStatus = AocItemStatus_AocItemsResponseWait;
}
break;
}
case AocItemStatus_AocItemsResponseWait:
if (!m_RequestAocItems.IsCompleted())
{
if (isCancelButtonPushed == true)
{
// Canceling.
m_RequestAocItems.Cancel();
}
}
else
{
try
{
if (m_RequestAocItems.WaitFinish(pSceneData->GetUserContext()->userHandle) == false)
{
//If an error or cancellation occurs.
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
if (m_RequestAocItems.ConvertResponse() == false)
{
// Display an error when the parse fails.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
}
else
{
// Confirm that length matches the number of items.
if (m_RequestAocItems.GetResponse().GetLength() != m_RequestAocItems.GetResponse().GetAocItems().size())
{
// length does not match the number of items.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_ItemCountDifferent);
SceneBase::SetStatus(DisplayStatus_Error);
NN_LOG("AOC Items length error.\n");
break;
}
SetAocInstalled();
// Next, load the AocPrices data.
m_AocItemStatus = AocItemStatus_ContentPricesInitialize;
}
}
}
catch (std::bad_alloc& e)
{
// Processes cancellation when memory allocation fails.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
catch (std::exception& e)
{
// Although only bad_alloc is expected in the sample, this processing is used when other exceptions are thrown.
// Process cancellation.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
}
break;
case AocItemStatus_ContentPricesInitialize:
{
// Create an nsUid list from all of the AocItems data.
nn::ec::NsUid nsUidList[SceneData::NsUidListMax];
int i;
for (i = 0; i < m_RequestAocItems.GetResponse().GetAocItems().size() && i < SceneData::NsUidListMax; i++)
{
nsUidList[i].value = m_RequestAocItems.GetResponse().GetAocItems()[i].GetNsUid().value;
}
const auto isSuccess = m_RequestContentPrices.SetupRequest(pSceneData->GetUserContext()->id, nsUidList,i);
if (isSuccess == false)
{
// Return to the previous screen when an error occurs.
NN_LOG("Request or initialization failed.\n");
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
m_AocItemStatus = AocItemStatus_ContentPricesResponseWait;
}
break;
}
case AocItemStatus_ContentPricesResponseWait:
if (!m_RequestContentPrices.IsCompleted())
{
if (isCancelButtonPushed == true)
{
// Canceling.
m_RequestContentPrices.Cancel();
}
}
else
{
try
{
if (m_RequestContentPrices.WaitFinish(pSceneData->GetUserContext()->userHandle) == false)
{
//If an error or cancellation occurs.
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
if (m_RequestContentPrices.ConvertResponse() == false)
{
// Display an error when the parse fails.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
}
else
{
// Confirm that length matches the number of items.
if (m_RequestContentPrices.GetResponse().GetLength() != m_RequestContentPrices.GetResponse().GetContentPrices().size())
{
// length does not match the number of items.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_ItemCountDifferent);
SceneBase::SetStatus(DisplayStatus_Error);
NN_LOG("Content Prices length Error.\n");
break;
}
// Assign ContentPrices array numbers to AocItems, and store the AocItems array numbers to display on the screen.
int displayListSize = 0;
for (int i = 0; i < m_RequestAocItems.GetResponse().GetAocItems().size(); i++)
{
nn::ec::NsUid itemNsUid;
itemNsUid.value = m_RequestAocItems.GetResponse().GetAocItems()[i].GetNsUid().value;
// Search ContentPrices for items with the same NsUid.
bool isExsist = false;
for (int j = 0; j < m_RequestContentPrices.GetResponse().GetContentPrices().size(); j++)
{
if (itemNsUid.value == m_RequestContentPrices.GetResponse().GetContentPrices()[j].GetNsUid().value)
{
// not_onsale items are handled by not being displayed on the list.
if (m_RequestContentPrices.GetResponse().GetContentPrices()[j].GetPrice().GetSalesStatus().compare("not_onsale") != 0)
{
// Store the ContentPrices array numbers in an array with the same structure as AocItems, such that the AocItems array numbers match up.
m_ContentPricesIndexs.push_back(j);
isExsist = true;
m_DisplayList[displayListSize] = i;
displayListSize++;
break;
}
}
}
// If there are AocItems objects but no ContentPrices objects, perform push_back to match the number of m_ContentPricesIndexs arrays with AocItems.
if (isExsist == false)
{
m_ContentPricesIndexs.push_back(std::numeric_limits<int>::max());
}
}
if (displayListSize == 0)
{
// Display an error if there are no items to display.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_NoneItem);
SceneBase::SetStatus(DisplayStatus_Error);
}
else
{
// The initial settings for the choices.
const int listMax = displayListSize; // The number of items to choose among.
constexpr int Row = 11; // The number of rows to display for the selection.
constexpr int OffsetX = 360; // The amount to shift the selection horizontally.
constexpr int OffsetY = 45; // The amount to shift the selection vertically.
constexpr int PositionX = 10; // The coordinate for the left position of the selection.
constexpr int PositionY = 100; // The coordinate for the upper position of the selection.
constexpr float CursorSize = 1.5; // The size of the cursor.
constexpr int CursorOffsetX = 5; // Fine tuning the x-position of the cursor.
constexpr int CursorOffsetY = -6; // Fine tuning the y-position of the cursor.
m_Select.SetInitialValue(
listMax
, Row
, OffsetX
, OffsetY
, PositionX
, PositionY
, pSceneData->GetSelectedPositionAocValue() // The initial position of the cursor.
, CursorSize
, CursorOffsetX
, CursorOffsetY
);
// Processing for informing the main screen that loading completed and deleting the previous screen.
pSceneData->SetLoadCompleted(true);
m_AocItemStatus = AocItemStatus_AocItemListStart;
}
}
}
}
catch (std::bad_alloc& e)
{
// Processes cancellation when memory allocation fails.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
catch (std::exception& e)
{
// Although only bad_alloc is expected in the sample, this processing is used when other exceptions are thrown.
// Process cancellation.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
}
break;
case AocItemStatus_AocItemListStart:
{
int selected;
Select::SelectStatus selectStatus;
// Execute the selection process.
m_Select.InputSelectionList(&selectStatus, &selected, gamePad);
pSceneData->SetSelectedPositionAocKey(selected);
if (selectStatus == Select::SelectStatus_Cancel)
{
// If the process has been canceled.
SceneBase::SetStatus(DisplayStatus_AocSalesStatusSelection);
}
if (selectStatus == Select::SelectStatus_Decision)
{
// Get the nsUid from the selected number.
const int aocItemIndex = m_DisplayList[selected];
nn::ec::NsUid nsUid;
nsUid.value = m_RequestAocItems.GetResponse().GetAocItems()[aocItemIndex].GetNsUid().value;
pSceneData->SetNsUid(nsUid);
SceneBase::SetStatus(DisplayStatus_ShowShopProductDetails);
}
break;
}
default:
break;
}
}//NOLINT(impl/function_size)
