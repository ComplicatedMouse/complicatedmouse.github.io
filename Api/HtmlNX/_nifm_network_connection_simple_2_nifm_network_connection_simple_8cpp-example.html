// Socket Communication Processing Sample
// Perform HTTP communication with example.com and output the obtained content and results to a log.
// Returns true if communication initialized successfully, or false otherwise.
// For more information about <tt>libcurl</tt> functions, see the libcurl documentation.
bool Program::ExecuteTcpIpCommunication() NN_NOEXCEPT
{
nn::os::SleepThread(nn::TimeSpan::FromSeconds(10));
nn::socket::Initialize(g_SocketConfigWithMemory);
curl_global_init(CURL_GLOBAL_DEFAULT);
CURL* curl = curl_easy_init();
if (curl == nullptr)
{
nn::socket::Finalize();
Printf("curl_easy_init failed.\n");
return false;
}
curl_easy_setopt(curl, CURLOPT_URL, "http://example.com");
// In the Debug and Develop versions, the content received from the server is displayed.
CURLcode curlCode = curl_easy_perform(curl);
Printf("curl_easy_perform returned %d.\n\n", curlCode);
char *pUrl;
curl_easy_getinfo(curl, CURLINFO_EFFECTIVE_URL, &pUrl);
Printf("url: %s\n", pUrl);
long responseCode = -1;
curl_easy_getinfo(curl, CURLINFO_RESPONSE_CODE, &responseCode);
Printf("response code: %d\n", static_cast<int>(responseCode));
curl_easy_cleanup(curl);
curl_global_cleanup();
nn::socket::Finalize();
return curlCode == CURLE_OK;
}
// The main function of the application.
void Program::Execute() NN_NOEXCEPT
{
// The NIFM library must be initialized before the NIFM API can be used.
nn::nifm::Initialize();
while (NN_STATIC_CONDITION(1))
{
// Submit a request to use network connection before communication processing,
// and confirm availability of the network connection.
#if 1
// If blocking APIs are being used:
nn::nifm::SubmitNetworkRequestAndWait();
#else
// If non-blocking APIs are being used and if checking for completion by polling:
nn::nifm::SubmitNetworkRequest();
while (nn::nifm::IsNetworkRequestOnHold())
{
Printf("Network request is on hold...\n");
nn::os::SleepThread(nn::TimeSpan::FromSeconds(1));
}
#endif
if (nn::nifm::IsNetworkAvailable())
{
Printf("Network is available.\n");
break;
}
// Handle use requests that result in failure.
if (nn::nifm::HandleNetworkRequestErrorResult())
{
// Resubmitting the use request may result in it being accepted.
Printf("Network is not available. SubmitNetworkRequest again.\n");
continue;
}
Printf("Network is not available.\n");
break;
}
// Perform the communications process if a network connection is available.
if (nn::nifm::IsNetworkAvailable())
{
// Communication processing.
bool isSuccessful = ExecuteTcpIpCommunication();
if (!isSuccessful)
{
// If the process fails because the network connection failed, display a message about that failure.
if (nn::nifm::IsNetworkAvailable())
{
Printf("Communication failed.\n");
}
else
{
Printf("Lost network.\n");
// This function may be called again to display the reason for disconnection (not required).
nn::nifm::HandleNetworkRequestErrorResult();
}
}
}
// Withdraw a request for a network connection if it is no longer needed.
// If the network connection is lost for unexpected reasons,
// you do not need to explicitly withdraw the request,
// but there is no problem if you do.
nn::nifm::CancelNetworkRequest();
NN_ASSERT(!nn::nifm::IsNetworkAvailable());
}
