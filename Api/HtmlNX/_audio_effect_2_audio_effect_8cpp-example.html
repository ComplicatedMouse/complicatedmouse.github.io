std::size_t ReadAdpcmFile(nn::audio::AdpcmHeaderInfo* header, void** adpcmData, const char* filename)
{
nn::fs::FileHandle handle;
nn::Result result = nn::fs::OpenFile(&handle, filename, nn::fs::OpenMode_Read);
NN_ABORT_UNLESS(result.IsSuccess());
int64_t size;
uint8_t adpcmheader[nn::audio::AdpcmHeaderSize];
result = nn::fs::GetFileSize(&size, handle);
NN_ABORT_UNLESS(result.IsSuccess());
*adpcmData = g_WaveBufferAllocator.Allocate(static_cast<std::size_t>(size) - sizeof(adpcmheader));
NN_ABORT_UNLESS_NOT_NULL(*adpcmData);
result = nn::fs::ReadFile(handle, 0, adpcmheader, sizeof(adpcmheader));
NN_ABORT_UNLESS(result.IsSuccess());
result = nn::fs::ReadFile(handle, sizeof(adpcmheader), *adpcmData, static_cast<size_t>(size) - sizeof(adpcmheader));
NN_ABORT_UNLESS(result.IsSuccess());
nn::fs::CloseFile(handle);
nn::audio::ParseAdpcmHeader(header, adpcmheader, sizeof(adpcmheader));
return static_cast<std::size_t>(size) - sizeof(adpcmheader);
}
void InitializeFileSystem()
{
nn::fs::SetAllocator(Allocate, Deallocate);
size_t cacheSize = 0;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::QueryMountRomCacheSize(&cacheSize));
g_MountRomCacheBuffer = new(std::nothrow) char[cacheSize];
NN_ABORT_UNLESS_NOT_NULL(g_MountRomCacheBuffer);
NN_ABORT_UNLESS_RESULT_SUCCESS(
nn::fs::MountRom("asset", g_MountRomCacheBuffer, cacheSize)
);
}
void FinalizeFileSystem()
{
nn::fs::Unmount("asset");
delete[] g_MountRomCacheBuffer;
g_MountRomCacheBuffer = NULL;
}

for (int i = 0, channelCountMax = nn::audio::GetLightLimiterChannelCountMax(pLightLimiter); i < channelCountMax; ++i)

void ProcessUserEffect(nn::audio::AuxType aux, int32_t* readBuffer, int sampleCount, int channelCount, int auxInputChannelCount, float balance, float volume)
{
// Process the obtained sample data in accordance with the aux effect.
// At this point, just simply adjust the volume and the pan.
// Load sample data from the audio renderer.
int readCount = nn::audio::ReadAuxSendBuffer(&aux, readBuffer, channelCount * sampleCount * 2);
while( readCount )
{
// The sample data resulting from the aux effect is block-interleaved by channel.
// Each block is the number of samples maintained by FinalMix / SubMix with Aux effects added.
// At this point, the aux effect has been added to FinalMix, so the number of samples included in each block matches parameter.sampleCount.
int samplesPerFrame = sampleCount;
// The included number of channels is the number set by the SetAuxInputOutput() function.
int channelBlockCount = auxInputChannelCount;
// The sample loaded by the ReadAuxSendBuffer() function gets the number of frames it is equivalent to.
int frameCount = readCount / (samplesPerFrame * channelCount);
for( auto frame = 0; frame < frameCount; ++frame )
{
int32_t* bufferBase = readBuffer + frame * samplesPerFrame * channelBlockCount;
for( auto ch = 0; ch < channelBlockCount; ++ch )
{
int32_t* addr = bufferBase + ch * samplesPerFrame;
float vol = (ch == 0) ? (balance * volume) : ((1.0f - balance) * volume);
for (auto i = 0; i < samplesPerFrame; ++i)
{
addr[i] = static_cast<int32_t>(addr[i] * vol);
}
}
}
// Write back the processed samples to the audio renderer.
int writeCount = nn::audio::WriteAuxReturnBuffer(&aux, readBuffer, readCount);
if( writeCount != readCount )
{
// Could not write back all of the samples to the audio renderer.
// Consider the following points as the general cause.
//
// - The loading frequency of nn::audio::ReadAuxSendBuffer() is lower for reasons such as increased processing load.
// - As a result, the number of samples loaded with nn::audio::ReadAuxSendBuffer() increased.
// - The system attempts to use nn::audio::WriteAuxReturnBuffer() to write all of the increased samples back at the same time.
// - As a result, the amount of buffer area in ReturnBuffer becomes insufficient, and it is not possible to write back all of the samples specified by readCount.
//
// To avoid this problem, consider the following improvements.
// - Reduce the number of samples loaded with nn::audio::ReadAuxSendBuffer().
// - Increase the frequency at which nn::audio::ReadAuxSendBuffer() is called (increase the update frequency for processing Aux effects)
//
NNS_LOG("Notice: Could not write back whole sample\n");
}
readCount = nn::audio::ReadAuxSendBuffer(&aux, readBuffer, channelCount * sampleCount);
}
}
void UpdateUserEffectParameter(const nn::hid::NpadButtonSet& currentButtons, const nn::hid::NpadButtonSet& downButtons, float& volume, float& balance, bool& isProcessEnabled)
{
if (!currentButtons.Test< ::nn::hid::NpadButton::L >())
{
return;
}
if (downButtons.Test< ::nn::hid::NpadButton::Y >())
{
isProcessEnabled = !isProcessEnabled;
NNS_LOG("[Aux] Effect process %s\n", isProcessEnabled ? "enabled" : "disabled");
return;
}
if ( !(currentButtons.Test< ::nn::hid::NpadButton::Up >() ||
currentButtons.Test< ::nn::hid::NpadButton::Down >() ||
currentButtons.Test< ::nn::hid::NpadButton::Left >() ||
currentButtons.Test< ::nn::hid::NpadButton::Right >()))
{
return;
}
if(currentButtons.Test< ::nn::hid::NpadButton::Up >())
{
volume = std::min(1.0f, volume + VolumeDelta);
}
if(currentButtons.Test< ::nn::hid::NpadButton::Down >())
{
volume = std::max(0.0f, volume - VolumeDelta);
}
if(currentButtons.Test< ::nn::hid::NpadButton::Left >())
{
balance = std::min(1.0f, balance + BalanceDelta);
}
if(currentButtons.Test< ::nn::hid::NpadButton::Right >())
{
balance = std::max(0.0f, balance - BalanceDelta);
}
NNS_LOG("UserEffect: Volume:%.2f Position:%.2f\n", volume, balance);
}
int32_t ProcessUserAnalyzer(nn::audio::CaptureType capture, int32_t* readBuffer, int sampleCount, int channelCount, int captureChannelCount)
{
// Analyze the sample data obtained with the Capture effect.
// Calculate the peak value here.
int32_t result = 0;
// Read the sample data from the Capture effect.
int readCount = nn::audio::ReadCaptureBuffer(&capture, readBuffer, channelCount * sampleCount * 2);
while (readCount)
{
// The sample data obtained with the Capture effect is block-interleaved, channel by channel.
// Each block is the number of samples maintained by FinalMix / SubMix with Aux effects added.
// In this sample program, the effect is added to FinalMix, so the number of samples included in each block matches parameter.sampleCount.
int samplesPerFrame = sampleCount;
// The included number of channels is the number set by the SetCaptureInput() function.
int channelBlockCount = captureChannelCount;
// The samples read by the ReadCaptureBuffer() function correspond to a certain number of frames of data.
int frameCount = readCount / (samplesPerFrame * channelCount);
for (auto frame = 0; frame < frameCount; ++frame)
{
int32_t* bufferBase = readBuffer + frame * samplesPerFrame * channelBlockCount;
for (auto ch = 0; ch < channelBlockCount; ++ch)
{
int32_t* addr = bufferBase + ch * samplesPerFrame;
for (auto i = 0; i < samplesPerFrame; ++i)
{
result = std::max(result, std::abs(addr[i]));
}
}
}
readCount = nn::audio::ReadCaptureBuffer(&capture, readBuffer, channelCount * sampleCount);
}
return result;
}

int UpdateReverbParameter(nn::audio::ReverbType* reverb, int updateMode, std::string& message)
{
// Update the reverb effect parameters.
ReverbParameterSet* parameterSet = nullptr;
switch (updateMode)
{
case 0:
parameterSet = &reverbParameterSets[updateMode];
message = "SmallRoom";
break;
case 1:
parameterSet = &reverbParameterSets[updateMode];
message = "Hall";
break;
case 2:
// Disable the reverb effect.
nn::audio::SetReverbEnabled(reverb, false);
message = "disabled";
break;
case 3:
// When enabled, the internal state of the reverb effect is reset.
// Clear all reverberant sounds that played before it was disabled.
nn::audio::SetReverbEnabled(reverb, true);
message = "re-enabled";
break;
default:
break;
}
if (parameterSet != nullptr)
{
// Update the reverb effect parameters.
// The updates are applied in the audio renderer when the
// nn::audio::RequestUpdateAudioRenderer() function is next called.
// Settings for early reflection.
nn::audio::SetReverbEarlyMode(reverb, parameterSet->earlyMode);
nn::audio::SetReverbEarlyGain(reverb, parameterSet->earlyGain);
nn::audio::SetReverbPredelayTime(reverb, parameterSet->predelayTime);
// Settings for late reflection.
nn::audio::SetReverbLateMode(reverb, parameterSet->lateMode);
nn::audio::SetReverbLateGain(reverb, parameterSet->lateGain);
nn::audio::SetReverbDecayTime(reverb, parameterSet->decayTime);
nn::audio::SetReverbHighFrequencyDecayRatio(reverb, parameterSet->highFrequencyDecayRatio);
nn::audio::SetReverbColoration(reverb, parameterSet->coloration);
// Adjust the overall gain for the effect.
nn::audio::SetReverbReverbGain(reverb, parameterSet->reverbGain);
nn::audio::SetReverbOutGain(reverb, parameterSet->outGain);
nn::audio::SetReverbDryGain(reverb, parameterSet->dryGain);
}
return (updateMode + 1) % 4;
}

break;
default:
break;
}

}

nn::audio::SetAuxInputOutput(aux, targetBus,targetBus, busCount);
}
extern "C" void nnMain()
{
g_Allocator.Initialize(g_WorkBuffer, sizeof(g_WorkBuffer));
g_WaveBufferAllocator.Initialize(g_WaveBufferPoolMemory, sizeof(g_WaveBufferPoolMemory));
g_EffectBufferAllocator.Initialize(g_EffectBufferPoolMemory, sizeof(g_EffectBufferPoolMemory));
InitializeFileSystem();
InitializeHidDevices();
AudioEffectSampleStatus sampleStatus = { AudioEffectIndex_Reverb };
// Specify a renderer parameter.
nn::audio::AudioRendererParameter parameter;
nn::audio::InitializeAudioRendererParameter(&parameter);
parameter.sampleRate = RenderRate;
parameter.sampleCount = RenderCount;
parameter.mixBufferCount = 6;
parameter.voiceCount = 24;
parameter.subMixCount = 0;
parameter.sinkCount = 1;
parameter.effectCount = 9; // Reverb, I3dl2Reverb, Delay, LightLimiter, Aux, Capture, BufferMixer * 2, BiquadFilter
parameter.performanceFrameCount = 0;
// Define the relationship between the mix buffer and the audio bus.
int channelCount = 2;
int8_t mainBus[2];
mainBus[nn::audio::ChannelMapping_FrontLeft] = 4;
mainBus[nn::audio::ChannelMapping_FrontRight] = 5;
int8_t subBusA[2];
subBusA[nn::audio::ChannelMapping_FrontLeft] = 2;
subBusA[nn::audio::ChannelMapping_FrontRight] = 3;
int8_t subBusB[2];
subBusB[nn::audio::ChannelMapping_FrontLeft] = 0;
subBusB[nn::audio::ChannelMapping_FrontRight] = 1;
// Check whether the parameter is supported by the system.
NN_ABORT_UNLESS(
nn::audio::IsValidAudioRendererParameter(parameter),
"Invalid AudioRendererParameter specified."
);
// Prepare the renderer's work buffer and register it to AudioRenderer.
size_t workBufferSize = nn::audio::GetAudioRendererWorkBufferSize(parameter);
void* workBuffer = g_Allocator.Allocate(workBufferSize, nn::os::MemoryPageSize);
NN_ABORT_UNLESS_NOT_NULL(workBuffer);
nn::os::SystemEvent systemEvent;
// Initialize the renderer.
nn::audio::AudioRendererHandle handle;
NN_ABORT_UNLESS(
nn::audio::OpenAudioRenderer(&handle, &systemEvent, parameter, workBuffer, workBufferSize).IsSuccess(),
"Failed to open AudioRenderer"
);
// Initialize AudioRendererConfig.
size_t configBufferSize = nn::audio::GetAudioRendererConfigWorkBufferSize(parameter);
void* configBuffer = g_Allocator.Allocate(configBufferSize, nn::os::MemoryPageSize);
NN_ABORT_UNLESS_NOT_NULL(configBuffer);
nn::audio::AudioRendererConfig config;
nn::audio::InitializeAudioRendererConfig(&config, parameter, configBuffer, configBufferSize);
nn::audio::FinalMixType finalMix;
nn::audio::AcquireFinalMix(&config, &finalMix, 4);
// Specify the renderer output destination.
nn::audio::DeviceSinkType deviceSink;
// The output channel is determined based on the mix buffer index specified for mainBus.
// Output from mainBus[nn::audio::ChannelMapping_FrontLeft] is sent to the left channel and
// output from mainBus[nn::audio::ChannelMapping_FrontRight] is sent to the right channel.
nn::Result result = nn::audio::AddDeviceSink(&config, &deviceSink, &finalMix, mainBus, channelCount, "MainAudioOut");
NN_ABORT_UNLESS(result.IsSuccess());
// Prepare a memory pool for the effect buffer.
// MemoryPool is never overwritten with the region corresponding to the address and size that are passed using nn::audio::AcquireMemoryPool().
// As a result, the entire region prepared as g_EffectBufferPoolMemory can be used for the memory pool.
// This sample also has separate allocators ready for WaveBuffer, Effect, and the memory.
// MemoryPool attaches and detaches entire regions, so
// We recommend dividing memory into regions with a granularity required for MemoryPool operations.
// For more information about using MemoryPool, see the
// @link Samples/Sources/Applications/AudioMemoryPool Samples/Sources/Applications/AudioMemoryPool/ @endlink sample.
nn::audio::MemoryPoolType effectBufferPool;
bool rt = nn::audio::AcquireMemoryPool(&config, &effectBufferPool, g_EffectBufferPoolMemory, sizeof(g_EffectBufferPoolMemory));
NN_ABORT_UNLESS(rt);
// Attach the memory pool. The settings are applied every time the nn::audio::RequestUpdateAudioRenderer() function is called.
rt = nn::audio::RequestAttachMemoryPool(&effectBufferPool);
NN_ABORT_UNLESS(rt);
// Prepare a reverb effect.
nn::audio::ReverbType reverb;
size_t reverbSize = nn::audio::GetRequiredBufferSizeForReverb(parameter.sampleRate, channelCount);
void* reverbBuffer = g_EffectBufferAllocator.Allocate(reverbSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS(reverbBuffer != nullptr);
// Add reverb effect to FinalMix.
result = nn::audio::AddReverb(&config, &reverb, reverbBuffer, reverbSize, &finalMix, 2);
NN_ABORT_UNLESS(result.IsSuccess());
// Set the input/output buffers for the reverb effect.
// Write back to the input source buffer.
nn::audio::SetReverbInputOutput(&reverb, subBusA, subBusA, sizeof(subBusA));
// Set the various parameters that determine the characteristics of the reverb effect.
// The configured parameters are applied the next time the nn::audio::RequestUpdateAudioRenderer() function is called. They are not applied until then.
// Make sure that you do not change any settings while nn::audio::RequestUpdateAudioRenderer() is being called.
nn::TimeSpan decayTime = nn::TimeSpan::FromSeconds(1);
nn::TimeSpan predecayTime = nn::TimeSpan::FromMilliSeconds(10);
// Early reflected sound.
nn::audio::SetReverbEarlyMode(&reverb, nn::audio::ReverbType::EarlyMode_Hall);
nn::audio::SetReverbEarlyGain(&reverb, 0.8f);
nn::audio::SetReverbPredelayTime(&reverb, predecayTime);
// Late reverberation.
nn::audio::SetReverbLateMode(&reverb, nn::audio::ReverbType::LateMode_Hall);
nn::audio::SetReverbLateGain(&reverb, 0.5f);
nn::audio::SetReverbDecayTime(&reverb, decayTime);
nn::audio::SetReverbHighFrequencyDecayRatio(&reverb, 0.5f);
nn::audio::SetReverbColoration(&reverb, 0.0f);
nn::audio::SetReverbReverbGain(&reverb, 0.8f);
nn::audio::SetReverbOutGain(&reverb, 1.0f);
nn::audio::SetReverbEnabled(&reverb, true);
nn::audio::I3dl2ReverbType i3dl2Reverb;
size_t i3dl2ReverbSize = nn::audio::GetRequiredBufferSizeForI3dl2Reverb(parameter.sampleRate, channelCount);
void* i3dl2ReverbBuffer = g_EffectBufferAllocator.Allocate(i3dl2ReverbSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS(i3dl2ReverbBuffer != nullptr);
// Add reverb effect to FinalMix.
result = nn::audio::AddI3dl2Reverb(&config, &i3dl2Reverb, i3dl2ReverbBuffer, i3dl2ReverbSize, &finalMix, channelCount);
NN_ABORT_UNLESS(result.IsSuccess());
// Set the input/output buffers for the reverb effect.
// Write back to the input source buffer.
nn::audio::SetI3dl2ReverbInputOutput(&i3dl2Reverb, subBusA, subBusA, sizeof(subBusA));
nn::audio::I3dl2ReverbParameterSet i3dl2Parameter;
nn::audio::LoadI3dl2ReverbPreset(&i3dl2Parameter, nn::audio::I3dl2ReverbType::Preset_SmallRoom);
nn::audio::SetI3dl2ReverbParameters(&i3dl2Reverb, &i3dl2Parameter);
// Prepare a delay effect.
nn::audio::DelayType delay;
nn::TimeSpan delayTimeMax = nn::TimeSpan::FromMilliSeconds(500);
size_t delaySize = nn::audio::GetRequiredBufferSizeForDelay(delayTimeMax, parameter.sampleRate, channelCount);
void* delayBuffer = g_EffectBufferAllocator.Allocate(delaySize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS_NOT_NULL(delayBuffer);
// Add delay effect to FinalMix.
result = nn::audio::AddDelay(&config, &delay, delayBuffer, delaySize, &finalMix, delayTimeMax, channelCount);
NN_ABORT_UNLESS(result.IsSuccess());
// Set the input/output buffers for the delay effect.
nn::audio::SetDelayInputOutput(&delay, subBusA, subBusA, sizeof(subBusA));
DelayParameterSet* parameterSet = &delayParameterSets[1];
nn::audio::SetDelayTime(&delay, parameterSet->delayTime);
nn::audio::SetDelayChannelSpread(&delay, parameterSet->channelSpread);
nn::audio::SetDelayDryGain(&delay, parameterSet->dryGain);
nn::audio::SetDelayFeedbackGain(&delay, parameterSet->feedbackGain);
nn::audio::SetDelayInGain(&delay, parameterSet->inGain);
nn::audio::SetDelayLowPassAmount(&delay, parameterSet->lowPassAmount);
nn::audio::SetDelayEnabled(&delay, true);
// Prepare the LightLimiter effects.
nn::audio::LightLimiterType lightLimiter;
auto lookAheadTimeMax = nn::TimeSpan::FromMicroSeconds(500);
auto lightLimiterSize = nn::audio::GetRequiredBufferSizeForLightLimiter(lookAheadTimeMax, parameter.sampleRate, channelCount);
auto lightLimiterBuffer = g_EffectBufferAllocator.Allocate(lightLimiterSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS_NOT_NULL(lightLimiterBuffer);
// Add LightLimiter to FinalMix.
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::audio::AddLightLimiter(&config, &lightLimiter, lightLimiterBuffer, lightLimiterSize, &finalMix, lookAheadTimeMax, channelCount));
nn::audio::SetLightLimiterInputOutput(&lightLimiter, subBusA, subBusA, sizeof(subBusA));
{
LightLimiterParameterSet* lightLimiterParameter = &lightLimiterParameterSets[1];
nn::audio::SetLightLimiterAttackTime(&lightLimiter, lightLimiterParameter->attackTime);
nn::audio::SetLightLimiterReleaseTime(&lightLimiter, lightLimiterParameter->releaseTime);
nn::audio::SetLightLimiterLookAheadTime(&lightLimiter, lightLimiterParameter->lookAheadTime);
nn::audio::SetLightLimiterThreshold(&lightLimiter, lightLimiterParameter->threshold);
nn::audio::SetLightLimiterInputGain(&lightLimiter, lightLimiterParameter->inputGain);
nn::audio::SetLightLimiterOutputGain(&lightLimiter, lightLimiterParameter->outputGain);
nn::audio::SetLightLimiterEnabled(&lightLimiter, true);
}
// Enables the feature for getting LightLimiter statistics.
nn::audio::SetLightLimiterStatisticsEnabled(&lightLimiter, true);
// Prepare an aux effect.
nn::audio::AuxType aux;
// Allocate the send and return buffers used by the aux effect.
// You are allocating buffers for four audio frames, so there could be a maximum delay of 5 msec × 4 frames = 20 msec.
// Making the buffers smaller would reduce the delay, but it would require a greater frequency for processing on the user end, as noted below.
const int AuxBufferFrameCount = 4;
size_t auxBufferSize = nn::audio::GetRequiredBufferSizeForAuxSendReturnBuffer(&parameter, AuxBufferFrameCount, sizeof(subBusA));
void* sendBuffer = g_EffectBufferAllocator.Allocate(auxBufferSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS_NOT_NULL(sendBuffer);
void* returnBuffer = g_EffectBufferAllocator.Allocate(auxBufferSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS_NOT_NULL(returnBuffer);
// Add the Aux effect to FinalMix.
result = nn::audio::AddAux(&config, &aux, &finalMix, sendBuffer, returnBuffer, auxBufferSize);
NN_ABORT_UNLESS(result.IsSuccess());
nn::audio::SetAuxEnabled(&aux, false);
// Set the input channels for the aux effect.
int auxInputChannelCount = sizeof(subBusA);
nn::audio::SetAuxInputOutput(&aux, subBusA, subBusA, auxInputChannelCount);
// Prepare the Capture effect.
nn::audio::CaptureType capture;
// Allocate the buffer to use with the Capture effect.
// Secure a buffer for six audio frames of data (30 milliseconds). Unlike with the Aux effect, there is no delay.
// If the buffer is small, the following user process must be performed more frequently.
const int CaptureBufferFrameCount = 6;
size_t captureBufferSize = nn::audio::GetRequiredBufferSizeForCapture(&parameter, CaptureBufferFrameCount, sizeof(subBusA));
void* captureBuffer = g_EffectBufferAllocator.Allocate(captureBufferSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS_NOT_NULL(captureBuffer);
// Add the Capture effect to FinalMix.
result = nn::audio::AddCapture(&config, &capture, &finalMix, captureBuffer, captureBufferSize);
nn::audio::SetCaptureEnabled(&capture, false);
NN_ABORT_UNLESS(result.IsSuccess());
// Set the input channels for the Capture effect.
int captureInputChannelCount = sizeof(subBusA);
nn::audio::SetCaptureInput(&capture, subBusA, captureInputChannelCount);
// Prepare a BiquadFilter effect.
nn::audio::BiquadFilterType biquadFilter;
// Add the BiquadFilter effect to FinalMix.
result = nn::audio::AddBiquadFilter(&config, &biquadFilter, &finalMix);
NN_ABORT_UNLESS(result.IsSuccess());
// Set the input/output buffers for the BiquadFilter effect.
// Write back to the input source buffer.
nn::audio::SetBiquadFilterInputOutput(&biquadFilter, subBusA, subBusA, sizeof(subBusA));
// Set a 2048-Hz cutoff low-pass filter for BiquadFilter.
nn::audio::BiquadFilterParameter filterParameter = { true, {720, 1439, 720}, {22684, -9350} };
nn::audio::SetBiquadFilterParameter(&biquadFilter, filterParameter);
// Prepare a BufferMixer effect.
nn::audio::BufferMixerType mixerA;
nn::audio::BufferMixerType mixerB;
nn::audio::AddBufferMixer(&config, &mixerA, &finalMix);
nn::audio::AddBufferMixer(&config, &mixerB, &finalMix);
nn::audio::SetBufferMixerVolume(&mixerA, 0, 1.0f);
nn::audio::SetBufferMixerVolume(&mixerA, 1, 1.0f);
nn::audio::SetBufferMixerVolume(&mixerB, 0, 1.0f);
nn::audio::SetBufferMixerVolume(&mixerB, 1, 1.0f);
nn::audio::SetBufferMixerInputOutput(&mixerA, subBusA, mainBus, sizeof(mainBus));
nn::audio::SetBufferMixerInputOutput(&mixerB, subBusB, mainBus, sizeof(mainBus));
// Apply the configured parameters in the renderer.
result = nn::audio::RequestUpdateAudioRenderer(handle, &config);
NN_ABORT_UNLESS(result.IsSuccess());
// Start rendering.
result = nn::audio::StartAudioRenderer(handle);
NN_ABORT_UNLESS(result.IsSuccess());
// Prepare a memory pool to maintain the sample data added to WaveBuffer.
nn::audio::MemoryPoolType waveBufferMemoryPool;
rt = nn::audio::AcquireMemoryPool(&config, &waveBufferMemoryPool, g_WaveBufferPoolMemory, sizeof(g_WaveBufferPoolMemory));
NN_ABORT_UNLESS(rt);
rt = nn::audio::RequestAttachMemoryPool(&waveBufferMemoryPool);
NN_ABORT_UNLESS(rt);
// Load the sound effects.
nn::audio::VoiceType voiceSe[SeCount];
nn::audio::WaveBuffer waveBufferSe[SeCount];
nn::audio::AdpcmHeaderInfo* header[SeCount];
void* dataSe[SeCount];
for (int i = 0; i < SeCount; ++i)
{
header[i] = reinterpret_cast<nn::audio::AdpcmHeaderInfo*>(g_WaveBufferAllocator.Allocate(sizeof(nn::audio::AdpcmHeaderInfo), NN_ALIGNOF(nn::audio::AdpcmHeaderInfo)));
std::size_t dataSeSize = ReadAdpcmFile(header[i], &dataSe[i], g_SeFileNames[i]);
nn::audio::AcquireVoiceSlot(&config, &voiceSe[i], header[i]->sampleRate, 1, nn::audio::SampleFormat_Adpcm, nn::audio::VoiceType::PriorityHighest, &header[i]->parameter, sizeof(nn::audio::AdpcmParameter));
nn::audio::SetVoiceDestination(&config, &voiceSe[i], &finalMix);
waveBufferSe[i].buffer = dataSe[i];
waveBufferSe[i].size = dataSeSize;
waveBufferSe[i].startSampleOffset = 0;
waveBufferSe[i].endSampleOffset = header[i]->sampleCount;
waveBufferSe[i].loop = false;
waveBufferSe[i].isEndOfStream = false;
waveBufferSe[i].pContext = &header[i]->loopContext;
waveBufferSe[i].contextSize = sizeof(nn::audio::AdpcmContext);
nn::audio::AppendWaveBuffer(&voiceSe[i], &waveBufferSe[i]);
nn::audio::SetVoicePlayState(&voiceSe[i], nn::audio::VoiceType::PlayState_Play);
auto bus = (i == 0) ? subBusA : subBusB;
nn::audio::SetVoiceMixVolume(&voiceSe[i], &finalMix, 0.707f / 2, 0, bus[nn::audio::ChannelMapping_FrontLeft]);
nn::audio::SetVoiceMixVolume(&voiceSe[i], &finalMix, 0.707f / 2, 0, bus[nn::audio::ChannelMapping_FrontRight]);
}
// Prepare the buffers for getting sample data from each kind of effect
int32_t* readAuxBuffer = static_cast<int32_t*>(g_Allocator.Allocate(auxBufferSize));
NN_ABORT_UNLESS_NOT_NULL(readAuxBuffer);
memset(readAuxBuffer, 0, auxBufferSize);
int32_t* readCaptureBuffer = static_cast<int32_t*>(g_Allocator.Allocate(captureBufferSize));
NN_ABORT_UNLESS_NOT_NULL(readCaptureBuffer);
memset(readCaptureBuffer, 0, captureBufferSize);
sampleStatus.volume = 1.0f;
sampleStatus.position = 0.5f;
sampleStatus.reverbState = "default";
sampleStatus.i3dl2ReverbState = "default";
sampleStatus.delayState = "default";
sampleStatus.lightLimiterState = "default";
sampleStatus.isBiquadFilterEnabled = true;
sampleStatus.isCaptureProcessEnabled = true;
sampleStatus.isAuxProcessEnabled = true;
bool running = true;
int32_t capturePeak = 0;
PrintUsage();
PrintEffectStatus(sampleStatus);
// Enable Aux effects and Capture effects that include proprietary processing just before the main loop.
nn::audio::SetAuxEnabled(&aux, true);
nn::audio::SetCaptureEnabled(&capture, true);
while (running)
{
systemEvent.Wait();
nn::hid::NpadButtonSet npadButtonCurrent = {};
nn::hid::NpadButtonSet npadButtonDown = {};
nn::hid::AnalogStickState analogStickStateL = {};
nn::hid::AnalogStickState analogStickStateR = {};
// Get the Npad input.
if (nn::hid::GetNpadStyleSet(nn::hid::NpadId::No1).Test<nn::hid::NpadStyleFullKey>())
{
static nn::hid::NpadFullKeyState npadFullKeyState = {};
nn::hid::NpadFullKeyState state;
nn::hid::GetNpadState(&state, nn::hid::NpadId::No1);
npadButtonCurrent |= state.buttons;
npadButtonDown |= state.buttons & ~npadFullKeyState.buttons;
analogStickStateL.x += state.analogStickL.x;
analogStickStateL.y += state.analogStickL.y;
analogStickStateR.x += state.analogStickR.x;
analogStickStateR.y += state.analogStickR.y;
npadFullKeyState = state;
}
if (nn::hid::GetNpadStyleSet(nn::hid::NpadId::Handheld).Test<nn::hid::NpadStyleHandheld>())
{
static nn::hid::NpadHandheldState npadHandheldState = {};
nn::hid::NpadHandheldState state;
nn::hid::GetNpadState(&state, nn::hid::NpadId::Handheld);
npadButtonCurrent |= state.buttons;
npadButtonDown |= state.buttons & ~npadHandheldState.buttons;
analogStickStateL.x += state.analogStickL.x;
analogStickStateL.y += state.analogStickL.y;
analogStickStateR.x += state.analogStickR.x;
analogStickStateR.y += state.analogStickR.y;
npadHandheldState = state;
}
// Get the input from the DebugPad.
{
static nn::hid::DebugPadState debugPadState = {};
nn::hid::DebugPadButtonSet debugPadButtonCurrent = {};
nn::hid::DebugPadButtonSet debugPadButtonDown = {};
nn::hid::DebugPadState state;
nn::hid::GetDebugPadState(&state);
debugPadButtonCurrent |= state.buttons;
debugPadButtonDown |= state.buttons & ~debugPadState.buttons;
analogStickStateL.x += state.analogStickL.x;
analogStickStateL.y += state.analogStickL.y;
analogStickStateR.x += state.analogStickR.x;
analogStickStateR.y += state.analogStickR.y;
debugPadState = state;
nns::audio::ConvertDebugPadButtonsToNpadButtons(&npadButtonCurrent, debugPadButtonCurrent);
nns::audio::ConvertDebugPadButtonsToNpadButtons(&npadButtonDown, debugPadButtonDown);
}
bool isEffectStatusPrintNeeded = false;
// Update the various effect parameters.
{
if (!npadButtonCurrent.Test<::nn::hid::NpadButton::L>())
{
// Update the parameters for the selected effect.
if (npadButtonDown.Test<::nn::hid::NpadButton::X>())
{
switch (sampleStatus.selectedIndex)
{
case AudioEffectIndex_Reverb:
sampleStatus.reverbParameterIndex = UpdateReverbParameter(&reverb, sampleStatus.reverbParameterIndex, sampleStatus.reverbState);
break;
case AudioEffectIndex_I3dl2Reverb:
sampleStatus.i3dl2ReverbParameterIndex = UpdateI3dl2ReverbParameter(&i3dl2Reverb, sampleStatus.i3dl2ReverbParameterIndex, sampleStatus.i3dl2ReverbState);
break;
case AudioEffectIndex_Delay:
sampleStatus.delayParameterIndex = UpdateDelayParameter(&delay, sampleStatus.delayParameterIndex, sampleStatus.delayState);
break;
case AudioEffectIndex_LightLimiter:
sampleStatus.lightLimiterParameterIndex = UpdateLightLimiterParameter(&lightLimiter, sampleStatus.lightLimiterParameterIndex, sampleStatus.lightLimiterState);
break;
case AudioEffectIndex_BiquadFilter:
sampleStatus.isBiquadFilterEnabled = UpdateBiquadFilterEnabled(&biquadFilter);
break;
default:
NN_UNEXPECTED_DEFAULT;
}
isEffectStatusPrintNeeded = true;
}
// Update InputOutput for the entire effect.
if (npadButtonDown.Test<::nn::hid::NpadButton::Y>())
{
int8_t* targetBus = (sampleStatus.seSourceFlag) ? subBusA : subBusB;
UpdateEffectInputOutput(&i3dl2Reverb, &reverb, &delay, &lightLimiter, &aux, targetBus, channelCount);
sampleStatus.seSourceFlag = !sampleStatus.seSourceFlag;
isEffectStatusPrintNeeded = true;
}
// Update the selected effect.
if (npadButtonDown.Test<::nn::hid::NpadButton::Left>())
{
if (sampleStatus.selectedIndex <= 0)
{
sampleStatus.selectedIndex = static_cast<AudioEffectIndex>(AudioEffectIndex_Count - 1);
}
else
{
sampleStatus.selectedIndex = static_cast<AudioEffectIndex>(sampleStatus.selectedIndex - 1);
}
isEffectStatusPrintNeeded = true;
}
if (npadButtonDown.Test<::nn::hid::NpadButton::Right>())
{
if (sampleStatus.selectedIndex + 1 >= AudioEffectIndex_Count)
{
sampleStatus.selectedIndex = static_cast<AudioEffectIndex>(0);
}
else
{
sampleStatus.selectedIndex = static_cast<AudioEffectIndex>(sampleStatus.selectedIndex + 1);
}
isEffectStatusPrintNeeded = true;
}
}
// Update the aux effect parameters.
UpdateUserEffectParameter(npadButtonCurrent, npadButtonDown, sampleStatus.volume, sampleStatus.position, sampleStatus.isAuxProcessEnabled);
// Aux also processes user-defined effects.
if (sampleStatus.isAuxProcessEnabled)
{
ProcessUserEffect(aux, readAuxBuffer, parameter.sampleCount, channelCount, auxInputChannelCount, sampleStatus.position, sampleStatus.volume);
}
// Update the information about whether to perform Capture effect processing.
// Use this to check the number of Capture effect lost samples.
if (npadButtonCurrent.Test<::nn::hid::NpadButton::L>() && npadButtonDown.Test<::nn::hid::NpadButton::X>())
{
UpdateCaptureProcessEnabled(&sampleStatus);
}
// Perform custom-defined waveform analysis with the Capture effect.
if (sampleStatus.isCaptureProcessEnabled)
{
capturePeak = std::max(capturePeak, ProcessUserAnalyzer(capture, readCaptureBuffer, parameter.sampleCount, channelCount, captureInputChannelCount));
}
}
//Play sound effects. (The same SE is not overlaid.)
for (int i = 0; i < SeCount; ++i)
{
// SE numbers {0, 1} correspond to the buttons {A, B}.
if(npadButtonDown.Test(i))
{
if (nn::audio::GetReleasedWaveBuffer(&voiceSe[i]) != nullptr)
{
nn::audio::AppendWaveBuffer(&voiceSe[i], &waveBufferSe[i]);
}
isEffectStatusPrintNeeded = true;
}
}
// Output to log.
if (npadButtonDown.Test<::nn::hid::NpadButton::Minus>())
{
if (npadButtonCurrent.Test<::nn::hid::NpadButton::L>())
{
PrintStatisticsAndLostSampleCount(&lightLimiter, &aux, &capture, capturePeak);
// Reset the statistics for effects and the cumulative number of samples lost in Capture effect processing.
nn::audio::ResetLightLimiterStatistics(&lightLimiter);
nn::audio::ResetAuxBufferStatistics(&aux);
nn::audio::ResetCaptureLostSampleCount(&capture);
capturePeak = 0;
}
else
{
PrintUsage();
}
}
if (isEffectStatusPrintNeeded)
{
PrintEffectStatus(sampleStatus);
}
// Apply changes made to the parameters for the various effects in the audio renderer.
// The changes to the parameters are not enabled until they are applied.
// Avoid accessing any effects while nn::audio::RequestUpdateAudioRenderer() is being called.
result = nn::audio::RequestUpdateAudioRenderer(handle, &config);
NN_ABORT_UNLESS(result.IsSuccess());
// End the sample.
if(npadButtonDown.Test<::nn::hid::NpadButton::Plus>())
{
running = false;
}
}
// End rendering.
nn::audio::StopAudioRenderer(handle);
nn::audio::CloseAudioRenderer(handle);
nn::os::DestroySystemEvent(systemEvent.GetBase());
// Free memory.
for (int i = 0; i < SeCount; ++i)
{
if (dataSe[i])
{
g_WaveBufferAllocator.Free(dataSe[i]);
dataSe[i] = nullptr;
}
if (header[i])
{
g_WaveBufferAllocator.Free(header[i]);
header[i] = nullptr;
}
}
if (configBuffer)
{
g_Allocator.Free(configBuffer);
configBuffer = nullptr;
}
if (workBuffer)
{
g_Allocator.Free(workBuffer);
workBuffer = nullptr;
}
if (reverbBuffer)
{
g_EffectBufferAllocator.Free(reverbBuffer);
reverbBuffer = nullptr;
}
if (delayBuffer)
{
g_EffectBufferAllocator.Free(delayBuffer);
delayBuffer = nullptr;
}
if (lightLimiterBuffer)
{
g_EffectBufferAllocator.Free(lightLimiterBuffer);
lightLimiterBuffer = nullptr;
}
if (sendBuffer)
