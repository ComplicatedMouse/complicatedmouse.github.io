void GraphicsTextureManager::Initialize(GraphicsSystem* pGraphicsSystem, nvn::MemoryPoolFlags memPoolFlags, int numDescriptors, size_t textureMemoryAreaSize, const char* memoryPoolDebugLabel) NN_NOEXCEPT
{
// numDescriptors is the max number of application texture/sampler pairs to make available for this GraphicsTextureManager object.
if (!m_IsInitialized)
{
// create memory pool used for both texture data, as well as TexturePool and SamplerPools to hold texture/sampler descriptors
const GraphicsSystem::ResourceInfo& resourceInfo = pGraphicsSystem->GetResourceInfo();
int totalNumTextureDescriptors = numDescriptors + resourceInfo.numReservedTextureDescriptors;
int totalNumSamplerDescriptors = numDescriptors + resourceInfo.numReservedSamplerDescriptors;
NN_ASSERT(totalNumTextureDescriptors <= resourceInfo.maxTextureDescriptorsInPool);
NN_ASSERT(totalNumSamplerDescriptors <= resourceInfo.maxSamplerDescriptorsInPool);
m_TextureDescriptorAreaSize = totalNumTextureDescriptors * resourceInfo.textureDescriptorSize;
m_SamplerDescriptorAreaSize = totalNumSamplerDescriptors * resourceInfo.samplerDescriptorSize;
size_t textureMemPoolSize = m_TextureDescriptorAreaSize + m_SamplerDescriptorAreaSize + textureMemoryAreaSize;
#if defined(NN_BUILD_TARGET_PLATFORM_WIN)
// On Windows, CPU_UNCACHED is not supported for non-buffer/non-linear textures, so we clear out any such bit and
// replace it with CPU_CACHED instead. Flushing the memory pool does not appear to be necessary after loading
// texture data into the CPU_CACHED memory pool. In fact, flushing after initializing a Texture with
// SetPackagedTextureData can cause texture data corruption.
// See [NVN Programming Guide > 3.5. Memory Pools on Windows > Packaged Resource Sets] for more details.
if (memPoolFlags & nvn::MemoryPoolFlags::CPU_UNCACHED)
{
nvn::MemoryPoolFlags temp = ~nvn::MemoryPoolFlags::CPU_UNCACHED;
memPoolFlags = memPoolFlags & temp;
memPoolFlags |= nvn::MemoryPoolFlags::CPU_CACHED;
}
#endif
m_MemoryPool.Initialize(pGraphicsSystem, memPoolFlags, textureMemPoolSize, memoryPoolDebugLabel);
// allocate memory pool storage for texture/sampler descriptors and create TexturePool and SamplerPool objects
ptrdiff_t textureDescriptorsOffset = 0;
ptrdiff_t samplerDescriptorsOffset = 0;
AllocateDescriptorArea(&textureDescriptorsOffset, &samplerDescriptorsOffset);
if (!m_TexturePool.Initialize(m_MemoryPool.GetMemoryPool(), textureDescriptorsOffset, totalNumTextureDescriptors))
{
NN_ASSERT(false, "nvnTexturePoolInitialize failed.\n");
}
if (!m_SamplerPool.Initialize(m_MemoryPool.GetMemoryPool(), samplerDescriptorsOffset, totalNumSamplerDescriptors))
{
NN_ASSERT(false, "nvnSamplerPoolInitialize failed.\n");
}
// allocate set of texture and sampler buffers
m_Textures = new nvn::Texture [numDescriptors];
m_Samplers = new nvn::Sampler [numDescriptors];
m_FirstTextureId = resourceInfo.numReservedTextureDescriptors; // reserved descriptors reside at the beginning so we must skip over those
m_FirstSamplerId = resourceInfo.numReservedSamplerDescriptors;
m_NumDescriptors = numDescriptors;
m_NumTextures = 0;
m_NumSamplers = 0;
m_pGraphicsSystem = pGraphicsSystem;
m_IsInitialized = true;
}
}
void GraphicsTextureManager::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_TextureHandleMap.clear();
for (int i = 0; i < m_NumTextures; ++i)
{
m_Textures[i].Finalize();
}
for (int i = 0; i < m_NumSamplers; ++i)
{
m_Samplers[i].Finalize();
}
m_TexturePool.Finalize();
m_SamplerPool.Finalize();
m_MemoryPool.Finalize();
m_TextureDescriptorAreaSize = 0;
m_SamplerDescriptorAreaSize = 0;
m_NumDescriptors = 0;
m_NumTextures = 0;
m_NumSamplers = 0;
m_FirstTextureId = 0;
m_FirstSamplerId = 0;
m_IsInitialized = false;
}
}
void* GraphicsTextureManager::AllocateTextureMemory(ptrdiff_t* pOutMemoryPoolOffset, size_t texelDataSize) NN_NOEXCEPT
{
return AllocateTextureMemory(pOutMemoryPoolOffset, texelDataSize, DefaultAlignment);
}
void* GraphicsTextureManager::AllocateTextureMemory(ptrdiff_t* pOutMemoryPoolOffset, size_t texelDataSize, size_t alignment) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT(pOutMemoryPoolOffset != nullptr);
ptrdiff_t textureStartOffset = m_MemoryPool.Allocate(texelDataSize, alignment);
void* textureCpuAddr = m_MemoryPool.GetBaseCpuAddress() + textureStartOffset;
*pOutMemoryPoolOffset = textureStartOffset;
return textureCpuAddr;
}
int GraphicsTextureManager::AddTexture(const nvn::TextureBuilder& textureBuilder) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT(m_NumTextures < m_NumDescriptors);
// initialize new texture
int newTextureIndex = m_NumTextures;
if (!m_Textures[newTextureIndex].Initialize(&textureBuilder))
{
NN_ASSERT(false, "nvnTextureInitialize failed.\n");
}
// assign id (descriptor) for new texture
int textureId = m_FirstTextureId + newTextureIndex;
m_TexturePool.RegisterTexture(textureId, &m_Textures[newTextureIndex], nullptr); // TextureView not used
// increment counter
++m_NumTextures;
return textureId;
}
int GraphicsTextureManager::AddTexture(TextureAsset* pTextureAsset) NN_NOEXCEPT
{
// adds a texture based on a TextureAsset
NN_ASSERT(m_IsInitialized);
// Note on loading texel data from file:
// We cannot load data from file directly into a buffer that is inside of an NVN memory pool
// (nn::svc::ResultInvalidCurrentMemory occurs within nn::fs::ReadFile - checked under NintendoSDK/NX Addon 4.4.0).
// So we instead have the caller read the TextureAsset from file themselves and pass in the result.
// This method copies over the texture data in textureAsset so textureAsset can be discarded after this method returns.
// create TextureBuilder based on TextureAsset header information
nvn::TextureBuilder textureBuilder;
textureBuilder.SetDefaults()
.SetDevice(m_pGraphicsSystem->GetDevice())
.SetTarget(pTextureAsset->header.target)
.SetFormat(pTextureAsset->header.format)
.SetSize2D(pTextureAsset->header.width, pTextureAsset->header.height)
.SetDepth(pTextureAsset->header.depth)
.SetLevels(pTextureAsset->header.numMipMapLevels);
// allocate space in our memory pool for the texel data and copy the data over
ptrdiff_t textureStartOffset = 0;
size_t textureAlignment = textureBuilder.GetStorageAlignment();
void* textureCpuAddr = AllocateTextureMemory(&textureStartOffset, pTextureAsset->header.texelsDataSize, textureAlignment);
void* textureSrcAddr = reinterpret_cast<char*>(pTextureAsset) + pTextureAsset->header.texelsOffset;
memcpy(textureCpuAddr, textureSrcAddr, pTextureAsset->header.texelsDataSize);
textureBuilder.SetPackagedTextureData(textureCpuAddr); // for Windows support
textureBuilder.SetStorage(m_MemoryPool.GetMemoryPool(), textureStartOffset);
// register texture
return AddTexture(textureBuilder);
}
int GraphicsTextureManager::AddSampler(const nvn::SamplerBuilder& samplerBuilder) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT(m_NumSamplers < m_NumDescriptors);
// initialize new sampler
int newSamplerIndex = m_NumSamplers;
if (!m_Samplers[newSamplerIndex].Initialize(&samplerBuilder))
{
NN_ASSERT(false, "nvnSamplerInitialize failed.\n");
}
// assign id (descriptor) for new sampler
int samplerId = m_FirstSamplerId + newSamplerIndex;
m_SamplerPool.RegisterSampler(samplerId, &m_Samplers[newSamplerIndex]);
// increment counter
++m_NumSamplers;
return samplerId;
}
nvn::TextureHandle GraphicsTextureManager::RegisterTextureHandle(int textureId, int samplerId) NN_NOEXCEPT
{
nvn::TextureHandle handle = m_pGraphicsSystem->GetDevice()->GetTextureHandle(textureId, samplerId);
TextureHandleLookupInfo& info = m_TextureHandleMap[handle];
info.textureId = textureId;
info.samplerId = samplerId;
return handle;
}
nvn::Texture* GraphicsTextureManager::GetTexture(int textureId) NN_NOEXCEPT
{
int index = textureId - m_FirstTextureId;
NN_ASSERT(index >= 0 && index < m_NumTextures);
return &m_Textures[index];
}
nvn::Sampler* GraphicsTextureManager::GetSampler(int samplerId) NN_NOEXCEPT
{
int index = samplerId - m_FirstSamplerId;
NN_ASSERT(index >= 0 && index < m_NumSamplers);
return &m_Samplers[index];
}
void GraphicsTextureManager::LookupTextureAndSamplerIds(nvn::TextureHandle handle) NN_NOEXCEPT
{
std::map<nvn::TextureHandle, TextureHandleLookupInfo>::const_iterator iter;
iter = m_TextureHandleMap.find(handle);
if (iter == m_TextureHandleMap.end())
{
NN_ASSERT(0, "LookupTextureAndSamplerIds: nvn::TextureHandle %ld does not exist in map.", handle);
}
}
void GraphicsTextureManager::SetTextureAndSamplerPools(nvn::CommandBuffer* commandBuffer) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT(commandBuffer != nullptr);
// set our texture and sampler pools for the command buffer
commandBuffer->SetTexturePool(&m_TexturePool);
commandBuffer->SetSamplerPool(&m_SamplerPool);
}
void GraphicsTextureManager::Clear() NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
for (int i = 0; i < m_NumTextures; ++i)
{
m_Textures[i].Finalize();
m_Samplers[i].Finalize();
}
// reset the memory pool's internal pointer so that it's pointing at the start of the texture memory area
m_MemoryPool.Reset();
AllocateDescriptorArea(nullptr, nullptr); // this effectively skips over the descriptor area
m_NumTextures = 0;
m_NumSamplers = 0;
}
GraphicsMemoryPool* GraphicsTextureManager::GetMemoryPool() NN_NOEXCEPT
{
return &m_MemoryPool;
}
//-------------------------------------------------------------------------------------------------
// Private methods
//-------------------------------------------------------------------------------------------------
void GraphicsTextureManager::AllocateDescriptorArea(ptrdiff_t* pOutTextureDescriptorsOffset, ptrdiff_t* pOutSamplerDescriptorsOffset) NN_NOEXCEPT
{
NN_ASSERT(m_TextureDescriptorAreaSize > 0);
NN_ASSERT(m_SamplerDescriptorAreaSize > 0);
ptrdiff_t textureOffset = m_MemoryPool.Allocate(m_TextureDescriptorAreaSize);
ptrdiff_t samplerOffset = m_MemoryPool.Allocate(m_SamplerDescriptorAreaSize);
if (pOutTextureDescriptorsOffset != nullptr)
{
*pOutTextureDescriptorsOffset = textureOffset;
}
if (pOutSamplerDescriptorsOffset != nullptr)
{
*pOutSamplerDescriptorsOffset = samplerOffset;
}
