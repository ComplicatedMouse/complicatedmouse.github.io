inline void LoadMatrix44(void* pMatrix4Dst, const nn::util::Matrix4x4fType& matrixSrc) NN_NOEXCEPT
{
nn::util::Float4x4 mtxStorage;
nn::util::MatrixStore(&mtxStorage, matrixSrc);
memcpy(pMatrix4Dst, &mtxStorage, sizeof(nn::util::Float4x4));
}
inline void LoadMatrix43(void* pMatrix4Dst, const nn::util::Matrix4x3fType& matrixSrc) NN_NOEXCEPT
{
nn::util::Matrix4x4fType temp;
nn::util::MatrixConvert(&temp, matrixSrc);
LoadMatrix44(pMatrix4Dst, temp);
}
