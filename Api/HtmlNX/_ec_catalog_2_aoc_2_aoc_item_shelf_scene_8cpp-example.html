DisplayKeyValueListLanguageSet::DisplayKeyValueListLanguageSet() NN_NOEXCEPT : m_Ja("ja"), m_enUS("en-US")
{
// ja
{
auto& keyArrayForJa = *(m_Ja.GetKeys());
keyArrayForJa[0] = AocListCorrespondenceTable(u8"equipment", u8"装備の種類から選択");
keyArrayForJa[1] = AocListCorrespondenceTable(u8"available_time", u8"ランクから選択");
auto& valueArrayForJa = *(m_Ja.GetValues());
NN_ASSERT(valueArrayForJa.size() <= std::extent<decltype(AocItemTagStrings_ja)>::value);
for (int i = 0; i < valueArrayForJa.size(); ++i)
{
const auto& preset = AocItemTagStrings_ja[i];
valueArrayForJa[i] = AocListCorrespondenceTable(preset.pNameForKey, preset.pNameForValue, preset.pNameForValueShelf);
}
}
// en-US
{
auto& keyArrayForEnUs = *(m_enUS.GetKeys());
keyArrayForEnUs[0] = AocListCorrespondenceTable("equipment", "equipment");
keyArrayForEnUs[1] = AocListCorrespondenceTable("available_time", "available time");
auto& valueArrayForEnUs = *(m_enUS.GetValues());
NN_ASSERT(valueArrayForEnUs.size() <= std::extent<decltype(AocItemTagStrings_enUS)>::value);
for (int i = 0; i < valueArrayForEnUs.size(); ++i)
{
const auto& preset = AocItemTagStrings_enUS[i];
valueArrayForEnUs[i] = AocListCorrespondenceTable(preset.pNameForKey, preset.pNameForValue, preset.pNameForValueShelf);
}
}
}

// Screen rendering.
void AocItemShelfScene::Draw() NN_NOEXCEPT
{
switch (m_AocItemShelfStatus)
{
case AocItemShelfStatus_None:
break;
case AocItemShelfStatus_AocItemsInitialize:
case AocItemShelfStatus_AocItemsResponseWait:
ImGui::SetWindowFontScale(1.8f);
ImGui::SetCursorPos(ImVec2(550, 675));
ImGui::TextColored(White, "Now Loading...");
break;
case AocItemShelfStatus_KeyListStart:
case AocItemShelfStatus_ValueListStart:
{
// Display the selection screen.
m_Select.DrawSelectionList(DrawSelectListCallBack, this);
// Note: There is no way to ensure that the language of the response is one of the system languages.
// Do not treat this condition as an error. Continue to display product information as is.
nn::settings::LanguageCode lang = nn::oe::GetDesiredLanguage();
if (::nn::util::Strncmp(lang.string, m_RequestAocItems.GetResponse().GetLang().c_str(), sizeof(lang.string)) != 0)
{
// Notation displays (translation):
// Product information cannot be displayed using the languages configured for this application. Display using an available language.
ImGui::SetWindowFontScale(1.1f);
ImGui::SetCursorPos(ImVec2(10, 570));
ImGui::TextColored(Red, "Product information cannot be displayed in this language currently set for this application. The supported language will be displayed.");
}
// Description of operations.
ImGui::SetWindowFontScale(1.8f);
ImGui::SetCursorPos(ImVec2(10, 600));
ImGui::PushStyleColor(ImGuiCol_Text, White);
if (m_AocItemShelfStatus == AocItemShelfStatus_KeyListStart)
{
ImGui::TextWrapped("Selection of major item list.\nSelect an item by up or down button and Push A button to retrieve each item.\nPush B button to go back to the start scene.");
}
else if (m_AocItemShelfStatus == AocItemShelfStatus_ValueListStart)
{
ImGui::TextWrapped("Selection of middle item list.\nSelect an item by up or down button and Push A button to jump to item shelf.\nPush B button to go back to the major item list.");
}
ImGui::PopStyleColor();
break;
}
default:
NN_LOG("Status that does not exist.\n");
break;
}
}
// Create a list from the ordered list that will be displayed on the screen and initialize the choices.
bool AocItemShelfScene::CreateKeyList() NN_NOEXCEPT
{
// Get the language.
const std::string lang = m_RequestAocItems.GetResponse().GetLang();
// Check for the presence of items in the ordered list that will be displayed on the screen.
for (int i = 0; i < m_TagList.size(); i++)
{
const auto pFoundItem = m_DisplayKeyValueListLanguageSet.Find(lang.c_str())->GetKeys()->FindTagName(m_TagList[i].GetTag()->GetKey());
if (nullptr != pFoundItem)
{
pFoundItem->SetIsPresent(true);
}
}
// Create the key list to actually display on the screen.
m_DisplayListSize = 0;
m_DisplayKeyValueListLanguageSet.Find(lang.c_str())->GetKeys()->GetDisplayNameList(m_DisplayList, &m_DisplayListSize);
if (m_DisplayListSize < 1)
{
return false;
}
// The initial settings for the choices.
int listMax = m_DisplayListSize; // The number of items to choose among.
int row = 7; // The number of rows to display for the selection.
int offsetX = 200; // The amount to shift the selection horizontally.
int offsetY = 70; // The amount to shift the selection vertically.
int positionX = 50; // The coordinate for the left position of the selection.
int positionY = 50; // The coordinate for the upper position of the selection.
float cursorSize = 2.5; // The size of the cursor.
int cursorOffsetX = -20; // Fine tuning the x-position of the cursor.
int cursorOffsetY = 0; // Fine tuning the y-position of the cursor.
m_Select.SetInitialValue(
listMax
, row
, offsetX
, offsetY
, positionX
, positionY
, pSceneData->GetSelectedPositionAocKey() // The initial position of the cursor.
, cursorSize
, cursorOffsetX
, cursorOffsetY
);
return true;
}
// Create a list in ascending order of the key, value elements from the AocItems data.
void AocItemShelfScene::CreateTagList() NN_NOEXCEPT
{
for (int i = 0; i < m_RequestAocItems.GetResponse().GetAocItems().size(); i++)
{
// Create a tagList from AOC items in ascending order of the key, value elements. Also store the array numbers for AOC items with duplicate key, value elements.
for (int j = 0; j < m_RequestAocItems.GetResponse().GetAocItems()[i].GetTagCollection().GetItemTags().size(); j++)
{
ItemTag itemTag = m_RequestAocItems.GetResponse().GetAocItems()[i].GetTagCollection().GetItemTags()[j];
// Create item tag data in ascending order of the key, value elements.
bool isExist = false;
for (auto itr = m_TagList.begin(); itr != m_TagList.end(); ++itr)
{
if (itr->GetTag()->GetKey().compare(itemTag.GetKey()) == 0)
{
if (itr->GetTag()->GetValue().compare(itemTag.GetValue()) == 0)
{
// If the key, value elements are the same, store the array number for aocItem.
itr->PushBackAocItemIndex(i);
isExist = true;
break;
}
else if (itr->GetTag()->GetValue().compare(itemTag.GetValue()) > 0)
{
// If the same key, value elements are not found, add in ascending order of key, value.
ItemTag tag(itemTag.GetKey(), itemTag.GetValue());
AocItemIndexCorrespondingToTag aocTag(tag);
aocTag.PushBackAocItemIndex(i);
m_TagList.insert(itr, aocTag);
isExist = true;
break;
}
}
else if (itr->GetTag()->GetKey().compare(itemTag.GetKey()) > 0)
{
// If the same key is not found, add in ascending order of the key elements.
ItemTag tag(itemTag.GetKey(), itemTag.GetValue());
AocItemIndexCorrespondingToTag aocTag(tag);
aocTag.PushBackAocItemIndex(i);
m_TagList.insert(itr, aocTag);
isExist = true;
break;
}
}
if (false == isExist)
{
// Add to the end.
ItemTag tag(itemTag.GetKey(), itemTag.GetValue());
AocItemIndexCorrespondingToTag aocTag(tag);
aocTag.PushBackAocItemIndex(i);
m_TagList.push_back(aocTag);
}
}
}

// Input management processing.
void AocItemShelfScene::Control(const nns::hid::GamePad& gamePad) NN_NOEXCEPT
{
// B Button for canceling loading.
bool isCancelButtonPushed = gamePad.HasAnyButtonsUp(nns::hid::Button::B::Mask);
switch (m_AocItemShelfStatus)
{
case AocItemShelfStatus_AocItemsInitialize:
{
const auto isSuccess = m_RequestAocItems.SetupRequest(pSceneData->GetUserContext()->id, SceneData::SalesStatus_OnsaleOnly);
if (isSuccess == false)
{
// Return to the previous screen when an error occurs.
NN_LOG("Request or initialization failed.\n");
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
m_AocItemShelfStatus = AocItemShelfStatus_AocItemsResponseWait;
}
break;
}
case AocItemShelfStatus_AocItemsResponseWait:
if (!m_RequestAocItems.IsCompleted())
{
if (isCancelButtonPushed == true)
{
// Canceling.
m_RequestAocItems.Cancel();
}
}
else
{
try
{
if (m_RequestAocItems.WaitFinish(pSceneData->GetUserContext()->userHandle) == false)
{
//If an error or cancellation occurs.
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
if (m_RequestAocItems.ConvertResponse() == false)
{
// Display an error when the parse fails.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
}
else
{
// Confirm that length matches the number of items.
if (m_RequestAocItems.GetResponse().GetLength() != m_RequestAocItems.GetResponse().GetAocItems().size())
{
// length does not match the number of items.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_ItemCountDifferent);
SceneBase::SetStatus(DisplayStatus_Error);
NN_LOG("AOC Items length error.\n");
break;
}
// Create a list in ascending order of the key, value elements from the AocItems data.
CreateTagList();
// Create a list of key elements to display on the screen.
if (false == CreateKeyList())
{
// Display an error if there are no items to display.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_NoneItem);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
// Confirm that length matches the number of items.
if (m_RequestAocItems.GetResponse().GetLength() != m_RequestAocItems.GetResponse().GetAocItems().size())
{
// length does not match the number of items.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_ItemCountDifferent);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
// Processing for informing the main screen that loading completed and deleting the previous screen.
pSceneData->SetLoadCompleted(true);
m_AocItemShelfStatus = AocItemShelfStatus_KeyListStart;
}
}
}
catch (std::bad_alloc& e)
{
// Processes cancellation when memory allocation fails.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
catch (std::exception& e)
{
// Although only bad_alloc is expected in the sample, this processing is used when other exceptions are thrown.
// Process cancellation.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
}
break;
case AocItemShelfStatus_KeyListStart:
{
int selected;
Select::SelectStatus selectStatus;
// Execute the selection process.
m_Select.InputSelectionList(&selectStatus, &selected, gamePad);
pSceneData->SetSelectedPositionAocKey(selected);
if (selectStatus == Select::SelectStatus_Cancel)
{
// If the process has been canceled.
SceneBase::SetStatus(DisplayStatus_Start);
}
if (selectStatus == Select::SelectStatus_Decision)
{
// Get the name of the selected item.
std::string displayName = m_DisplayList[selected];
// Get the language.
const std::string lang = m_RequestAocItems.GetResponse().GetLang();
// Get the list in that language.
auto pDisplayKeyValueList = m_DisplayKeyValueListLanguageSet.Find(lang.c_str());
// Initialize the existence of Value items.
pDisplayKeyValueList->GetValues()->InvalidateIndexes();
// Get the item corresponding to the selected item name.
auto pFoundItem = pDisplayKeyValueList->GetKeys()->FindDisplayName(displayName);
if (nullptr != pFoundItem)
{
// Get the key.
m_SelectedKey = pFoundItem->GetResponseName();
}
else
{
// Unexpected behavior, so return to the start screen.
SceneBase::SetStatus(DisplayStatus_Start);
break;
}
// Create a list of value elements corresponding to the key elements to actually display on the screen.
int keyIndex = 0;
// Find the beginning of the selected key.
for (int i = 0; i < m_TagList.size(); i++)
{
if (m_TagList[i].GetTag()->GetKey().compare(m_SelectedKey) == 0)
{
keyIndex = i;
break;
}
}
std::string valueName;
for (int i = keyIndex; i < m_TagList.size(); i++)
{
// If it is a different key, end the extraction of value.
if (m_TagList[i].GetTag()->GetKey().compare(m_SelectedKey) != 0)
{
break;
}
valueName = m_TagList[i].GetTag()->GetValue();
// Select the check box for items that exist.
const auto pFoundItem = m_DisplayKeyValueListLanguageSet.Find(lang.c_str())->GetValues()->FindTagName(valueName);
if (nullptr != pFoundItem)
{
pFoundItem->SetIsPresent(true);
}
}
// Create the list of value elements to actually display on the screen.
m_DisplayListSize = 0;
m_DisplayKeyValueListLanguageSet.Find(lang.c_str())->GetValues()->GetDisplayNameList(m_DisplayList, &m_DisplayListSize);
if (m_DisplayListSize < 1)
{
// Unexpected behavior, so return to the start screen.
SceneBase::SetStatus(DisplayStatus_Start);
break;
}
// The initial settings for the choices.
int listMax = m_DisplayListSize; // The number of items to choose among.
int row = 7; // The number of rows to display for the selection.
int offsetX = 200; // The amount to shift the selection horizontally.
int offsetY = 70; // The amount to shift the selection vertically.
int positionX = 50; // The coordinate for the left position of the selection.
int positionY = 50; // The coordinate for the upper position of the selection.
float cursorSize = 2.5; // The size of the cursor.
int cursorOffsetX = -20; // Fine tuning the x-position of the cursor.
int cursorOffsetY = 0; // Fine tuning the y-position of the cursor.
m_Select.SetInitialValue(
listMax
, row
, offsetX
, offsetY
, positionX
, positionY
, pSceneData->GetSelectedPositionAocValue() // The initial position of the cursor.
, cursorSize
, cursorOffsetX
, cursorOffsetY
);
// Transition the screen to the Value list.
m_AocItemShelfStatus = AocItemShelfStatus_ValueListStart;
}
break;
}
case AocItemShelfStatus_ValueListStart:
{
int selected;
Select::SelectStatus selectStatus;
// Execute the selection process.
m_Select.InputSelectionList(&selectStatus, &selected, gamePad);
pSceneData->SetSelectedPositionAocValue(selected);
if (selectStatus == Select::SelectStatus_Cancel)
{
// If the process has been canceled:
// Create a list of key elements to display on the screen.
CreateKeyList();
// Transition to the key list screen.
m_AocItemShelfStatus = AocItemShelfStatus_KeyListStart;
}
if (selectStatus == Select::SelectStatus_Decision)
{
// Get the displayed characters from the item name that was selected.
std::string displayName = m_DisplayList[selected];
const std::string lang = m_RequestAocItems.GetResponse().GetLang();
auto pFoundKeyItem = m_DisplayKeyValueListLanguageSet.Find(lang.c_str())->GetValues()->FindDisplayName(displayName);
std::string valueString;
std::string shelfName;
if (pFoundKeyItem != nullptr)
{
valueString = pFoundKeyItem->GetResponseName();
shelfName = pFoundKeyItem->GetShelfName();
}
else
{
// Unexpected behavior, so return to the start screen.
SceneBase::SetStatus(DisplayStatus_Start);
break;
}
// Get the list of hit-on key, value elements to actually display on the screen.
// Find the selected key, value elements.
int keyIndex = std::numeric_limits<int>::max();
for (int i = 0; i < m_TagList.size(); i++)
{
if (m_TagList[i].GetTag()->GetKey().compare(m_SelectedKey) == 0
&& (m_TagList[i].GetTag()->GetValue().compare(valueString) == 0))
{
keyIndex = i;
break;
}
}
if (keyIndex != std::numeric_limits<int>::max())
{
nn::ec::NsUid nsUidList[SceneData::NsUidListMax];
int i = 0;
for (i = 0; i < m_TagList[keyIndex].GetAocItemIndexs().size() && i < SceneData::NsUidListMax; i++)
{
int itemIndex = m_TagList[keyIndex].GetAocItemIndexs()[i];
nsUidList[i].value = m_RequestAocItems.GetResponse().GetAocItems()[itemIndex].GetNsUid().value;
}
if (i > 0)
{
// Specification of the shelf name (the listName to specify in ec::ShowShopProductList).
pSceneData->SetAocShelfList(nsUidList, i, shelfName);
SceneBase::SetStatus(DisplayStatus_AocItemShelfShop);
}
else
{
// Unexpected behavior, so return to the start screen.
SceneBase::SetStatus(DisplayStatus_Start);
break;
}
}
}
break;
}
default:
break;
}
