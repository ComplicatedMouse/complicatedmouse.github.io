#include <stdint.h>
#include <nn/nn_Macro.h>
#include <nn/nn_Log.h>
#include <nn/nn_Abort.h>
#include <nn/os.h>
//-----------------------------------------------------------------------------
namespace {
const size_t ThreadStackSize = 8192; // Stack size of the message queue operation thread.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack1[ ThreadStackSize ]; // The first thread stack.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack2[ ThreadStackSize ]; // The second thread stack.
const size_t BufferSize = 3; // Message queue size.
uintptr_t g_messageBuffer[ BufferSize ]; // Message queue buffer.
nn::os::MessageQueueType g_MessageQueue; // Message queue.
} // Namespace.
//-----------------------------------------------------------------------------
//
// Wait + string transmission.
//
inline void SendString(const char* message)
{
nn::os::SendMessageQueue( &g_MessageQueue, reinterpret_cast<uintptr_t>( message ) );
}
//
// First thread (side sending to the message queue).
//
void ThreadFunction1(void *arg)
{
NN_UNUSED(arg);
SendString("Hello!\n");
SendString("This is sample program for MessageQueue OS-API.\n");
SendString("The MessageQueue can send and receive a data of the type uintptr_t,\n");
SendString("so it's suitable to send and receive an address value.\n");
// Endmark.
nn::os::SendMessageQueue( &g_MessageQueue, 0x00 );
}
//
// Second thread (side receiving from the message queue).
//
void ThreadFunction2(void *arg)
{
NN_UNUSED(arg);
uintptr_t data;
// Generate characters until the endmark appears.
for (;;)
{
nn::os::ReceiveMessageQueue( &data, &g_MessageQueue );
if (data == 0x00)
{
return;
}
NN_LOG( reinterpret_cast<char*>( data ) );
}
}
//
// The main function.
//
extern "C" void nnMain()
{
nn::os::ThreadType thread1;
nn::os::ThreadType thread2;
nn::Result result;
// Initialize the message queue.
nn::os::InitializeMessageQueue( &g_MessageQueue, g_messageBuffer, BufferSize );
// Generate threads.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread1, ThreadFunction1, NULL, g_ThreadStack1, ThreadStackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread2, ThreadFunction2, NULL, g_ThreadStack2, ThreadStackSize, nn::os::DefaultThreadPriority ) );
// Start thread execution.
nn::os::StartThread( &thread1 );
nn::os::StartThread( &thread2 );
// Wait for the thread to end.
nn::os::WaitThread( &thread1 );
nn::os::WaitThread( &thread2 );
// Free threads.
nn::os::DestroyThread( &thread1 );
nn::os::DestroyThread( &thread2 );
// Finalize the message queue.
nn::os::FinalizeMessageQueue( &g_MessageQueue );
}
