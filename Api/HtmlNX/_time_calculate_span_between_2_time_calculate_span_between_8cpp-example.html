#include <nn/time.h>
#include <nn/nn_Log.h>
#include <nn/nn_Abort.h>
#include <nn/os/os_ThreadApi.h>
extern "C" void nnMain()
{
// Initialize the time library.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::time::Initialize() );
NN_LOG("nn::time::Initialize\n");
// Run the sample that measures the elapsed time.
NN_LOG("---- Sample of getting nn::TimeSpan from two ClockSnapshot's.\n");
{
// Get the ClockSnapshot for the starting point.
nn::time::ClockSnapshot snapshot1;
nn::time::ClockSnapshot::CreateWithStandardSystemClock(&snapshot1);
// If you save the obtained ClockSnapshot value in a location such as save data such that it is retained permanently, you can load that value
// on the next occasion to calculate the elapsed time spanning events like shutting down the application and turning off the system.
NN_LOG("Waiting three seconds.\n");
nn::os::SleepThread(nn::TimeSpan::FromSeconds(3));
// Get the ClockSnapshot for the end point.
nn::time::ClockSnapshot snapshot2;
nn::time::ClockSnapshot::CreateWithStandardSystemClock(&snapshot2);
// Calculate the elapsed time from the two ClockSnapshot.
nn::TimeSpan elapsed;
bool isSucceeded = nn::time::CalculateSpanBetween(
&elapsed,
snapshot1,
snapshot2);
if(isSucceeded)
{
// Comes to here if the elapsed time has been calculated successfully.
NN_LOG("nn::time::CalculateSpanBetween succeeded. %lld seconds elapsed.\n", elapsed.GetSeconds());
// By determining whether this calculated elapsed time exceeds a specified
// amount of time, you can code actions like generating an event again
// if more than x amount of time has passed since the last event.
// The time provided by nn::time is not immune to hardware spoofing attacks.
// In addition, nn::time::CalculateSpanBetween() may be affected by network clocks without assuring a monotonic increase,
// and the elapsed time may become a negative value.
// Implement your application such that it does not stop when this occurs.
// When the elapsed time becomes a negative value, you can wait for actual time to pass without performing any actions so that the time eventually becomes positive, in cases where the negative value is small.
// Alternatively, you can also use methods such as making the latest ClockSnapshot persistent, and retry the default elapsed time.
}
else
{
// This function fails to calculate the elapsed time and returns false if all of the following conditions are satisfied.
// - The monotonically increasing clock values that store the start time and the end time are discontinuous.
// - Either the start time or the end time does not store a network clock time.
NN_LOG("nn::time::CalculateSpanBetween failed.\n");
// As normal behavior, exercise care when the nn::time::CalculateSpanBetween() function fails.
//
// For example, to prevent the nn::time::CalculateSpanBetween() function from continually failing
// when perpetuating the ClockSnapshot starting point, you must do something like
// get the latest ClockSnapshot and then perpetuate it
// so that you can use that value the next time the application starts.
}
}
// Finalize the time library.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::time::Finalize() );
NN_LOG("nn::time::Finalize\n");
}
