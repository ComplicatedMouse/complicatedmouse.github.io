{
"asset:/MultiProgramApplicationMain/SampleBgm0-2ch.wav",
};
const std::string g_ProgramName = "Main";
void MainLoop(const AccountManager::UserInfo* pUserInfo) NN_NOEXCEPT
{
g_AudioAllocator.Initialize(g_AudioWorkBuffer, sizeof(g_AudioWorkBuffer));
g_WaveBufferAllocator.Initialize(g_WaveBufferPoolMemory, sizeof(g_WaveBufferPoolMemory));
nn::fs::SetAllocator(FsAllocate, FsDeallocate);
ApplicationHeap applicationHeap;
GraphicsSystem graphicsSystem;
FontSystem fontSystem;
// Create a class to display and manipulate screens.
UserInterface scene = UserInterface(pUserInfo, g_ProgramName);
// Use the block scope because the AudioRenderer destructor is called before switching the programIndex.
{
AudioRenderer audioRenderer = AudioRenderer(
&g_AudioAllocator,
&g_WaveBufferAllocator,
g_WaveBufferPoolMemory,
sizeof(g_WaveBufferPoolMemory),
g_BgmFileNames
);
audioRenderer.AudioParamSetting();
nv::SetGraphicsAllocator(NvAllocate, NvFree, NvReallocate, NULL);
nv::SetGraphicsDevtoolsAllocator(NvAllocate, NvFree, NvReallocate, NULL);
nv::InitializeGraphics(aligned_alloc(4096, GraphicsMemorySize), GraphicsMemorySize);
nns::hid::ControllerManager controllerManager;
nns::hid::util::SetControllerManagerWithDefault(&controllerManager);
applicationHeap.Initialize(ApplicationHeapSize);
graphicsSystem.Initialize(
&applicationHeap, FrameBufferWidth, FrameBufferHeight);
fontSystem.Initialize(&applicationHeap, &graphicsSystem);
nn::gfx::util::DebugFontTextWriter& textWriter = fontSystem.GetDebugFontTextWriter();
while (NN_STATIC_CONDITION(true))
{
audioRenderer.AudioMain();
// Update input.
controllerManager.Update();
nns::hid::Controller* pInput = controllerManager.GetController(nns::hid::ControllerId_GamePad, 0);
// Input management processing.
scene.Input(reinterpret_cast<nns::hid::GamePad*>(pInput));
// Get the screen change information.
if (scene.GetStatus() == UserInterface::MainStatus::MainStatus_Jump)
{
// Exit the main loop, using MainStatus::MainStatus_Jump, with a jump after displaying the mid-jump appearance
// with MainStatus::MainStatus_JumpRequested.
break;
}
// Screen output content.
scene.Draw(&textWriter);
graphicsSystem.BeginDraw();
fontSystem.Draw();
graphicsSystem.EndDraw();
graphicsSystem.Synchronize(
nn::TimeSpan::FromNanoSeconds(1000 * 1000 * 1000 / FrameRate));
}
}
// The AudioRenderer destructor is called, and audio is stopped.
// Sound does not play before switching the programIndex. Failure to do so cuts off sound during playback.
if (scene.GetStatus() == UserInterface::MainStatus::MainStatus_Jump)
{
auto indexNum = scene.GetIndexNumber();
auto launchParam = scene.GetInputLaunchParam();
// Run PushOpenUsers() and jump to another program.
ExecuteProgram(indexNum, &launchParam);
}
// It is not called after running nn::oe::ExecuteProgram(), but all resources for the program, prior to running, are released.
fontSystem.Finalize();
graphicsSystem.Finalize();
applicationHeap.Finalize();
}//NOLINT(impl/function_size)
} // Namespace.
extern "C" void nnMain()
{
nn::account::Initialize();
AccountManager::UserInfo userInfo;
// Get users who were opened for the previous program, using account pop processing.
if (AccountManager::GetInstance().PopOpenUsers() == true)
{
AccountManager::UserList userList;
userList = AccountManager::GetInstance().GetOpenUserList();
for (auto&& info : userList.userInfo)
{
// User ID exists:
if (info.IsExist() == true)
{
// Maintain open user data.
// Only repeated once, because only one user is open.
userInfo = info;
break;
}
}
}
else
{
// Select the user.
if (!AccountManager::GetInstance().OpenUserWithSelectorApplet(&userInfo))
{
// Quit when no user is selected.
NN_LOG("Because the user was not selected, it ends.");
return;
}
}
MainLoop(&userInfo);
// User closing.
AccountManager::GetInstance().CloseUser(userInfo.id);
return;
}
