void DynamicResolution::Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
int64_t shaderBinFileSize = 0;
// read shader binary, create GraphicsShader object from it, and get uniform block reflection info
m_GlslcData = ReadFile(&shaderBinFileSize, "rom:/NvnSimple07.glslc", m_pGraphicsSystem->GetAllocator());
m_Shader.Initialize(m_pGraphicsSystem, static_cast<GLSLCoutput*>(m_GlslcData));
m_pModelDataUniformBlockInfo = &m_Shader.LookupUniformBlockInfo("VertexShaderUniformBlock");
m_pLightDataUniformBlockInfo = &m_Shader.LookupUniformBlockInfo("LightDataUniformBlock");
// create memory pool for vertex data
size_t attributeDataSize = sizeof(g_VertexPositions) + sizeof(g_VertexNormals) + sizeof(g_VertexUvs);
size_t indexDataSize = sizeof(g_VertexIndices) * 6;
size_t uniformAlignment = m_pGraphicsSystem->GetResourceInfo().uniformBufferAlignment;
size_t alignedModelDataUniformBlockSize = nn::util::align_up(m_pModelDataUniformBlockInfo->GetSize(), uniformAlignment);
size_t alignedLightDataUniformBlockSize = nn::util::align_up(m_pLightDataUniformBlockInfo->GetSize(), uniformAlignment);
size_t memPoolSize = attributeDataSize + indexDataSize + alignedModelDataUniformBlockSize * CubeCount + alignedLightDataUniformBlockSize;
m_VertexAndUniformDataMemPool.Initialize(m_pGraphicsSystem, nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED, memPoolSize, "DynamicResolution MemPool");
// allocate blocks of memory for vertex and uniform data
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_VertexDataBuffer, attributeDataSize);
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_IndexDataBuffer, indexDataSize);
for (int i = 0; i < CubeCount; i++)
{
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_ModelDataBuffer[i], alignedModelDataUniformBlockSize, uniformAlignment);
}
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_LightDataBuffer, alignedLightDataUniformBlockSize, uniformAlignment);
// load vertex and index data into memory pool
const int numFaces = 6;
const int numVertices = 4 * numFaces;
Vertex* vertices = static_cast<Vertex*>(m_VertexDataBuffer.Map());
// we use a single vertex buffer with interleaved attributes
for (int i = 0; i < numVertices; ++i)
{
Vertex* pVertex = vertices + i;
float x = g_VertexPositions[3 * i];
float y = g_VertexPositions[3 * i + 1];
float z = g_VertexPositions[3 * i + 2];
float normalX = g_VertexNormals[3 * i];
float normalY = g_VertexNormals[3 * i + 1];
float normalZ = g_VertexNormals[3 * i + 2];
float u = g_VertexUvs[2 * i];
float v = g_VertexUvs[2 * i + 1];
pVertex->m_Pos.Set(x, y, z);
pVertex->m_Normal.Set(normalX, normalY, normalZ);
pVertex->m_Uv.Set(u, v);
}
// indices
m_NumIndices = 6 * numFaces;
uint32_t* indices = static_cast<uint32_t*>(m_IndexDataBuffer.Map());
for (uint32_t i = 0; i < m_NumIndices; ++i)
{
uint32_t delta = 4 * (i / 6);
uint32_t index = i % 6;
indices[i] = g_VertexIndices[index] + delta;
}
// setup NVN VertexStreamState and VertexAttribState objects based on vertex attribute names and shader
// the ordering of the attribute names must logically match up with the attribute ordering in our vertex data structure (see above)
const char* attributeNames[NumVertexAttributes] = { "attr_position", "attr_normal", "attr_uv" };
// we use an interleaved vertex buffer layout for this demo
SetupVertexAttribAndStreamStateInterleaved(attributeNames, NumVertexAttributes, m_Shader, m_VertexAttribStates, &m_VertexStreamState);
// projection transform
nn::util::MatrixPerspectiveFieldOfViewRightHanded(&m_ProjMtx, nn::util::FloatPi / 3.0f, 16.0f / 9.0f, 0.1f, 1000.0f);
// view transform
nn::util::Vector3f cameraPos(0, 0, 28);
nn::util::Vector3f targetPos(0, 0, 0);
nn::util::Vector3f up(0, 1, 0);
nn::util::MatrixLookAtRightHanded(&m_ViewMtx, cameraPos, targetPos, up);
// The modelView transform will update per frame, so we only load projection transform here.
const UniformBlockMemberInfo& projMtxInfo = m_pModelDataUniformBlockInfo->LookupMemberInfo("proj");
for (int i = 0; i < CubeCount; i++)
{
// compute and load uniform data into memory pool
char* baseUniformBlockAddr = static_cast<char*>(m_ModelDataBuffer[i].Map());
char* projMtxAddr = baseUniformBlockAddr + projMtxInfo.GetOffset();
LoadMatrix44(projMtxAddr, m_ProjMtx);
}
// setup uniform data for lights
const UniformBlockMemberInfo& lightNumInfo = m_pLightDataUniformBlockInfo->LookupMemberInfo("lightNum");
const UniformBlockMemberInfo& lightColorInfo = m_pLightDataUniformBlockInfo->LookupMemberInfo("lightColorArray[0]");
const UniformBlockMemberInfo& lightPosInfo = m_pLightDataUniformBlockInfo->LookupMemberInfo("lightPosArray[0]");
char* baseUniformBlockAddr = static_cast<char*>(m_LightDataBuffer.Map());
m_LightColorAddr = baseUniformBlockAddr + lightColorInfo.GetOffset();
m_LightPosAddr = baseUniformBlockAddr + lightPosInfo.GetOffset();
m_LightNumAddr = baseUniformBlockAddr + lightNumInfo.GetOffset();
for (int i = 0; i < LightCount; i++)
{
int z = i / (RowLength * ColumnLength);
int y = (i - z * RowLength * ColumnLength) / RowLength;
int x = i % RowLength;
// light color
int colorIndex = (x + z) % 5;
Color lightColor = g_LightColorTable[colorIndex];
m_LightColorArray[i] = lightColor;
// light position
nn::util::Vector3f lightPos = nn::util::Vector3f((x - 2.0f) * 4.5f, (y - 1.5f) * 3.5f, (DepthLength - z - 1) * 5.0f + 2.0f);
// compute light transform and compute view space light position for lighting calculation in fragment shader
nn::util::Matrix4x3fType translateMtx;
nn::util::Vector3f translate(lightPos);
nn::util::MatrixIdentity(&translateMtx);
nn::util::MatrixSetTranslate(&translateMtx, translate);
nn::util::Matrix4x3fType lightViewMtx;
nn::util::MatrixMultiply(&lightViewMtx, translateMtx, m_ViewMtx);
nn::util::Vector3f lightViewPos;
nn::util::MatrixGetAxisW(&lightViewPos, lightViewMtx);
m_LightPosArray[i] = nn::util::Vector4f(lightViewPos, 0.0f);
}
// lights number start at 0
m_ActivedLightNumber = 0;
// load uniform data into memory pool
memcpy(m_LightNumAddr, &m_ActivedLightNumber, sizeof(uint32_t));
memcpy(m_LightPosAddr, m_LightPosArray, lightPosInfo.GetSize() * LightCount);
memcpy(m_LightColorAddr, m_LightColorArray, lightColorInfo.GetSize() * LightCount);
// load texture files
const int NumTextureAssets = 2;
const char* textureFiles[NumTextureAssets] = { "rom:/CubeTexture.tex", "rom:/BrickTexture.tex" };
TextureAsset* textureAssets[NumTextureAssets];
int totalSize = 0; // calculate total size required to store the textures to be loaded
for (int i = 0; i < NumTextureAssets; ++i)
{
textureAssets[i] = ReadTexture(textureFiles[i], m_pGraphicsSystem->GetAllocator());
totalSize += textureAssets[i]->header.texelsDataSize;
}
// create texture manager instance
m_TextureManager.Initialize(m_pGraphicsSystem, nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED,
32, totalSize, "DynamicResolution Texture Pool");
// initialize nvn::Textures from texture asset files
int textureIds[NumTextureAssets];
for (int i = 0; i < NumTextureAssets; ++i)
{
textureIds[i] = m_TextureManager.AddTexture(textureAssets[i]);
}
// initialize nvn::Samplers
int samplerId;
nvn::SamplerBuilder samplerBuilder;
samplerBuilder.SetDefaults();
samplerBuilder.SetDevice(m_pGraphicsSystem->GetDevice());
samplerBuilder.SetMinMagFilter(nvn::MinFilter::LINEAR, nvn::MagFilter::LINEAR);
samplerId = m_TextureManager.AddSampler(samplerBuilder);
// generate texture handles
m_TextureHandles[0] = m_TextureManager.RegisterTextureHandle(textureIds[0], samplerId);
m_TextureHandles[1] = m_TextureManager.RegisterTextureHandle(textureIds[1], samplerId);
// free memory for texture files since texel data was loaded into TextureManager MemoryPool
for (int i = 0; i < NumTextureAssets; ++i)
{
m_pGraphicsSystem->GetAllocator()->Free(textureAssets[i]);
}
// get shader binding location for sampler uniform
m_SamplerLocation = m_Shader.LookupUniformInfo("sampler").GetBindingLocations(nvn::ShaderStage::FRAGMENT);
// initialize render state
m_RenderState.Initialize();
m_RenderState.GetColorState()->SetBlendEnable(0, true);
m_RenderState.GetBlendState()->SetBlendTarget(0);
m_RenderState.GetBlendState()->SetBlendEquation(nvn::BlendEquation::ADD, nvn::BlendEquation::ADD);
m_RenderState.GetBlendState()->SetBlendFunc(nvn::BlendFunc::SRC_ALPHA, nvn::BlendFunc::ONE_MINUS_SRC_ALPHA, nvn::BlendFunc::SRC_ALPHA, nvn::BlendFunc::ONE_MINUS_SRC_ALPHA);
m_RenderState.GetPolygonState()->SetFrontFace(nvn::FrontFace::CW);
m_RenderState.GetPolygonState()->SetCullFace(nvn::Face::BACK);
m_RenderState.GetPolygonState()->SetPolygonMode(nvn::PolygonMode::FILL);
m_RenderState.GetDepthStencilState()->SetDepthTestEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthWriteEnable(true);
m_RenderState.GetDepthStencilState()->SetDepthFunc(nvn::DepthFunc::LESS);
// Initialize text renderer
m_Font.InitializeFromNxSystemFont(m_pGraphicsSystem->GetAllocator(), nn::pl::SharedFontType_Standard);
m_TextRenderer.Initialize(m_pGraphicsSystem);
m_TextRenderer.SetScalingFactor(1.5f); // 1080 / 720
m_TextRenderer.SetCanvasDimensions(m_pGraphicsSystem->GetRenderTargetWidth() * 1.0f, m_pGraphicsSystem->GetRenderTargetHeight() * 1.0f);
m_InfoArea = m_TextRenderer.CreateTextArea(1000, 500, &m_Font, 30, m_Font.GetLineSpacing() * 1.5f);
m_GpuTimeArea = m_TextRenderer.CreateTextArea(1000, 100, &m_Font, 30, m_Font.GetLineSpacing() * 1.5f);
m_TextCommandBuffer.Initialize(m_pGraphicsSystem);
// default resolution scale
m_ResolutionScale = 1.0f;
m_CurTextureHandleIndex = 0;
// build our command buffer to hold GPU commands to draw our triangles up front once
m_CommandBuffer.Initialize(m_pGraphicsSystem);
GenerateCommandBuffer();
m_IsInitialized = true;
}
}//NOLINT(impl/function_size)
void DynamicResolution::DrawFrame() NN_NOEXCEPT
{
m_IsAppResolutionChanged = false;
// The actived lights number automatically increase from 0 to 80 and decrease from 80 to 0 to change load dynamically.
static bool isLightNumberIncreasing = true;
static int frame = 0;
bool isLightNumberChanged = false;
if (frame >= LightIncreaseInterval)
{
frame = 0;
m_ActivedLightNumber = isLightNumberIncreasing ? ++m_ActivedLightNumber : --m_ActivedLightNumber;
isLightNumberChanged = true;
if (m_ActivedLightNumber > LightCount)
{
isLightNumberIncreasing = false;
}
else if (m_ActivedLightNumber < 0)
{
isLightNumberIncreasing = true;
}
m_ActivedLightNumber = m_ActivedLightNumber > LightCount ? LightCount : m_ActivedLightNumber;
m_ActivedLightNumber = m_ActivedLightNumber < 0 ? 0 : m_ActivedLightNumber;
// load light numer into memory pool
memcpy(m_LightNumAddr, &m_ActivedLightNumber, sizeof(uint32_t));
}
frame++;
if (IsTriggered(PadButton_Y))
{
m_IsDynamicResolutionOn = !m_IsDynamicResolutionOn;
// when toggle dynamic resolution off, set resolution scale to 1 to render with default resolution
if (m_IsDynamicResolutionOn == false)
{
m_ResolutionScale = 1.0f;
ScaleScreenResolution(m_ResolutionScale);
}
}
// get gpu draw time in microsecond
float gpuTime = m_pGraphicsSystem->GetLastGpuDrawTime().GetMicroSeconds() / 1000.0f;
if (m_IsDynamicResolutionOn)
{
ComputeDynamicResolution(gpuTime);
}
// check to see if the texture handle is to be updated
bool updateTextureHandle = false;
if (IsTriggered(PadButton_X))
{
m_CurTextureHandleIndex = (m_CurTextureHandleIndex + 1) % NumTextureHandles;
updateTextureHandle = true;
}
UpdateCubeUniform();
m_TextRenderer.ClearText(m_InfoArea);
m_TextRenderer.ClearText(m_GpuTimeArea);
m_TextRenderer.AppendTextPrintf(m_InfoArea, u8"draw GPU time(ms): \n");
m_TextRenderer.AppendTextPrintf(m_GpuTimeArea, u8"%.2f \n", gpuTime);
m_TextRenderer.AppendTextPrintf(m_InfoArea, u8"Screen Width: %d | Screen Height: %d \n", static_cast<int>(m_pGraphicsSystem->GetScreenWidth() * m_ResolutionScale), static_cast<int>(m_pGraphicsSystem->GetScreenHeight() * m_ResolutionScale));
m_TextRenderer.AppendTextPrintf(m_InfoArea, u8"Light Count: %d \n", m_ActivedLightNumber);
m_TextRenderer.AppendTextPrintf(m_InfoArea, u8"Dynamic Resolution: %s ", (m_IsDynamicResolutionOn) ? u8"ON " : u8"OFF");
m_TextRenderer.AppendTextPrintf(m_InfoArea, u8"(Press Y Button)");
// generate the command buffer if necessary and execute it
if (!m_IsPrebuiltCommandBufferUsed || updateTextureHandle || m_IsScreenResolutionChanged || m_IsAppResolutionChanged || isLightNumberChanged)
{
GenerateCommandBuffer();
}
m_pGraphicsSystem->GetQueue()->SubmitCommands(1, &m_CommandHandle);
nvn::CommandBuffer* pTextCommandBuffer = m_TextCommandBuffer.StartRecording();
{
m_TextRenderer.StartDrawing(pTextCommandBuffer);
m_TextRenderer.DrawTextArea(m_InfoArea, 20, 100, Color(255, 255, 255, 255), pTextCommandBuffer);
m_TextRenderer.DrawTextArea(m_GpuTimeArea, 350, 100, GetGpuTimeTextColor(gpuTime), pTextCommandBuffer);
}
m_TextCommandBuffer.EndRecordingAndExecute();
}
void DynamicResolution::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_Font.Finalize();
m_TextRenderer.Finalize();
m_TextCommandBuffer.Finalize();
m_LightDataBuffer.Finalize();
for (int i = 0; i < CubeCount; i++)
{
m_ModelDataBuffer[i].Finalize();
}
m_TextureManager.Finalize();
m_CommandBuffer.Finalize();
m_VertexDataBuffer.Finalize();
m_IndexDataBuffer.Finalize();
m_VertexAndUniformDataMemPool.Finalize();
m_Shader.Finalize();
m_pGraphicsSystem->GetAllocator()->Free(m_GlslcData);
m_GlslcData = nullptr;
m_pGraphicsSystem = nullptr;
m_CommandHandle = 0;
m_Theta = 0.0f;
m_IsInitialized = false;
}

void DynamicResolution::ScaleScreenResolution(float scale) NN_NOEXCEPT
{
int newWidth = static_cast<int>(scale * m_pGraphicsSystem->GetScreenWidth());
int newHeight = static_cast<int>(scale * m_pGraphicsSystem->GetScreenHeight());
// nn::vi::SetLayerCrop serves the same purpose as nvn::Window::SetCrop to change screen resolution,
// and can be used with other Graphics APIs like OpenGl and Vulkan.
m_pGraphicsSystem->GetNvnWindow()->SetCrop(0, 0, newWidth, newHeight);
m_IsAppResolutionChanged = true;
}
void DynamicResolution::ComputeDynamicResolution(float gpuTime) NN_NOEXCEPT
{
// To make the resolution not change too fast, we change resolution only when gpu processing load continuously exceeds/belows the thresholds.
// Note that the thresholds are suitable only for running this application on NX, it need to be changed for different applications and platforms.
static int increaseResolutionCounter = 0;
static int decreaseResolutionCounter = 0;
if (gpuTime > m_GpuLoadThresholdMax)
{
decreaseResolutionCounter++;
increaseResolutionCounter = 0;
}
else if (gpuTime < m_GpuLoadThresholdMin)
{
increaseResolutionCounter++;
decreaseResolutionCounter = 0;
}
else
{
increaseResolutionCounter = 0;
decreaseResolutionCounter = 0;
}
// decrease resolution when gpu load keeps exceeding the maximum load threshold
if (decreaseResolutionCounter > 20 && m_ResolutionScale > m_ResolutionScaleMin)
{
decreaseResolutionCounter = 0;
m_ResolutionScale -= m_ResolutionScaleIncrement;
m_ResolutionScale = m_ResolutionScale < m_ResolutionScaleMin ? m_ResolutionScaleMin : m_ResolutionScale;
ScaleScreenResolution(m_ResolutionScale);
}
// increase resolution when gpu load keeps going below the minimum load threshold
else if (increaseResolutionCounter > 40 && m_ResolutionScale < m_ResolutionScaleMax)
{
increaseResolutionCounter = 0;
m_ResolutionScale += m_ResolutionScaleIncrement;
m_ResolutionScale = m_ResolutionScale > m_ResolutionScaleMax ? m_ResolutionScaleMax : m_ResolutionScale;
ScaleScreenResolution(m_ResolutionScale);
}
}
void DynamicResolution::UpdateCubeUniform() NN_NOEXCEPT
{
// update modelView uniform
nn::util::Vector3f rotate(0, 0, 0);
nn::util::Matrix4x3fType rotateMtx;
rotate.SetX(-m_Theta);
nn::util::MatrixIdentity(&rotateMtx);
nn::util::MatrixSetRotateXyz(&rotateMtx, rotate);
const UniformBlockMemberInfo& modelViewInvTransposeMtxInfo = m_pModelDataUniformBlockInfo->LookupMemberInfo("modelViewInvTranspose");
const UniformBlockMemberInfo& modelViewMtxInfo = m_pModelDataUniformBlockInfo->LookupMemberInfo("modelView");
for (int i = 0; i < CubeCount; i++)
{
int z = i / (RowLength * ColumnLength);
int y = (i - z * RowLength * ColumnLength) / RowLength;
int x = i % RowLength;
// compute and load uniform data into memory pool
char* baseUniformBlockAddr = static_cast<char*>(m_ModelDataBuffer[i].Map());
char* modelViewMtxAddr = baseUniformBlockAddr + modelViewMtxInfo.GetOffset();
// inverse transpose model view transform for normal calculation
char* modelViewInvTransposeMtxAddr = baseUniformBlockAddr + modelViewInvTransposeMtxInfo.GetOffset();
// compute model transform and load into memory pool
nn::util::Matrix4x3fType translateMtx;
nn::util::Vector3f translate((x - 2.0f) * 4.5f, (y - 1.5f) * 3.5f, z * 5.0f);
nn::util::MatrixIdentity(&translateMtx);
nn::util::MatrixSetTranslate(&translateMtx, translate);
nn::util::Matrix4x3fType modelMtx;
nn::util::MatrixIdentity(&modelMtx);
nn::util::MatrixMultiply(&modelMtx, translateMtx, modelMtx);
if (x == 1 || x == 3)
{
nn::util::MatrixMultiply(&modelMtx, rotateMtx, modelMtx);
}
nn::util::Matrix4x3fType modelViewMtx;
nn::util::MatrixMultiply(&modelViewMtx, modelMtx, m_ViewMtx);
LoadMatrix43(modelViewMtxAddr, modelViewMtx);
nn::util::Matrix4x3fType modelViewInvTransposeMtx;
nn::util::MatrixInverseTranspose(&modelViewInvTransposeMtx, modelViewMtx);
LoadMatrix43(modelViewInvTransposeMtxAddr, modelViewInvTransposeMtx);
}
// update per frame for rotation
m_Theta += 0.02f;
if (m_Theta > 2 * nn::util::FloatPi)
{
m_Theta -= (2 * nn::util::FloatPi);
}
}
Color DynamicResolution::GetGpuTimeTextColor(float gpuTime) NN_NOEXCEPT
{
Color gpuTimeTextColor;
if (gpuTime > m_GpuLoadThresholdMax)
{
gpuTimeTextColor.Set(255, 0, 0, 255);
}
else if (gpuTime < m_GpuLoadThresholdMin)
{
gpuTimeTextColor.Set(0, 255, 0, 255);
}
else
{
gpuTimeTextColor.Set(255, 255, 0, 255);
}
return gpuTimeTextColor;
}
void DynamicResolution::GenerateCommandBuffer() NN_NOEXCEPT
{
nvn::CommandBuffer* pCommandBuffer = m_CommandBuffer.StartRecording();
{
// scissor/viewport
int renderWidth = static_cast<int>(m_pGraphicsSystem->GetScreenWidth() * m_ResolutionScale);
int renderHeight = static_cast<int>(m_pGraphicsSystem->GetScreenHeight() * m_ResolutionScale);
pCommandBuffer->SetScissor(0, 0, renderWidth, renderHeight);
pCommandBuffer->SetViewport(0, 0, renderWidth, renderHeight);
// clear color and depth/stencil buffers
static float color[4] = { 0.2f, 0.2f, 0.2f, 1.0f };
pCommandBuffer->ClearColor(0, color, nvn::ClearColorMask::RGBA);
pCommandBuffer->ClearDepthStencil(1.0f, true, 0, 0);
// render state
m_RenderState.Bind(pCommandBuffer);
// bind shader
m_Shader.Bind(pCommandBuffer);
// bind vertex buffer data
pCommandBuffer->BindVertexAttribState(NumVertexAttributes, m_VertexAttribStates);
pCommandBuffer->BindVertexStreamState(1, &m_VertexStreamState);
BindVertexBuffer(0, &m_VertexDataBuffer, pCommandBuffer);
// set texture and sampler pools
m_TextureManager.SetTextureAndSamplerPools(pCommandBuffer);
// bind texture
pCommandBuffer->BindTexture(nvn::ShaderStage::FRAGMENT, m_SamplerLocation, m_TextureHandles[m_CurTextureHandleIndex]);
// bind light uniform buffer
int32_t bindingLocationLight = m_pLightDataUniformBlockInfo->GetBindingLocations(nvn::ShaderStage::FRAGMENT);
BindUniformBuffer(nvn::ShaderStage::FRAGMENT, bindingLocationLight, &m_LightDataBuffer, pCommandBuffer);
for (int i = 0; i < CubeCount; i++)
{
// bind model uniform buffer
int32_t bindingLocationModel = m_pModelDataUniformBlockInfo->GetBindingLocations(nvn::ShaderStage::VERTEX);
BindUniformBuffer(nvn::ShaderStage::VERTEX, bindingLocationModel, &m_ModelDataBuffer[i], pCommandBuffer);
// draw call
pCommandBuffer->DrawElements(nvn::DrawPrimitive::TRIANGLES, nvn::IndexType::UNSIGNED_INT, m_NumIndices, m_IndexDataBuffer.GetAddress());
}
}
m_CommandHandle = m_CommandBuffer.EndRecording();
}
