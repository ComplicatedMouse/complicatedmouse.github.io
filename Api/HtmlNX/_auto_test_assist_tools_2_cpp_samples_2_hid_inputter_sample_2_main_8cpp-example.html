void ControllerConnection()
{
constexpr int controllerCount = 4;
int controllerId[controllerCount] = {};
// Connect controllers.
{
hidInputter::ControllerDeviceInfo deviceInfo = {
hidInputter::ControllerDeviceType_JoyConLeft,
hidInputter::ControllerInterfaceType_Bluetooth,
{0,0,0}, {255,0,0}
};
hidInputter::AddControllerWithDeviceInfo(&(controllerId[0]), deviceInfo);
}
{
hidInputter::ControllerDeviceInfo deviceInfo = {
hidInputter::ControllerDeviceType_JoyConRight,
hidInputter::ControllerInterfaceType_Bluetooth,
{0,0,0}, {255,0,0}
};
hidInputter::AddControllerWithDeviceInfo(&(controllerId[1]), deviceInfo);
}
{
hidInputter::ControllerDeviceInfo deviceInfo = {
hidInputter::ControllerDeviceType_JoyConLeft,
hidInputter::ControllerInterfaceType_Rail,
{0,0,0}, {255,0,0}
};
hidInputter::AddControllerWithDeviceInfo(&(controllerId[2]), deviceInfo);
}
{
hidInputter::ControllerDeviceInfo deviceInfo = {
hidInputter::ControllerDeviceType_JoyConRight,
hidInputter::ControllerInterfaceType_Rail,
{0,0,0}, {255,0,0}
};
hidInputter::AddControllerWithDeviceInfo(&(controllerId[3]), deviceInfo);
}

for (int i = 0; i < controllerCount; i++)

}

for (int i = 0; i < controllerCount; i++)
{
hidInputter::AddController(&(controllerId[i]));
}
Sleep(3000);
// Disconnect controllers.
for (int i = 0; i < controllerCount; i++)
{
hidInputter::ReleaseController(controllerId[i]);
}
Sleep(1000);
// Add the Joy-Con (R).
for (int i = 0; i < controllerCount; i++)
{

hidInputter::ControllerDeviceType_JoyConRight,
hidInputter::ControllerInterfaceType_Bluetooth,
{0,0,0}, {255,0,0}
};
hidInputter::AddControllerWithDeviceInfo(&(controllerId[i]), deviceInfo);
}

for (int i = 0; i < controllerCount; i++)
{

}

{0,0,0}, {255,0,0}

hidInputter::PushButton(controllerId[0], hidInputter::Button_Plus, 120);
Sleep(1000);
hidInputter::PushButton(controllerId[0], hidInputter::Button_Minus, 120);
Sleep(1000);
hidInputter::PushButton(controllerId[0], hidInputter::Button_Home, 120);
Sleep(1000);
hidInputter::PushButton(controllerId[0], hidInputter::Button_Home, 120);

hidInputter::Button_A,

hidInputter::Button_Y

hidInputter::Button seqButton[] = {
hidInputter::Button_A, hidInputter::Button_B, hidInputter::Button_X, hidInputter::Button_Y,
hidInputter::Button_Left, hidInputter::Button_Right, hidInputter::Button_Up, hidInputter::Button_Down,

controllerId[0], seqButton, sizeof(seqButton) / sizeof(hidInputter::Button), 120, 90);

}
void StickControl()
{
constexpr int controllerCount = 4;
int controllerId[controllerCount] = {};
for (int i = 0; i < controllerCount; i++)
{
hidInputter::AddController(&(controllerId[i]));
}
printf("Tilt an analog stick.\n");
hidInputter::TiltAnalogStick(
controllerId[0], hidInputter::Stick_L, 180.0f, 1.0f, 1000);
Sleep(1);
hidInputter::TiltAnalogStick(
controllerId[0], hidInputter::Stick_L, -90.0f, 1.0f, 2000);
Sleep(1);
hidInputter::TiltAnalogStick(
controllerId[0], hidInputter::Stick_R, 90.0f, 1.0f, 1000);
Sleep(1);
printf("Tilt an analog stick all around.\n");
for (float degree = 0.0f; degree < 360.0f; degree += 5.0f)
{
hidInputter::HoldAnalogStick(
controllerId[0], hidInputter::Stick_R, degree, 1.0f);
Sleep(10);
}
hidInputter::ReleaseAnalogStick(controllerId[0], hidInputter::Stick_R);
for (float degree = 0.0f; degree > -360.0f; degree -= 5.0f)
{
hidInputter::HoldAnalogStick(
controllerId[0], hidInputter::Stick_L, degree, 0.5f);
Sleep(10);
}
hidInputter::ReleaseAnalogStick(controllerId[0], hidInputter::Stick_L);
printf("Tilt multi controllers' sticks all around.\n");
for (float degree = 0.0f; degree < 360.0f; degree += 5.0f)
{
for (int i = 0; i < controllerCount; i++)
{
hidInputter::HoldAnalogStick(
controllerId[i], hidInputter::Stick_R, degree, 1.0f);
hidInputter::HoldAnalogStick(
controllerId[i], hidInputter::Stick_L, -degree, 1.0f);
}
Sleep(10);
}
for (int i = 0; i < controllerCount; i++)
{
hidInputter::ReleaseController(controllerId[i]);
}
}

int main()
{
int targetCount = 0;
tmUtility::GetConnectedTargetCount(&targetCount);
if (targetCount < 1)
{
printf("Capture device is not found.\n");
return 0;
}
SerialNumberString* serialListBuffer = new SerialNumberString[targetCount];
for (int i = 0; i < targetCount; i++)
{
serialListBuffer[i].value[0] = '\0';
}
tmUtility::GetConnectedTargetSerialNumbers(serialListBuffer, targetCount);
std::vector<SerialNumberString> serialList;
for (int i = 0; i < targetCount; i++)
{
if (strlen(serialListBuffer[i].value) > 0)
{
serialList.push_back(serialListBuffer[i]);
}
}
delete[] serialListBuffer;
printf("Please select target.\n");
for (int i = 0; i < serialList.size(); i++)
{
printf("[%d] : %s\n", i, serialList.at(i).value);
}
int target = 0;
scanf_s("%d", &target);
if (target < 0 || target >= targetCount)
{
printf("Invalid input.");
return 0;
}
SerialNumberString targetSerial = serialList.at(target);
// Initialize HidInputter.
{
hidInputter::HidInputterResult result = hidInputter::Initialize(targetSerial);
if (result != hidInputter::HidInputterResult_Success)
{
printf("Initialize HidInputter is failed.\n");
return 0;
}
}
// Initialize CaptureProcessor.
{
captureProcessor::CaptureResult result = captureProcessor::InitializeForRemoteVideo(targetSerial, true);
if (result != captureProcessor::CaptureResult_Success)
{
printf("Initialize CaptureProcessor is failed.\n");
return 0;
}
}
RunSample();
// Finalize CaptureProcessor.
captureProcessor::Finalize();
// Finalize HidInputter.
hidInputter::Finalize();
return 0;
}
