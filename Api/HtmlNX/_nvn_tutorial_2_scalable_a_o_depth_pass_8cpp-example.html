/*
* ScalableAODepthPass Constructor
* -------------------------------
* Sets up default member data values.
*/
ScalableAODepthPass::ScalableAODepthPass() : m_DepthTexHandle(0)
{
m_Name = "ScalableAODepthPass";
}
/*
* ScalableAODepthPass Destructor
* ------------------------------
* Empty destructor.
*/
ScalableAODepthPass::~ScalableAODepthPass()
{
}
/*
* ScalableAODepthPass::Init
* -------------------------
* Sets the asset data to use for drawing.
*/
void ScalableAODepthPass::Init(AssetFileDataHolder* pData)
{
m_pAssetData = pData;
}
/*
* ScalableAODepthPass::SetClipInfo
* --------------------------------
* Calculates the clip info using the near and far values.
*/
void ScalableAODepthPass::SetClipInfo(float near, float far)
{
m_ClipInfo[0] = near * far;
m_ClipInfo[1] = near - far;
m_ClipInfo[2] = far;
}
/*
* ScalableAODepthPass::SetupUniforms
* ----------------------------------
* Sets up the uniform buffer.
*/
void ScalableAODepthPass::SetupUniforms(UniformBufferManager* pUniformBufferManager)
{
m_pBlockSamplerFS = pUniformBufferManager->CreateUniformBuffer(sizeof(ScalableAODepth::BlockSamplerFSUniformBlockData));
}
/*
* ScalableAODepthPass::SetTextureHandle
* -------------------------------------
* Sets up the texture handle for the texture needed to generate SAO texture.
*/
void ScalableAODepthPass::SetTextureHandle(const NVNtextureHandle& depthHandle)
{
m_DepthTexHandle = depthHandle;
}
/*
* ScalableAODepthPass::UpdateUniforms
* -----------------------------------
* Updates the uniform buffer.
*/
void ScalableAODepthPass::UpdateUniforms()
{
ScalableAODepth::BlockSamplerFSUniformBlockData* blockSamplerFS = reinterpret_cast<ScalableAODepth::BlockSamplerFSUniformBlockData*>(m_pBlockSamplerFS->GetMappedPointer());
blockSamplerFS->SetUniform_u_DepthTex(m_DepthTexHandle);
blockSamplerFS->SetUniform_u_ClipInfo(m_ClipInfo);
}
/*
* ScalableAODepthPass::Draw
* -------------------------
* Draws a screen space quad that converts depth values into a view space depth buffer
*/
void ScalableAODepthPass::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
ScalableAODepth::BlockSamplerFSUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pBlockSamplerFS->GetCurrentBufferAddress(),
sizeof(ScalableAODepth::BlockSamplerFSUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
