void Shape::CreateOctahedron(SimplePrimitiveRenderer* pRenderer, int maxModelViewTransforms) NN_NOEXCEPT
{
NN_ASSERT(!m_IsInitialized);
m_pRenderer = pRenderer;
nn::util::Vector3f vtx[6];
vtx[0].Set(0, 2, 0);
vtx[1].Set(1, 0, 1);
vtx[2].Set(-1, 0, 1);
vtx[3].Set(1, 0, -1);
vtx[4].Set(-1, 0, -1);
vtx[5].Set(0, -2, 0);
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[0], vtx[1], vtx[2]));
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[0], vtx[3], vtx[1]));
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[0], vtx[4], vtx[3]));
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[0], vtx[2], vtx[4]));
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[5], vtx[2], vtx[1]));
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[5], vtx[1], vtx[3]));
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[5], vtx[3], vtx[4]));
m_Primitives.push_back(m_pRenderer->AddTriangle(vtx[5], vtx[4], vtx[2]));
CreateShape(maxModelViewTransforms);
}
void Shape::CreateCone(SimplePrimitiveRenderer* pRenderer, int maxTransforms) NN_NOEXCEPT
{
NN_ASSERT(!m_IsInitialized);
m_pRenderer = pRenderer;
m_Primitives.push_back(m_pRenderer->AddCone(3, 1.5, 100));
CreateShape(maxTransforms);
}
void Shape::Destroy() NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
m_Primitives.clear();
m_ModelViews.clear();
m_Materials.clear();
m_pRenderer = nullptr;
}
void Shape::SetMaterial(size_t partIndex, const MaterialSettings& materialSettings) NN_NOEXCEPT
{
NN_ASSERT(partIndex >= 0 && partIndex < m_Primitives.size());
m_pRenderer->UpdateMaterial(m_Materials[partIndex], materialSettings);
}
void Shape::SetModelTransform(size_t modelViewTransformIndex, const nn::util::Matrix4x3fType& modelTransform) NN_NOEXCEPT
{
nn::util::Matrix4x3fType modelViewTransform;
nn::util::MatrixIdentity(&modelViewTransform);
nn::util::MatrixMultiply(&modelViewTransform, modelTransform, *m_pViewMtx);
m_pRenderer->UpdateModelViewTransform(m_ModelViews[modelViewTransformIndex], modelViewTransform);
}
void Shape::SetModelViewTransformAsCurrent(size_t modelViewTransformIndex) NN_NOEXCEPT
{
NN_ASSERT(modelViewTransformIndex >= 0 && modelViewTransformIndex < m_ModelViews.size());
for (size_t i = 0; i < m_Primitives.size(); ++i)
{
m_pRenderer->SetPrimitiveModelView(m_Primitives[i], m_ModelViews[modelViewTransformIndex]);
}
}
void Shape::StartDrawing(nvn::CommandBuffer* pCommandBuffer, size_t partIndex) NN_NOEXCEPT
{
NN_ASSERT(partIndex >= 0 && partIndex < m_Primitives.size());
m_pRenderer->BindPrimitive(m_Primitives[partIndex], pCommandBuffer);
}
void Shape::Draw(nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
m_pRenderer->DrawBoundPrimitive(pCommandBuffer);
}
//-------------------------------------------------------------------------------------------------
// Private methods
//-------------------------------------------------------------------------------------------------
void Shape::CreateShape(int maxModelViewTransforms) NN_NOEXCEPT
{
NN_ASSERT(maxModelViewTransforms > 0);
for (int i = 0; i < maxModelViewTransforms; ++i)
{
m_ModelViews.push_back(m_pRenderer->AddModelViewTransform());
}
for (size_t i = 0; i < m_Primitives.size(); ++i)
{
Material newMaterial = m_pRenderer->AddMaterial();
m_pRenderer->SetPrimitiveMaterial(m_Primitives[i], newMaterial);
m_Materials.push_back(newMaterial);
}
m_IsInitialized = true;
}
