void DrawPixelData(Buffer* pPngImagePixelData, nn::image::Dimension pngImageDimension) NN_NOEXCEPT
{
// Initialize the graphics system.
static const int RenderWidth = 1280;
static const int RenderHeight = 720;
GraphicsSystem graphicsSystem;
graphicsSystem.Initialize(RenderWidth, RenderHeight);
nn::gfx::Texture pngTexture;
{
// Set up the texture.
graphicsSystem.SetupTextureWithPixelData(
&pngTexture,
pngImageDimension.width,
pngImageDimension.height,
nn::gfx::ImageFormat_R8_G8_B8_A8_Unorm,
pPngImagePixelData->GetDataPtr(),
pPngImagePixelData->GetDataSize());
}
// Render on the screen.
int x = 0;
int increamentX = 1;
int y = 0;
int increamentY = 1;
for (int frame = 0; frame < 600000; ++frame)
{
graphicsSystem.BeginDraw();
graphicsSystem.ClearColorTarget((frame & 0xFF) * (1.0f / 256.0f), 0.4f, 0.4f, 1.0f);
graphicsSystem.DrawScreenRect(x, y, pngImageDimension.width, pngImageDimension.height, &pngTexture);
x = x + increamentX;
y = y + increamentY;
increamentX = (x + pngImageDimension.width > RenderWidth || x < 0) ? -increamentX : increamentX;
increamentY = (y + pngImageDimension.height > RenderHeight || y < 0) ? -increamentY : increamentY;
graphicsSystem.EndDraw();
graphicsSystem.Synchronize(nn::TimeSpan::FromNanoSeconds(1000 * 1000 * 1000 / 60));
}
// Destroy the texture.
pngTexture.Finalize(&graphicsSystem.GetDevice());
// Destroy the graphics system.
graphicsSystem.Finalize();
}
bool DecodePngData(
Buffer *pOutPixelBuffer,
nn::image::Dimension* pOutDimension,
const FileData *pngData,
nn::image::PixelFormat format) NN_NOEXCEPT
{
// Initialize the PNG decoder.
nn::image::PngDecoder decoder;
decoder.Initialize(Allocate, nullptr, Free, nullptr);
// Register the PNG data.
decoder.SetImageData(pngData->GetDataPtr(), pngData->GetDataSize());
decoder.SetOutputPixelFormat(format);
// Analyze and convert the PNG data.
nn::image::PngStatus status = decoder.Analyze();
switch (status)
{
case nn::image::PngStatus_OutOfMemory:
NN_LOG("Insufficient memory during analysis.\n");
return false;
case nn::image::PngStatus_InternalError:
NN_LOG("Internal error during analysis.\n");
return false;
default:
NN_ASSERT(status == nn::image::PngStatus_Ok);
}
// Get the width and height of the image.
*pOutDimension = decoder.GetDimension();
// Get the pixel data stride.
size_t rowStride = decoder.GetRowStrideMin();
// Calculate the output size.
size_t pixelDataSize = rowStride * pOutDimension->height;
// Allocate the output buffer.
pOutPixelBuffer->Initialize(pixelDataSize);
// Decode.
status = decoder.Decode(pOutPixelBuffer->GetDataPtr(), pOutPixelBuffer->GetDataSize(), rowStride);
switch (status)
{
case nn::image::PngStatus_OutOfMemory:
NN_LOG("Insufficient memory during decoding.\n");
return false;
case nn::image::PngStatus_InternalError:
NN_LOG("Internal error during decoding.\n");
return false;
case nn::image::PngStatus_ShortOfOutputBuffer:
NN_LOG("Insufficient output buffer size.\n");
return false;
default:
NN_ASSERT(status == nn::image::PngStatus_Ok);
}
return true;
}
extern "C" void nnMain()
{
// Initialize the file system.
size_t cacheSize = 0;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::QueryMountRomCacheSize(&cacheSize));
char* mountRomCacheBuffer = new(std::nothrow) char[cacheSize];
NN_ABORT_UNLESS_NOT_NULL(mountRomCacheBuffer);
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::MountRom("Asset", mountRomCacheBuffer, cacheSize));
// Load a PNG file.
FileData pngImageData("Asset:/ExampleInput.png");
// Decode the PNG file.
Buffer pngImagePixelData;
nn::image::Dimension pngImageDimension;
if (!DecodePngData(&pngImagePixelData, &pngImageDimension, &pngImageData, nn::image::PixelFormat_Rgba32))
{
NN_LOG("Failed to decode PNG data. \n");
return;
}
// Render the pixel data on the screen.
DrawPixelData(&pngImagePixelData, pngImageDimension);
// Finalize the file system.
nn::fs::Unmount("Asset");
delete[] mountRomCacheBuffer;
return;
}
