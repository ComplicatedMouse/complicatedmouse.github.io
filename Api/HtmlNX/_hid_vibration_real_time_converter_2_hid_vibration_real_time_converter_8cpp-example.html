int16_t clamp(int32_t x)
{
const int32_t max = static_cast<int32_t>(std::numeric_limits<int16_t>::max());
const int32_t min = static_cast<int32_t>(std::numeric_limits<int16_t>::min());
x = x < max ? x : max;
x = x > min ? x : min;
return static_cast<int16_t>(x);
}

std::size_t ReadAdpcmFile(nn::audio::AdpcmHeaderInfo* header, void** adpcmData, const char* filename)
{
nn::fs::FileHandle handle;
nn::Result result = nn::fs::OpenFile(&handle, filename, nn::fs::OpenMode_Read);
NN_ABORT_UNLESS(result.IsSuccess());
int64_t size;
uint8_t adpcmheader[nn::audio::AdpcmHeaderSize];
result = nn::fs::GetFileSize(&size, handle);
NN_ABORT_UNLESS(result.IsSuccess());
*adpcmData = g_WaveBufferAllocator.Allocate(static_cast<std::size_t>(size) - sizeof(adpcmheader));
NN_ABORT_UNLESS_NOT_NULL(*adpcmData);
result = nn::fs::ReadFile(handle, 0, adpcmheader, sizeof(adpcmheader));
NN_ABORT_UNLESS(result.IsSuccess());
result = nn::fs::ReadFile(handle, sizeof(adpcmheader), *adpcmData, static_cast<size_t>(size) - sizeof(adpcmheader));
NN_ABORT_UNLESS(result.IsSuccess());
nn::fs::CloseFile(handle);
nn::audio::ParseAdpcmHeader(header, adpcmheader, sizeof(adpcmheader));
return static_cast<std::size_t>(size) - sizeof(adpcmheader);
}

nn::hid::SetSupportedNpadIdType(npadIds, padIdCount);

{

{

}

int UpdateI3dl2ReverbParameter(nn::audio::I3dl2ReverbType* pReverb, int updateMode, std::string& message)
{
// Update the reverb effect parameters.
nn::audio::I3dl2ReverbParameterSet parameterSet;
parameterSet = nn::audio::GetI3dl2ReverbParameters(pReverb); // Use the original parameter set when you re-enable.
switch (updateMode)
{
case nn::audio::I3dl2ReverbType::Preset_SmallRoom:
nn::audio::LoadI3dl2ReverbPreset(&parameterSet, nn::audio::I3dl2ReverbType::Preset_SmallRoom);
message = "SmallRoom";
break;
case nn::audio::I3dl2ReverbType::Preset_LargeRoom:
nn::audio::LoadI3dl2ReverbPreset(&parameterSet, nn::audio::I3dl2ReverbType::Preset_LargeRoom);
message = "LargeRoom";
break;
case nn::audio::I3dl2ReverbType::Preset_Hall:
nn::audio::LoadI3dl2ReverbPreset(&parameterSet, nn::audio::I3dl2ReverbType::Preset_Hall);
message = "Hall";
break;
case nn::audio::I3dl2ReverbType::Preset_CavernousCathedral:
nn::audio::LoadI3dl2ReverbPreset(&parameterSet, nn::audio::I3dl2ReverbType::Preset_CavernousCathedral);
message = "Cathedral";
break;
case nn::audio::I3dl2ReverbType::Preset_MetalCorridor:
nn::audio::LoadI3dl2ReverbPreset(&parameterSet, nn::audio::I3dl2ReverbType::Preset_MetalCorridor);
message = "Corridor";
break;
case nn::audio::I3dl2ReverbType::Preset_Count:
parameterSet = i3dl2ReverbParameterUserSet;
message = "UserDefine ";
break;
case nn::audio::I3dl2ReverbType::Preset_Count + 1:
// Disable the reverb effect.
nn::audio::SetI3dl2ReverbEnabled(pReverb, false);
message = "disabled";
break;
case nn::audio::I3dl2ReverbType::Preset_Count + 2:
// When enabled, the internal state of the reverb effect is reset.
// Clear all reverberant sounds that played before it was disabled.
nn::audio::SetI3dl2ReverbEnabled(pReverb, true);
message = "re-enabled";
break;
default:
break;
}
nn::audio::SetI3dl2ReverbParameters(pReverb, &parameterSet);
return (updateMode + 1) % 8;
}
void InitializeAudio()
{
nn::Result result;
bool ret;
g_WaveBufferAllocator.Initialize(g_WaveBufferPoolMemory, sizeof(g_WaveBufferPoolMemory));
g_EffectBufferAllocator.Initialize(g_EffectBufferPoolMemory, sizeof(g_EffectBufferPoolMemory));
// Specify a renderer parameter.
nn::audio::AudioRendererParameter parameter;
nn::audio::InitializeAudioRendererParameter(&parameter);
parameter.sampleRate = RenderRate;
parameter.sampleCount = RenderCount;
parameter.mixBufferCount = 4;
parameter.voiceCount = 4;
parameter.subMixCount = 0;
parameter.sinkCount = 1;
parameter.effectCount = 3;
parameter.performanceFrameCount = 0;
// Define the relationship between the mix buffer and the audio bus.
const int ChannelCount = 2;
int8_t mainBus[ChannelCount];
mainBus[nn::audio::ChannelMapping_FrontLeft] = 0;
mainBus[nn::audio::ChannelMapping_FrontRight] = 1;
int8_t subBusA[ChannelCount];
subBusA[nn::audio::ChannelMapping_FrontLeft] = 2;
subBusA[nn::audio::ChannelMapping_FrontRight] = 3;
// Check whether the parameter is supported by the system.
NN_ABORT_UNLESS(
nn::audio::IsValidAudioRendererParameter(parameter),
"Invalid AudioRendererParameter specified."
);
// Prepare the renderer's work buffer and register it to AudioRenderer.
size_t workBufferSize = nn::audio::GetAudioRendererWorkBufferSize(parameter);
g_RendererWorkBuffer = g_Allocator.Allocate(workBufferSize, nn::os::MemoryPageSize);
NN_ABORT_UNLESS_NOT_NULL(g_RendererWorkBuffer);
NNS_LOG("Open AudioRenderer\n");
// Initialize the renderer.
NN_ABORT_UNLESS(
nn::audio::OpenAudioRenderer(&g_AudioHandle, &g_AudioSystemEvent, parameter, g_RendererWorkBuffer, workBufferSize).IsSuccess(),
"Failed to open AudioRenderer"
);
// Initialize AudioRendererConfig.
size_t configBufferSize = nn::audio::GetAudioRendererConfigWorkBufferSize(parameter);
g_RendererConfigBuffer = g_Allocator.Allocate(configBufferSize, nn::os::MemoryPageSize);
NN_ABORT_UNLESS_NOT_NULL(g_RendererConfigBuffer);
nn::audio::InitializeAudioRendererConfig(&g_AudioConfig, parameter, g_RendererConfigBuffer, configBufferSize);
nn::audio::FinalMixType finalMix;
nn::audio::AcquireFinalMix(&g_AudioConfig, &finalMix, 4);
// Specify the renderer output destination.
nn::audio::DeviceSinkType deviceSink;
// The output channel is determined based on the mix buffer index specified for mainBus.
// Output from mainBus[nn::audio::ChannelMapping_FrontLeft] is sent to the left channel and
// output from mainBus[nn::audio::ChannelMapping_FrontRight] is sent to the right channel.
result = nn::audio::AddDeviceSink(&g_AudioConfig, &deviceSink, &finalMix, mainBus, ChannelCount, "MainAudioOut");
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Prepare a memory pool for the effect buffer.
nn::audio::MemoryPoolType effectBufferPool;
ret = AcquireMemoryPool(&g_AudioConfig, &effectBufferPool, g_EffectBufferPoolMemory, sizeof(g_EffectBufferPoolMemory));
NN_ABORT_UNLESS(ret);
// Attach the memory pool. The settings are applied every time the nn::audio::RequestUpdateAudioRenderer() function is called.
ret = RequestAttachMemoryPool(&effectBufferPool);
NN_ABORT_UNLESS(ret);
nn::audio::I3dl2ReverbParameterSet i3dl2Parameter;
//Settings for various i3dl2Reverb parameters.
size_t i3dl2ReverbSize = nn::audio::GetRequiredBufferSizeForI3dl2Reverb(parameter.sampleRate, ChannelCount);
g_I3dl2ReverbBuffer = g_EffectBufferAllocator.Allocate(i3dl2ReverbSize);
NN_ABORT_UNLESS_NOT_NULL(g_I3dl2ReverbBuffer);
// Add reverb effect to FinalMix.
result = nn::audio::AddI3dl2Reverb(&g_AudioConfig, &g_I3dl2Reverb, g_I3dl2ReverbBuffer, i3dl2ReverbSize, &finalMix, ChannelCount);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Set the input/output buffers for the reverb effect.
// Write back to the input source buffer.
nn::audio::SetI3dl2ReverbInputOutput(&g_I3dl2Reverb, subBusA, subBusA, sizeof(subBusA));
nn::audio::LoadI3dl2ReverbPreset(&i3dl2Parameter, nn::audio::I3dl2ReverbType::Preset_SmallRoom);
nn::audio::SetI3dl2ReverbParameters(&g_I3dl2Reverb, &i3dl2Parameter);
// Allocate the send and return buffers used by the aux effect.
const int AuxBufferFrameCount = 4;
size_t auxBufferSize = nn::audio::GetRequiredBufferSizeForAuxSendReturnBuffer(&parameter, AuxBufferFrameCount, sizeof(subBusA));
void* sendBuffer = g_EffectBufferAllocator.Allocate(auxBufferSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS_NOT_NULL(sendBuffer);
void* returnBuffer = g_EffectBufferAllocator.Allocate(auxBufferSize, nn::audio::BufferAlignSize);
NN_ABORT_UNLESS_NOT_NULL(returnBuffer);
// Add the Aux effect to FinalMix.
result = nn::audio::AddAux(&g_AudioConfig, &g_AuxType, &finalMix, sendBuffer, returnBuffer, auxBufferSize);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Set the input channels for the aux effect.
int auxInputChannelCount = sizeof(subBusA);
nn::audio::SetAuxInputOutput(&g_AuxType, subBusA, subBusA, auxInputChannelCount);
// Prepare the buffer for getting sample data from the audio renderer.
g_ReadBuffer = static_cast<int32_t*>(g_Allocator.Allocate(auxBufferSize));
g_ReducedDataLengthBuffer = static_cast<int16_t*>(g_Allocator.Allocate(auxBufferSize));
NN_ABORT_UNLESS_NOT_NULL(g_ReadBuffer);
NN_ABORT_UNLESS_NOT_NULL(g_ReducedDataLengthBuffer);
memset(g_ReadBuffer, 0, auxBufferSize);
memset(g_ReducedDataLengthBuffer, 0, auxBufferSize);
// Prepare a BufferMixer effect.
nn::audio::BufferMixerType mixer;
result = nn::audio::AddBufferMixer(&g_AudioConfig, &mixer, &finalMix);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
nn::audio::SetBufferMixerVolume(&mixer, 0, 1.0f);
nn::audio::SetBufferMixerVolume(&mixer, 1, 1.0f);
nn::audio::SetBufferMixerInputOutput(&mixer, subBusA, mainBus, ChannelCount);
// Apply the configured parameters in the renderer.
result = nn::audio::RequestUpdateAudioRenderer(g_AudioHandle, &g_AudioConfig);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Start rendering.
result = nn::audio::StartAudioRenderer(g_AudioHandle);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Prepare a memory pool to maintain the sample data added to WaveBuffer.
nn::audio::MemoryPoolType waveBufferMemoryPool;
ret = AcquireMemoryPool(&g_AudioConfig, &waveBufferMemoryPool, g_WaveBufferPoolMemory, sizeof(g_WaveBufferPoolMemory));
NN_ABORT_UNLESS(ret);
ret = RequestAttachMemoryPool(&waveBufferMemoryPool);
NN_ABORT_UNLESS(ret);
// Load the sound effects.
for (int i = 0; i < SeCount; ++i)
{
g_Header[i] = reinterpret_cast<nn::audio::AdpcmHeaderInfo*>(g_WaveBufferAllocator.Allocate(sizeof(nn::audio::AdpcmHeaderInfo), NN_ALIGNOF(nn::audio::AdpcmHeaderInfo)));
std::size_t dataSeSize = ReadAdpcmFile(g_Header[i], &g_DataSe[i], g_SeFileNames[i]);
nn::audio::AcquireVoiceSlot(&g_AudioConfig, &g_VoiceSe[i], g_Header[i]->sampleRate, 1, nn::audio::SampleFormat_Adpcm, nn::audio::VoiceType::PriorityHighest, &g_Header[i]->parameter, sizeof(nn::audio::AdpcmParameter));
// Connect voice and finalMix.
nn::audio::SetVoiceDestination(&g_AudioConfig, &g_VoiceSe[i], &finalMix);
g_WaveBufferSe[i].buffer = g_DataSe[i];
g_WaveBufferSe[i].size = dataSeSize;
g_WaveBufferSe[i].startSampleOffset = 0;
g_WaveBufferSe[i].endSampleOffset = g_Header[i]->sampleCount;
g_WaveBufferSe[i].loop = false;
g_WaveBufferSe[i].isEndOfStream = false;
g_WaveBufferSe[i].pContext = &g_Header[i]->loopContext;
g_WaveBufferSe[i].contextSize = sizeof(nn::audio::AdpcmContext);
nn::audio::AppendWaveBuffer(&g_VoiceSe[i], &g_WaveBufferSe[i]);
nn::audio::SetVoicePlayState(&g_VoiceSe[i], nn::audio::VoiceType::PlayState_Play);
// Mix the content of voice with FinalMix.
nn::audio::SetVoiceMixVolume(&g_VoiceSe[i], &finalMix, 1.0f, 0, subBusA[nn::audio::ChannelMapping_FrontLeft]);
nn::audio::SetVoiceMixVolume(&g_VoiceSe[i], &finalMix, 1.0f, 0, subBusA[nn::audio::ChannelMapping_FrontRight]);
}
}//NOLINT(impl/function_size)
void FinalizeAudio()
{
// End rendering.
nn::audio::StopAudioRenderer(g_AudioHandle);
nn::audio::CloseAudioRenderer(g_AudioHandle);
nn::os::DestroySystemEvent(g_AudioSystemEvent.GetBase());
// Free memory.
for (int i = 0; i < SeCount; ++i)
{
if (g_DataSe[i])
{
g_WaveBufferAllocator.Free(g_DataSe[i]);
g_DataSe[i] = nullptr;
g_WaveBufferAllocator.Free(g_Header[i]);
g_Header[i] = nullptr;
}
}
if (g_RendererConfigBuffer)
{
g_Allocator.Free(g_RendererConfigBuffer);
g_RendererConfigBuffer = nullptr;
}
if (g_RendererWorkBuffer)
{
g_Allocator.Free(g_RendererWorkBuffer);
g_RendererWorkBuffer = nullptr;
}
if (g_I3dl2ReverbBuffer)
{
g_EffectBufferAllocator.Free(g_I3dl2ReverbBuffer);
}
if (g_ReadBuffer)
{
g_Allocator.Free(g_ReadBuffer);
g_ReadBuffer = nullptr;
}
if (g_ReducedDataLengthBuffer)
{
g_Allocator.Free(g_ReducedDataLengthBuffer);
g_ReducedDataLengthBuffer = nullptr;
}
}
nn::hid::NpadButtonSet GetPadButtonDown()
{
nn::hid::NpadButtonSet buttonDown = {};
if (nn::hid::GetNpadStyleSet(nn::hid::NpadId::No1).Test<nn::hid::NpadStyleFullKey>())
{
static nn::hid::NpadFullKeyState npadFullKeyState = {};
nn::hid::NpadFullKeyState state;
nn::hid::GetNpadState(&state, nn::hid::NpadId::No1);
buttonDown |= state.buttons & ~npadFullKeyState.buttons;
npadFullKeyState = state;
}
else if (nn::hid::GetNpadStyleSet(nn::hid::NpadId::Handheld).Test<nn::hid::NpadStyleHandheld>())
{
static nn::hid::NpadHandheldState npadHandheldState = {};
nn::hid::NpadHandheldState state;
nn::hid::GetNpadState(&state, nn::hid::NpadId::Handheld);
buttonDown |= state.buttons & ~npadHandheldState.buttons;
npadHandheldState = state;
}
return buttonDown;
}
void UpdateButton()
{
auto buttonDown = GetPadButtonDown();
//Play sound effects. (The same SE is not overlaid.)
for (int i = 0; i < SeCount; ++i)
{
// SE numbers {0, 1, 2, 3} correspond to the {A, B, X, Y} Buttons.
if (buttonDown.Test(i))
{
if (nn::audio::GetReleasedWaveBuffer(&g_VoiceSe[i]))
{
nn::audio::AppendWaveBuffer(&g_VoiceSe[i], &g_WaveBufferSe[i]);
NNS_LOG("Playing Sound: %s\n", g_SeFileNames[i]);
}
if ((nn::audio::GetVoicePlayState(&g_VoiceSe[i]) == nn::audio::VoiceType::PlayState_Stop) || (nn::audio::GetVoicePlayState(&g_VoiceSe[i]) == nn::audio::VoiceType::PlayState_Pause))
{
nn::audio::SetVoicePlayState(&g_VoiceSe[i], nn::audio::VoiceType::PlayState_Play);
NNS_LOG("Playing Sound: %s\n", g_SeFileNames[i]);
}
}
}
// Enables and disables AuxType.
if (buttonDown.Test<::nn::hid::NpadButton::R>())
{
if (nn::audio::IsAuxEnabled(&g_AuxType))
{
nn::audio::SetAuxEnabled(&g_AuxType, false);
//Clear out vibration.
nn::hid::VibrationValue vib = nn::hid::VibrationValue::Make();
for (int j = 0; j < g_VibrationDeviceCount; ++j)
{
nn::hid::SendVibrationValue(g_VibrationHandles[j], vib);
}
NNS_LOG("RealTime Converter Off\n");
}
else
{
nn::audio::SetAuxEnabled(&g_AuxType, true);
NNS_LOG("RealTime Converter On\n");
}
}
if (buttonDown.Test<::nn::hid::NpadButton::L>())
{
static int index = 0;
index = UpdateI3dl2ReverbParameter(&g_I3dl2Reverb, index, g_I3dl2ReverbState);
NNS_LOG("I3dl2Reverb:%-15s\n", g_I3dl2ReverbState.c_str());
}
// End the sample.
if (buttonDown.Test<::nn::hid::NpadButton::Plus>())
{
g_IsRunning = false;
}
if (buttonDown.Test<::nn::hid::NpadButton::Minus>())
{
PrintUsage();
}
}

extern "C" void nnMain()
{
NN_LOG("%s Start.\n", Title);
g_Allocator.Initialize(g_WorkBuffer, sizeof(g_WorkBuffer));
InitializeFileSystem();
InitializeHidDevices();
InitializeAudio();
nn::hid::VibrationConverter vibrationConverter(RenderRate);
PrintUsage();
while (g_IsRunning)
{
// Periodically signaled with the audio frame interval.
g_AudioSystemEvent.Wait();
// Button operations.
UpdateButton();
// Convert the sample data received with the Aux effect to vibration values.
if (nn::audio::IsAuxEnabled(&g_AuxType))
{
// Load sample data from the audio renderer.
const int readCount = nn::audio::ReadAuxSendBuffer(&g_AuxType, g_ReadBuffer, nn::audio::GetAuxSampleCount(&g_AuxType));
const int chunkLengthOfAudio = static_cast<int>((RenderRate * nn::hid::VibrationConverter::ChunkDuration.GetMilliSeconds()) / 1000);
for (auto i = 0; i < readCount; ++i)
{
g_ReducedDataLengthBuffer[i] = clamp(g_ReadBuffer[i]);
}
for (auto i = 0; i < readCount; i += chunkLengthOfAudio)
{
nn::hid::VibrationValue vib;
// Convert from sample data to vibration values.
vibrationConverter.ProcessChunk(&vib, (g_ReducedDataLengthBuffer + i), chunkLengthOfAudio);
// Send the calculated vibration values to all vibration handles.
for (int j = 0; j < g_VibrationDeviceCount; ++j)
{
nn::hid::SendVibrationValue(g_VibrationHandles[j], vib);
}
}
// Rewrite the sample to the audio renderer.
nn::audio::WriteAuxReturnBuffer(&g_AuxType, g_ReadBuffer, readCount);
}
// Apply changes made to the parameters for the various effects in the audio renderer.
auto result = nn::audio::RequestUpdateAudioRenderer(g_AudioHandle, &g_AudioConfig);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
}
FinalizeAudio();
FinalizeFileSystem();
g_Allocator.Finalize();
NN_LOG("%s End.\n", Title);
}
