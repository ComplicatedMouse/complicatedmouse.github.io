#include <nn/account/account_ApiForApplications.h>
#include <nn/nifm/nifm_Api.h>
#include <nns/console/console_ConsoleManager.h>
#include <nns/console/console_InputManager.h>
#include "AccountApplication_RootSequence.h"
#include "AccountApplication_ScreenIo.h"
namespace
{
// The region for managing the console and the drawing to the console. Because there are restrictions on size and alignment, this is basically placed directly under the namespace.
nns::console::DefaultConsoleManagerHolder g_ManagerHolder = NNS_CONSOLE_CONSOLE_MANAGER_HOLDER_INITIALIZER;
} // ~namespace <anonymous>
extern "C" void nnMain()
{
// Initialize the dependency module.
nn::nifm::Initialize();
// Initialize the nn::account features.
nn::account::Initialize();
// Initialize the drawing features and HID.
auto& manager = g_ManagerHolder.GetManager();
nns::console::DefaultInputManager im;
im.Initialize();
// Initialize the sequence.
Window window;
RootSequence root(window);
// The drawing settings for the sequence.
const auto Margin = 40;
const auto Width = static_cast<int>(manager.Settings.ScreenWidth);
const auto TotalHeight = (static_cast<int>(manager.Settings.ScreenHeight) - Margin);
manager.Register(&window, 0, 0, Width, TotalHeight);
im.SetPadEventListener(&window);
// Process the sequence and draw.
while (NN_STATIC_CONDITION(true))
{
// Update HID.
im.Update();
// Update the sequence.
auto* subSeq = root.GetSubSequence();
if (subSeq && !subSeq->IsClosed())
{
subSeq->Update();
}
else
{
root.Update();
}
// Confirm the content to draw to the screen.
window.Update();
// Draw.
manager.Update();
manager.Draw();
}
}
