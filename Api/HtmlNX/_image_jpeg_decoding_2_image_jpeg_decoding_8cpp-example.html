struct MetaData
{
// 0th TIFF IFD category metadata.
const char *maker; // Manufacturer name.
size_t makerSize; // Number of bytes of the manufacturer name including the terminating character.
const char *model; // Image capturing device name.
size_t modelSize; // Number of bytes of the image capturing device name including the terminating character.
bool hasOrientation; // Specifies whether there is information about image direction.
nn::image::ExifOrientation orientation; // Image direction.
const char *software; // Software name.
size_t softwareSize; // Number of bytes of the software name including the terminating character.
const char *dateTime; // Imaging date and time (20 bytes including the main single character).
// 0th Exif IFD category metadata.
const void *makerNote; // Manufacturer's notes.
size_t makerNoteSize; // Manufacturer's note byte count.
bool hasEffectiveDimension; // Specifies whether there is an effective image size.
nn::image::Dimension effectiveDim; // Effective image size.
const char *uniqueId; // Image unique ID (33 bytes including the terminating character).
// First Tiff IFD category metadata.
const void *thumbnail; // Thumbnail byte count.
size_t thumbnailSize; // Thumbnail image byte count.
};
void PrintMetaData(const MetaData &metaData)
{
LogIfNotNull(" - Manufacturer name.", metaData.maker);
LogIfNotNull(" - Image capturing device name.", metaData.model);
if (metaData.hasOrientation)
{
const char *orientationStr;
switch (metaData.orientation)
{
case nn::image::ExifOrientation_Normal:
orientationStr = "Upright.";
break;
case nn::image::ExifOrientation_FlipHorizontal:
orientationStr = "Inverted in the horizontal direction.";
break;
case nn::image::ExifOrientation_Rotate180:
orientationStr = "Rotated 180 degrees clockwise.";
break;
case nn::image::ExifOrientation_FlipVertical:
orientationStr = "Inverted in the vertical direction.";
break;
case nn::image::ExifOrientation_FlipTopRightToLeftBottom:
orientationStr = "Inverted around an angled line from the right down so that the right end of the top side goes to the bottom end of the left side.";
break;
case nn::image::ExifOrientation_Rotate270:
orientationStr = "Rotated 270 degrees clockwise.";
break;
case nn::image::ExifOrientation_FlipTopLeftToRightBottom:
orientationStr = "Inverted around an angled line from the left down so that the left end of the top side goes to the bottom end of the right side.";
break;
case nn::image::ExifOrientation_Rotate90:
orientationStr = "Rotated 90 degrees clockwise.";
break;
default:
NN_UNEXPECTED_DEFAULT;
}
LogIfNotNull(" - Image direction.", orientationStr);
}
LogIfNotNull(" - Software name.", metaData.software);
LogIfNotNull(" - Imaging date and time.", metaData.dateTime);
if (metaData.makerNote != nullptr)
{
NN_LOG(" - Manufacturer's note: %zu bytes.\n", metaData.makerNoteSize);
}
if (metaData.hasEffectiveDimension)
{
NN_LOG(" - Effective image size: Width %zu px, Height %zu px.\n", metaData.effectiveDim.width, metaData.effectiveDim.height);
}
LogIfNotNull(" - Unique ID.", metaData.uniqueId);
if (metaData.thumbnail != nullptr)
{
NN_LOG(" - Thumbnail: %zu bytes.\n", metaData.thumbnailSize);
}
}
void ExtractExifData(MetaData *pOutMetaData, const void *exifData, size_t exifSize)
{
// Exif information analysis class instance.
Buffer workBuf(nn::image::ExifExtractor::GetWorkBufferSize());
nn::image::ExifExtractor extractor(workBuf.GetDataPtr(), workBuf.GetDataSize());
/* ----------------------------------------------------------------------------------------
Analysis phase.
*/
// Set Exif data.
extractor.SetExifData(exifData, exifSize);
// Analyze.
nn::image::JpegStatus jpegStatus = extractor.Analyze();
switch (jpegStatus)
{
case nn::image::JpegStatus_WrongFormat:
NN_LOG("[This Exif information is corrupted.]\n ]\n");
// Processed as though no Exif information exists.
return;
default:
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok); // Unknown error code.
}
/* ----------------------------------------------------------------------------------------
Metadata expansion phase.
*/
// Get the 0th TIFF IFD information.
pOutMetaData->maker = extractor.ExtractMaker(&pOutMetaData->makerSize);
pOutMetaData->model = extractor.ExtractModel(&pOutMetaData->modelSize);
pOutMetaData->hasOrientation = extractor.ExtractOrientation(&pOutMetaData->orientation);
pOutMetaData->software = extractor.ExtractSoftware(&pOutMetaData->softwareSize);
// Per the Exif information specifications, the "imaging date and time" string is 20 bytes including the terminating character.
size_t dateTimeSize;
pOutMetaData->dateTime = extractor.ExtractDateTime(&dateTimeSize);
NN_ASSERT(pOutMetaData->dateTime == nullptr || dateTimeSize == 20);
// 0th Exif IFD.
pOutMetaData->makerNote = extractor.ExtractMakerNote(&pOutMetaData->makerNoteSize);
pOutMetaData->hasEffectiveDimension = extractor.ExtractEffectiveDimension(&pOutMetaData->effectiveDim);
// Per the Exif information specifications, the Unique ID string is 33 bytes including the terminating character.
size_t uniqueIdSize;
pOutMetaData->uniqueId = extractor.ExtractUniqueId(&uniqueIdSize);
NN_ASSERT(pOutMetaData->uniqueId == nullptr || uniqueIdSize == 33);
// First TIFF IFD.
pOutMetaData->thumbnail = extractor.ExtractThumbnail(&pOutMetaData->thumbnailSize);
}
bool DecodeJpegData(
nn::image::Dimension *pOutDim,
Buffer *pOutBuf,
const int alignment,
const FileData &jpegData)
{
/* ----------------------------------------------------------------------------------------
Decode Setting Phase
*/
// Decoder instance.
nn::image::JpegDecoder decoder(nn::util::pmr::NewDeleteResource());
// Set JPEG data.
decoder.SetImageData(jpegData.GetDataPtr(), jpegData.GetDataSize());
// [Input Example]
// Specify denominator (2) when decoding with a 1/2 scaling for both height and width. If not scaling, this specification may be omitted.
decoder.SetResolutionDenominator(2);
/* ----------------------------------------------------------------------------------------
JPEG Data Analysis Phase
*/
nn::image::JpegStatus jpegStatus = decoder.Analyze();
switch (jpegStatus)
{
case nn::image::JpegStatus_WrongFormat:
NN_LOG("[This JPEG data is corrupted.]\n ]\n");
return false;
case nn::image::JpegStatus_UnsupportedFormat:
NN_LOG("[This JPEG data is in an unsupported format. ]\n");
return false;
default:
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok); // Unknown error code.
}
// Get the image width and height.
*pOutDim = decoder.GetAnalyzedDimension();
NN_LOG("Output Image Size: Width %u px, Height %u px.\n", pOutDim->width, pOutDim->height);
/* ----------------------------------------------------------------------------------------
Exif Information Analysis Phase
(Skip this section if you are not handling Exif information.)
*/
// Get the location of Exif data.
size_t exifSize;
const void *exifData = decoder.GetAnalyzedExifData(&exifSize);
if (exifData != nullptr)
{
NN_LOG("EXIF information cannot be found.");
// Get Exif information content and display the content.
MetaData metaData;
ExtractExifData(&metaData, exifData, exifSize);
PrintMetaData(metaData);
if (metaData.thumbnail != nullptr)
{
// [Input Example]
// The decode target is replaced if there is a thumbnail image.
NN_LOG("-> Change the decode target to the thumbnail.");
decoder.SetImageData(metaData.thumbnail, metaData.thumbnailSize);
// [Input Example]
// Decode the thumbnail without shrinking. (â†‘ The value set to 2 previously is reset to 1.)
decoder.SetResolutionDenominator(1);
jpegStatus = decoder.Analyze();
switch (jpegStatus)
{
case nn::image::JpegStatus_WrongFormat:
NN_LOG("[This thumbnail is destroyed. ]\n");
return false;
case nn::image::JpegStatus_UnsupportedFormat:
NN_LOG("[This thumbnail is in an unsupported format. ]\n");
return false;
default:
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok); // Unknown error code.
}
// Get the image width and height.
*pOutDim = decoder.GetAnalyzedDimension();
NN_LOG("Output Image Size: Width %u px, Height %u px.\n", pOutDim->width, pOutDim->height);
}
}
/* ----------------------------------------------------------------------------------------
JPEG Data Expansion Phase
*/
NN_LOG("Decodes JPEG data.");
// Calculate the output size.
const size_t AlignedWidth = pOutDim->width % alignment == 0? pOutDim->width: pOutDim->width - pOutDim->width % alignment + alignment;
if (AlignedWidth * pOutDim->height * sizeof(nn::Bit32) > pOutBuf->GetDataSize())
{
NN_LOG(
"[The output data (%zu bytes) exceeds the buffer size (%zu bytes). ]\n",
AlignedWidth * pOutDim->height * sizeof(nn::Bit32),
pOutBuf->GetDataSize());
return false;
}
// Decode processing.
jpegStatus = decoder.Decode(
pOutBuf->GetDataPtr(),
pOutBuf->GetDataSize(),
alignment);
switch (jpegStatus)
{
case nn::image::JpegStatus_WrongFormat:
NN_LOG("[This JPEG data is corrupted.]\n ]\n");
return false;
case nn::image::JpegStatus_UnsupportedFormat:
NN_LOG("[This JPEG data is in an unsupported format. ]\n");
return false;
case nn::image::JpegStatus_OutOfMemory:
NN_LOG("[The work memory required for decoding is insufficient. Data in an unknown format may have been entered. ]\n");
return false;
default:
NN_ASSERT(jpegStatus == nn::image::JpegStatus_Ok); // Unknown error code.
}
// The decoder object is unnecessary at this time.
return true;
}
} // End of unnamed namespace.
extern "C" void nnMain()
{
// Initialize the file system.
nn::fs::SetAllocator(Allocate, Deallocate);
size_t cacheSize = 0;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::QueryMountRomCacheSize(&cacheSize));
char* mountRomCacheBuffer = new(std::nothrow) char[cacheSize];
NN_ABORT_UNLESS_NOT_NULL(mountRomCacheBuffer);
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::MountRom("Asset", mountRomCacheBuffer, cacheSize));
// Load the JPEG file.
// [Input Example]
// Set an appropriate JPEG file to load.
FileData jpegData("Asset:/ExampleInput.jpg");
// Allocate the output buffer.
// This sample handles up to 2,073,600 pixels (1080p).
Buffer pixBuf(1920 * 1080 * sizeof(nn::Bit32)); // The pixel count is multiplied by 4 to expand with 32-bit RGBA.
// JPEG decoding process.
// [Input Example]
// When decoding as a texture, require the width to be 4-byte aligned.
static const int AlignmentSize = 4;
nn::image::Dimension dim; // The size before alignment is stored in dim.
if (!DecodeJpegData(&dim, &pixBuf, AlignmentSize, jpegData))
{
NN_LOG("[Failed to decode JPEG data. ]\n");
}
else
{
// Using JPEG Data: Use within the application as RGBA data.
// Pixel data is stored in R, G, B, and A order regardless of endianness.
NN_LOG("The process ended normally.");
}
// FS finalization process.
nn::fs::Unmount("Asset");
delete[] mountRomCacheBuffer;
return;
}
