#pragma once
#include <GraphicsSystem.h>
class SceneBaseClass
{
public:
SceneBaseClass() NN_NOEXCEPT :
m_IsPrebuiltCommandBufferUsed(true),
m_IsScreenResolutionChanged(false)
{
}
virtual ~SceneBaseClass() NN_NOEXCEPT
{
}
virtual void Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT = 0;
virtual void DrawFrame() NN_NOEXCEPT = 0;
virtual void Finalize() NN_NOEXCEPT = 0;
virtual void TogglePrebuiltCommandBufferUse() NN_NOEXCEPT
{
m_IsPrebuiltCommandBufferUsed = !m_IsPrebuiltCommandBufferUsed;
}
virtual bool IsPrebuiltCommandBufferUsed() const NN_NOEXCEPT
{
return m_IsPrebuiltCommandBufferUsed;
}
virtual void NotifyScreenResolutionChange() NN_NOEXCEPT
{
m_IsScreenResolutionChanged = true;
}
protected:
bool m_IsPrebuiltCommandBufferUsed;
bool m_IsScreenResolutionChanged;
};
