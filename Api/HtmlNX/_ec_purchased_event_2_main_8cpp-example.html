// Do the two course IDs match?
bool IsMatchCourseId(const nn::ec::CourseId& courseId1, const nn::ec::CourseId& courseId2) NN_NOEXCEPT
{
bool isMatch = true;
for (int i = 0; i < nn::ec::MaxCourseIdStringLength; i++)
{
if (courseId1.value[i] != courseId2.value[i])
{
isMatch = false;
break;
}
}
return isMatch;
}

// Helper class for managing the expected values of the blink conditions.
class BlinkConditionHelper
{
public:
// The blink conditions being managed.
enum Condition
{
Condition_Visible,
Condition_Invisible,
};
// The default blink interval.
static const nn::TimeSpan DefaultInterval;
// Constructor.
explicit BlinkConditionHelper(const nn::TimeSpan& blinkInterval = DefaultInterval) NN_NOEXCEPT
: m_Interval(blinkInterval)
, m_Origin(nn::os::GetSystemTick())
, m_Condition(Condition_Visible)
{
}
// Reset the managed blink condition to the default.
void Reset(Condition condition = Condition_Visible) NN_NOEXCEPT
{
m_Condition = condition;
m_Origin = nn::os::GetSystemTick();
}
// Get the currently expected blink condition.
Condition GetCondition() const NN_NOEXCEPT
{
return m_Condition;
}
// Update the elapsed time of the blink condition.
// Return the currently expected blink condition in the return value.
Condition Update() NN_NOEXCEPT
{
const auto now = nn::os::GetSystemTick();
if ((now - m_Origin).ToTimeSpan() >= m_Interval)
{
m_Origin = now;
m_Condition = (Condition_Visible == m_Condition) ? Condition_Invisible : Condition_Visible;
}
return GetCondition();
}
private:
const nn::TimeSpan m_Interval;
nn::os::Tick m_Origin;
Condition m_Condition;
};
// The default blink interval of the characters shown when a purchase is made.
const nn::TimeSpan BlinkConditionHelper::DefaultInterval = nn::TimeSpan::FromMilliSeconds(500);
extern "C" void nnMain()
{
// The CourseId to use for the ShowShopSubscriptionList() function.
const constexpr nn::ec::CourseId sampleCourse1Id = {"aa5e5131d43ba0d4"};
const constexpr nn::ec::CourseId sampleCourse2Id = {"d6d733334c51f61a"};
ShopSelectorInterface shopSelector(sampleCourse1Id, sampleCourse2Id);
if (shopSelector.Initialize() == false)
{
// End the process if the initial settings are not successful.
return;
}
nv::SetGraphicsAllocator(NvAllocate, NvFree, NvReallocate, NULL);
nv::SetGraphicsDevtoolsAllocator(NvAllocate, NvFree, NvReallocate, NULL);
nv::InitializeGraphics(aligned_alloc(4096, GraphicsMemorySize), GraphicsMemorySize);
ApplicationHeap applicationHeap;
applicationHeap.Initialize(ApplicationHeapSize);
GraphicsSystem* pGraphicsSystem = new GraphicsSystem();
pGraphicsSystem->Initialize(
&applicationHeap, FrameBufferWidth, FrameBufferHeight);
FontSystem* pFontSystem = new FontSystem();
pFontSystem->Initialize(&applicationHeap, pGraphicsSystem);
nn::gfx::util::DebugFontTextWriter& textWriter =
pFontSystem->GetDebugFontTextWriter();
// Initialize the feature for detecting the purchase of items on the shop.
nn::ec::InitializePurchasedEvent();
// Get the event signaling the purchase of an item on the shop and then clear it to the nonsignaled state.
nn::ec::GetPurchasedEvent().Clear();
// Store the information for the purchased item as vector type data.
std::vector <nn::ec::PurchasedItemInfo> allPurchasedInfo;
BlinkConditionHelper blinkConditionHelper;
while (NN_STATIC_CONDITION(true))
{
// Input processing.
// If A is pressed, conduct a shop jump inside the shopSelector.Control() function.
const nns::hid::GamePad* pInput = shopSelector.Control();
if (pInput->HasAnyButtonsUp(nns::hid::Button::B::Mask) == true)
{
// If B is pressed, clear the purchase report information.
allPurchasedInfo.clear();
}
// Get the purchase report event information when the purchase report event information event is signaled.
if (true == nn::ec::GetPurchasedEvent().TryWait())
{
// Clear the system event to the nonsignaled state.
nn::ec::GetPurchasedEvent().Clear();
blinkConditionHelper.Reset();
// Get all purchase information.
// We suggest calling this function repeatedly until it returns false so you don't miss out on any information.
nn::ec::PurchasedItemInfo purchasedInfo;
while (true == nn::ec::PopPurchasedItemInfo(&purchasedInfo))
{
allPurchasedInfo.push_back(purchasedInfo);
}
}
// Display blinking characters when a purchase is made.
textWriter.SetScale(3.0, 3.0);
textWriter.SetTextColor(Color::Red);
if (BlinkConditionHelper::Condition_Visible == blinkConditionHelper.Update())
{
bool isSubscription1 = false;
bool isSubscription2 = false;
bool isConsumable = false;
for (const auto info : allPurchasedInfo)
{
if (false == isConsumable && info.type == nn::ec::PurchasedItemInfo::Type::Consumable)
{
textWriter.SetCursor(700, 100);
textWriter.Print("Purchased!");
isConsumable = true;
}
if ((false == isSubscription1)
&& (info.type == nn::ec::PurchasedItemInfo::Type::Subscription)
&& (IsMatchCourseId(info.GetCourseId(), sampleCourse1Id)))
{
textWriter.SetCursor(700, 160);
textWriter.Print("Purchased!");
isSubscription1 = true;
}
if ((false == isSubscription2)
&& (info.type == nn::ec::PurchasedItemInfo::Type::Subscription)
&& (IsMatchCourseId(info.GetCourseId(), sampleCourse2Id)))
{
textWriter.SetCursor(700, 220);
textWriter.Print("Purchased!");
isSubscription2 = true;
}
if (true == isConsumable && true == isSubscription1 && true == isSubscription2)
{
break;
}
}
}
// Display the list of information when a purchase is made.
textWriter.SetScale(2.0, 2.0);
textWriter.SetCursor(0, 500);
textWriter.SetTextColor(Color::White);
textWriter.Print(" ---------------------------- Purchased Event Info ----------------------------- \n");
textWriter.SetScale(1.5, 1.5);
int startIndex = allPurchasedInfo.size() < 5 ? 0 : allPurchasedInfo.size() - 5;
for (int i = startIndex; i < allPurchasedInfo.size(); i++)
{
nn::Bit32 value[2];
value[0] = static_cast<nn::Bit32>(allPurchasedInfo[i].nsaId.id >> 32);
value[1] = static_cast<nn::Bit32>(allPurchasedInfo[i].nsaId.id);
textWriter.SetCursorX(0);
textWriter.Print("%3d:",i);
textWriter.SetCursorX(250);
textWriter.Print("NSAID :0x%08x_%08x", value[0], value[1]);
switch (allPurchasedInfo[i].type)
{
case nn::ec::PurchasedItemInfo::Type::Subscription:
textWriter.SetCursorX(60);
textWriter.Print("Subscription");
textWriter.SetCursorX(700);
textWriter.Print("CourseId :%s \n", allPurchasedInfo[i].GetCourseId().value);
break;
case nn::ec::PurchasedItemInfo::Type::Consumable:
textWriter.SetCursorX(60);
textWriter.Print("Consumable \n");
break;
default:
textWriter.Print("\n");
break;
}
}
shopSelector.Draw(&textWriter);
pGraphicsSystem->BeginDraw();
pFontSystem->Draw();
pGraphicsSystem->EndDraw();
pGraphicsSystem->Synchronize(
nn::TimeSpan::FromNanoSeconds(1000 * 1000 * 1000 / FrameRate));
}
pFontSystem->Finalize();
delete pFontSystem;
pGraphicsSystem->Finalize();
delete pGraphicsSystem;
applicationHeap.Finalize();
}//NOLINT(impl/function_size)
