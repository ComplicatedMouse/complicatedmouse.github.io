NN_ALIGNAS(4096) char g_NgcWorkBuffer[nn::ngc::ProfanityFilter::WorkMemorySize];
void CountNumbersSample()
{
NN_LOG("\n### Sample count for characters that represent numbers.\n");
// Specify a string that includes characters representing numbers and execute the count processing.
const char* CountCheckText = "あ123い４５６う⑦ⅷⅨえ⓪お";
int num = nn::ngc::CountNumbers(CountCheckText);
// Display the count result.

}
void CheckProfanityWordsSample()
{
NN_LOG("\n### Profanity text check sample.\n");
// Create and initialize an instance of the <tt>ProfanityFilter</tt> class.
nn::ngc::ProfanityFilter filter;
nn::Result result = filter.Initialize(g_NgcWorkBuffer,nn::ngc::ProfanityFilter::WorkMemorySize);
if(result.IsFailure())
{
NN_LOG("Failed to initialize the ProfanityFilter class.\n");
return;
}
// Get the version information for the profanity word patterns installed on the system.
uint32_t nVersion = filter.GetContentVersion();
if(nVersion == 0)
{
NN_LOG("The pattern list version is invalid.\n");
return;
}
NN_LOG( "The pattern list version is %d.\n", nVersion );
// Assuming that text data can move between regions, reference the
// system's current region and language settings and check for profanity.
const char* ppWords[] =
{
"mario",
"@mario",
"badword",
"badwordj",
"sbadwordjl",
"testdisney.ne.jptest",
"クッパ",
"直アド",
"hoge直アドhoge",
"badwordc",
"bbwczh",
"tchinesenewsnett",
"badwordp",
"bbwpru",
"badwordc",
"bbwcom",
};
const int WordNum = sizeof(ppWords) / sizeof(ppWords[0]);
nn::Bit32 bResults[ WordNum ];
// Add Japanese and Chinese to the pattern list and execute the check processing.
nn::Bit32 patterns
= nn::ngc::ProfanityFilterPatternList_Japanese
| nn::ngc::ProfanityFilterPatternList_SimplifiedChinese;
result = filter.CheckProfanityWords( bResults, patterns, ppWords, WordNum );
if(result.IsFailure())
{
NN_LOG("Failed the check processing for profanity strings.\n");
return;
}
// Display the check results.
for ( int i = 0; i < WordNum; i++ )
{
if ( bResults[ i ] == 0 )
{
NN_LOG("\"%s\" does not have a match in the pattern list.\n", ppWords[i]);
}
else
{
NN_LOG("\"%s\" has a match in the pattern list.\n", ppWords[i]);
}
}
}
void MaskProfanityWordsSample()
{
NN_LOG("\n### String mask sample.\n");
// Create and initialize an instance of the <tt>ProfanityFilter</tt> class.
nn::ngc::ProfanityFilter filter;
nn::Result result = filter.Initialize(g_NgcWorkBuffer,nn::ngc::ProfanityFilter::WorkMemorySize);
if(result.IsFailure())
{
NN_LOG("Failed to initialize the ProfanityFilter class.\n");
return;
}
int nWordCount;
char checkText[] =
"こんにちは。はじめまして。\n"
"一緒にゲームをしていて あなたのことが(以下略)\n"
"もしよかったらhoge@foobar.netまでメールをくれませんか？\n"
"偽物のアドレスhoge@foobarはマスクされてはいけません！\n"
"直アドはマスクされます！";
char convertBuffer[nn::ngc::TextLengthMax];
std::memset(convertBuffer,0x00,sizeof(convertBuffer));
nn::util::Strlcpy(convertBuffer, checkText, sizeof(convertBuffer));
nn::Bit32 patterns
= nn::ngc::ProfanityFilterPatternList_Japanese;
NN_LOG("## Replace all characters in the profanity string with asterisks.\n");
// Set the masking method to <strong>Overwrite</strong>.
filter.SetMaskMode( nn::ngc::ProfanityFilter::MaskMode_OverWrite );
// Add Japanese to the pattern list that is being used and execute mask processing.
result = filter.MaskProfanityWordsInText( &nWordCount, convertBuffer ,patterns);
if(result.IsFailure())
{
NN_LOG("Failed the mask processing for profanity strings.\n");
return;
}
// Display the number of strings that were masked.
NN_LOG("The number of masked strings is %d.\n",nWordCount);
// Display the strings after mask processing.
NN_LOG("%s\n",convertBuffer);
NN_LOG("## Replace the profanity string with a single asterisk character.\n");
std::memset(convertBuffer,0x00,sizeof(convertBuffer));
nn::util::Strlcpy(convertBuffer, checkText, sizeof(convertBuffer));
// Set the mask method to "one asterisk character."
filter.SetMaskMode( nn::ngc::ProfanityFilter::MaskMode_ReplaceByOneCharacter );
// Add Japanese to the pattern list that is being used and execute mask processing.
result = filter.MaskProfanityWordsInText( &nWordCount, convertBuffer ,patterns);
if(result.IsFailure())
{
NN_LOG("Failed the mask processing for profanity strings.\n");
return;
}
// Display the number of strings that were masked.
NN_LOG("The number of masked strings is %d.\n",nWordCount);
// Display the strings after mask processing.
NN_LOG("%s\n",convertBuffer);
return;
}
void CheckAtSignSample()
{
NN_LOG("\n### At sign (@) check sample.\n");
// Create and initialize an instance of the <tt>ProfanityFilter</tt> class.
nn::ngc::ProfanityFilter filter;
nn::Result result = filter.Initialize(g_NgcWorkBuffer,nn::ngc::ProfanityFilter::WorkMemorySize);
if(result.IsFailure())
{
NN_LOG("Failed to initialize the ProfanityFilter class.\n");
return;
}
const char* ppWords[] =
{
"@mario",
"mario",
};
const int WordNum = sizeof(ppWords) / sizeof(ppWords[0]);
nn::Bit32 bResults[ WordNum ];
nn::Bit32 patterns
= 1 << nn::ngc::ProfanityFilterPatternList_Japanese;
// Configure the setting for skipping the at sign (@) check.
NN_LOG("## Enable skipping the at sign (@).\n");
filter.SkipAtSignCheck( nn::ngc::ProfanityFilter::SkipMode_SkipAtSign );
// Perform the at sign (@) check.
result = filter.CheckProfanityWords( bResults, patterns, ppWords, WordNum );
if(result.IsFailure())
{
NN_LOG("Failed the check processing for profanity strings.\n");
return;
}
// Display the check results.
for ( int i = 0; i < WordNum; i++ )
{
if ( bResults[ i ] == 0 )
{
NN_LOG("\"%s\" does not contain an at sign (@). \n", ppWords[i]);
}
else
{
NN_LOG("\"%s\" contains an at sign (@). \n", ppWords[i]);
}
}
NN_LOG("## Disable skipping the at sign (@).\n");
// Configure the setting for not skipping the at sign (@) check.
filter.SkipAtSignCheck( nn::ngc::ProfanityFilter::SkipMode_NotSkip );
// Perform the at sign (@) check.
result = filter.CheckProfanityWords( bResults, patterns, ppWords, WordNum );
if(result.IsFailure())
{
NN_LOG("Failed the check processing for profanity strings.\n");
return;
}
// Display the check results.
for ( int i = 0; i < WordNum; i++ )
{
if ( bResults[ i ] == 0 )
{
NN_LOG("\"%s\" does not contain an at sign (@). \n", ppWords[i]);
}
else
{
NN_LOG("\"%s\" contains an at sign (@). \n", ppWords[i]);
}
}
}
// Initialize the memory management system of the application.
extern "C" void nninitStartup()
{
NN_ABORT_UNLESS_RESULT_SUCCESS(
nn::os::SetMemoryHeapSize(MemoryHeapSize));
uintptr_t address = uintptr_t();
NN_ABORT_UNLESS_RESULT_SUCCESS(
nn::os::AllocateMemoryBlock(&address, MallocHeapSize));
nn::init::InitializeAllocator(
reinterpret_cast<void*>(address), MallocHeapSize);
}
// Entry point of the application.
extern "C" void nnMain()
{
// Initializes nn::fs.
nn::fs::SetAllocator(Allocate, Deallocate);
// Sample count processing for characters that represent numbers.
CountNumbersSample();
// Profanity text check sample.
CheckProfanityWordsSample();
// Profanity character mask sample.
MaskProfanityWordsSample();
// At sign (@) sample.
CheckAtSignSample();
}
