struct GuiState
{
bool isExitRequested;
bool isShowDemoWindow;
};
void CalculateCallback(nns::gfx::GraphicsFramework* pGraphicsFramework, void* pUserData)
{
NN_UNUSED(pGraphicsFramework);
GuiState* pGuiState = reinterpret_cast<GuiState*>(pUserData);
nns::dbgui::NewFrame();
ImGui::Text("Hello, world!");
if (ImGui::Button("Demo Window"))
{
pGuiState->isShowDemoWindow = !pGuiState->isShowDemoWindow;
}
if (ImGui::Button("Exit"))
{
pGuiState->isExitRequested = true;
}
if (pGuiState->isShowDemoWindow)
{
// Normally user code does not need or want to call it because positions are saved in and INI file anyway.
// Here we just want to make the demo initial state a bit more friendly!
ImGui::SetNextWindowPos(ImVec2(650, 20), ImGuiCond_FirstUseEver);
ImGui::ShowDemoWindow(&pGuiState->isShowDemoWindow);
}
}
void MakeCommandCallback(nns::gfx::GraphicsFramework* pGfw, int bufferIndex, void* pUserData)
{
NN_UNUSED(pUserData);
pGfw->BeginFrame(bufferIndex);
nn::gfx::CommandBuffer* pCommandBuffer = pGfw->GetRootCommandBuffer(bufferIndex);
nn::gfx::ColorTargetView* pScreenTarget = pGfw->GetColorTargetView();
pCommandBuffer->ClearColor(pScreenTarget, 0.45f, 0.55f, 0.60f, 1.00f, nullptr);
pCommandBuffer->SetRenderTargets(1, &pScreenTarget, nullptr);
pCommandBuffer->SetViewportScissorState(pGfw->GetViewportScissorState());
nns::dbgui::Render(pCommandBuffer);
pGfw->EndFrame(bufferIndex);
}
extern "C" void nnMain()
{
const size_t GraphicsSystemMemorySize = 8 * 1024 * 1024;
nns::gfx::GraphicsFramework::InitializeGraphicsSystem(GraphicsSystemMemorySize);
nn::hid::InitializeDebugPad();
nn::hid::InitializeKeyboard();
nn::hid::InitializeMouse();
#if defined(NN_BUILD_TARGET_PLATFORM_OS_NN)
const int DisplayWidth = 1920;
const int DisplayHeight = 1080;
#else
const int DisplayWidth = 1280;
const int DisplayHeight = 720;
#endif // defined(NN_BUILD_TARGET_PLATFORM_OS_NN)
const int ScreenBpp = 4;
const int BufferCount = 2;
nns::gfx::GraphicsFramework::FrameworkInfo gfwInfo;
gfwInfo.SetDefault();
gfwInfo.SetDisplayWidth(DisplayWidth);
gfwInfo.SetDisplayHeight(DisplayHeight);
gfwInfo.SetBufferCount(BufferCount);
gfwInfo.SetSwapChainBufferCount(BufferCount);
gfwInfo.SetDepthStencilBufferEnabled(false);
gfwInfo.SetRootCommandBufferCommandMemorySize(1 * 1024 * 1024);
gfwInfo.SetMemoryPoolSize(
nns::gfx::GraphicsFramework::MemoryPoolType_CommandBuffer,
4 * 1024 * 1024);
gfwInfo.SetMemoryPoolSize(
nns::gfx::GraphicsFramework::MemoryPoolType_RenderTarget,
nn::util::align_up(DisplayWidth * DisplayHeight * ScreenBpp, 4096) * (BufferCount * 2));
gfwInfo.SetMemoryPoolSize(
nns::gfx::GraphicsFramework::MemoryPoolType_ConstantBuffer,
2 * 1024 * 1024);
gfwInfo.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_BufferView, 16);
gfwInfo.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_TextureView, 1024);
gfwInfo.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_Sampler, 1024);
nns::gfx::DebugGraphicsFramework gfw;
gfw.Initialize(gfwInfo);
nns::dbgui::GraphicsFrameworkInteroperabilityData graphicsFrameworkInteroperabilityData;
nns::dbgui::InitializeWithGraphicsFramework(&graphicsFrameworkInteroperabilityData, &gfw);
GuiState guiState = { false, false };
gfw.SetCalculateCallback(&CalculateCallback, &guiState);
gfw.SetMakeCommandCallback(&MakeCommandCallback, &guiState);
while (!guiState.isExitRequested)
{
nn::hid::KeyboardState keyboardState;
nn::hid::GetKeyboardState(&keyboardState);
if (keyboardState.attributes.Test<nn::hid::KeyboardAttribute::IsConnected>())
{
nns::dbgui::SetKeyboardState(keyboardState);
nns::dbgui::AddInputCharactersFromKeyboardState(keyboardState);
}
nn::hid::DebugPadState debugPadState;
nn::hid::GetDebugPadState(&debugPadState);
if (debugPadState.attributes.Test<nn::hid::DebugPadAttribute::IsConnected>())
{
nns::dbgui::SetDebugPadState(debugPadState.buttons, debugPadState.analogStickL);
}
nn::hid::MouseState mouseState;
nn::hid::GetMouseState(&mouseState);
if (mouseState.attributes.Test<nn::hid::MouseAttribute::IsConnected>())
{
nns::dbgui::SetMouseState(mouseState);
}
gfw.ProcessFrame();
}
nns::dbgui::FinalizeWithGraphicsFramework(&graphicsFrameworkInteroperabilityData, &gfw);
gfw.Finalize();
}
