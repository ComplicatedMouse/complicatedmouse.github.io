nn::g3d::ModelObj g_ModelObj; // The model instance to which the shape animation will be applied.
// Initialize the model for shape animation.
void InitializeModel(nns::gfx::GraphicsFramework* pGfxFramework) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pGfxFramework);
nn::gfx::Device* pDevice = pGfxFramework->GetDevice();
// Get the model resource.
nn::g3d::ResModel* pResModel = FindResModelFromHolder(g_pModelName);
NN_ASSERT_NOT_NULL(pResModel);
// Construct RenderModel to simplify the rendering process.
nns::g3d::RenderModel* pRenderModel = SetupRenderModel(pDevice, pResModel);
NN_ASSERT_NOT_NULL(pRenderModel);
g_Holder.renderModels.push_back(pRenderModel);
// Put the vertex attribute required for shape animation in a dynamic vertex buffer.
pResModel->ActivateDynamicVertexAttrForShapeAnim();
// Construct the model instance.
nn::g3d::ModelObj* pModelObj = &g_ModelObj;
{
nn::g3d::ModelObj::Builder builder(pResModel);
{
builder.ShapeBufferingCount(g_BufferingCount);
builder.SkeletonBufferingCount(g_BufferingCount);
builder.MaterialBufferingCount(g_BufferingCount);
}
builder.CalculateMemorySize();
size_t workMemorySize = builder.GetWorkMemorySize();
void* pWorkMemory = g3ddemo::AllocateMemory(workMemorySize, nn::g3d::ModelObj::Alignment_Buffer);
NN_ASSERT_NOT_NULL(pWorkMemory);
bool isSucceeded = builder.Build(pModelObj, pWorkMemory, workMemorySize);
NN_ASSERT(isSucceeded);
NN_UNUSED(isSucceeded);
}
NN_ASSERT_NOT_NULL(pModelObj);
// Construct the block buffer.
{
size_t size = pModelObj->CalculateBlockBufferSize(pDevice);
size_t alignment = pModelObj->GetBlockBufferAlignment(pDevice);
// Allocate the memory pool area specified by CpuCached_GpuCached to efficiently read and
// write to the buffer when the CPU dynamically updates the vertex buffer.
const nns::gfx::GraphicsFramework::MemoryPoolType poolType = nns::gfx::GraphicsFramework::MemoryPoolType_Data;
ptrdiff_t memoryOffset = pGfxFramework->AllocatePoolMemory(poolType, size, alignment);
bool isSucceeded = pModelObj->SetupBlockBuffer(pDevice, pGfxFramework->GetMemoryPool(poolType), memoryOffset, size);
NN_ASSERT(isSucceeded);
NN_UNUSED(isSucceeded);
}
// In this demo, the following information does not need to be updated in every frame so the update process is performed only once.
// - Skeleton
// - Materials
{
pModelObj->CalculateWorld(nn::util::Matrix4x3f::Identity());
for (int bufferIndex = 0; bufferIndex < g_BufferingCount; ++bufferIndex)
{
pModelObj->CalculateSkeleton(bufferIndex);
pModelObj->CalculateMaterial(bufferIndex);
}
}
g_Holder.modelObjs.push_back(pModelObj);
// Enable or disable automatically applying the shape animation calculation result.
if (g_IsGpuCalculationEnabled)
{
pModelObj->SetShapeAnimCalculationDisabled();
}
else
{
pModelObj->SetShapeAnimCalculationEnabled();
}
// Construct RenderModelObj to simplify the rendering process.
nns::g3d::RenderModelObj* pRenderModelObj = SetupRenderModelObj(pDevice, pModelObj, pRenderModel);
NN_ASSERT_NOT_NULL(pRenderModel);
g_Holder.renderModelObjs.push_back(pRenderModelObj);
}
nn::g3d::ShapeAnimObj g_ShapeAnimObj; // The shape animation instance.
// Initialize the shape animation.
void InitializeShapeAnimObj() NN_NOEXCEPT
{
// Get the resource.
const nn::g3d::ResModel* pResModel = g_ModelObj.GetResource();
nn::g3d::ResShapeAnim* pResShapeAnim = FindResShapeAnimFromHolder(g_pShapeAnimName);
NN_ASSERT_NOT_NULL(pResShapeAnim);
// Construct the shape animation instance.
nn::g3d::ShapeAnimObj* pShapeAnimObj = &g_ShapeAnimObj;
{
nn::g3d::ShapeAnimObj::Builder builder;
builder.Reserve(pResModel);
builder.Reserve(pResShapeAnim);
builder.CalculateMemorySize();
size_t workMemorySize = builder.GetWorkMemorySize();
void* pWorkMemory = g3ddemo::AllocateMemory(workMemorySize, nn::g3d::ShapeAnimObj::Alignment_Buffer);
NN_ASSERT_NOT_NULL(pWorkMemory);
bool isSucceeded = builder.Build(pShapeAnimObj, pWorkMemory, workMemorySize);
NN_ASSERT(isSucceeded);
NN_UNUSED(isSucceeded);
}
NN_ASSERT_NOT_NULL(pShapeAnimObj);
// Bind the shape animation to the model.
{
NN_ASSERT(pShapeAnimObj->IsAcceptable(pResShapeAnim));
NN_ASSERT(pShapeAnimObj->IsAcceptable(pResModel));
pShapeAnimObj->SetResource(pResShapeAnim);
pShapeAnimObj->Bind(&g_ModelObj);
}
}
// Finalize the shape animation.
void FinalizeShapeAnimObj() NN_NOEXCEPT
{
void* pWorkMemory = g_ShapeAnimObj.GetBufferPtr();
g3ddemo::FreeMemory(pWorkMemory);
}

{

g_ShapeAnimObj.Calculate();
// Apply the calculation result to the model.
g_ShapeAnimObj.ApplyTo(&g_ModelObj);
// Update the frame.
g_ShapeAnimObj.GetFrameCtrl().UpdateFrame();
}

// The buffer storing the blend weight value.
struct BlendWeightBuffer
{
nn::gfx::Buffer buffer; // The buffer.
ptrdiff_t offset; // The offset from the start of the memory pool.
size_t stride; // The stride of the buffer.
};
BlendWeightBuffer g_BlendWeightBuffer;
// Initialize the buffer storing the blend weight.
void InitializeBlendWeightBuffer(nns::gfx::GraphicsFramework* pGfxFramework) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pGfxFramework);
nn::gfx::Device* pDevice = pGfxFramework->GetDevice();
NN_ASSERT_EQUAL(g_ModelObj.GetShapeCount(), 1);
int keyShapeCount = g_ModelObj.GetShape(0)->GetKeyShapeCount();
nn::gfx::Buffer::InfoType info;
info.SetDefault();
info.SetGpuAccessFlags(nn::gfx::GpuAccess_UnorderedAccessBuffer);
size_t alignment = nn::gfx::Buffer::GetBufferAlignment(pDevice, info);
size_t stride = nn::util::align_up(sizeof(float) * keyShapeCount, alignment);
size_t bufferSize = stride * g_BufferingCount;
info.SetSize(bufferSize);
nns::gfx::GraphicsFramework::MemoryPoolType poolType = nns::gfx::GraphicsFramework::MemoryPoolType_ConstantBuffer;
ptrdiff_t offset = pGfxFramework->AllocatePoolMemory(poolType, bufferSize, alignment);
NN_ASSERT_NOT_EQUAL(offset, nn::gfx::util::MemoryPoolAllocator::InvalidOffset);
g_BlendWeightBuffer.buffer.Initialize(pDevice, info, pGfxFramework->GetMemoryPool(poolType), offset, bufferSize);
g_BlendWeightBuffer.stride = stride;
g_BlendWeightBuffer.offset = offset;
}
// Finalize the buffer storing the blend weight.
void FinalizeBlendWeightBuffer(nns::gfx::GraphicsFramework* pGfxFramework) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pGfxFramework);
nn::gfx::Device* pDevice = pGfxFramework->GetDevice();
nns::gfx::GraphicsFramework::MemoryPoolType poolType = nns::gfx::GraphicsFramework::MemoryPoolType_ConstantBuffer;
g_BlendWeightBuffer.buffer.Finalize(pDevice);
pGfxFramework->FreePoolMemory(poolType, g_BlendWeightBuffer.offset);
}
// Update the buffer storing the blend weight.
void UpdateBlendWeightBuffer(int bufferIndex) NN_NOEXCEPT
{
NN_ASSERT_RANGE(bufferIndex, 0, g_BufferingCount);
const nn::g3d::ShapeObj* pShapeObj = g_ModelObj.FindShape(g_pShapeName);
NN_ASSERT_NOT_NULL(pShapeObj);
int keyShapeCount = pShapeObj->GetKeyShapeCount();
// | keyshape0 / buffer0 | keyShape1 / buffer0 | : | keyShape0 / buffer1 | keyShape1 / buffer1 | :
ptrdiff_t offset = g_BlendWeightBuffer.stride * bufferIndex;
nn::util::BytePtr pMapedWeightArray(g_BlendWeightBuffer.buffer.Map());
pMapedWeightArray.Advance(offset);
float* pWeight = pMapedWeightArray.Get<float>();
for (int keyShapeIndex = 0; keyShapeIndex < keyShapeCount; ++keyShapeIndex)
{
*pWeight = pShapeObj->GetBlendWeight(keyShapeIndex);
++pWeight;
}
g_BlendWeightBuffer.buffer.FlushMappedRange(offset, sizeof(float) * keyShapeCount);
g_BlendWeightBuffer.buffer.Unmap();
}
nn::gfx::VertexState g_BlendShapeVertexState; // The vertex state to use when blending shapes in the GPU.
// Initialize the vertex state to use when blending.
void InitializeBlendShapeVertexState(nn::gfx::Device* pDevice) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pDevice);
NN_ASSERT(nn::gfx::IsInitialized(*pDevice));
const nn::g3d::ResShadingModel* pResShadingModel = FindResShadingModelFromHolder(g_pBlendShapeShadingModelName);
NN_ASSERT_NOT_NULL(pResShadingModel);
const nn::g3d::ShapeObj* pShapeObj = g_ModelObj.FindShape(g_pShapeName);
NN_ASSERT_NOT_NULL(pShapeObj);
const int vertexAttrCount = pResShadingModel->GetAttrCount();
const int vertexBufferCount = pShapeObj->GetVertexBufferCount();
// Allocate a temporary memory space necessary for constructing the vertex state.
nn::util::MemorySplitter::MemoryBlock attributeStateBlock, bufferStateBlock;
void* pWorkMemory = nullptr;
{
attributeStateBlock.SetSizeBy<nn::gfx::VertexAttributeStateInfo>(1, vertexAttrCount);
attributeStateBlock.SetAlignment(NN_ALIGNOF(nn::gfx::VertexAttributeStateInfo));
bufferStateBlock.SetSizeBy<nn::gfx::VertexBufferStateInfo>(1, vertexBufferCount);
bufferStateBlock.SetAlignment(NN_ALIGNOF(nn::gfx::VertexBufferStateInfo));
nn::util::MemorySplitter splitter;
splitter.Append(&attributeStateBlock);
splitter.Append(&bufferStateBlock);
pWorkMemory = g3ddemo::AllocateMemory(splitter.GetSize(), splitter.GetAlignment());
}
NN_ASSERT_NOT_NULL(pWorkMemory);
// Create the vertex state.
nn::gfx::VertexState::InfoType vertexStateInfo;
vertexStateInfo.SetDefault();
// Set the vertex attribute state.
{
nn::gfx::VertexAttributeStateInfo* pVertexAttributeStateInfo = attributeStateBlock.Get<nn::gfx::VertexAttributeStateInfo>(pWorkMemory);
int validVertexAttrCount = 0;
for (int vertexAttrIndex = 0; vertexAttrIndex < vertexAttrCount; ++vertexAttrIndex)
{
// Get the ID written in the shader annotation.
const char* attrId = pResShadingModel->GetAttrName(vertexAttrIndex);
// Normally, the information stored in the shader assignment is used to bind the vertex attribute defined in the shader with the vertex attribute in the model.
// But because this shader is not a material shader, we make the ID name the same as the vertex attribute name and configure this directly.
const nn::g3d::ResVertexAttr* pResVertexAttr = pShapeObj->FindResVertexAttr(attrId);
if (!pResVertexAttr)
{
continue;
}
const nn::g3d::ResAttrVar* pResAttrVar = pResShadingModel->GetAttr(vertexAttrIndex);
NN_ASSERT_NOT_NULL(pResAttrVar);
// Set the vertex attribute information.
pVertexAttributeStateInfo[validVertexAttrCount].SetDefault();
pVertexAttributeStateInfo[validVertexAttrCount].SetBufferIndex(pResVertexAttr->GetBufferIndex());
pVertexAttributeStateInfo[validVertexAttrCount].SetFormat(pResVertexAttr->GetFormat());
pVertexAttributeStateInfo[validVertexAttrCount].SetOffset(pResVertexAttr->GetOffset());
pVertexAttributeStateInfo[validVertexAttrCount].SetShaderSlot(pResAttrVar->GetLocation());
++validVertexAttrCount;
}
vertexStateInfo.SetVertexAttributeStateInfoArray(pVertexAttributeStateInfo, validVertexAttrCount);
}
// Set the vertex buffer state.
{
// Set all vertex buffers that are inside ShapeObj.
const nn::g3d::ResVertex* pResVertex = pShapeObj->GetResVertex();
nn::gfx::VertexBufferStateInfo* pVertexBufferStateInfo = bufferStateBlock.Get<nn::gfx::VertexBufferStateInfo>(pWorkMemory);
for (int vertexBufferIndex = 0; vertexBufferIndex < vertexBufferCount; ++vertexBufferIndex)
{
pVertexBufferStateInfo[vertexBufferIndex].SetDefault();
pVertexBufferStateInfo[vertexBufferIndex].SetStride(pResVertex->GetVertexBufferStride(vertexBufferIndex));
}
vertexStateInfo.SetVertexBufferStateInfoArray(pVertexBufferStateInfo, vertexBufferCount);
}
size_t memorySize = nn::gfx::VertexState::GetRequiredMemorySize(vertexStateInfo);
void* pMemory = g3ddemo::AllocateMemory(memorySize, nn::gfx::VertexState::RequiredMemoryInfo_Alignment);
NN_ASSERT_NOT_NULL(pMemory);
g_BlendShapeVertexState.SetMemory(pMemory, memorySize);
g_BlendShapeVertexState.Initialize(pDevice, vertexStateInfo);
g3ddemo::FreeMemory(pWorkMemory);
}
// Destroy the vertex state used for blending.
void FinalizeBlendShapeVertexState(nn::gfx::Device* pDevice) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pDevice);
NN_ASSERT(nn::gfx::IsInitialized(*pDevice));
g_BlendShapeVertexState.Finalize(pDevice);
void* pWorkMemory = g_BlendShapeVertexState.GetMemory();
g3ddemo::FreeMemory(pWorkMemory);
}
// Create a command for blending shapes.
void MakeBlendShapeCommand(nn::gfx::CommandBuffer* pCommandBuffer, int bufferIndex) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pCommandBuffer);
NN_ASSERT(nn::gfx::IsInitialized(*pCommandBuffer));
NN_ASSERT_RANGE(bufferIndex, 0, g_BufferingCount);
nn::g3d::ResShadingModel* pBlendShapeShader = FindResShadingModelFromHolder(g_pBlendShapeShadingModelName);
NN_ASSERT_NOT_NULL(pBlendShapeShader);
// Get the shader program.
nn::g3d::ResShaderProgram* pResShaderProgram = nullptr;
{
int programIndex = pBlendShapeShader->GetDefaultProgramIndex();
pResShaderProgram = pBlendShapeShader->GetShaderProgram(programIndex);
}
NN_ASSERT_NOT_NULL(pResShaderProgram);
NN_ASSERT(pResShaderProgram->IsInitialized());
// Set the shader.
pCommandBuffer->SetShader(pResShaderProgram->GetShader(), nn::gfx::ShaderStageBit_Vertex);
const nn::g3d::ShapeObj* pShapeObj = g_ModelObj.FindShape(g_pShapeName);
NN_ASSERT_NOT_NULL(pShapeObj);
//Set the vertex buffer.
{
const nn::g3d::ResVertex* pResVertex = pShapeObj->GetResVertex();
for (int vertexBufferIndex = 0, vertexBufferCount = pShapeObj->GetVertexBufferCount(); vertexBufferIndex < vertexBufferCount; ++vertexBufferIndex)
{
const nn::gfx::Buffer* pVertexBuffer = pResVertex->GetVertexBuffer(vertexBufferIndex);
const nn::gfx::Buffer::InfoType* pBufferInfo = pResVertex->GetVertexBufferInfo(vertexBufferIndex);
nn::gfx::GpuAddress gpuAddress;
pVertexBuffer->GetGpuAddress(&gpuAddress);
pCommandBuffer->SetVertexBuffer(vertexBufferIndex, gpuAddress, pResVertex->GetVertexBufferStride(vertexBufferIndex), pBufferInfo->GetSize());
}
}
// Set the vertex state.
NN_ASSERT(nn::gfx::IsInitialized(g_BlendShapeVertexState));
pCommandBuffer->SetVertexState(&g_BlendShapeVertexState);
// Set the dynamic vertex buffer as the shader storage block.
{
// Get the dynamic vertex buffer and the buffer information.
const nn::gfx::Buffer* pDynamicVertexBuffer = nullptr;
const nn::gfx::BufferInfo* pBufferInfo = nullptr;
{
const int vertexBufferIndex = 0;
pDynamicVertexBuffer = pShapeObj->GetDynamicVertexBuffer(vertexBufferIndex, bufferIndex);
pBufferInfo = pShapeObj->GetVertexBufferInfo(vertexBufferIndex);
}
NN_ASSERT_NOT_NULL(pDynamicVertexBuffer);
NN_ASSERT_NOT_NULL(pBufferInfo);
// Bind the dynamic vertex buffer.
nn::gfx::GpuAddress gpuAddress;
pDynamicVertexBuffer->GetGpuAddress(&gpuAddress);
const char* dynamicVertexBufferId = "dynamicVertexBuffer";
SetUnorderedAccessBuffer(pCommandBuffer, dynamicVertexBufferId, gpuAddress, pBufferInfo->GetSize(), pBlendShapeShader, pResShaderProgram);
}
// Bind the blend weight buffer.
{
int keyShapeCount = pShapeObj->GetKeyShapeCount();
nn::gfx::GpuAddress gpuAddress;
g_BlendWeightBuffer.buffer.GetGpuAddress(&gpuAddress);
gpuAddress.Offset(g_BlendWeightBuffer.stride * bufferIndex);
const char* id = "weight";
SetUnorderedAccessBuffer(pCommandBuffer, id, gpuAddress, sizeof(float) * keyShapeCount, pBlendShapeShader, pResShaderProgram);
}
// Write the draw command.
pShapeObj->GetResMesh()->Draw(pCommandBuffer);
}

// Initialize the environment model.
void InitializeEnvModel(nns::gfx::GraphicsFramework* pGfxFramework) NN_NOEXCEPT
{
NN_ASSERT_NOT_NULL(pGfxFramework);
nn::gfx::Device* pDevice = pGfxFramework->GetDevice();
// Get the model resource.
const char* pModelName = "env";
nn::g3d::ResModel* pResModel = FindResModelFromHolder(pModelName);
NN_ASSERT_NOT_NULL(pResModel);
// Bind the shader parameter to the material.
const int envMaterialIndex = 0;
{
NN_ASSERT_EQUAL(1, pResModel->GetMaterialCount());
nn::g3d::ResMaterial* pResMaterial = pResModel->GetMaterial(envMaterialIndex);
const nn::g3d::ResShaderAssign* pResShaderAssign = pResMaterial->GetShaderAssign();
const nn::g3d::ResShaderArchive* pShaderArchive = FindResShaderArchiveFromHolder(pResShaderAssign->GetShaderArchiveName());
const nn::g3d::ResShadingModel* pResShadingModel = pShaderArchive->FindShadingModel(pResShaderAssign->GetShadingModelName());
nn::g3d::ShaderUtility::BindShaderParam(pResMaterial, pResShadingModel);
}
// Construct the model instance.
nn::g3d::ModelObj* pEnvModel = &g_EnvModel;
{
nn::g3d::ModelObj::Builder builder(pResModel);
builder.MaterialBufferingCount(1);
builder.CalculateMemorySize();
size_t workMemorySize = builder.GetWorkMemorySize();
void* pWorkMemory = g3ddemo::AllocateMemory(workMemorySize, nn::g3d::ModelObj::Alignment_Buffer);
bool isSucceeded = builder.Build(pEnvModel, pWorkMemory, workMemorySize);
NN_ASSERT(isSucceeded);
NN_UNUSED(isSucceeded);
}
NN_ASSERT_NOT_NULL(pEnvModel);
// Construct the block buffer.
{
size_t size = pEnvModel->CalculateBlockBufferSize(pDevice);
size_t alignment = pEnvModel->GetBlockBufferAlignment(pDevice);
const nns::gfx::GraphicsFramework::MemoryPoolType poolType = nns::gfx::GraphicsFramework::MemoryPoolType_Data;
ptrdiff_t memoryOffset = pGfxFramework->AllocatePoolMemory(poolType, size, alignment);
bool isSucceeded = pEnvModel->SetupBlockBuffer(pDevice, pGfxFramework->GetMemoryPool(poolType), memoryOffset, size);
NN_ASSERT(isSucceeded);
NN_UNUSED(isSucceeded);
}
g_Holder.modelObjs.push_back(pEnvModel);
// In this demo, the material does not need to be updated in every frame so the update process is performed only once.
pEnvModel->CalculateMaterial(0);
// Bind the environment model's material uniform block to the model getting the shape animation.
const nn::gfx::Buffer* buffers[g_BufferingCount];
for (int bufferIndex = 0; bufferIndex < g_BufferingCount; ++bufferIndex)
{
buffers[bufferIndex] = pEnvModel->GetMaterial(envMaterialIndex)->GetMaterialBlock(0);
}
nns::g3d::RenderModelObj* pRenderModelObj = FindRenderModelObjFromHolder(g_pModelName);
NN_ASSERT_NOT_NULL(pRenderModelObj);
pRenderModelObj->BindUniformBlock("env", buffers, g_BufferingCount);
}
// Initialize the view.
void InitializeRenderView(nn::gfx::Device* pDevice) NN_NOEXCEPT
{
g_RenderView.Initialize(pDevice, 1, g_BufferingCount);
// The projection matrix does not need to be updated in every frame, so perform the update process only once.
nn::util::Matrix4x4fType matrix;
MatrixPerspectiveFieldOfViewRightHanded(&matrix,
nn::util::DegreeToRadian(45.0f), 16.0f / 9.0f, 10.0f, 40000.0f);
g_RenderView.SetProjectionMtx(0, matrix);
// Set the uniform block for the view.
{

void SetRenderStateCommand(nn::gfx::CommandBuffer* pCommandBuffer, nns::gfx::GraphicsFramework* pGfxFramework) NN_NOEXCEPT
{

NN_ASSERT(nn::gfx::IsInitialized(*pCommandBuffer));
NN_ASSERT_NOT_NULL(pGfxFramework);

// Clear the depth stencil.
nn::gfx::DepthStencilView* pDepth = pGfxFramework->GetDepthStencilView();

// Set a viewport scissor state.

}
void MakeCommand(nns::gfx::GraphicsFramework* pGfxFramework, int bufferIndex, void* pUserData)
{
int uniformBufferIndex = *reinterpret_cast<int*>(pUserData);
// Generate the commands.
pGfxFramework->BeginFrame(bufferIndex);
{
nn::gfx::CommandBuffer* pCommandBuffer = pGfxFramework->GetRootCommandBuffer(bufferIndex);
// Set the color target and state.
SetRenderStateCommand(pCommandBuffer, pGfxFramework);
#ifdef APPLY_ANIMATION_MODE_GPU
// Execute the commands to write the blend result in the dynamic vertex buffer.
if (g_IsGpuCalculationEnabled)
{
// Execute the blending process.
{
const MeasureInfo& measureInfo = g_MeasureInfoArray[SectionType_BlendShapeCommand];
NN_PERF_SET_COLOR_GPU(measureInfo.color);
NN_PERF_SET_COLOR(measureInfo.color);
NN_PERF_BEGIN_MEASURE_NAME_GPU(pCommandBuffer,measureInfo.name);
NN_PERF_AUTO_MEASURE_NAME(measureInfo.name);
MakeBlendShapeCommand(pCommandBuffer, bufferIndex);
NN_PERF_END_MEASURE_GPU(pCommandBuffer);
}
{
const MeasureInfo& measureInfo = g_MeasureInfoArray[SectionType_BarrierCommand];
NN_PERF_SET_COLOR_GPU(measureInfo.color);
NN_PERF_SET_COLOR(measureInfo.color);
NN_PERF_BEGIN_MEASURE_NAME_GPU(pCommandBuffer, measureInfo.name);
NN_PERF_AUTO_MEASURE_NAME(measureInfo.name);
// The dynamic vertex buffer is set as an SSBO and the values are overwritten, so wait for the writing process to complete.
pCommandBuffer->FlushMemory(nn::gfx::GpuAccess_UnorderedAccessBuffer);
pCommandBuffer->InvalidateMemory(nn::gfx::GpuAccess_VertexBuffer | nn::gfx::GpuAccess_ShaderCode);
NN_PERF_END_MEASURE_GPU(pCommandBuffer);
}
}
#endif // defined APPLY_ANIMATION_GPU
// Generate model display commands.
for (nns::g3d::RenderModelObj* pRenderModelObj : g_Holder.renderModelObjs)
{
pRenderModelObj->Draw(pCommandBuffer, 0, uniformBufferIndex);
}
// Generate the commands to render the menu.
nns::dbgui::Render(pCommandBuffer);
}
pGfxFramework->EndFrame(bufferIndex);
}

int ShapeAnimationMain()
{
nns::gfx::DebugGraphicsFramework* pGfxFramework = g3ddemo::GetGfxFramework();
// Initialize the load measurement feature.
{
pGfxFramework->InitializePerf();
#if defined( NN_BUILD_CONFIG_OS_WIN )
NN_PERF_SET_GET_CORE_NUMBER_FUNCTION(g3ddemo::GetFixedCoreNumber);
#endif
// Assign the current main thread to core 0.
nn::os::SetThreadCoreMask(nn::os::GetCurrentThread(), 0, 1);
}
// Initialization.
Initialize(pGfxFramework);
int uniformBufferIndex = 0;
pGfxFramework->SetCalculateCallback(Calculate, &uniformBufferIndex);
pGfxFramework->SetMakeCommandCallback(MakeCommand, &uniformBufferIndex);
pGfxFramework->SetFrameworkMode(nns::gfx::GraphicsFramework::FrameworkMode_DeferredSubmission);
pGfxFramework->ResetFrame();
// Main loop.
while (NN_STATIC_CONDITION(1))
{
g3ddemo::UpdateController();
if (g3ddemo::IsButtonTriggered<nns::hid::Button::Start>())
{
break;
}
// Execute frame processing.
pGfxFramework->ProcessFrame();
uniformBufferIndex = (uniformBufferIndex + 1) % g_BufferingCount;
}
pGfxFramework->QueueFinish();
pGfxFramework->SetCalculateCallback(nullptr, nullptr);
pGfxFramework->SetMakeCommandCallback(nullptr, nullptr);
// Finalize.
Finalize(pGfxFramework);
// Finalize the load measurement feature.
pGfxFramework->FinalizePerf();
return EXIT_SUCCESS;
} // NOLINT
