This tutorial recreates Tutorial05 using HLSL shaders.

Overview

This tutorial recreates Tutorial05 using HLSL shaders.

The DirectX Shader Compiler (DXC) is used to convert the HLSL to SPIR-V. The command line used for this can be found in Visua Studio in Project -> Properties -> Build Events -> Pre-Build Event -> Command Line

To be able to use the Texture2D and SamplerState objects from the HLSL, the functions nvnCommandBufferBindSeparateTexture and nvnCommandBufferBindSeparateSampler are used instead of the bindless uniform version from GLSL. NVN_DEVICE_FLAG_ENABLE_SEPARATE_SAMPLER_TEXTURE_SUPPORT_BIT device flag is needed to enable this feature in NVN.

809 /*
810  * HLSL::UpdateCommandBuffer
811  * -------------------------
812  * Builds the command buffer to render the cube.
813  */
814 void HLSL::UpdateCommandBuffer()
815 {
816  /* Sync to safely overwrite memory. */
817  nvnQueueFenceSync(&m_Queue, &m_CommandBufferSync, NVN_SYNC_CONDITION_ALL_GPU_COMMANDS_COMPLETE, 0);
818  nvnQueueFlush(&m_Queue);
819  nvnSyncWait(&m_CommandBufferSync, NVN_WAIT_TIMEOUT_MAXIMUM);
820 
821  /* Reset the command buffer memory. */
822  nvnCommandBufferAddCommandMemory(&m_CommandBuffer, m_pCommandMemoryPool->GetMemoryPool(), m_CommandPoolOffset, g_CommandMemorySize);
823  nvnCommandBufferAddControlMemory(&m_CommandBuffer, m_pControlPool, g_ControlMemorySize);
824 
825  /* Starts the recording of a new set of commands for the given command buffer. */
826  nvnCommandBufferBeginRecording(&m_CommandBuffer);
827  {
828  /* Bind the texture and sampler descriptor pools. */
829  m_pTextureIDManager->SetSamplerPool(&m_CommandBuffer);
830  m_pTextureIDManager->SetTexturePool(&m_CommandBuffer);
831 
832  /* Sets the scissor rectangle and viewport to the full screen */
833  nvnCommandBufferSetScissor(&m_CommandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
834  nvnCommandBufferSetViewport(&m_CommandBuffer, 0, 0, m_ScreenWidth, m_ScreenHeight);
835 
836  /* Clears the currently set render target at a given index. */
837  float clear_color[4] = { 0.4f, 0.55f, 0.6f, 1.0f };
838  nvnCommandBufferClearColor(&m_CommandBuffer, 0, clear_color, NVN_CLEAR_COLOR_MASK_RGBA);
839  nvnCommandBufferClearDepthStencil(&m_CommandBuffer, 1.0, NVN_TRUE, 0, 255);
840 
841  std::vector<NVNModelData*>& modelData = m_pDataHolder->GetModelData();
842  std::vector<NVNProgramData*>& programData = m_pDataHolder->GetProgramData();
843 
844  if (m_pShaderScratchMemoryPool)
845  {
846  nvnCommandBufferSetShaderScratchMemory(&m_CommandBuffer, m_pShaderScratchMemoryPool->GetMemoryPool(), 0, m_ShaderScratchSize);
847  }
848 
849  /* Bind the render state objects. */
850  nvnCommandBufferBindBlendState(&m_CommandBuffer, &m_BlendState);
851  nvnCommandBufferBindChannelMaskState(&m_CommandBuffer, &m_ChannelMaskState);
852  nvnCommandBufferBindColorState(&m_CommandBuffer, &m_ColorState);
853  nvnCommandBufferBindDepthStencilState(&m_CommandBuffer, &m_DepthStencilState);
854  nvnCommandBufferBindMultisampleState(&m_CommandBuffer, &m_MultisampleState);
855  nvnCommandBufferBindPolygonState(&m_CommandBuffer, &m_PolygonState);
856  nvnCommandBufferSetSampleMask(&m_CommandBuffer, static_cast<uint32_t>(~0));
857 
858  /* Bind the vertex buffer(s). */
859  Model* model = &modelData[0]->m_Model;
860  NVNbufferAddress vboAddress = nvnBufferGetAddress(&modelData[0]->m_VertexBuffer);
861  for(size_t i = 0; i < model->m_VertexAttributes.size(); ++i)
862  {
863  VertexAttribute& attribute = model->m_VertexAttributes[i];
864  nvnCommandBufferBindVertexBuffer(&m_CommandBuffer, attribute.m_Location, vboAddress + modelData[0]->m_VertexAttributeBufferOffsets[i], attribute.m_DataSize);
865  }
866 
867  /* Bind the vertex states. */
868  nvnCommandBufferBindVertexAttribState(&m_CommandBuffer, static_cast<int>(modelData[0]->m_VertexAttributeStates.size()), &modelData[0]->m_VertexAttributeStates[0]);
869  nvnCommandBufferBindVertexStreamState(&m_CommandBuffer, static_cast<int>(modelData[0]->m_VertexStreamStates.size()), &modelData[0]->m_VertexStreamStates[0]);
870 
871  /* Bind the shader program and uniforms. */
872  int vertBindingLoc = SimpleTexturedModelAlphaHLSL::BlockVSUniformBlockData::GetBinding(NVN_SHADER_STAGE_VERTEX);
873  nvnCommandBufferBindUniformBuffer(&m_CommandBuffer, NVN_SHADER_STAGE_VERTEX, vertBindingLoc, m_UniformBlockVertex.GetBufferAddress(), sizeof(SimpleTexturedModelAlphaHLSL::BlockVSUniformBlockData));
874 
875  /* Bind the texture and sampler as a separate texture object. */
876  int fragBindingLocTexture = SimpleTexturedModelAlphaHLSL::STMTextureUniformData::GetBinding(NVN_SHADER_STAGE_FRAGMENT);
877  nvnCommandBufferBindSeparateTexture(&m_CommandBuffer, NVN_SHADER_STAGE_FRAGMENT, fragBindingLocTexture, m_SeparateTextureHandle);
878 
879  int fragBindingLocSampler = SimpleTexturedModelAlphaHLSL::STMSamplerUniformData::GetBinding(NVN_SHADER_STAGE_FRAGMENT);
880  nvnCommandBufferBindSeparateSampler(&m_CommandBuffer, NVN_SHADER_STAGE_FRAGMENT, fragBindingLocSampler, m_SeparateSamplerHandle);
881 
882  programData[0]->BindShaderProgram(&m_CommandBuffer);
883 
884  /* Draw the primitives. */
885  NVNindexType indexType = static_cast<NVNindexType>(modelData[0]->m_Model.m_IndexData.m_IndexType);
886  uint32_t numIndices = modelData[0]->m_Model.m_IndexData.m_DataSize / modelData[0]->m_Model.m_IndexData.m_Stride;
887  nvnCommandBufferDrawElements(&m_CommandBuffer,
888  (NVNdrawPrimitive)modelData[0]->m_Model.m_NvnDrawPrimitiveType,
889  indexType,
890  numIndices,
891  nvnBufferGetAddress(&modelData[0]->m_IndexBuffer));
892  }
893 
894  m_CommandHandle = nvnCommandBufferEndRecording(&m_CommandBuffer);
895 }

898  * HLSL::Draw
899  * ----------
900  * This method updates uniform data and submits the commands
901  * recorded in the command buffer to the queue and presents
902  * the render target to the screen.

905 {

908  ++s_Frame;

911  int index = UpdateRenderTargets();

914 
915  /* Update the uniform data. */
916  nn::util::Matrix4x3fType simpleTranslateMat;
917  nn::util::MatrixIdentity(&simpleTranslateMat);
918  nn::util::Vector3fType simpleTranslateVector;
919  nn::util::VectorSet(&simpleTranslateVector, -0.0f, 0.0f, -3.0f);
920  nn::util::MatrixSetTranslate(&simpleTranslateMat, simpleTranslateVector);
921 
922  nn::util::Matrix4x3fType simpleRotateXMat;
923  nn::util::MatrixIdentity(&simpleRotateXMat);
924  nn::util::Vector3fType simpleRotateXVector;
925  nn::util::VectorSet(&simpleRotateXVector, 30.0f * 3.1459f / 180.0f, 0.0f, 0.0f);
926  nn::util::MatrixSetRotateXyz(&simpleRotateXMat, simpleRotateXVector);
927 
928  nn::util::Matrix4x3fType simpleRotateYMat;
929  nn::util::MatrixIdentity(&simpleRotateYMat);
930  nn::util::Vector3fType simpleRotateYVector;
931  nn::util::VectorSet(&simpleRotateYVector, 0.0f, s_Frame * 3.14159f / 180.0f, 0.0f);
932  nn::util::MatrixSetRotateXyz(&simpleRotateYMat, simpleRotateYVector);
933 
934  nn::util::Matrix4x3fType simpleModelMat;
935  nn::util::Matrix4x3fType tempMat1;
936  nn::util::MatrixMultiply(&tempMat1, simpleRotateYMat, simpleRotateXMat);
937  nn::util::MatrixMultiply(&simpleModelMat, tempMat1, simpleTranslateMat);
938  nn::util::Matrix4x4fType simpleModelMat44;
939  nn::util::MatrixConvert(&simpleModelMat44, simpleModelMat);
940 
941  nn::util::Matrix4x4fType projMat44;
942  nn::util::MatrixPerspectiveFieldOfViewRightHanded(&projMat44, 60.0f * 3.14159f / 180.0f, static_cast<float>(m_ScreenWidth) / static_cast<float>(m_ScreenHeight), 1.0f, 1000.0f);
943 
944  UpdateUniformBlock(simpleModelMat44, projMat44);

947  SetStates(s_Frame);
948 
949  /* If the state changed, re-record the command buffer. */
950  if (m_ChangeState)
951  {
952  UpdateCommandBuffer();
953  }
954 
959  nvnSyncWait(&m_WindowSync, NVN_WAIT_TIMEOUT_MAXIMUM);
960  nvnQueueSubmitCommands(&m_Queue, 1, &m_CommandHandle);
961 
962  /* Print out the frame counter on the screen. */
963  m_pDebugTextRenderer->Printf(0, 0, "Frames: %i", s_Frame);
964 
965  /* Draw the accumulated debug text for the frame. */
966  m_pDebugTextRenderer->Draw(m_pRenderTargets[index], m_ScreenWidth, m_ScreenHeight);
967

969  nvnQueuePresentTexture(&m_Queue, m_pWindow, index);
970 }

1004 /*
1005  * HLSL::Resize
1006  * ------------
1007  * This method is called everytime the window is resized and is passed
1008  * the new size of the window. It frees the old render target and creates a new
1009  * render target with the new screen size. A new set of commands for the command buffer
1010  * is recorded.
1011  */
1012 void HLSL::Resize(int width, int height)
1013 {
1014  /* Check for the window being minimized or having no visible surface. */
1015  if (width == 0 || height == 0)
1016  {
1017  return;
1018  }
1019 
1020  /* Save the current height and width of the screen. */
1021  m_ScreenWidth = width;
1022  m_ScreenHeight = height;
1023 
1024  /* Sync to safely rewrite the command buffer. */
1025  nvnQueueFenceSync(&m_Queue, &m_CommandBufferSync, NVN_SYNC_CONDITION_ALL_GPU_COMMANDS_COMPLETE, 0);
1026  nvnQueueFlush(&m_Queue);
1027  nvnSyncWait(&m_CommandBufferSync, NVN_WAIT_TIMEOUT_MAXIMUM);
1028 
1029  /* If it's the first time Resize is called, allocate the NVNwindow. */
1030  if (m_pWindow == NULL)
1031  {
1032  m_pWindow = new NVNwindow;
1033  }
1034  /*
1035  * Otherwise finalize (free) the NVNwindow used for the previous window size.
1036  * The NVNWindow must be finalized before a render target it owns is finalized.
1037  */
1038  else
1039  {
1040  nvnWindowFinalize(m_pWindow);
1041  }
1042 
1043  /* Set up the builder for the render target. */
1044  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
1045  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_DISPLAY_BIT | NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
1046  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
1047  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
1048  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_RGBA8);
1049

1051  {

1053  if (!m_pRenderTargets[i])

1056  }
1057  /* Otherwise finalize (free) the render target used for the previous window size. */
1058  else
1059  {
1060  nvnTextureFinalize(m_pRenderTargets[i]);
1061  }
1062 
1063  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * i);
1064 
1065  /* Create the texture using the current state of the texture builder. */
1066  if (!nvnTextureInitialize(m_pRenderTargets[i], &m_RenderTargetBuilder))
1067  {
1068  NN_ASSERT(0, "Failed to initialize render target");
1069  }
1070  }
1071 
1072  if (!m_pDepthTarget)
1073  {
1074  m_pDepthTarget = new NVNtexture;
1075  }
1076  else
1077  {
1078  nvnTextureFinalize(m_pDepthTarget);
1079  }
1080 
1081  /* Initialize depth buffer for render target. */
1082  nvnTextureBuilderSetDefaults(&m_RenderTargetBuilder);
1083  nvnTextureBuilderSetFlags(&m_RenderTargetBuilder, NVN_TEXTURE_FLAGS_COMPRESSIBLE_BIT);
1084  nvnTextureBuilderSetSize2D(&m_RenderTargetBuilder, m_ScreenWidth, m_ScreenHeight);
1085  nvnTextureBuilderSetTarget(&m_RenderTargetBuilder, NVN_TEXTURE_TARGET_2D);
1086  nvnTextureBuilderSetFormat(&m_RenderTargetBuilder, NVN_FORMAT_DEPTH32F);
1087 
1088  nvnTextureBuilderSetStorage(&m_RenderTargetBuilder, m_pRenderTargetMemoryPool->GetMemoryPool(), m_ColorTargetSize * g_NumColorBuffers);
1089  nvnTextureInitialize(m_pDepthTarget, &m_RenderTargetBuilder);
1090 
1091  /* Pass off the render targets to the window. */
1092  nvnWindowBuilderSetTextures(&m_WindowBuilder, g_NumColorBuffers, m_pRenderTargets);
1093  nvnWindowInitialize(m_pWindow, &m_WindowBuilder);
1094 
1095  /* Update the cube rendering command buffer. */
1096  UpdateCommandBuffer();
1097 }
