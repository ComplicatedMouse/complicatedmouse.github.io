#include <nn/nn_Macro.h>
#include <nn/nn_Log.h>

#include <nn/os.h>
//-----------------------------------------------------------------------------
namespace {
nn::os::Tick g_StartTick; // Tick value at the start time.
nn::os::TimerEventType g_TimerEvent1; // Timer event 1.
nn::os::TimerEventType g_TimerEvent2; // Timer event 2.
const size_t StackSize = 8192; // Stack size.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack1[StackSize]; // Stack for the timer event 1 wait thread.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack2[StackSize]; // Stack for the timer event 2 wait thread.
} // Namespace.
//-----------------------------------------------------------------------------
inline int64_t GetMilliSecondsFromStart()
{
nn::os::Tick currentTick = nn::os::GetSystemTick();
return nn::os::ConvertToTimeSpan(currentTick - g_StartTick).GetMilliSeconds();
}
//-----------------------------------------------------------------------------
// First thread (wait for the periodic timer event).
//
void ThreadFunction1(void *arg)
{
NN_UNUSED(arg);
for (int i=0; i<10; ++i)
{
nn::os::WaitTimerEvent( &g_TimerEvent1 );
NN_LOG("PeriodicTimerEvent: count=%d time=%5lld (msec)\n", i, GetMilliSecondsFromStart());
}
nn::os::StopTimerEvent( &g_TimerEvent1 );
}
//-----------------------------------------------------------------------------
// Second thread (wait for one-shot timer event).
//
void ThreadFunction2(void *arg)
{
NN_UNUSED(arg);
nn::os::WaitTimerEvent( &g_TimerEvent2 );
NN_LOG("OneShotTimerEvent: time=%5lld (msec)\n", GetMilliSecondsFromStart());
}
//-----------------------------------------------------------------------------
// The main function.
//
extern "C" void nnMain()
{
nn::os::ThreadType thread1;
nn::os::ThreadType thread2;
nn::Result result;
// Initialize the timer event.
nn::os::InitializeTimerEvent( &g_TimerEvent1, nn::os::EventClearMode_AutoClear );
nn::os::InitializeTimerEvent( &g_TimerEvent2, nn::os::EventClearMode_AutoClear );
// Generate threads.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread1, ThreadFunction1, NULL, g_ThreadStack1, StackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread2, ThreadFunction2, NULL, g_ThreadStack2, StackSize, nn::os::DefaultThreadPriority ) );
// Start time.
g_StartTick = nn::os::GetSystemTick();
// Start thread execution.
nn::os::StartThread( &thread1 );
nn::os::StartThread( &thread2 );
// Start timer event 1 as a periodic timer event.
const nn::TimeSpan interval = nn::TimeSpan::FromMilliSeconds(100);
nn::os::StartPeriodicTimerEvent( &g_TimerEvent1, interval, interval );
// Start timer event 2 as a one-shot timer event.
const nn::TimeSpan oneShotTime = nn::TimeSpan::FromMilliSeconds(1500);
nn::os::StartOneShotTimerEvent( &g_TimerEvent2, oneShotTime );
// Wait for the thread to end.
nn::os::WaitThread( &thread1 );
nn::os::WaitThread( &thread2 );
// Free threads.
nn::os::DestroyThread( &thread1 );
nn::os::DestroyThread( &thread2 );
// Finalize the timer event.
nn::os::FinalizeTimerEvent( &g_TimerEvent1 );
nn::os::FinalizeTimerEvent( &g_TimerEvent2 );
}
