#include <nvntutorial/InputManager.Windows.h>
InputManagerWin::InputManagerWin()
{
m_MouseDeltaX = 0;
m_MouseDeltaY = 0;
memset(&m_OldKeyboardState, 0, sizeof(nn::hid::KeyboardState));
memset(&m_CurrentKeyboardState, 0, sizeof(nn::hid::KeyboardState));
memset(&m_OldMouseState, 0, sizeof(nn::hid::MouseState));
memset(&m_CurrentMouseState, 0, sizeof(nn::hid::MouseState));
memset(m_CurrentMouseButtonState, 0, sizeof(m_CurrentMouseButtonState));
}
InputManagerWin::~InputManagerWin()
{
}
void InputManagerWin::Init()
{
nn::hid::InitializeKeyboard();
nn::hid::InitializeMouse();
nn::hid::GetKeyboardState(&m_CurrentKeyboardState);
nn::hid::GetMouseState(&m_CurrentMouseState);
m_OldKeyboardState = m_CurrentKeyboardState;
m_OldMouseState = m_OldMouseState;
m_KeyboardBitflagMapping.insert(std::pair<char, int>('w', nn::hid::KeyboardKey::W::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('a', nn::hid::KeyboardKey::A::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('s', nn::hid::KeyboardKey::S::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('d', nn::hid::KeyboardKey::D::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('q', nn::hid::KeyboardKey::Q::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('e', nn::hid::KeyboardKey::E::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('R', nn::hid::KeyboardKey::RightArrow::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('L', nn::hid::KeyboardKey::LeftArrow::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('X', nn::hid::KeyboardKey::X::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('C', nn::hid::KeyboardKey::C::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('Y', nn::hid::KeyboardKey::Y::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('H', nn::hid::KeyboardKey::H::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('K', nn::hid::KeyboardKey::K::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('U', nn::hid::KeyboardKey::U::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>('J', nn::hid::KeyboardKey::J::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>(CHAR_RIGHT_SHIFT, nn::hid::KeyboardKey::RightShift::Index));
m_KeyboardBitflagMapping.insert(std::pair<char, int>(CHAR_LEFT_SHIFT, nn::hid::KeyboardKey::LeftShift::Index));
m_MouseBitflagMapping.insert(std::pair<MouseButton, int>(MouseButton_Left, nn::hid::MouseButton::Left::Index));
m_MouseBitflagMapping.insert(std::pair<MouseButton, int>(MouseButton_Right, nn::hid::MouseButton::Right::Index));
m_MouseBitflagMapping.insert(std::pair<MouseButton, int>(MouseButton_Middle, nn::hid::MouseButton::Middle::Index));
m_MouseBitflagMapping.insert(std::pair<MouseButton, int>(MouseButton_Forward, nn::hid::MouseButton::Forward::Index));
m_MouseBitflagMapping.insert(std::pair<MouseButton, int>(MouseButton_Back, nn::hid::MouseButton::Back::Index));
}
void InputManagerWin::Update()
{
nn::hid::GetKeyboardState(&m_CurrentKeyboardState);
nn::hid::GetMouseState(&m_CurrentMouseState);
UpdateMouseButtonState();
UpdateKeyboardButtonState();
m_MouseDeltaX = m_CurrentMouseState.x - m_OldMouseState.x;
m_MouseDeltaY = m_CurrentMouseState.y - m_OldMouseState.y;
m_OldKeyboardState = m_CurrentKeyboardState;
m_OldMouseState = m_CurrentMouseState;
}
/*
* InputManager::GetKeyboardButtonState
* ------------------------------------
* Gets the state of a particular registered key on the keyboard.
*/
InputManager::ButtonState InputManagerWin::GetKeyboardButtonState(char key)
{
int bitflagIndex = m_KeyboardBitflagMapping.at(key);
return m_CurrentKeyboardButtonState[bitflagIndex];
}
/*
* InputManager::GetMouseButtonState
* ---------------------------------
* Gets the state of a given mouse button.
*/
InputManager::ButtonState InputManagerWin::GetMouseButtonState(InputManagerWin::MouseButton mouseButton)
{
int bitflagIndex = m_MouseBitflagMapping.at(mouseButton);
return m_CurrentMouseButtonState[bitflagIndex];
}
/*
* InputManager::GetMouseDeltas
* ----------------------------
* Gets change in position of the mouse.
*/
void InputManagerWin::GetMouseDeltas(int& mouseX, int& mouseY)
{
mouseX = m_MouseDeltaX;
mouseY = m_MouseDeltaY;
}
/*
* InputManager::UpdateMouseButtonState
* ------------------------------------
* Updates the current state of the mouse buttons.
*/
void InputManagerWin::UpdateMouseButtonState()
{
for (int i = 0; i < 5; ++i)
{
bool currentButtonClickState = m_CurrentMouseState.buttons.Test(i);
bool oldButtonClickState = m_OldMouseState.buttons.Test(i);
m_CurrentMouseButtonState[i] = GetButtonState(oldButtonClickState, currentButtonClickState);
}
}
/*
* InputManager::UpdateKeyboardButtonState
* ---------------------------------------
* Updates the current state of the keyboard.
*/
void InputManagerWin::UpdateKeyboardButtonState()
{
for (int i = 0; i < 256; ++i)
{
bool currentButtonClickState = m_CurrentKeyboardState.keys.Test(i);
bool oldButtonClickState = m_OldKeyboardState.keys.Test(i);
m_CurrentKeyboardButtonState[i] = GetButtonState(oldButtonClickState, currentButtonClickState);
}
}
