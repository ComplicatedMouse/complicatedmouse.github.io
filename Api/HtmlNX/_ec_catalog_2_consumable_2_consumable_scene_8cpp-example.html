// The list to display during the selection process.
void ConsumableScene::DrawSelectConsumableCallBack(int listIndex, SceneBase* pScene, int x, int y, bool isSelected) NN_NOEXCEPT
{
NN_UNUSED(isSelected);
NN_ASSERT_NOT_NULL(pScene);
ConsumableScene* pConsumableScene = dynamic_cast<ConsumableScene*>(pScene);
Consumable consumable = pConsumableScene->m_RequestConsumable.GetResponse().GetConsumables()[listIndex];
ImGui::SetCursorPos(ImVec2(x, y));
ImGui::SetWindowFontScale(1.5f);
ImGui::Text("name:%s", consumable.GetName().c_str());
ImGui::SetWindowFontScale(1.2f);
ImGui::SetCursorPosX(x); // Reconfigure because the x-coordinate is reset when you display with Text. The same goes for the rest.
ImGui::Text("description:%s", consumable.GetDescription().c_str());
ImGui::SetCursorPosX(x);
ImGui::Text("disclaimer:%s", consumable.GetDisclaimer().c_str());
}
// Screen rendering.
void ConsumableScene::Draw() NN_NOEXCEPT
{
switch (m_ConsumableStatus)
{
case ConsumableStatus_Initialize:
case ConsumableStatus_ResponseWait:
ImGui::SetWindowFontScale(1.8f);
ImGui::SetCursorPos(ImVec2(550, 675));
ImGui::TextColored(White, "Now Loading...");
break;
case ConsumableStatus_Start:
{
// Display the selection screen.
m_Select.DrawSelectionList(DrawSelectConsumableCallBack, this);
// Note: There is no way to ensure that the language of the response is one of the system languages.
// Do not treat this condition as an error. Continue to display product information as is.
nn::settings::LanguageCode lang = nn::oe::GetDesiredLanguage();
if (::nn::util::Strncmp(lang.string, m_RequestConsumable.GetResponse().GetLang().c_str(), sizeof(lang.string)) != 0)
{
// Notation displays (translation):
// Product information cannot be displayed using the languages configured for this application. Display using an available language.
ImGui::SetWindowFontScale(1.1f);
ImGui::SetCursorPos(ImVec2(10, 570));
ImGui::TextColored(Red, "Product information cannot be displayed in this language currently set for this application. The supported language will be displayed.");
}
// Description of operations.
ImGui::SetWindowFontScale(2.0f);
ImGui::SetCursorPos(ImVec2(10, 600));
ImGui::TextColored(White, "Select an item by up or down button and Push A button to retrieve each item.\nPush B button to go back to the start scene.");
break;
}
default:
break;
}
}
// Input management processing.
void ConsumableScene::Control(const nns::hid::GamePad& gamePad) NN_NOEXCEPT
{
switch (m_ConsumableStatus)
{
case ConsumableStatus_Initialize:
{
const auto isSuccess = m_RequestConsumable.SetupRequest(pSceneData->GetUserContext()->id);
if (isSuccess == false)
{
// Return to the previous screen when an error occurs.
NN_LOG("Request or initialization failed.\n");
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
m_ConsumableStatus = ConsumableStatus_ResponseWait;
}
break;
}
case ConsumableStatus_ResponseWait:
if (!m_RequestConsumable.IsCompleted())
{
if (gamePad.HasAnyButtonsUp(nns::hid::Button::B::Mask) == true)
{
// Canceling.
m_RequestConsumable.Cancel();
}
}
else
{
try
{
if (m_RequestConsumable.WaitFinish(pSceneData->GetUserContext()->userHandle) == false)
{
//If an error or cancellation occurs.
SceneBase::SetStatus(DisplayStatus_Cancel);
}
else
{
if (m_RequestConsumable.ConvertResponse() == false)
{
// Display an error when the parse fails.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_InvalidServerResponse);
SceneBase::SetStatus(DisplayStatus_Error);
}
else
{
// The initial settings for the choices.
int listMax = m_RequestConsumable.GetResponse().GetConsumables().size();
if (0 == listMax)
{
// Display an error if there are no items to display.
pSceneData->SetSceneErrorCode(SceneData::ErrorStatus_NoneItem);
SceneBase::SetStatus(DisplayStatus_Error);
break;
}
int row = 5; // Number of lines
int offsetX = 1200; // The amount to shift the selection horizontally.
int offsetY = 100; // The amount to shift the selection vertically.
int positionX = 10; // The coordinate for the left position of the selection.
int positionY = 50; // The coordinate for the upper position of the selection.
float cursorSize = 1.5; // The size of the cursor.
int cursorOffsetX = 0; // Fine tuning the x-position of the cursor.
int cursorOffsetY = 20; // Fine tuning the y-position of the cursor.
m_Select.SetInitialValue(
listMax
, row
, offsetX
, offsetY
, positionX
, positionY
, pSceneData->GetSelectedPositionConsumable()
, cursorSize
, cursorOffsetX
, cursorOffsetY
);
// Processing for informing the main screen that loading completed and deleting the previous screen.
pSceneData->SetLoadCompleted(true);
m_ConsumableStatus = ConsumableStatus_Start;
}
}
}
catch (std::bad_alloc& e)
{
// Processes cancellation when memory allocation fails.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
catch (std::exception& e)
{
// Although only bad_alloc is expected in the sample, this processing is used when other exceptions are thrown.
// Process cancellation.
NN_LOG("%s \n", e.what());
SceneBase::SetStatus(DisplayStatus_Cancel);
}
}
break;
case ConsumableStatus_Start:
int selected;
Select::SelectStatus selectStatus;
// Execute the selection process.
m_Select.InputSelectionList(&selectStatus, &selected, gamePad);
pSceneData->SetSelectedPositionConsumable(selected);
if (selectStatus == Select::SelectStatus_Cancel)
{
// If the process has been canceled.
SceneBase::SetStatus(DisplayStatus_Start);
}
if (selectStatus == Select::SelectStatus_Decision)
{
// If A is pressed.
pSceneData->SetConsumableId(m_RequestConsumable.GetResponse().GetConsumables()[selected].GetConsumableId().c_str());
SceneBase::SetStatus(DisplayStatus_ConsumableItem);
}
break;
default:
break;
}
}//NOLINT(impl/function_size)
