#include <stdint.h>
#include <nn/nn_Macro.h>
#include <nn/nn_Log.h>
#include <nn/nn_Assert.h>
#include <nn/nn_TimeSpan.h>
#include <nn/os.h>
#include <nn/htc.h>
#include <nn/TargetConfigs/build_Platform.h>
#if defined(NN_BUILD_TARGET_PLATFORM_OS_WIN)
#include <nn/socket.h>
#endif
//-----------------------------------------------------------------------------
//
// The main function.
//
extern "C" void nnMain()
{
nn::Result result;
size_t length;
#if defined(NN_BUILD_TARGET_PLATFORM_OS_WIN)
// For Windows applications, you must call nn::socket::Initialize() to use htc.
static nn::socket::ConfigDefaultWithMemory s_SocketConfig;
nn::socket::Initialize(s_SocketConfig);
#endif
nn::htc::Initialize();
// Get the notification event for connecting to Target Manager.
nn::os::SystemEvent connectionEvent;
nn::htc::BindHostConnectionEvent(&connectionEvent);
// Get the notification event for disconnecting from Target Manager.
nn::os::SystemEvent disconnectionEvent;
nn::htc::BindHostDisconnectionEvent(&disconnectionEvent);
// Wait for a connection to Target Manager.
NN_LOG("Waiting for a connection from the host.");
connectionEvent.Wait();
// Get the size of the NINTENDO_SDK_ROOT environment variable from the host PC.
result = nn::htc::GetEnvironmentVariableLength(&length, "NINTENDO_SDK_ROOT");
if (result.IsSuccess())
{
NN_LOG("length = %d", length);
}
else
{
NN_LOG("GetEnvironmentVariableLength failed (%08x)", result.GetInnerValueForDebug());
}
// Wait for disconnection from Target Manager.
NN_LOG("Waiting for disconnection from the host.");
disconnectionEvent.Wait();
nn::htc::Finalize();
#if defined(NN_BUILD_TARGET_PLATFORM_OS_WIN)
nn::socket::Finalize();
#endif
}
