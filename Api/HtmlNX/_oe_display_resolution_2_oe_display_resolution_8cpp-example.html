#include <nn/nn_Common.h>
#include <nn/nn_Log.h>
#include <nn/oe.h>
//-----------------------------------------------------------------------------
//
// Main function.
//
extern "C" void nnMain()
{
int width, height;
// Initialize the application operating environment library.
nn::oe::Initialize();
// Get the current default display resolution.
nn::oe::GetDefaultDisplayResolution( &width, &height );
NN_LOG("Default Display Resolution : Width=%d, Height=%d\n", width, height);
// Get the system event when the default display resolution is changed.
nn::os::SystemEvent defaultDisplayResolutionChangeEvent;
nn::oe::GetDefaultDisplayResolutionChangeEvent( &defaultDisplayResolutionChangeEvent );
for (;;)
{
// Wait for the default display resolution to change.
defaultDisplayResolutionChangeEvent.Wait();
// Get the current default display resolution.
nn::oe::GetDefaultDisplayResolution( &width, &height );
NN_LOG("Default Display Resolution : Width=%d, Height=%d\n", width, height);
}
}
