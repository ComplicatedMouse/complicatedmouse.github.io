class Color
{
public:
Color() NN_NOEXCEPT {}
Color(uint8_t r, uint8_t g, uint8_t b, uint8_t a) NN_NOEXCEPT
{
Set(r, g, b, a);
}
NN_IMPLICIT Color(uint32_t rgb) NN_NOEXCEPT
{
Set(rgb);
}
void Set(uint8_t r, uint8_t g, uint8_t b, uint8_t a) NN_NOEXCEPT
{
m_Color[0] = static_cast<float>(r) / 255.0f;
m_Color[1] = static_cast<float>(g) / 255.0f;
m_Color[2] = static_cast<float>(b) / 255.0f;
m_Color[3] = static_cast<float>(a) / 255.0f;
}
void Set(uint32_t rgb) NN_NOEXCEPT
{
uint8_t hr = static_cast<uint8_t>( (rgb & 0xff0000) >> 16 );
uint8_t hg = static_cast<uint8_t>( (rgb & 0x00ff00) >> 8 );
uint8_t hb = static_cast<uint8_t>( rgb & 0x0000ff );
Set(hr, hg, hb, 255);
}
const float* Get() NN_NOEXCEPT
{
return m_Color;
}
private:
float m_Color[4] = { 0 };
};
