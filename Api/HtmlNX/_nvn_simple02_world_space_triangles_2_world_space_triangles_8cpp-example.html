// two triangles' worth of position data
float g_VertexPositions[WorldSpaceTriangles::NumTriangles][9] =
{
{
-384.0f, 378.0f, 0.0f,
192.0f, -378.0f, 0.0f,
384.0f, 0.0f, 0.0f
},
{
-384.0f, 378.0f, 0.0f,
384.0f, -378.0f, 0.0f,
384.0f, 378.0f, 0.0f
},
};
// use the same vertex colors for both triangles
float g_VertexColors[] =
{
1.0f, 0.0f, 1.0f, 1.0f,
1.0f, 1.0f, 0.0f, 1.0f,
0.0f, 1.0f, 1.0f, 1.0f,
};
// render the two triangles separately
uint32_t g_VertexIndices[] =
{
0, 1, 2,
};
float g_triOffset = 288.0f;
}

void WorldSpaceTriangles::Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
int64_t shaderBinFileSize = 0;
// read shader binary, create GraphicsShader object from it, and get uniform block reflection info
m_GlslcData = ReadFile(&shaderBinFileSize, "rom:/NvnSimple02.glslc", m_pGraphicsSystem->GetAllocator());
m_Shader.Initialize(m_pGraphicsSystem, static_cast<GLSLCoutput*>(m_GlslcData));
m_pUniformBlockInfo = &m_Shader.LookupUniformBlockInfo("VertexShaderUniformBlock");
// create memory pool for vertex data and two shader uniforms blocks, one for each triangle
size_t positionDataSize = sizeof(g_VertexPositions);
size_t colorDataSize = sizeof(g_VertexColors);
size_t indexDataSize = sizeof(g_VertexIndices);
size_t uniformAlignment = m_pGraphicsSystem->GetResourceInfo().uniformBufferAlignment;
size_t memPoolSize = positionDataSize + colorDataSize + indexDataSize + (nn::util::align_up(m_pUniformBlockInfo->GetSize(), uniformAlignment) * NumTriangles);
m_VertexAndUniformDataMemPool.Initialize(m_pGraphicsSystem, nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED, memPoolSize, "WorldSpaceTriangles MemPool");
// allocate blocks of memory for vertex and uniform data
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_PositionDataBuffer, positionDataSize);
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_ColorDataBuffer, colorDataSize);
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_IndexDataBuffer, indexDataSize);
for (int i = 0; i < NumTriangles; ++i)
{
m_VertexAndUniformDataMemPool.AllocateBuffer(&m_UniformBlockBuffers[i], m_pUniformBlockInfo->GetSize(), uniformAlignment);
}
// load vertex data into memory pool
memcpy(m_PositionDataBuffer.Map(), g_VertexPositions, positionDataSize);
memcpy(m_ColorDataBuffer.Map(), g_VertexColors, colorDataSize);
memcpy(m_IndexDataBuffer.Map(), g_VertexIndices, indexDataSize);
// compute and load data for uniform blocks into memory pool, and initialize triangle state
nn::util::Matrix4x4fType projMtx;
nn::util::Matrix4x4fType modelViewMtx;
nn::util::MatrixIdentity(&projMtx);
nn::util::MatrixOrthographicOffCenterRightHanded(&projMtx, -960.0f, 960.0f, -540.0f, 540.0f, -2000.0f, 2000.0f);
nn::util::MatrixIdentity(&modelViewMtx);
const UniformBlockMemberInfo& modelViewMtxInfo = m_pUniformBlockInfo->LookupMemberInfo("modelViewMtx");
const UniformBlockMemberInfo& projMtxInfo = m_pUniformBlockInfo->LookupMemberInfo("projMtx");
for (int i = 0; i < NumTriangles; ++i)
{
char* baseAddr = static_cast<char*>(m_UniformBlockBuffers[i].Map());
char* modelViewMtxAddr = baseAddr + modelViewMtxInfo.GetOffset();
char* projMtxAddr = baseAddr + projMtxInfo.GetOffset();
LoadMatrix44(projMtxAddr, projMtx);
LoadMatrix44(modelViewMtxAddr, modelViewMtx);
m_TriangleState[i].SetModelViewAddress(modelViewMtxAddr); // we use this modelViewAddr to later update the modelview uniform data per frame
m_TriangleState[i].SetRotateAngle(0.0f);
m_TriangleState[i].SetTranslateVector(nn::util::Vector3f(-g_triOffset + (2 * g_triOffset * i), 0, 0));
}
// set up NVN VertexStreamState and VertexAttribState objects based on vertex attribute names and shader
// the ordering of the attribute names must logically match up with the ordering of the vertex streams being setup and bound
const char* attributeNames[NumVertexAttributes] = { "attr_position", "attr_color" };
SetupVertexAttribAndStreamStateSeparate(attributeNames, NumVertexAttributes, m_Shader, m_VertexAttribStates, m_VertexStreamStates);
// initialize render state
m_RenderState.Initialize();
m_RenderState.GetPolygonState()->SetCullFace(nvn::Face::NONE);
m_RenderState.GetPolygonState()->SetPolygonMode(nvn::PolygonMode::FILL);
m_RenderState.GetDepthStencilState()->SetDepthTestEnable(false);
m_RenderState.GetDepthStencilState()->SetDepthWriteEnable(false);
// build our command buffer to hold GPU commands to draw our triangles
m_CommandBuffer.Initialize(m_pGraphicsSystem);
GenerateCommandBuffer();
m_IsInitialized = true;
}
}
void WorldSpaceTriangles::DrawFrame() NN_NOEXCEPT
{
nn::util::Matrix4x3fType modelViewMtx;
nn::util::Matrix4x3fType rotateMtx;
// compute and load updated model view transform for each triangle
for (int i = 0; i < NumTriangles; ++i)
{
nn::util::MatrixIdentity(&modelViewMtx);
nn::util::MatrixSetTranslate(&modelViewMtx, m_TriangleState[i].GetTranslateVector());
nn::util::MatrixIdentity(&rotateMtx);
nn::util::MatrixSetRotateXyz(&rotateMtx, nn::util::Vector3f(0, m_TriangleState[i].GetRotateAngle(), 0));
nn::util::MatrixMultiply(&modelViewMtx, rotateMtx, modelViewMtx);
LoadMatrix43(m_TriangleState[i].GetModelViewAddress(), modelViewMtx);
}
if (!m_IsPrebuiltCommandBufferUsed || m_IsScreenResolutionChanged)
{
GenerateCommandBuffer();
}
// execute the command buffer
m_pGraphicsSystem->GetQueue()->SubmitCommands(1, &m_CommandHandle);
// update per frame state
float delta = 0.005f;
for (int i = 0; i < NumTriangles; ++i)
{
float theta = m_TriangleState[i].GetRotateAngle();
if (theta > 2 * nn::util::FloatPi)
{
theta -= (2 * nn::util::FloatPi);
}
float dx = (theta > nn::util::FloatPi / 2 && theta <= 3 * nn::util::FloatPi / 2) ? g_triOffset : -g_triOffset;
m_TriangleState[i].SetTranslateVector(nn::util::Vector3f(dx - (2 * dx * i), 0, 0));
m_TriangleState[i].SetRotateAngle(theta + delta);
}
m_IsScreenResolutionChanged = false;
}
void WorldSpaceTriangles::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_CommandBuffer.Finalize();
m_PositionDataBuffer.Finalize();
m_ColorDataBuffer.Finalize();
m_IndexDataBuffer.Finalize();
for (int i = 0; i < NumTriangles; ++i)
{
m_UniformBlockBuffers[i].Finalize();
}
m_VertexAndUniformDataMemPool.Finalize();
m_Shader.Finalize();
m_pGraphicsSystem->GetAllocator()->Free(m_GlslcData);
m_GlslcData = nullptr;
m_pGraphicsSystem = nullptr;
m_CommandHandle = 0;
m_IsInitialized = false;
}
}

void WorldSpaceTriangles::GenerateCommandBuffer() NN_NOEXCEPT
{
// generate GPU commands to draw the triangles
nvn::CommandBuffer* pCommandBuffer = m_CommandBuffer.StartRecording();
{
// scissor/viewport
m_pGraphicsSystem->SetDefaultViewportAndScissor(pCommandBuffer);
// clear color and depth/stencil buffers
static float color[4] = { 0.8f, 0.8f, 0.8f, 0.8f };
pCommandBuffer->ClearColor(0, color, nvn::ClearColorMask::RGBA);
pCommandBuffer->ClearDepthStencil(1.0f, true, 0, 0);
// render state
m_RenderState.Bind(pCommandBuffer);
// bind shader
m_Shader.Bind(pCommandBuffer);
// bind vertex stream and attribute state
pCommandBuffer->BindVertexAttribState(NumVertexAttributes, m_VertexAttribStates);
pCommandBuffer->BindVertexStreamState(NumVertexAttributes, m_VertexStreamStates);
size_t triPositionsSize = m_PositionDataBuffer.GetSize() / 2;
// bind color vertex buffer (common to both triangles)
BindVertexBuffer(1, &m_ColorDataBuffer, pCommandBuffer);
// for each triangle
for (int i = 0; i < NumTriangles; ++i)
{
// Bind the triangle's uniform buffer data to the uniform block (VertexShaderUniformBlock).
// With NVN, most uniform data (outside of texture samplers) is bound through uniform blocks as opposed to individually.
int32_t bindingLocation = m_pUniformBlockInfo->GetBindingLocations(nvn::ShaderStage::VERTEX);
BindUniformBuffer(nvn::ShaderStage::VERTEX, bindingLocation, &m_UniformBlockBuffers[i], pCommandBuffer);
// bind position vertex buffer
pCommandBuffer->BindVertexBuffer(0, m_PositionDataBuffer.GetAddress() + (triPositionsSize * i), triPositionsSize);
// draw call
int numIndices = sizeof(g_VertexIndices) / sizeof(*g_VertexIndices);
pCommandBuffer->DrawElements(nvn::DrawPrimitive::TRIANGLES, nvn::IndexType::UNSIGNED_INT, numIndices, m_IndexDataBuffer.GetAddress());
}
}
m_CommandHandle = m_CommandBuffer.EndRecording();
}
