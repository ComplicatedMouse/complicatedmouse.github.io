// The size of the screenshot to save.
// The supported sizes for screenshots depends on the firmware.
const nn::album::ImageSize ScreenShotSize = nn::album::ImageSize_1280x720;
// The setting for reporting the result of the save operation.
// You can select the conditions under which the user will be automatically notified about the result of saving an image to the album.
// Even if you choose not to automatically notify the user, the application can be independently notified using the value returned by the function.
const nn::album::AlbumReportOption ReportOption = nn::album::AlbumReportOption_ReportAlways;
namespace {
// Creates image data to save as a screenshot.
// The color format for the image data is R8G8B8A8, but the alpha channel is ignored.
uint8_t* CreateImageData(size_t* pOutSize, int width, int height) NN_NOEXCEPT
{
const int PixelSize = 4;
size_t size = PixelSize * width * height;
uint8_t* pImage = reinterpret_cast<uint8_t*>(malloc(size));
NN_ASSERT_NOT_NULL(pImage);
for(int y = 0; y < height; y++)
{
uint8_t* pLine = pImage + PixelSize * width* y;
for(int x = 0; x < width; x++)
{
uint8_t* pPixel = pLine + PixelSize * x;
int v = ((x / 32) + (y / 32)) % 2;
if(v == 0)
{
pPixel[0] = 255; // R
pPixel[1] = 255; // G
pPixel[2] = 255; // B
pPixel[3] = 255; // A (ignored)
}
else
{
// +---------+---------+
// | Red. | Green. |
// +---------+---------+
// | Blue. | Black |
// +---------+---------+
if(y < height / 2)
{
if(x < width / 2)
{
pPixel[0] = 255; // R
pPixel[1] = 0; // G
pPixel[2] = 0; // B
pPixel[3] = 255; // A (ignored)
}
else
{
pPixel[0] = 0; // R
pPixel[1] = 255; // G
pPixel[2] = 0; // B
pPixel[3] = 255; // A (ignored)
}
}
else
{
if(x < width / 2)
{
pPixel[0] = 0; // R
pPixel[1] = 0; // G
pPixel[2] = 255; // B
pPixel[3] = 255; // A (ignored)
}
else
{
pPixel[0] = 0; // R
pPixel[1] = 0; // G
pPixel[2] = 0; // B
pPixel[3] = 255; // A (ignored)
}
}
}
}
}
*pOutSize = size;
return pImage;
}
void DestroyImageData(uint8_t* pImage) NN_NOEXCEPT
{
free(pImage);
}
}
// Saves a screenshot to an album.
void ExecuteSaveScreenshot() NN_NOEXCEPT
{
NN_LOG("Sample: SaveScreenShot\n");
// Initialize the album library.
nn::album::Initialize();
int width = nn::album::GetImageWidth(ScreenShotSize);
int height = nn::album::GetImageHeight(ScreenShotSize);
NN_LOG("Screenshot width = %d, height = %d\n", width, height);
// Creates the image data to save.
size_t imageDataSize = 0;
auto pImageData = CreateImageData(&imageDataSize, width, height);
{
NN_LOG("SaveScreenshot\n");
auto result = nn::album::SaveScreenshot(
pImageData,
imageDataSize,
ScreenShotSize,
ReportOption
);
if(result.IsFailure())
{
if(nn::album::ResultAlbumUnknown::Includes(result))
{
if(nn::album::ResultAlbumError::Includes(result))
{
// This error is returned if creating or saving the screenshot failed.
NN_LOG(" -> Screenshot Failed(AlbumError)\n");
}
else if(nn::album::ResultAlbumFull::Includes(result))
{
// This error is returned if the storage where the screenshot will be saved is full.
NN_LOG(" -> Screenshot Failed(AlbumFull)\n");
}
else if(nn::album::ResultAlbumFileCountLimit::Includes(result))
{
// This error is returned if the maximum number of screenshots that can be saved has been reached.
NN_LOG(" -> Screenshot Failed(AlbumFileCountLimit)\n");
}
else
{
NN_LOG(" -> Screenshot Failed(Unknown)\n");
}
}
else if(nn::album::ResultNotSupported::Includes(result))
{
NN_LOG(" -> NotSupported\n");
}
else
{
NN_LOG(" -> Error(Unknown)\n");
}
return;
}
}
NN_LOG(" -> OK\n");
// The image data can be destroyed after the function call.
DestroyImageData(pImageData);
// Finalize the album library.
nn::album::Finalize();
// Wait for the automatic notification of the save result to go away.
// This sample program performs a series of save operations, so it waits for the notification to go away.
// However, your application does not need to sleep when it uses the save function.
NN_LOG("Waiting for five seconds...\n");
nn::os::SleepThread(nn::TimeSpan::FromSeconds(5));
}

// Saves a screenshot to an album and posts the screenshot to an SNS.
void ExecuteSaveAndShareScreenshot() NN_NOEXCEPT
{
NN_LOG("Sample: SaveAndShareScreenShot\n");
// Initialize the album library.
nn::album::Initialize();
int width = nn::album::GetImageWidth(ScreenShotSize);
int height = nn::album::GetImageHeight(ScreenShotSize);
NN_LOG("Screenshot width = %d, height = %d\n", width, height);
// Creates the image data to save.
size_t imageDataSize = 0;
auto pImageData = CreateImageData(&imageDataSize, width, height);
// Specifies the account to use for posting.
// If nn::account::InvalidUid has been specified for Uid, a user selection screen will appear when it is time to post.
// The an account has been selected in the application already and the selected Uid is specified,
// the user selection screen will be skipped and the post screen will be displayed.
nn::account::Uid uid = nn::account::InvalidUid;
{
NN_LOG("SaveAndShareScreenshot\n");
auto result = nn::album::SaveAndShareScreenshot(
pImageData,
imageDataSize,
ScreenShotSize,
ReportOption,
uid
);
if(result.IsFailure())
{
if(nn::album::ResultAlbumUnknown::Includes(result))
{
if(nn::album::ResultAlbumError::Includes(result))
{
// This error is returned if creating or saving the screenshot failed.
NN_LOG(" -> Screenshot Failed(AlbumError)\n");
}
else if(nn::album::ResultAlbumFull::Includes(result))
{
// This error is returned if the storage where the screenshot will be saved is full.
NN_LOG(" -> Screenshot Failed(AlbumFull)\n");
}
else if(nn::album::ResultAlbumFileCountLimit::Includes(result))
{
// This error is returned if the maximum number of screenshots that can be saved has been reached.
NN_LOG(" -> Screenshot Failed(AlbumFileCountLimit)\n");
}
else
{
NN_LOG(" -> Screenshot Failed(Unknown)\n");
}
}
else if(nn::album::ResultShareUnknown::Includes(result))
{
if(nn::album::ResultShareCanceled::Includes(result))
{
// This error is returned if the user has canceled the post.
NN_LOG(" -> Share Failed(ShareCanceled)\n");
}
else if(nn::album::ResultShareFailed::Includes(result))
{
// This error is returned if an error occurs during posting.
NN_LOG(" -> Share Failed(ShareFailed)\n");
}
else
{
NN_LOG(" -> Share Failed(Unknown)\n");
}
}
else if(nn::album::ResultNotSupported::Includes(result))
{
NN_LOG(" -> NotSupported\n");
}
else
{
NN_LOG(" -> Error(Unknown)\n");
}
return;
}
}
NN_LOG(" -> OK\n");
// The image data can be destroyed after the function call.
DestroyImageData(pImageData);
// Finalize the album library.
nn::album::Finalize();
}
extern "C" void nnMain() NN_NOEXCEPT
{
NN_LOG("Start the sample program.\n");
// Executes the features for saving an image as a screenshot.
ExecuteSaveScreenshot();
// Executes the features for saving an image as a screenshot (including image orientation).
ExecuteSaveScreenshotWithOrientaiton(nn::album::ImageOrientation_None);
ExecuteSaveScreenshotWithOrientaiton(nn::album::ImageOrientation_Rotate90);
ExecuteSaveScreenshotWithOrientaiton(nn::album::ImageOrientation_Rotate180);
ExecuteSaveScreenshotWithOrientaiton(nn::album::ImageOrientation_Rotate270);
// Executes the features for saving an image as a screenshot and posting that screenshot.
ExecuteSaveAndShareScreenshot();
NN_LOG("Complete the sample program.\n");
}
