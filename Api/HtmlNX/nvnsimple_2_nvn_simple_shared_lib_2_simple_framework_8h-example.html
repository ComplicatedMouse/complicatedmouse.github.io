#pragma once
#include "GraphicsStructures.h"
#include "GraphicsSystem.h"
#include "SceneBaseClass.h"
//-------------------------------------------------------------------------------------------------
// Simple framework functions
//-------------------------------------------------------------------------------------------------
void SimpleFrameworkInitialize(GraphicsSystem* pGraphicsSystem, const GraphicsSystemParams& graphicsSystemParams) NN_NOEXCEPT;
void SimpleFrameworkFinalize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT;
bool SimpleFrameworkUpdate() NN_NOEXCEPT; // returns false when user requested application to exit (only on NX)
/**************************************************************************************************
*
* SimpleFrameworkMain:
*
*
* Each Application should calls this function in nnMain() to run.
* The class object passed in must derives SceneBaseClass and implements the
* following methods:
*
* - void Initialize(GraphicsSystem* pGraphicsSystem)
* - void DrawFrame()
* - void Finalize()
*
* For a more custom implementation, applications can use the above SimpleFramework* functions
* directly.
*
**************************************************************************************************/
void SimpleFrameworkMain(SceneBaseClass* pScene, const GraphicsSystemParams& graphicsSystemParams) NN_NOEXCEPT;
//-------------------------------------------------------------------------------------------------
// SimpleFrameworkMainWithDebugInfo:
//
// Working just like SimpleFrameworkMain but with debug information of frame time,
// CPU time of frame drawing, and command buffer reuse status.
//-------------------------------------------------------------------------------------------------
void SimpleFrameworkMainWithDebugInfo(SceneBaseClass* pScene, const GraphicsSystemParams& graphicsSystemParams, const char* sampleName, const Color& textColor) NN_NOEXCEPT;
