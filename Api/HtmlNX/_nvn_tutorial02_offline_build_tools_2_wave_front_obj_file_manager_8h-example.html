#pragma once
#include <tiny_obj_loader.h>
#include <vector>
#include <string>
#include <sstream>
namespace PBRModelBinary
{
struct DataHeader;
}
struct WaveFrontMeshData
{
uint32_t m_NumIndices;
uint32_t m_MaterialIndex;
std::vector<float> m_Positions;
std::vector<float> m_Normals;
std::vector<float> m_TexCoords;
std::vector<float> m_Tangents;
std::vector<float> m_Binormals;
std::vector<uint32_t> m_Indices;
void SetupData(const tinyobj::shape_t& shape, const tinyobj::attrib_t& attrib);
};
struct WaveFrontObjConfig
{
tinyobj::attrib_t m_Attrib;
std::vector<tinyobj::shape_t> m_Shapes;
std::vector<tinyobj::material_t> m_Materials;
std::vector<WaveFrontMeshData*> m_Meshes;
};
class WaveFrontObjFileManager
{
public:
WaveFrontObjFileManager();
~WaveFrontObjFileManager();
bool LoadObjFile(const std::string& pathName,
const std::string& objFileName,
tinyobj::attrib_t& attrib,
std::vector<tinyobj::shape_t>& shapes,
std::vector<tinyobj::material_t>& materials);
void ProcessData(const tinyobj::attrib_t& attrib,
const std::vector<tinyobj::shape_t>& shapes,
std::vector<WaveFrontMeshData*>& meshes);
bool WriteBinaryFile(const std::string& pathName,
const std::string& objFileName,
const WaveFrontObjConfig& object);
private:
void SetupDataHeader(PBRModelBinary::DataHeader& dataHeader, const uint64_t& dataSize, const void* data);
};
