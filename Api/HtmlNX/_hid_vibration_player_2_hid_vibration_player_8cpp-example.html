{
const char ProgramName[] = "HidVibrationPlayer";
nns::gfx::DebugGraphicsFramework g_Gfw;
nns::hid::Demo* g_pDemo;
void DrawTitle(nn::gfx::CommandBuffer* pCommand) NN_NOEXCEPT
{
auto pTextWriter = &g_Gfw.GetDebugFont()->object;
pTextWriter->Draw(pCommand);
pTextWriter->SetScale(2.0f, 2.0f);
pTextWriter->SetTextColor(nns::hid::Color::LimeGreen);
pTextWriter->SetCursor(10.0f, 10.0f);
pTextWriter->Print("%s", ProgramName);
pTextWriter->Draw(pCommand);
}
void Calculate(nns::gfx::GraphicsFramework* pGraphicsFramework, void* pUserData)
{
NN_UNUSED(pGraphicsFramework);
NN_UNUSED(pUserData);
g_pDemo->Update();
}
void MakeCommand(nns::gfx::GraphicsFramework* pGraphicsFramework, int bufferIndex, void* pUserData)
{
NN_UNUSED(pUserData);
pGraphicsFramework->BeginFrame(bufferIndex);
{
auto pCommandBuffer = pGraphicsFramework->GetRootCommandBuffer(bufferIndex);
// Disable cache.
pCommandBuffer->InvalidateMemory(nn::gfx::GpuAccess_ShaderCode | nn::gfx::GpuAccess_Descriptor | nn::gfx::GpuAccess_Texture);
// Set the render target.
{
auto pColorTargetView = pGraphicsFramework->GetColorTargetView();
auto clearColor = nn::util::Color4f::Black();
pCommandBuffer->ClearColor(pColorTargetView, clearColor.GetR(), clearColor.GetG(), clearColor.GetB(), 1.0f, nullptr);
pCommandBuffer->SetRenderTargets(1, &pColorTargetView, nullptr);
pCommandBuffer->SetViewportScissorState(g_Gfw.GetViewportScissorState());
}
DrawTitle(pCommandBuffer);
g_pDemo->Draw(pCommandBuffer, bufferIndex);
}
pGraphicsFramework->EndFrame(bufferIndex);
}
void InitializeGraphics() NN_NOEXCEPT
{
const size_t GraphicsSystemMemorySize = 8 * 1024 * 1024;
nns::gfx::GraphicsFramework::InitializeGraphicsSystem(GraphicsSystemMemorySize);
const auto DisplayWidth = 1280;
const auto DisplayHeight = 720;
const auto ScreenBpp = 4;
const auto BufferCount = 2;
const auto windowOriginMode = nn::gfx::util::ScalableViewport::OriginMode::OriginMode_LowerLeft;
nns::gfx::GraphicsFramework::FrameworkInfo gfwInfo;
gfwInfo.SetDefault();
gfwInfo.SetDisplayWidth(DisplayWidth);
gfwInfo.SetDisplayHeight(DisplayHeight);
gfwInfo.SetBufferCount(BufferCount);
gfwInfo.SetSwapChainBufferCount(BufferCount);
gfwInfo.SetDepthStencilBufferEnabled(false);
gfwInfo.SetRootCommandBufferCommandMemorySize(1 * 1024 * 1024);
nn::gfx::util::ScalableViewport::WindowCoordinate virtualWindowCoordinate, physicalWindowCoord;
{
float windowWidth, windowHeight;
windowWidth = static_cast<float>(DisplayWidth);
windowHeight = static_cast<float>(DisplayHeight);
virtualWindowCoordinate.SetWidth(windowWidth);
virtualWindowCoordinate.SetHeight(windowHeight);
virtualWindowCoordinate.SetOriginMode(nn::gfx::util::ScalableViewport::OriginMode_UpperLeft);
physicalWindowCoord.SetWidth(windowWidth);
physicalWindowCoord.SetHeight(windowHeight);
physicalWindowCoord.SetOriginMode(windowOriginMode);
}
gfwInfo.SetPhysicalWindowCoordinate(physicalWindowCoord);
gfwInfo.SetVirtualWindowCoordinate(virtualWindowCoordinate);
gfwInfo.SetMemoryPoolSize(
nns::gfx::GraphicsFramework::MemoryPoolType_CommandBuffer,
4 * 1024 * 1024);
gfwInfo.SetMemoryPoolSize(
nns::gfx::GraphicsFramework::MemoryPoolType_RenderTarget,
nn::util::align_up(DisplayWidth * DisplayHeight * ScreenBpp, 4096) * (BufferCount * 2));
gfwInfo.SetMemoryPoolSize(
nns::gfx::GraphicsFramework::MemoryPoolType_ConstantBuffer,
2 * 1024 * 1024);
gfwInfo.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_BufferView, 16);
gfwInfo.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_TextureView, 1024);
gfwInfo.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_Sampler, 1024);
g_Gfw.Initialize(gfwInfo);
g_Gfw.SetCalculateCallback(Calculate, nullptr);
g_Gfw.SetMakeCommandCallback(MakeCommand, nullptr);
}
void FinalizeGraphics() NN_NOEXCEPT
{
g_Gfw.SetCalculateCallback(nullptr, nullptr);
g_Gfw.SetMakeCommandCallback(nullptr, nullptr);
g_Gfw.Finalize();
}
void InitializeDemo() NN_NOEXCEPT
{
auto pTextWriter = &g_Gfw.GetDebugFont()->object;
auto pRenderer = g_Gfw.GetPrimitiveRenderer();
g_pDemo = new nns::hid::Demo(pTextWriter, pRenderer);
g_pDemo->Initialize();
}
void FinalizeDemo() NN_NOEXCEPT
{
g_pDemo->Finalize();
delete g_pDemo;
}
} // Anonymous namespace.
extern "C" void nnMain()
{
NN_LOG("%s Start.\n", ProgramName);
// Initialize graphics features.
InitializeGraphics();
// Initialize DEMO.
InitializeDemo();
while (NN_STATIC_CONDITION(true))
{
g_Gfw.ProcessFrame();
//Check the exit conditions.
if (g_pDemo->IsQuitRequired())
{
break;
}
}
// Finalize.
FinalizeDemo();
FinalizeGraphics();
NN_LOG("%s Done\n", ProgramName);
}
