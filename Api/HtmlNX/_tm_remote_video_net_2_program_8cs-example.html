using System;
using System.Threading;
namespace TmRemoteVideo
{
class Program
{
static readonly string videoOutDirectory = $"{Environment.GetEnvironmentVariable("TEMP")}\\Nintendo";
static readonly int retryInterval = 1000;
static readonly int retryCount = 10;
static void PrintRemoteVideoUpdate()
{
Console.WriteLine("OnRemoteVideoUpdate Signaled");
}
static void RegisterRemoteVideoUpdateEvent(Nintendo.Tm.Target target)
{
Console.WriteLine("Register a remote video update event.");
Nintendo.Tm.Target.RemoteVideoUpdateHandler handler = PrintRemoteVideoUpdate;
target.OnRemoteVideoUpdate += handler;
}
static void LaunchSampleProgram(Nintendo.Tm.Target target)
{
Console.WriteLine("Launch the sample program.");
var sdkRootPath = System.Environment.GetEnvironmentVariable("NINTENDO_SDK_ROOT");
var sampleProgramPath = sdkRootPath + "\\Resources\\Firmwares\\NX\\Sample\\Binaries\\GfxPrimitiveRenderer.nsp";
target.LaunchProgram(sampleProgramPath);
}
static void StartRemoteVideoRecording(Nintendo.Tm.Target target)
{
Console.WriteLine("Start remote video recording.");
var fileName = $"@{target.GetName()}_@{DateTime.Now.ToString("yyyyMMdd_HHmmss")}";
target.StartVideoRecording(videoOutDirectory, fileName, (int)Nintendo.Tm.VideoRecordingTime.Infinite);
}
static void StopRemoteVideoRecording(Nintendo.Tm.Target target)
{
Console.WriteLine("Stop remote video recording.");
target.StopVideoRecording();
}
static void OpenRemoteVideoStream(Nintendo.Tm.Target target)
{
Console.WriteLine("Open the remote video stream.");
target.OpenVideoStream();
}
static void SaveRecentlyRecordedRemoteVideo(Nintendo.Tm.Target target)
{
Console.WriteLine("Save recently recorded remote video.");
var fileName = $"@{target.GetName()}_@{DateTime.Now.ToString("yyyyMMdd_HHmmss")}";
target.SaveRecentVideo(videoOutDirectory, fileName);
}
static void PrintScrennImageInformation(Nintendo.Tm.Target target)
{
Console.WriteLine("Get the screen image size.");
int screenImageWidth;
int screenImageHeight;
target.GetScreenImageSize(out screenImageWidth, out screenImageHeight);
Console.WriteLine($"Screen Image : WidthSize = {screenImageWidth} HeightSize = {screenImageHeight}");
Console.WriteLine("Get the screen image data.");
var data = target.GetScreenImageData();
var dataSize = screenImageWidth * screenImageHeight * 3;
Console.WriteLine($"Top left pixel :R[ {data[0]} ] G[ {data[1]} ] B[ {data[2]} ]");
Console.WriteLine($"Bottom Right pixel:R[ {data[dataSize - 3]} ] G[ {data[dataSize - 2]} ] B[ {data[dataSize - 1]} ]");
}
static void CloseRemoteVideoStream(Nintendo.Tm.Target target)
{
Console.WriteLine("Close the remote video stream.");
target.CloseVideoStream();
}
static void KillSampleProgram(Nintendo.Tm.Target target)
{
Console.WriteLine("Kill the sample program.");
target.KillProgram();
}
static void UseVideoRecording(Nintendo.Tm.Target target)
{
StartRemoteVideoRecording(target);
Thread.Sleep(5000);
StopRemoteVideoRecording(target);
}
static void UseVideoStream(Nintendo.Tm.Target target)
{
// Because Nintendo.Tm.Target.SaveRecentVideo() sometimes fails at Nintendo.Tm.Error.InternalError,
// CloseRemoteVideoStream() is called if SaveRecentlyRecordedRemoteVideo() fails, and the process is retried from OpenRemoteVideoStream().
for (var i = 0; i < retryCount; i++)
{
OpenRemoteVideoStream(target);
Thread.Sleep(5000);
try
{
try
{
SaveRecentlyRecordedRemoteVideo(target);
}
catch (Exception e)
{
Console.Error.WriteLine(e.ToString());
Thread.Sleep(retryInterval);
continue;
}
PrintScrennImageInformation(target);
return;
}
finally
{
CloseRemoteVideoStream(target);
}
}
throw new Exception($"Could not save recently recorded remote video in {retryCount} times.");
}
static void Main(string[] args)
{
Nintendo.Tm.Target target;
try
{
if (!Nintendo.Tm.TargetManager.TryGetDefaultTarget(out target))
{
Console.Error.WriteLine("Could not get default target.");
return;
}
if (!target.GetIsConnected())
{
Console.Error.WriteLine("Default target is not connected.");
return;
}
RegisterRemoteVideoUpdateEvent(target);
LaunchSampleProgram(target);
try
{
UseVideoRecording(target);
UseVideoStream(target);
}
finally
{
KillSampleProgram(target);
}
}
catch (Exception e)
{
Console.Error.WriteLine(e.ToString());
}
}
}
}
