/**************************************************************************************************
*
* WorldSpaceTriangles:
*
* A class that implements the drawing of some animated triangles in world space coordinates.
* Its shader code takes in modelview and projection matrices as uniform data as a means of
* applying transformations to the triangles.
*
* Dependent shader code:
* - NvnSimple02.{vs|fs}
*
* Dependent projects:
* - NvnSimpleSharedLib
* - NvnSimpleShaderCompiler
*
* Compared with ScreenSpaceTriangles, the main addition that this class makes is the usage
* of uniform blocks. Illustrated is how to lookup uniform reflection information via the
* GraphicsShader class, how to prepare blocks of memory to contain uniform data, and how to
* bind those blocks of uniform data when generating GPU rendering commands.
*
**************************************************************************************************/
class WorldSpaceTriangles : public SceneBaseClass
{
public:
WorldSpaceTriangles() NN_NOEXCEPT :
m_IsInitialized(false),
m_GlslcData(nullptr),
m_pGraphicsSystem(nullptr),
m_CommandBuffer(),
m_CommandHandle(0),
m_Shader(),
m_VertexAndUniformDataMemPool()
{}
void Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT;
void DrawFrame() NN_NOEXCEPT;
void Finalize() NN_NOEXCEPT;
static const int NumTriangles = 2;
private:
void GenerateCommandBuffer() NN_NOEXCEPT;
static const int NumVertexAttributes = 2;
bool m_IsInitialized;
void* m_GlslcData;
GraphicsSystem* m_pGraphicsSystem;
GraphicsCommandBuffer m_CommandBuffer;
nvn::CommandHandle m_CommandHandle;
GraphicsShader m_Shader;
GraphicsMemoryPool m_VertexAndUniformDataMemPool;
TriangleState m_TriangleState[NumTriangles];
nvn::Buffer m_UniformBlockBuffers[NumTriangles];
nvn::Buffer m_PositionDataBuffer;
nvn::Buffer m_ColorDataBuffer;
nvn::Buffer m_IndexDataBuffer;
GraphicsRenderState m_RenderState;
nvn::VertexAttribState m_VertexAttribStates[NumVertexAttributes];
nvn::VertexStreamState m_VertexStreamStates[NumVertexAttributes];
const UniformBlockInfo* m_pUniformBlockInfo;
};
