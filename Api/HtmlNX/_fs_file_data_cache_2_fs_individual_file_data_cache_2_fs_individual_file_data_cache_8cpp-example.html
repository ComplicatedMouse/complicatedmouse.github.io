void RomFile()
{
// The mount name of the resource data and the file path of the test target.
const char* mountName = "rom";
const char* testTargetPath = "rom:/dummy.bmp";
// The buffer size when reading.
const size_t readBufferSize = 4 * 1024;
// Allocate 32 MB of cache memory for file data cache.
// The optimal size of cache memory differs, depending on the application. For a normal application, consider allocating even more than
// this amount of cache memory.
const size_t fileDataCacheSize = 32 * 1024 * 1024;
void* fileDataCache = std::malloc(fileDataCacheSize);
NN_ABORT_UNLESS_NOT_NULL(fileDataCache);
// Allocate file system cache memory and mount the file system for the resource data.
// (This allocated cache memory is used to store file system management information. It has nothing to do with file data cache.)
size_t fileSystemCacheSize;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::QueryMountRomCacheSize(&fileSystemCacheSize));
void* fileSystemCache = std::malloc(fileSystemCacheSize);
NN_ABORT_UNLESS_NOT_NULL(fileSystemCache);
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::MountRom(mountName, fileSystemCache, fileSystemCacheSize));
// Begin the sequential read benchmarking process.
NN_LOG("ROM sequential reading test\n");
// Enable individual file data cache and read sequentially.
{
nn::fs::EnableIndividualFileDataCache(testTargetPath, fileDataCache, fileDataCacheSize);
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunSequentialRead(testTargetPath, readBufferSize);
});
NN_LOG(" Individual cache enabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
nn::fs::DisableIndividualFileDataCache(testTargetPath);
}
// Read sequentially without enabling file data cache.
{
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunSequentialRead(testTargetPath, readBufferSize);
});
NN_LOG(" Cache disabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
}
NN_LOG("\n");
// Begin the random read benchmarking process.
NN_LOG("ROM random reading test\n");
// Enable individual file data cache and read randomly.
{
nn::fs::EnableIndividualFileDataCache(testTargetPath, fileDataCache, fileDataCacheSize);
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunRandomRead(testTargetPath, readBufferSize);
});
NN_LOG(" Individual cache enabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
nn::fs::DisableIndividualFileDataCache(testTargetPath);
}
// Read randomly without enabling file data cache.
{
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunRandomRead(testTargetPath, readBufferSize);
});
NN_LOG(" Cache disabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
}
NN_LOG("\n");
// Unmount the file system and free the various memory.
nn::fs::Unmount(mountName);
std::free(fileSystemCache);
std::free(fileDataCache);
}
void SaveDataFile()
{
nn::Result result;
// The read/write buffer size.
const size_t readBufferSize = 4 * 1024;
const size_t writeBufferSize = 4 * 1024;
// Allocate 32 MB of cache memory for file data cache.
// The optimal size of cache memory differs, depending on the application. For a normal application, consider allocating even more than
// this amount of cache memory.
const size_t fileDataCacheSize = 32 * 1024 * 1024;
void* fileDataCache = std::malloc(fileDataCacheSize);
NN_ABORT_UNLESS_NOT_NULL(fileDataCache);
// Initialize the account library.
nn::account::Initialize();
// Get the user account at the beginning.
// Aborts if there are no user accounts.
nn::account::Uid user;
int userCount = 0;
result = nn::account::ListAllUsers(&userCount, &user, 1);
NN_ABORT_UNLESS(result.IsSuccess() && userCount > 0);
// Open the user selected at application startup.
// Keep it open for the duration of user operations.
nn::account::UserHandle userHandle;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::account::OpenUser(&userHandle, user));
NN_UTIL_SCOPE_EXIT
{
nn::account::CloseUser(userHandle);
};
const size_t testTargetSize = 4 * 1024 * 1024;
const char* mountName = "save";
const char* testTargetPath = "save:/file";
// Mount the save data and create the file.
{
NN_LOG("Mount save data as \"%s\".\n", mountName);
// Create the selected user save data.
// If data already exists, does nothing and returns nn::ResultSuccess.
result = nn::fs::EnsureSaveData(user);
if (nn::fs::ResultUsableSpaceNotEnough::Includes(result))
{
// Error handling when the application does not have enough memory is required for the nn::fs::EnsureSaveData() function.
// The system automatically displays a message indicating that there was not enough capacity to create save data in the error viewer.
// The application must offer options to cancel account selection and to return to the prior scene.
NN_ABORT("Usable space not enough.\n");
}
// No further error handling is necessary because
// the process is aborted in the library if it fails for any other reason.
// Mount the save data as "save."
result = nn::fs::MountSaveData(mountName, user);
// Always abort when a failure occurs.
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
nn::fs::DirectoryEntryType directoryEntryType;
result = nn::fs::GetEntryType(&directoryEntryType, testTargetPath);
if (result.IsSuccess())
{
nn::fs::DeleteFile(testTargetPath);
}
result = nn::fs::CreateFile(testTargetPath, testTargetSize);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
}
// Start the sequential write benchmark.
NN_LOG("Save data sequential writing test\n");
// Enable individual file data cache and perform sequential write.
{
nn::fs::EnableIndividualFileDataCache(testTargetPath, fileDataCache, fileDataCacheSize);
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunSequentialWrite(mountName, testTargetPath, writeBufferSize);
});
NN_LOG(" Individual cache enabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
nn::fs::DisableIndividualFileDataCache(testTargetPath);
}
// Perform sequential write without enabling file data cache.
{
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunSequentialWrite(mountName, testTargetPath, writeBufferSize);
});
NN_LOG(" Cache disabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
}
NN_LOG("\n");
// Begin the sequential read benchmarking process.
NN_LOG("Save data sequential reading test\n");
// Enable individual file data cache and read sequentially.
{
nn::fs::EnableIndividualFileDataCache(testTargetPath, fileDataCache, fileDataCacheSize);
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunSequentialRead(testTargetPath, readBufferSize);
});
NN_LOG(" Individual cache enabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
nn::fs::DisableIndividualFileDataCache(testTargetPath);
}
// Read sequentially without enabling file data cache.
{
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunSequentialRead(testTargetPath, readBufferSize);
});
NN_LOG(" Cache disabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
}
NN_LOG("\n");
// Start the random write benchmark.
NN_LOG("Save data random writing test\n");
// Enable individual file data cache and perform random write.
{
nn::fs::EnableIndividualFileDataCache(testTargetPath, fileDataCache, fileDataCacheSize);
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunRandomWrite(mountName, testTargetPath, writeBufferSize);
});
NN_LOG(" Individual cache enabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
nn::fs::DisableIndividualFileDataCache(testTargetPath);
}
// Perform random write without enabling file data cache.
{
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunRandomWrite(mountName, testTargetPath, writeBufferSize);
});
NN_LOG(" Cache disabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
}
NN_LOG("\n");
// Begin the random read benchmarking process.
NN_LOG("Save data random reading test\n");
// Enable individual file data cache and read randomly.
{
nn::fs::EnableIndividualFileDataCache(testTargetPath, fileDataCache, fileDataCacheSize);
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunRandomRead(testTargetPath, readBufferSize);
});
NN_LOG(" Individual cache enabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
nn::fs::DisableIndividualFileDataCache(testTargetPath);
}
// Read randomly without enabling file data cache.
{
nn::TimeSpan elapsed = MeasureExecutionTime([=]() {
RunRandomRead(testTargetPath, readBufferSize);
});
NN_LOG(" Cache disabled = %" PRId64 " ms.\n", elapsed.GetMilliSeconds());
}
NN_LOG("\n");
// Mount the file system and free the memory.
nn::fs::Unmount(mountName);
std::free(fileDataCache);
} // NOLINT(impl/function_size)
} // namespace unnamed
/*
* This is the definition for nninitStartup(), which initializes the memory management functionality of the application.
*
* Normally, global file data caching is automatically enabled when starting the application.
* To accurately measure the individual file data caching benchmark, this sample overwrites the
* definition of nninitStartup() so that global file data caching is automatically enabled.
*/
// Global file data caching is not automatically enabled in the Windows environment.
// nninitStartup() is only defined when the application operating environment is the operating system of the NX development hardware.
#if defined(NN_BUILD_TARGET_PLATFORM_OS_NN)
extern "C" void nninitStartup()
{
// Gets the information for the memory available to the application.
nn::os::MemoryInfo memInfo;
nn::os::QueryMemoryInfo(&memInfo);
size_t size = static_cast<size_t>(memInfo.totalAvailableMemorySize) - memInfo.totalUsedMemorySize;
// Along with the preconditions for nn::os::SetMemoryHeapSize(), rounds down size as an integer multiplier of nn::os::MemoryBlockUnitSize.
size = nn::util::align_down(size, nn::os::MemoryBlockUnitSize);
// Sets the memory heap size.
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::SetMemoryHeapSize(size));
// From the memory heap, allocates the memory region to be used by malloc().
uintptr_t p;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::os::AllocateMemoryBlock(&p, size));
// Sets a memory region for malloc() use.
nn::init::InitializeAllocator(reinterpret_cast<void*>(p), size, true);
// Registers the memory allocator to use by implementation of the thread_local specifier.
nn::os::SetMemoryAllocatorForThreadLocal(
[](size_t size, size_t) { return std::malloc(size); },
[](void* ptr, size_t) { std::free(ptr); }
);
}
#endif // defined(NN_BUILD_TARGET_PLATFORM_OS_NN)
/*
* The application entry point.
*/
extern "C" void nnMain()
{
RomFile();
SaveDataFile();
}
