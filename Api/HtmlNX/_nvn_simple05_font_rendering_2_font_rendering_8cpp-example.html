void FontRendering::Initialize(GraphicsSystem* pGraphicsSystem) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
m_Font.InitializeFromNxSystemFont(m_pGraphicsSystem->GetAllocator(), nn::pl::SharedFontType_Standard);
m_FontKorean.InitializeFromNxSystemFont(m_pGraphicsSystem->GetAllocator(), nn::pl::SharedFontType_Korean);
m_TextRenderer.Initialize(m_pGraphicsSystem);
m_TextRenderer.SetScalingFactor(1.5f); // 1080 / 720
m_TextRenderer.SetCanvasDimensions(1920.0f, 1080.0f);
m_TextArea1 = m_TextRenderer.CreateTextArea(700, 400, &m_Font);
m_TextArea2 = m_TextRenderer.CreateTextArea(800, 500, &m_Font, 70, 1.3f);
m_TextArea3 = m_TextRenderer.CreateTextArea(700, 100, &m_FontKorean);
// build command buffer
m_CommandBuffer.Initialize(m_pGraphicsSystem);
GenerateCommandBuffer();
m_IsInitialized = true;
}
}
void FontRendering::DrawFrame() NN_NOEXCEPT
{
if (!m_IsPrebuiltCommandBufferUsed || m_IsScreenResolutionChanged)
{
GenerateCommandBuffer();
}
m_pGraphicsSystem->GetQueue()->SubmitCommands(1, &m_CommandHandle);
m_IsScreenResolutionChanged = false;
}
void FontRendering::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_CommandBuffer.Finalize();
m_TextRenderer.Finalize();
m_Font.Finalize();
m_pGraphicsSystem = nullptr;
m_CommandHandle = 0;
m_IsInitialized = false;
}
}

// Private methods

void FontRendering::GenerateCommandBuffer() NN_NOEXCEPT
{
m_TextRenderer.ClearText(m_TextArea1);
m_TextRenderer.ClearText(m_TextArea2);
m_TextRenderer.ClearText(m_TextArea3);
//In the NintendoSDK, UTF-8 is used as standard character encoding.
//The SimpleTextRenderer recives UTF-8 strings and converts them to UTF-32 for supporting multi bytes character rendering
//Add the u8 prefix to string literals to ensuring the input strings use UTF-8 character encoding
m_TextRenderer.AppendText(m_TextArea1, u8"Testing nn::fontll + NX nn::pl::SharedFont\nHello, world!\n");
m_TextRenderer.AppendText(m_TextArea1, u8"お元気ですか？頑張ってね！");
m_TextRenderer.AppendText(m_TextArea2, u8"Nintendo");
m_TextRenderer.AppendText(m_TextArea2, u8"SDK");
m_TextRenderer.AppendText(m_TextArea2, u8"\nNVN");
m_TextRenderer.AppendText(m_TextArea3, u8"안녕하신가요? 화이팅입니다!");
nvn::CommandBuffer* pCommandBuffer = m_CommandBuffer.StartRecording();
{
m_pGraphicsSystem->SetDefaultViewportAndScissor(pCommandBuffer);
static float color[4] = { 1.0f, 1.0f, 1.0f, 1.0f };
pCommandBuffer->ClearColor(0, color, nvn::ClearColorMask::RGBA);
m_TextRenderer.StartDrawing(pCommandBuffer);
m_TextRenderer.DrawTextArea(m_TextArea1, 300, 330, Color(0x000000), pCommandBuffer);
m_TextRenderer.DrawTextArea(m_TextArea2, 100, 100, Color(0x044cae), pCommandBuffer);
m_TextRenderer.DrawTextArea(m_TextArea3, 300, 446, Color(0x000000), pCommandBuffer);
}
m_CommandHandle = m_CommandBuffer.EndRecording();
}

}
