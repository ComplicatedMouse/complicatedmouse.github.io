using System;
using System.Collections.Generic;
using Nintendo.AutoTestAssistTool;
using Nintendo.Tm;
namespace autoTestAssistTool.logReaderSample
{
class Program
{
LogStore m_LogStore;
Program(LogStore logStore)
{
m_LogStore = logStore;
}
void LogSample()
{
Console.WriteLine("-----------------------------");
Console.WriteLine("Print log");
for (int i = 0; i < 10; ++i)
{
if (m_LogStore.WaitForNextLine(TimeSpan.FromSeconds(1)))
{
m_LogStore.MoveToNextLine();
Console.WriteLine(m_LogStore.GetLine());
}
else
{
Console.WriteLine("New log was not found");
}
}
System.Threading.Thread.Sleep(TimeSpan.FromSeconds(5));
Console.WriteLine("-----------------------------");
Console.WriteLine("Print log after waiting for outputting specified text");
// Set the string to use for regular expression matching.
var keyString = @".*";
m_LogStore.MoveToNewestLine();
if (m_LogStore.WaitUntilMatched(keyString, TimeSpan.FromSeconds(10)))
{
Console.WriteLine(m_LogStore.GetLine());
}
else
{
Console.WriteLine("Matching timeout");
}
Console.WriteLine("-----------------------------");
Console.WriteLine("Move to 10 sec before");
m_LogStore.MoveToFormerLineByTime(TimeSpan.FromSeconds(10));
Console.WriteLine(m_LogStore.GetLine());
Console.WriteLine("-----------------------------");
Console.WriteLine("Move to newest log");
m_LogStore.MoveToNewestLine();
Console.WriteLine(m_LogStore.GetLine());
Console.WriteLine("-----------------------------");
Console.WriteLine("Search log before");
m_LogStore.SearchFormerLog(keyString);
Console.WriteLine(m_LogStore.GetLine());
}
static void Main(string[] args)
{
try
{
var targets = new List<Target>();
try
{
var allTargets = TargetManager.GetTargets();
foreach (var target in allTargets)
{
if (target.GetIsConnected())
{
targets.Add(target);
}
}
}
catch (TmException e)
{
Console.WriteLine(e.Message);
Console.WriteLine(e.StackTrace);
return;
}
if (targets.Count == 0)
{
Console.WriteLine("Connected target was not found.");
return;
}
Console.WriteLine("Please select target.");
int count = 0;
foreach (var target in targets)
{
Console.WriteLine($"{count++} : " + target.GetSerialNumber());
}
var input = Console.ReadLine();
int index;
if (!Int32.TryParse(input, out index))
{
Console.WriteLine("Invalid input.");
return;
}
if (index >= 0 && index < targets.Count)
{
var serial = targets[index].GetSerialNumber();
using (var logStore = new LogStore(serial, 1000))
{
var program = new Program(logStore);
program.RunSamples();
}
}
else
{
Console.WriteLine("Invalid number was input.");
}
}
catch (Exception e)
{
Console.WriteLine(e.Message);
}
}
void RunSamples()
{
LogSample();
}
}
