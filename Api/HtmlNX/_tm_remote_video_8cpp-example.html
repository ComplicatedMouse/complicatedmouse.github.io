bool ErrorCheckAndPrint(nn::tm::Error result, const char* file, int line, const char* operation)
{
if (result != nn::tm::Error_Ok)
{
nn::tm::ErrorText errorText;
nn::tm::GetErrorTextEn(&errorText, result);
printf("%s:%d %s failed because %s.\n", file, line, operation, errorText.text);
return false;
}
return true;
}

{

{
printf("Signal Success \n");
}

{

}

bool StopVideoRecord(nn::tm::TargetHandle target)
{
HANDLE eventHandle = OpenEvent(EVENT_ALL_ACCESS, FALSE, EventName);
printf("-------- StopVideoRecording -------- \n");
if (!ERROR_CHECK_AND_PRINT(nn::tm::StopVideoRecording(target)))
{
return false;
}
printf("StopVideoRecording After Wait Signal ... ");
DWORD stopRecordingSignalResult = WaitForSingleObject(eventHandle, 10000);
PrintSignalResult(stopRecordingSignalResult);
ResetEvent(eventHandle);
return true;
}

void VideoRecordingExitReasonResult(nn::tm::VideoRecordingExitReason exitReason)
{
if (exitReason == nn::tm::VideoRecordingExitReason::VideoRecordingExitReason_None)
{
printf("Recording Exit Reason : Video recording has not been requested. \n");
}
else if (exitReason == nn::tm::VideoRecordingExitReason::VideoRecordingExitReason_TimeCompleted)
{
printf("Recording Exit Reason : The specified time elapsed. \n");
}
else if (exitReason == nn::tm::VideoRecordingExitReason::VideoRecordingExitReason_StopTriggered)
{
printf("Recording Exit Reason : nn::tm::StopVideoRecording was called. \n");
}
else if (exitReason == nn::tm::VideoRecordingExitReason::VideoRecordingExitReason_GetVideoFailed)
{
printf("Recording Exit Reason : The target video was not available. \n");
}
else if (exitReason == nn::tm::VideoRecordingExitReason::VideoRecordingExitReason_ConnectionInterrupted)
{
printf("Recording Exit Reason : Connection to the target was interrupted. \n");
}
else
{
printf("Recording Exit Reason : Unexpected error. \n");
}
}

bool UseVideoRecording(nn::tm::TargetHandle target, const char* tempDirectoryPath)
{
if (!StartVideoRecord(target, tempDirectoryPath))
{
return false;
}
Sleep(5000);
if (!StopVideoRecord(target))
{
return false;
}
if (!GetLastVideoRecordingExitReason(target))
{
return false;
}
return true;
}
bool UseVideoStream(nn::tm::TargetHandle target, const char* tempDirectoryPath)
{
// Because nn::tm::SaveRecentVideo() sometimes fails at nn::tm::Error_InternalError,
// CloseTargetVideo() is called if SaveRecentVideo() fails, and the process is retried from OpenTargetVideo().
bool saveSuccess = false;
for (int i = 0; i < RetryCount; i++)
{
if (!OpenTargetVideo(target))
{
return false;
}
Sleep(5000);
if (SaveRecentVideo(target, tempDirectoryPath))
{
saveSuccess = true;
break;
}
if (!CloseTargetVideo(target))
{
return false;
}
}
if (!saveSuccess)
{
return false;
}
if (!GetScreenImageSizeAndData(target))
{
return false;
}
if (!CloseTargetVideo(target))
{
return false;
}
return true;
}
int main()
{
// Disable standard output buffering.
if (setvbuf(stdout, NULL, _IONBF, 0) != 0)
{
printf("Failed to disable log buffering\n");
return 1;
}
nn::tm::TargetHandle target;
nn::tm::GetDefaultTarget(&target);
HANDLE eventHandle = CreateEvent(NULL, false, false, EventName);
char tempDirectoryPath[PathStringBufferSize];
GetEnvironmentVariableA("TEMP", tempDirectoryPath, sizeof(tempDirectoryPath));
errno_t err = strcat_s(tempDirectoryPath, PathStringBufferSize, "\\Nintendo");
_ASSERT(err == 0);
if (SetRemoteVideoUpdateEvent(target, eventHandle))
{
if (LaunchSampleProgram(target))
{
if (UseVideoRecording(target, tempDirectoryPath))
{
UseVideoStream(target, tempDirectoryPath);
}
TerminateSampleProgram(target);
}
UnsetRemoteVideoUpdateEvent(target, eventHandle);
}
return 0;
}
