void GraphicsMemoryPool::Initialize(GraphicsSystem* pGraphicsSystem, nvn::MemoryPoolFlags flags, size_t poolSize,
const char* memoryPoolDebugLabel) NN_NOEXCEPT
{
Initialize(pGraphicsSystem, flags, poolSize, nullptr, 0, memoryPoolDebugLabel);
}
void GraphicsMemoryPool::Initialize(GraphicsSystem* pGraphicsSystem, nvn::MemoryPoolFlags flags, size_t poolSize,
void* initialBuffer, size_t initialBufferSize, const char* memoryPoolDebugLabel) NN_NOEXCEPT
{
// initialBuffer and initialBufferSize is provided as a convenience to copy over data into the buffer that the
// MemoryPool is initialized with prior to nvn::MemoryPool::Initialize.
if (!m_IsInitialized)
{
m_pGraphicsSystem = pGraphicsSystem;
m_PoolSize = nn::util::align_up<size_t>(poolSize, NVN_MEMORY_POOL_STORAGE_GRANULARITY);
m_PoolBuf = pGraphicsSystem->GetAllocator()->Allocate(m_PoolSize, NVN_MEMORY_POOL_STORAGE_ALIGNMENT);
if (initialBuffer != nullptr && initialBufferSize > 0)
{
memcpy(m_PoolBuf, initialBuffer, initialBufferSize);
}
nvn::MemoryPoolBuilder poolBuilder;
poolBuilder.SetDefaults();
poolBuilder.SetDevice(pGraphicsSystem->GetDevice());
poolBuilder.SetFlags(flags);
poolBuilder.SetStorage(m_PoolBuf, m_PoolSize);
if (!m_Pool.Initialize(&poolBuilder))
{
NN_ASSERT(0, "Failed to initialize buffer memory pool");
}
if (memoryPoolDebugLabel)
{
m_Pool.SetDebugLabel(memoryPoolDebugLabel);
}
m_CurOffset = 0;
m_IsInitialized = true;
}
}
void GraphicsMemoryPool::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_Pool.Finalize();
m_pGraphicsSystem->GetAllocator()->Free(m_PoolBuf);
m_PoolBuf = nullptr;
m_PoolSize = 0;
m_CurOffset = 0;
m_IsInitialized = false;
}
}
ptrdiff_t GraphicsMemoryPool::Allocate(size_t size) NN_NOEXCEPT
{
return Allocate(size, DefaultAlignment);
}
ptrdiff_t GraphicsMemoryPool::Allocate(size_t size, size_t alignment) NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT(size > 0);
m_CurOffset = nn::util::align_up(m_CurOffset, alignment);
ptrdiff_t newChunk = m_CurOffset;
m_CurOffset += size;
NN_ASSERT(static_cast<size_t>(m_CurOffset) <= m_PoolSize);
return newChunk;
}
void GraphicsMemoryPool::AllocateBuffer(nvn::Buffer* pBuffer, size_t size) NN_NOEXCEPT
{
AllocateBuffer(pBuffer, size, DefaultAlignment);
}
void GraphicsMemoryPool::AllocateBuffer(nvn::Buffer* pBuffer, size_t size, size_t alignment) NN_NOEXCEPT
{
NN_ASSERT(pBuffer != nullptr);
ptrdiff_t offset = Allocate(size, alignment);
nvn::BufferBuilder bufferBuilder;
bufferBuilder.SetDefaults();
bufferBuilder.SetDevice(m_pGraphicsSystem->GetDevice());
bufferBuilder.SetStorage(&m_Pool, offset, size);
if (!pBuffer->Initialize(&bufferBuilder))
{
NN_ASSERT(false, "nvn::Buffer initialization failed.\n");
}
}
nvn::Buffer* GraphicsMemoryPool::AllocateBuffer(size_t size) NN_NOEXCEPT
{
return AllocateBuffer(size, DefaultAlignment);
}
nvn::Buffer* GraphicsMemoryPool::AllocateBuffer(size_t size, size_t alignment) NN_NOEXCEPT
{
nvn::Buffer* pBuffer = new nvn::Buffer;
AllocateBuffer(pBuffer, size, alignment);
return pBuffer;
}
void GraphicsMemoryPool::Reset() NN_NOEXCEPT
{
m_CurOffset = 0;
}
void GraphicsMemoryPool::Flush() NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
// Flushes CPU cache back to GPU.
// Flushing is not necessary for CPU_UNCACHED memory pools.
if (m_Pool.GetFlags() & nvn::MemoryPoolFlags::CPU_CACHED)
{
m_Pool.FlushMappedRange(0, m_PoolSize);
}
}
void GraphicsMemoryPool::Invalidate() NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
// invalidates CPU cache so that latest data written to by the GPU is brought back into the CPU cache
m_Pool.InvalidateMappedRange(0, m_PoolSize);
}
size_t GraphicsMemoryPool::GetPoolSize() const NN_NOEXCEPT
{
return m_PoolSize;
}
size_t GraphicsMemoryPool::GetUsedSize() const NN_NOEXCEPT
{
return static_cast<size_t>(m_CurOffset);
}
nvn::MemoryPool* GraphicsMemoryPool::GetMemoryPool() NN_NOEXCEPT
{
return &m_Pool;
}
uint8_t* GraphicsMemoryPool::GetBaseCpuAddress() NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
NN_ASSERT((m_Pool.GetFlags() & nvn::MemoryPoolFlags::CPU_NO_ACCESS) != nvn::MemoryPoolFlags::CPU_NO_ACCESS);
// nvn::MemoryPool::Map returns NULL for CPU_NO_ACCESS MemoryPools so we add the above assertion.
// On NX platform, technically we could return the m_PoolBuf pointer and have the application safely access
// the memory using it, regardless of the CPU access flags. However, on Windows nvn::MemoryPool::Map must be used.
// cast to make pointer arithmetic easier
return static_cast<uint8_t*>(m_Pool.Map());
}
nvn::BufferAddress GraphicsMemoryPool::GetBaseGpuAddress() NN_NOEXCEPT
{
NN_ASSERT(m_IsInitialized);
return m_Pool.GetBufferAddress();
}
