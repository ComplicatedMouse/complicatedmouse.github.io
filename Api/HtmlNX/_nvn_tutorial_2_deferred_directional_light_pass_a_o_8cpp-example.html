/*
* DeferredDirectionalLightPassAO::UpdateUniforms
* --------------------------------------------
* Updates the lighting pass uniforms.
*/
void DeferredDirectionalLightPassAO::UpdateUniforms()
{
DeferredDirectionalLightAO::LightInfoUniformBlockData* uniformMapLightInfo = reinterpret_cast<DeferredDirectionalLightAO::LightInfoUniformBlockData*>(m_pLightUniformBlockLightInfo->GetMappedPointer());
uniformMapLightInfo->SetUniform_u_invScreenWidth(m_InvScreenWidth);
uniformMapLightInfo->SetUniform_u_invScreenHeight(m_InvScreenHeight);
uniformMapLightInfo->SetUniform_u_globalAmbient(m_GlobalAmbient);
float val[4];
nn::util::VectorStore(val, m_EyePos);
uniformMapLightInfo->SetUniform_u_eyePos(val);
nn::util::VectorStore(val, m_LightDir);
uniformMapLightInfo->SetUniform_u_lightDir(val);
nn::util::VectorStore(val, m_LightIntensity);
uniformMapLightInfo->SetUniform_u_lightIntensity(val);
uniformMapLightInfo->SetUniform_u_OutsideMapVal(m_OutsideShadowMapVal);
uniformMapLightInfo->SetUniform_u_invShadowMapWidth(1.0f / static_cast<float>(m_ShadowMapWidth));
uniformMapLightInfo->SetUniform_u_invShadowMapHeight(1.0f / static_cast<float>(m_ShadowMapHeight));
nn::util::Float4x4 temp;
nn::util::MatrixStore(&temp, m_LightViewMatrix);
uniformMapLightInfo->SetUniform_u_lightViewMatrix(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_LightProjMatrix);
uniformMapLightInfo->SetUniform_u_lightProjMatrix(*reinterpret_cast<float(*)[16]>(&temp));
nn::util::MatrixStore(&temp, m_InvViewProj);
uniformMapLightInfo->SetUniform_u_invViewProj(*reinterpret_cast<float(*)[16]>(&temp));
DeferredDirectionalLightAO::TextureHandlesUniformBlockData* uniformMapTextureHandles = reinterpret_cast<DeferredDirectionalLightAO::TextureHandlesUniformBlockData*>(m_pLightUniformBlockTextureHandles->GetMappedPointer());
uniformMapTextureHandles->SetUniform_u_diffuseTex(m_DiffuseTex);
uniformMapTextureHandles->SetUniform_u_specularTex(m_SpecularTex);
uniformMapTextureHandles->SetUniform_u_normalTex(m_NormalTex);
uniformMapTextureHandles->SetUniform_u_shadowTex(m_ShadowTex);
uniformMapTextureHandles->SetUniform_u_depthTex(m_DepthTex);
uniformMapTextureHandles->SetUniform_u_ssaoTex(m_SSAOTex);
DeferredDirectionalLightAO::ControlsUniformBlockData* uniformMapControls = reinterpret_cast<DeferredDirectionalLightAO::ControlsUniformBlockData*>(m_pLightUniformBlockControls->GetMappedPointer());
uniformMapControls->SetUniform_u_useSSAO(m_UseSSAO);
}
/*
* DeferredDirectionalLightPassAO::SetupUniforms
* -------------------------------------------
* Sets up the uniform buffers.
*/
void DeferredDirectionalLightPassAO::SetupUniforms(UniformBufferManager* pUniformBufferManager)
{
m_pLightUniformBlockLightInfo = pUniformBufferManager->CreateUniformBuffer(sizeof(DeferredDirectionalLightAO::LightInfoUniformBlockData));
m_pLightUniformBlockTextureHandles = pUniformBufferManager->CreateUniformBuffer(sizeof(DeferredDirectionalLightAO::TextureHandlesUniformBlockData));
m_pLightUniformBlockControls = pUniformBufferManager->CreateUniformBuffer(sizeof(DeferredDirectionalLightAO::ControlsUniformBlockData));
}
/*
* DeferredDirectionalLightPassAO::Draw
* ----------------------------------
* Draws a screen space quad and calculates the lighting
* contribution of the direction light for each fragment.
*/
void DeferredDirectionalLightPassAO::Draw(NVNcommandBuffer* pCommandBuffer) const
{
std::vector<NVNProgramData*>& programData = m_pAssetData->GetProgramData();
/* Bind the uniform buffer. */
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredDirectionalLightAO::LightInfoUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pLightUniformBlockLightInfo->GetCurrentBufferAddress(),
sizeof(DeferredDirectionalLightAO::LightInfoUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredDirectionalLightAO::TextureHandlesUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pLightUniformBlockTextureHandles->GetCurrentBufferAddress(),
sizeof(DeferredDirectionalLightAO::TextureHandlesUniformBlockData));
nvnCommandBufferBindUniformBuffer(pCommandBuffer,
NVN_SHADER_STAGE_FRAGMENT,
DeferredDirectionalLightAO::ControlsUniformBlockData::GetBinding(NVN_SHADER_STAGE_FRAGMENT),
m_pLightUniformBlockControls->GetCurrentBufferAddress(),
sizeof(DeferredDirectionalLightAO::ControlsUniformBlockData));
/* Bind the shader program. */
nvnCommandBufferBindProgram(pCommandBuffer, &programData[0]->m_Program, programData[0]->m_ShaderStages);
/* Draws a full screen covering triangle by giving vertex ID's to the shader */
nvnCommandBufferDrawArrays(pCommandBuffer, NVNdrawPrimitive::NVN_DRAW_PRIMITIVE_TRIANGLES, 0, 3);
}
/*
* DeferredDirectionalLightPassAO::SetUseSSAO
* -----------------------------------------
* Enables/disables screen space ambient occlusion.
*/
bool DeferredDirectionalLightPassAO::ToggleUseSSAO()
{
return m_UseSSAO = !m_UseSSAO;
}
