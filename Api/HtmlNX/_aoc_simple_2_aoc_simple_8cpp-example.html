#include <new>
#include <nn/aoc.h>
#include <nn/fs.h>
#include <nn/fs/fs_AddOnContent.h>
#include <nn/nn_Assert.h>
#include <nn/nn_Log.h>
#include <nn/nn_Abort.h>
namespace {
// Buffer for listing the downloadable content indices.
const int MaxListupCount = 256;
nn::aoc::AddOnContentIndex g_AocListupBuffer[MaxListupCount];
// Buffer for loading the content of files within downloadable content.
const int ReadBufferSize = 1024;
nn::Bit8 g_ReadBuffer[ReadBufferSize] = {};
}
extern "C" void nnMain()
{
nn::Result result;
NN_LOG("AocSimple\n");
// Get the number of downloadable content instances.
int aocCount = nn::aoc::CountAddOnContent();
NN_LOG("CountAddOnCountent -> %d\n", aocCount);
// List the number of downloadable content instances.
int listupCount = nn::aoc::ListAddOnContent(g_AocListupBuffer, 0, MaxListupCount);
NN_LOG("ListAddOnCountent -> %d\n", listupCount);
for (int i = 0; i < listupCount; ++i)
{
NN_LOG(" Index[%d]: %d\n", i, g_AocListupBuffer[i]);
}
// Load and display the listed downloadable content.
for (int i = 0; i < listupCount; ++i)
{
// The index to access.
auto index = g_AocListupBuffer[i];
// Mount downloadable content.
size_t cacheSize = 0;
result = nn::fs::QueryMountAddOnContentCacheSize(&cacheSize, index);
NN_ASSERT(result.IsSuccess());
char* mountCacheBuffer = new(std::nothrow) char[cacheSize];
NN_ASSERT_NOT_NULL(mountCacheBuffer);
const char MountName[] = "aoc";
result = nn::fs::MountAddOnContent(
MountName, index, mountCacheBuffer, cacheSize
);
NN_ABORT_UNLESS_RESULT_SUCCESS(result);
// Access files from mounted downloadable content.
nn::fs::FileHandle fileHandle;
result = nn::fs::OpenFile(&fileHandle, "aoc:/testdata", nn::fs::OpenMode::OpenMode_Read);
NN_ASSERT(result.IsSuccess());
size_t readSize;
result = nn::fs::ReadFile(&readSize, fileHandle, 0, g_ReadBuffer, sizeof(g_ReadBuffer));
NN_ASSERT(result.IsSuccess());
nn::fs::CloseFile(fileHandle);
g_ReadBuffer[readSize] = '\0';
NN_LOG("testdata(Index = %d)\n", index);
NN_LOG("%s\n", g_ReadBuffer);
// Unmount downloadable content.
nn::fs::Unmount(MountName);
delete[] mountCacheBuffer;
}
}
