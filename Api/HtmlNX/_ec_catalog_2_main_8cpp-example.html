void MountContents()
{
static constexpr size_t MountRomCacheBufferSize = 4 * 1024;
static constexpr size_t EntryListCapacity = 100;
static char s_MountRomCacheBuffer[MountRomCacheBufferSize];
size_t mountRomCacheUseSize = 0;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::QueryMountRomCacheSize(&mountRomCacheUseSize));
NN_ASSERT(mountRomCacheUseSize <= MountRomCacheBufferSize);
NN_LOG("Mount contents.\n");
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::MountRom(CONTENTS_MOUNT_POINT, s_MountRomCacheBuffer, MountRomCacheBufferSize));
nn::fs::DirectoryEntry entryList[EntryListCapacity];
nn::fs::DirectoryHandle h = {};
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::OpenDirectory(&h, CONTENTS_MOUNT_POINT":/", nn::fs::OpenDirectoryMode_All));
int64_t n = 0;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::ReadDirectory(&n, entryList, h, EntryListCapacity));
NN_LOG("%d entry found.\n", static_cast<int>(n));
for(int64_t i = 0; i < n; i++)
{
auto& e = entryList[i];
NN_LOG(" %s%s\n", e.name, (e.directoryEntryType == nn::fs::DirectoryEntryType_Directory ? "/" : ""));
}
nn::fs::CloseDirectory(h);
}
void UnmountContents()
{
nn::fs::Unmount(CONTENTS_MOUNT_POINT);
}
bool LoadFileToBuffer(void** ppOutFileBuffer, size_t* pOutFileBufferSize, nns::gfx::GraphicsFramework* pGfw, const char* pFilePath)
{
bool isSuccess = false;
size_t fileSize = 0;
void* pFileBuffer = nullptr;
nn::fs::FileHandle hFile;
NN_ABORT_UNLESS_RESULT_SUCCESS(nn::fs::OpenFile(&hFile, pFilePath, nn::fs::OpenMode_Read));
int64_t fileSizeAsInt64 = 0;
if(nn::fs::GetFileSize(&fileSizeAsInt64, hFile).IsFailure())
{
nn::fs::CloseFile(hFile);
NN_ABORT("Failed to get font file size");
}
fileSize = static_cast<size_t>(fileSizeAsInt64);
pFileBuffer = pGfw->AllocateMemory(fileSize, 8);
size_t readSize = 0;
if(nn::fs::ReadFile(&readSize, hFile, 0, pFileBuffer, fileSize).IsFailure())
{
nn::fs::CloseFile(hFile);
NN_ABORT("Failed to read font file");
}
if(fileSize == readSize)
{
isSuccess = true;
}
nn::fs::CloseFile(hFile);
if(!isSuccess && (pFileBuffer != nullptr))
{
pGfw->FreeMemory(pFileBuffer);
pFileBuffer = nullptr;
}
*ppOutFileBuffer = pFileBuffer;
*pOutFileBufferSize = fileSize;
return isSuccess;
}
// Initialize GraphicsFramework.
void InitializeGraphicsFramework(nns::gfx::GraphicsFramework* pGfw)
{
nns::gfx::GraphicsFramework::InitializeGraphicsSystem(GraphicsMemorySize);
nns::gfx::GraphicsFramework::FrameworkInfo info;
{
info.SetDefault();
info.SetDisplayWidth(::FrameBufferWidth);
info.SetDisplayHeight(::FrameBufferHeight);
info.SetDepthStencilBufferEnabled(false);
info.SetRootCommandBufferCommandMemorySize(::CommandBufferMemorySize);
nn::gfx::util::ScalableViewport::WindowCoordinate virtualWindowCoordinate;
nn::gfx::util::ScalableViewport::WindowCoordinate physicalWindowCoord;
{
float windowWidth, windowHeight;
windowWidth = static_cast<float>(::FrameBufferWidth);
windowHeight = static_cast<float>(::FrameBufferHeight);
virtualWindowCoordinate.SetWidth(windowWidth);
virtualWindowCoordinate.SetHeight(windowHeight);
virtualWindowCoordinate.SetOriginMode(nn::gfx::util::ScalableViewport::OriginMode_UpperLeft);
physicalWindowCoord.SetWidth(windowWidth);
physicalWindowCoord.SetHeight(windowHeight);
physicalWindowCoord.SetOriginMode(nn::gfx::util::ScalableViewport::OriginMode_UpperLeft);
}
info.SetPhysicalWindowCoordinate(physicalWindowCoord);
info.SetVirtualWindowCoordinate(virtualWindowCoordinate);
info.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_BufferView, 16);
info.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_TextureView, 1024);
info.SetDescriptorPoolSlotCount(nn::gfx::DescriptorPoolType_Sampler, 1024);
}
pGfw->Initialize(info);
}
// Release GraphicsFramework.
void FinalizeGraphicsFramework(nns::gfx::GraphicsFramework* pGfw)
{
pGfw->Finalize();
}
void* SetupApplicationCustomFont(
nns::gfx::GraphicsFramework* pGfw,
nns::dbgui::InterfaceInfo& dbguiInterfaceInfo)
{
void* pFontBuffer = nullptr;
size_t fontBufferSize = 0;
if(LoadFileToBuffer(&pFontBuffer, &fontBufferSize, pGfw, CONTENTS_MOUNT_POINT":/NotoSansCJKjp-Regular.otf"))
{
const float characterSize = 22.0f;
const bool isCompressed = false;
dbguiInterfaceInfo.SetCustomTtfWithRanges(pFontBuffer, fontBufferSize, isCompressed, characterSize, g_GlyphRanges);
}
return pFontBuffer;
}
void* InitializeDbgui(
nns::gfx::GraphicsFramework* pGfw,
int imguiDisplayWidth,
int imguiDisplayHeight,
int bufferCount)
{
nns::dbgui::InterfaceInfo dbguiInterfaceInfo;
dbguiInterfaceInfo.SetDefault();
dbguiInterfaceInfo.SetRenderingConfiguration(
imguiDisplayWidth,
imguiDisplayHeight,
static_cast<int>(pGfw->GetScalableViewport()->GetPhysicalWindowCoordinate()->GetWidth()),
static_cast<int>(pGfw->GetScalableViewport()->GetPhysicalWindowCoordinate()->GetHeight()));
dbguiInterfaceInfo.SetBufferCount(bufferCount);
dbguiInterfaceInfo.SetAllocator(
pGfw->GetAllocateFunction(),
pGfw->GetFreeFunction(),
pGfw->GetAllocateFunctionUserData());
dbguiInterfaceInfo.SetDescriptorPoolData(
pGfw->GetDescriptorPool(nn::gfx::DescriptorPoolType_Sampler),
pGfw->AllocateDescriptorSlot(nn::gfx::DescriptorPoolType_Sampler, nns::dbgui::GetRequiredSamplerDescriptorSlotCount()),
pGfw->GetDescriptorPool(nn::gfx::DescriptorPoolType_TextureView),
pGfw->AllocateDescriptorSlot(nn::gfx::DescriptorPoolType_TextureView, nns::dbgui::GetRequiredTextureViewDescriptorSlotCount()),
pGfw->GetDescriptorPool(nn::gfx::DescriptorPoolType_BufferView),
pGfw->AllocateDescriptorSlot(nn::gfx::DescriptorPoolType_BufferView, nns::dbgui::GetRequiredBufferViewDescriptorSlotCount()));
void* pOutFontBuffer = SetupApplicationCustomFont(pGfw, dbguiInterfaceInfo);
nns::dbgui::Initialize(pGfw->GetDevice(), pGfw->GetQueue(), dbguiInterfaceInfo);
return pOutFontBuffer;
}
void BeginCommand()
{
nns::dbgui::NewFrame(); // Perform EndFrame() inside nns::dbgui::Render() of EndCommand().
// Draw a full-screen window.
ImGui::SetNextWindowBgAlpha(1.0f);
ImGui::SetNextWindowPos(ImVec2(0, 0));
ImGui::SetNextWindowSize(ImVec2(::FrameBufferWidth, ::FrameBufferHeight));
ImGui::Begin(
"Window",
nullptr,
ImGuiWindowFlags_NoTitleBar |
ImGuiWindowFlags_NoResize |
ImGuiWindowFlags_NoMove |
ImGuiWindowFlags_NoScrollbar |
ImGuiWindowFlags_NoSavedSettings |
ImGuiWindowFlags_NoInputs);
}
void EndCommand(nns::gfx::GraphicsFramework* pGfw, int bufferIndex)
{
ImGui::End();
pGfw->BeginFrame(bufferIndex);
nn::gfx::CommandBuffer* pCommandBuffer = pGfw->GetRootCommandBuffer(bufferIndex);
nn::gfx::ColorTargetView* pScreenTarget = pGfw->GetColorTargetView();
pCommandBuffer->ClearColor(pScreenTarget, 0, 0, 0, 0, nullptr);
pCommandBuffer->SetRenderTargets(1, &pScreenTarget, nullptr);
pCommandBuffer->SetViewportScissorState(pGfw->GetViewportScissorState());
pCommandBuffer->ClearColor(
pGfw->GetScanBufferView(pGfw->GetNextScanBufferIndex()), 0, 0, 0, 0, nullptr);
nns::dbgui::Render(pCommandBuffer);
nn::gfx::Texture* pScanBuffer = pGfw->GetScanBuffer(pGfw->GetNextScanBufferIndex());
nn::gfx::Texture* pColorBuffer = pGfw->GetColorBuffer();
nn::gfx::TextureCopyRegion srcRegion;
srcRegion.SetDefault();
srcRegion.SetOffsetU(0);
srcRegion.SetOffsetV(0);
srcRegion.SetWidth(::FrameBufferWidth);
srcRegion.SetHeight(::FrameBufferHeight);
nn::gfx::TextureCopyRegion dstRegion;
dstRegion.SetDefault();
dstRegion.SetOffsetU(0);
dstRegion.SetOffsetV(0);
dstRegion.SetWidth(::FrameBufferWidth);
dstRegion.SetHeight(::FrameBufferHeight);
pCommandBuffer->BlitImage(pScanBuffer, dstRegion, pColorBuffer, srcRegion, 0);
pGfw->EndFrame(bufferIndex, false);
}
// The calculation callback function.
// Only accept input here.
void CalculateCallback(nns::gfx::GraphicsFramework* pGfw, void* pUserData)
{
NN_UNUSED(pGfw);
auto pArg = reinterpret_cast<::CallbackArgument*>(pUserData);
nns::hid::ControllerManager* pCtrlMgr = pArg->pControllerManager;
// Update input.
pCtrlMgr->Update();
nns::hid::Controller* pInput = pCtrlMgr->GetController(nns::hid::ControllerId_GamePad, 0);
NN_ASSERT_NOT_NULL(pInput);
// Input processing.
pArg->pScene->Control(*reinterpret_cast<nns::hid::GamePad*>(pInput));
}
// The screen display callback function.
void MakeCommandCallback(nns::gfx::GraphicsFramework* pGfw, int bufferIndex, void* pUserData)
{
auto pArg = reinterpret_cast<::CallbackArgument*>(pUserData);
SceneData* pSceneDataInfo = pArg->pSceneDataInfo;
BeginCommand();
// Get the screen change information.
SceneBase::DisplayStatus changeStatus = pArg->pScene->GetStatus();
// Generate the display content in each branch target scene.
if((changeStatus != pArg->status) &&
(changeStatus != SceneBase::DisplayStatus::DisplayStatus_None))
{
switch(changeStatus)
{
case SceneBase::DisplayStatus::DisplayStatus_Start:
// Do not save the previous scenes because screen transition is always available.
pArg->status = changeStatus;
delete pArg->pScene;
// Initialization settings when arriving at the start screen.
pArg->pSceneDataInfo->SetSelectedPositionSalesStatus(0);
pArg->pScene = new StartScene(pArg->pSceneDataInfo);
break;
case SceneBase::DisplayStatus::DisplayStatus_Consumable:
// Remember the previous screen (so you can return if a process is canceled or an error occurs).
pArg->pPreScene = pArg->pScene;
pArg->preStatus = pArg->status;
pArg->status = changeStatus;
pArg->pScene = new ConsumableScene(pArg->pSceneDataInfo);
break;
case SceneBase::DisplayStatus::DisplayStatus_ConsumableItem:
// Remember the previous screen (so you can return if a process is canceled or an error occurs).
pArg->pPreScene = pArg->pScene;
pArg->preStatus = pArg->status;
pArg->status = changeStatus;
pArg->pScene = new ConsumableItemScene(pArg->pSceneDataInfo);
break;
// Display the eShop screen for consumable items.
case SceneBase::DisplayStatus::DisplayStatus_ConsumableItemShop:
{
nn::ec::ConsumableId consumableId;
nn::ec::NsUid nsuid;
const nn::account::UserHandle userHandle =
pArg->pSceneDataInfo->GetUserContext()->userHandle;
nn::util::Strlcpy(
consumableId.value,
pArg->pSceneDataInfo->GetConsumableId().c_str(),
sizeof(nn::ec::ConsumableId));
nsuid.value = pArg->pSceneDataInfo->GetConsumableItemId();
nn::ec::ShowShopConsumableItemDetail(consumableId, nsuid, userHandle);
pArg->status = SceneBase::DisplayStatus::DisplayStatus_ConsumableItem;
pArg->pScene->SetStatus(SceneBase::DisplayStatus::DisplayStatus_None);
break;
}
case SceneBase::DisplayStatus::DisplayStatus_AocSalesStatusSelection:
// Do not save the previous scenes because screen transition is always available.
pArg->status = changeStatus;
delete pArg->pScene;
// Specifies the scene to jump to after selecting sale status.
pArg->pSceneDataInfo->SetSceneThatTransitionFromSalesStatus(SceneBase::DisplayStatus::DisplayStatus_AocItem);
pArg->pScene = new SelectSalesStatusScene(pArg->pSceneDataInfo);
break;
// Display the AOC item list.
case SceneBase::DisplayStatus::DisplayStatus_AocItem:
// Remember the previous screen (so you can return if a process is canceled or an error occurs).
pArg->pPreScene = pArg->pScene;
pArg->preStatus = pArg->status;
pArg->status = changeStatus;
pArg->pScene = new AocItemScene(pArg->pSceneDataInfo);
break;
// Display details about products based on nsuid and the purchase screen.
case SceneBase::DisplayStatus::DisplayStatus_ShowShopProductDetails:
{
const nn::account::UserHandle userHandle =
pArg->pSceneDataInfo->GetUserContext()->userHandle;
nn::ec::NsUid nsuid;
pArg->pSceneDataInfo->GetNsUid(&nsuid);
nn::ec::ShowShopProductDetails(nsuid, userHandle);
pArg->status = SceneBase::DisplayStatus::DisplayStatus_AocItem;
pArg->pScene->SetStatus(SceneBase::DisplayStatus::DisplayStatus_None);
break;
}
// Display the screen with the Aoc item key list. (In this scene, you also select a value corresponding to a key.)
case SceneBase::DisplayStatus::DisplayStatus_AocItemShelf:
// Remember the previous screen (so you can return if a process is canceled or an error occurs).
pArg->pPreScene = pArg->pScene;
pArg->preStatus = pArg->status;
pArg->status = changeStatus;
pArg->pScene = new AocItemShelfScene(pArg->pSceneDataInfo);
break;
// Display the eShop screen for the AOC item shelf list.
case SceneBase::DisplayStatus::DisplayStatus_AocItemShelfShop:
{
const nn::account::UserHandle userHandle =
pArg->pSceneDataInfo->GetUserContext()->userHandle;
nn::ec::NsUid nsuidList[SceneData::NsUidListMax];
int listSize;
std::string listName;
pSceneDataInfo->GetAocShelfList(nsuidList, &listSize, &listName);
nn::ec::ShowShopProductList(nsuidList, listSize, listName.c_str(), userHandle);
pArg->status = SceneBase::DisplayStatus::DisplayStatus_AocItemShelf;
pArg->pScene->SetStatus(SceneBase::DisplayStatus::DisplayStatus_None);
break;
}
// Show Related Title/Bundle List.
case SceneBase::DisplayStatus::DisplayStatus_RelatedTitleSalesStatusSelection:
{
// Do not save the previous scenes because screen transition is always available.
pArg->status = changeStatus;
delete pArg->pScene;
// Specifies the scene to jump to after selecting sale status.
pArg->pSceneDataInfo->SetSceneThatTransitionFromSalesStatus(SceneBase::DisplayStatus::DisplayStatus_RelatedTitleBundle);
pArg->pScene = new SelectSalesStatusScene(pArg->pSceneDataInfo);
break;
}
// Show Related Title/Bundle List.
case SceneBase::DisplayStatus::DisplayStatus_RelatedTitleBundle:
{
// Remember the previous screen (so you can return if a process is canceled or an error occurs).
pArg->pPreScene = pArg->pScene;
pArg->preStatus = pArg->status;
pArg->status = changeStatus;
pArg->pScene = new RelatedTitleBundleScene(pArg->pSceneDataInfo);
break;
}
// Display the Course list for the Subscription service item list.
case SceneBase::DisplayStatus::DisplayStatus_SubscriptionServiceItemCourse:
{
// Remember the previous screen (so you can return if a process is canceled or an error occurs).
pArg->pPreScene = pArg->pScene;
pArg->preStatus = pArg->status;
pArg->status = changeStatus;
pArg->pScene = new SubscriptionServiceItemCourseScene(pArg->pSceneDataInfo);
break;
}
// Display the Subscription service item screen.
case SceneBase::DisplayStatus::DisplayStatus_SubscriptionServiceItem:
{
// Remember the previous screen (so you can return if a process is canceled or an error occurs).
pArg->pPreScene = pArg->pScene;
pArg->preStatus = pArg->status;
pArg->status = changeStatus;
pArg->pScene = new SubscriptionServiceItemScene(pArg->pSceneDataInfo);
break;
}
// Based on CourseId and Nsuid, display details about a subscription service item and the purchase screen.
case SceneBase::DisplayStatus::DisplayStatus_ShowShopSubscriptionDetails:
{
const nn::account::UserHandle userHandle =
pArg->pSceneDataInfo->GetUserContext()->userHandle;
nn::ec::NsUid nsuid;
pSceneDataInfo->GetNsUid(&nsuid);
nn::ec::CourseId courseId;
nn::util::Strlcpy(
courseId.value,
pArg->pSceneDataInfo->GetCourseId().c_str(),
sizeof(nn::ec::CourseId));
nn::ec::ShowShopSubscriptionDetails(courseId, nsuid, userHandle);
pArg->status = SceneBase::DisplayStatus::DisplayStatus_AocItem;
pArg->pScene->SetStatus(SceneBase::DisplayStatus::DisplayStatus_None);
break;
}
// Display the error screen.
case SceneBase::DisplayStatus::DisplayStatus_Error:
// Delete the scene where the error occurred.
delete pArg->pScene;
pArg->status = changeStatus;
pArg->pScene = new ErrorScene(pArg->pSceneDataInfo);
// Cancel key operation in ErrorScene and transition to DisplayStatus_Cancel.
break;
// Return to the previous screen if canceled.
case SceneBase::DisplayStatus::DisplayStatus_Cancel:
delete pArg->pScene;
pArg->pScene = pArg->pPreScene;
pArg->pScene->SetStatus(SceneBase::DisplayStatus::DisplayStatus_None);
pArg->status = pArg->preStatus;
default:
break;
}
pArg->pScene->Draw();
}
// Delete the previous screen when loading of each list has finished and the list can be displayed.
if(pSceneDataInfo->GetLoadCompleted() == true)
{
pSceneDataInfo->SetLoadCompleted(false);
delete pArg->pPreScene;
pArg->pPreScene = nullptr;
}
pArg->pScene->Draw();
EndCommand(pGfw, bufferIndex);
} //NOLINT(impl/function_size)
extern "C" void nnMain()
{
// Network connection settings.
NN_ABORT_UNLESS(nn::nifm::Initialize().IsSuccess());
while (NN_STATIC_CONDITION(1))
{
nn::nifm::SubmitNetworkRequestAndWait();
if (nn::nifm::IsNetworkAvailable())
{
NN_LOG("Network is available.\n");
break;
}
// Handle use requests that result in failure.
if (nn::nifm::HandleNetworkRequestErrorResult())
{
// Resubmitting the use request may result in it being accepted.
NN_LOG("Network is not available. Submit the network request again.\n");
continue;
}
NN_LOG("Network is not available.\n");
nn::nifm::CancelNetworkRequest();
// Because the probability of successfully resubmitting the use request is low,
// this is treated as the network connection failing in this sample, and the application is terminated.
return;
}
// Select a user already associated with a Nintendo Account.
nn::account::Initialize();
nn::account::Uid user;
{
nn::account::UserSelectionSettings setting = nn::account::DefaultUserSelectionSettings;
setting.isNetworkServiceAccountRequired = true;
auto result = nn::account::ShowUserSelector(&user, setting);
if (nn::account::ResultCancelledByUser::Includes(result))
{
NN_LOG("User selection is canceled. A sample program ends.\n");
return;
}
}
// Open the user.
UserContext userContext(user);
// Use when converting PosixTime to calendar.
nn::time::Initialize();
// Initialize ShopServiceAccessors.
NN_ABORT_UNLESS(nn::ec::InitializeForShopServiceAccessors(::g_WorkMemory, sizeof(::g_WorkMemory)).IsSuccess());
SceneData sceneDataInfo(&userContext);
nns::hid::ControllerManager controllerManager;
nns::hid::util::SetControllerManagerWithDefault(&controllerManager);
// Mount the file system.
MountContents();
nns::gfx::GraphicsFramework gfw;
InitializeGraphicsFramework(&gfw);
// Dbgui initialization.
// (Takes around 3 seconds because it includes glyph generation.)
const auto pFontBuffer = InitializeDbgui(&gfw, FrameBufferWidth, FrameBufferHeight, BufferCount);
// Information to pass to the callback function.
::CallbackArgument calculateArg;
calculateArg.pControllerManager = &controllerManager;
calculateArg.pScene = new StartScene(&sceneDataInfo);
calculateArg.pPreScene = nullptr;
calculateArg.pSceneDataInfo = &sceneDataInfo;
calculateArg.status = SceneBase::DisplayStatus_Start;
calculateArg.preStatus = SceneBase::DisplayStatus_Start;
// Register the callback function to use in the drawing loop.
gfw.SetCalculateCallback(&CalculateCallback, &calculateArg);
gfw.SetMakeCommandCallback(&MakeCommandCallback, &calculateArg);
while (NN_STATIC_CONDITION(true))
{
gfw.ProcessFrame();
}
delete calculateArg.pScene;
nns::dbgui::Finalize(gfw.GetDevice());
gfw.FreeMemory(pFontBuffer);
FinalizeGraphicsFramework(&gfw);
UnmountContents();
NN_ABORT_UNLESS(nn::ec::FinalizeForShopServiceAccessors().IsSuccess());
}
