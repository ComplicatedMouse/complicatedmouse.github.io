void GraphicsRenderState::Initialize() NN_NOEXCEPT
{
m_BlendState.SetDefaults();
m_ChannelMaskState.SetDefaults();
m_DepthStencilState.SetDefaults();
m_ColorState.SetDefaults();
m_MultisampleState.SetDefaults();
m_PolygonState.SetDefaults();
}
void GraphicsRenderState::Bind(nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
pCommandBuffer->BindBlendState(&m_BlendState);
pCommandBuffer->BindChannelMaskState(&m_ChannelMaskState);
pCommandBuffer->BindColorState(&m_ColorState);
pCommandBuffer->BindDepthStencilState(&m_DepthStencilState);
pCommandBuffer->BindMultisampleState(&m_MultisampleState);
pCommandBuffer->BindPolygonState(&m_PolygonState);
}
nvn::BlendState* GraphicsRenderState::GetBlendState() NN_NOEXCEPT
{
return &m_BlendState;
}
nvn::ChannelMaskState* GraphicsRenderState::GetChannelMaskState() NN_NOEXCEPT
{
return &m_ChannelMaskState;
}
nvn::DepthStencilState* GraphicsRenderState::GetDepthStencilState() NN_NOEXCEPT

return &m_DepthStencilState;
}
nvn::ColorState* GraphicsRenderState::GetColorState() NN_NOEXCEPT
{
return &m_ColorState;
}
nvn::MultisampleState* GraphicsRenderState::GetMultisampleState() NN_NOEXCEPT
{
return &m_MultisampleState;
}
nvn::PolygonState* GraphicsRenderState::GetPolygonState() NN_NOEXCEPT
{
return &m_PolygonState;
}
