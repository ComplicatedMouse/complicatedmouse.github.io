// Echo the received data.
void Echo() NN_NOEXCEPT
{
nn::websocket::WebSocket webSocket;
nn::Result result = webSocket.Initialize();
if (result.IsFailure())
{
NN_LOG("nn::websocket::WebSocket::Initialize() failed. e = 0x%08X\n", result.GetInnerValueForDebug());
return;
}
NN_UTIL_SCOPE_EXIT
{
webSocket.Finalize();
};
webSocket.SetHandshakeCompletionEventHandler(HandshakeCompletionEventHandler, nullptr);
webSocket.SetOpenEventHandler(OpenEventHandler, nullptr);
webSocket.SetMessageEventHandler(MessageEventHandler, nullptr);
webSocket.SetCloseEventHandler(CloseEventHandler, nullptr);
webSocket.SetErrorEventHandler(ErrorEventHandler, nullptr);
#if defined(USE_HTCS)
nn::websocket::HtcsConnection connection("@WebSocketEcho");
#else
nn::nifm::SubmitNetworkRequestAndWait();
if (!nn::nifm::IsNetworkAvailable())
{
NN_LOG("Network connection could not be established.\n");
return;
}
NN_UTIL_SCOPE_EXIT
{
nn::nifm::CancelNetworkRequest();
};
nn::websocket::Connection connection;
#endif
webSocket.SetConnection(&connection);
// Connect to the server.
// Block until the connection process is complete.
result = webSocket.Open(Uri);
if (result.IsFailure())
{
NN_LOG("nn::websocket::WebSocket::Open() failed. e = 0x%08X\n", result.GetInnerValueForDebug());
return;
}
nn::os::TimerEvent timer(nn::os::EventClearMode_AutoClear);
int count = 0;
timer.StartPeriodic(0, nn::TimeSpan::FromSeconds(1));
while (webSocket.IsAvailable())
{
webSocket.Perform();
// Send "Hello" once per second. (Stop after sending five times.)
if (timer.TryWait())
{
if (count < 5)
{
char message[16] = {};
nn::util::SNPrintf(message, sizeof (message), "Hello(%d)", count + 1);
result = webSocket.SendText(message);
if (result.IsFailure())
{
NN_LOG("nn::websocket::WebSocket::SendText() failed. e = 0x%08X\n", result.GetInnerValueForDebug());
}
}
else if (count == 5)
{
// After adding the close request to the send queue, nn::websocket::WebSocket::Perform() must continue to be called until the request is actually received by the server.
// For this reason, never end this loop immediately.
result = webSocket.Close();
if (result.IsFailure())
{
NN_LOG("nn::websocket::WebSocket::Close() failed. e = 0x%08X\n", result.GetInnerValueForDebug());
}
}
count++;
}
// Sleep here to avoid a busy loop.
// The lower the nn::websocket::WebSocket::Perform() call frequency, the less responsive the communication process becomes.
// Adjust the call frequency based on the application's processing load and communication load.
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(10));
}
}
// Handle events when the handshake is completed.
bool HandshakeCompletionEventHandler(const nn::websocket::HeaderList& headers, const char*, void*) NN_NOEXCEPT
{
NN_LOG("Response Headers:\n");
for (auto&& pHeader : headers)
{
// The header format is name: value.
// There is one space after the colon, regardless of the actual server response. (Ensured by the library.)
NN_LOG(" - %s\n", pHeader);
// Include the colon in the search when searching for headers.
// We recommend using case-insensitive header names. (See RFC7230#section-3.2.)
if (nn::util::Strnicmp(pHeader, "Set-Cookie:", sizeof ("Set-Cookie:") - 1) == 0)
{
NN_LOG(" - Cookie => '%s'\n", pHeader + sizeof ("Set-Cookie:"));
}
}
// If you do not care about the server response, you can return false to drop the connection.
return true;
}
// Handle server connection events.

NN_LOG("Open:\n");
}
// Handle message receipt events.
void MessageEventHandler(void* pBuffer, size_t size, bool isText, void*) NN_NOEXCEPT
{
if (isText)
{
NN_LOG("TxtFrame: %.*s\n", static_cast<int>(size), reinterpret_cast<const char*>(pBuffer));
}
else
{
NN_LOG("BinFrame: size = %zu\n", size);
}
}
// Handle server disconnection events.
void CloseEventHandler(uint16_t code, const char* pReason, size_t length, void*) NN_NOEXCEPT
{
NN_LOG("Close: code = %u, reason = %.*s\n", code, static_cast<int>(length), pReason);
}
// Handle error events.
void ErrorEventHandler(nn::Result result, void*) NN_NOEXCEPT
{
NN_LOG("Error: 0x%08X\n", result.GetInnerValueForDebug());
}
