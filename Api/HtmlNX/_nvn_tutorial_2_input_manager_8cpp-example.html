/*
* InputManager::GetButtonState
* ----------------------------
* Helper function to compare the current and old button
* states and returns the type of the input.
*/
InputManager::ButtonState InputManager::GetButtonState(bool oldClickState, bool currentClickState)
{
ButtonState buttonState = ButtonState_Not_Pressed;
if (!currentClickState && oldClickState)
{
buttonState = ButtonState_Released;
}
else if (!currentClickState && !oldClickState)
{
buttonState = ButtonState_Not_Pressed;
}
else if (currentClickState && !oldClickState)
{
buttonState = ButtonState_Triggered;
}
else if (currentClickState && oldClickState)
{
buttonState = ButtonState_Pressed;
}
return buttonState;
}
