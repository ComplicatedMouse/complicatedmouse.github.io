void WaitHostCompleted(int sock)
{
nn::htcs::FdSet fd;
nn::htcs::FdSetZero(&fd);
nn::htcs::FdSetSet(sock, &fd);
NN_ABORT_UNLESS(nn::htcs::Select(sock + 1, &fd, nullptr, nullptr, nullptr) > 0, "Server: Select failed (%d)\n", nn::htcs::GetLastError());
if (nn::htcs::FdSetIsSet(sock, &fd))
{
char buf;
NN_ABORT_UNLESS(nn::htcs::Recv(sock, &buf, sizeof(buf), 0) == 0);
}
else
{
NN_ABORT();
}
}
// The sent data is sent back to the peer as is.
void EchoBack(int sock)
{
char buf[128] = {};
nn::htcs::ssize_t receivedBytes = nn::htcs::Recv(sock, buf, sizeof(buf), 0);
if (receivedBytes < 0)
{
NN_LOG("Receive error. %d\n", nn::htcs::GetLastError());
return;
}
else if (receivedBytes == 0)
{
NN_LOG("Gracefully closed.\n");
return;
}
NN_LOG("Received \"%s\".\n", buf);
nn::htcs::ssize_t sentBytes = nn::htcs::Send(sock, buf, receivedBytes, 0);
if (sentBytes < 0)
{
NN_LOG("Send error. %d\n", nn::htcs::GetLastError());
}
NN_ABORT_UNLESS(nn::htcs::Send(sock, "\n", 1, 0) != -1, "Server: Send failed (%d)\n", nn::htcs::GetLastError());
WaitHostCompleted(sock);
}
void NonBlockingEchoBack(int sock)
{
nn::htcs::ssize_t receivedBytes = 0;
char buf[128] = {};
nn::htcs::FdSet fd;
nn::htcs::TimeVal time;
while (true)
{
time.tv_sec = 0;
time.tv_usec = 0;
nn::htcs::FdSetZero(&fd);
nn::htcs::FdSetSet(sock, &fd);
int result = nn::htcs::Select(sock + 1, &fd, nullptr, nullptr, &time);
if (result > 0)
{
break;
}
else if (result < 0)
{
NN_LOG("Select error (%d)\n", nn::htcs::GetLastError());
return;
}
NN_LOG("Wait for the socket to be ready...\n");
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1000));
}
if (nn::htcs::FdSetIsSet(sock, &fd))
{
receivedBytes = nn::htcs::Recv(sock, buf, sizeof(buf) - 1, 0);
if (receivedBytes < 0)
{
NN_LOG("Receive error (%d)\n", nn::htcs::GetLastError());
return;
}
else if (receivedBytes == 0)
{
NN_LOG("Gracefully closed.\n");
return;
}
}
else
{
NN_ABORT();
}
NN_LOG("Received \"%s\".\n", buf);
nn::htcs::ssize_t sentBytes = 0;
time.tv_sec = 0;
time.tv_usec = 10;
nn::htcs::FdSetZero(&fd);
nn::htcs::FdSetSet(sock, &fd);
NN_ABORT_UNLESS(nn::htcs::Select(sock + 1, nullptr, &fd, nullptr, &time) != -1, "Server: Select failed (%d)\n", nn::htcs::GetLastError());
if (nn::htcs::FdSetIsSet(sock, &fd))
{
sentBytes = nn::htcs::Send(sock, buf, receivedBytes, 0);
if (sentBytes < 0)
{
NN_LOG("Send error (%d)\n", nn::htcs::GetLastError());
}
NN_ABORT_UNLESS(nn::htcs::Send(sock, "\n", 1, 0) != -1, "Server: Send failed (%d)\n", nn::htcs::GetLastError());
}
else
{
NN_ABORT();
}
WaitHostCompleted(sock);
}
void ServerSample()
{
// Register the service.
nn::htcs::SockAddrHtcs serviceAddr;
serviceAddr.family = nn::htcs::HTCS_AF_HTCS;
serviceAddr.peerName = nn::htcs::GetPeerNameAny();
std::strcpy(serviceAddr.portName.name, "ServerInTarget");
int serviceSocket = nn::htcs::Socket();
NN_ABORT_UNLESS(nn::htcs::Bind(serviceSocket, &serviceAddr) != -1, "Server: Bind failed (%d)\n", nn::htcs::GetLastError());
NN_ABORT_UNLESS(nn::htcs::Listen(serviceSocket, 1) != -1, "Server: Listen failed (%d)\n", nn::htcs::GetLastError());
// Wait for the connection.
NN_LOG("Waiting for the connection from the host.\n");
int sock = nn::htcs::Accept(serviceSocket, nullptr);
NN_ABORT_UNLESS(sock >= 0, "Server: Accept failed. Error(%d).\n", nn::htcs::GetLastError());
NN_LOG("Accepted the client on the host.\n");
// Echo back test.
EchoBack(sock);
// Disconnect.
nn::htcs::Close(sock);
nn::htcs::Close(serviceSocket);
NN_LOG("Closed.\n");
}
void ClientSample()
{
// Create the server address information for the host you want to connect to.
nn::htcs::SockAddrHtcs addr;
addr.family = nn::htcs::HTCS_AF_HTCS;
addr.peerName = nn::htcs::GetDefaultHostName();
std::strcpy(addr.portName.name, "ServerInHost");
// Try to connect until a connection is made.
int sock = nn::htcs::Socket();
while (NN_STATIC_CONDITION(true))
{
if (nn::htcs::Connect(sock, &addr) == 0)
{
NN_LOG("Connected to the server on the host.\n");
break;
}
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1000));
}
// Echo back test.
EchoBack(sock);
// Disconnect.
nn::htcs::Close(sock);
NN_LOG("Closed.\n");
}
void NonBlockingServerSample()
{
// Register the service.
nn::htcs::SockAddrHtcs address;
address.family = nn::htcs::HTCS_AF_HTCS;
address.peerName = nn::htcs::GetPeerNameAny();
std::strcpy(address.portName.name, "ServerInTarget");
nn::htcs::FdSet fd;
nn::htcs::FdSetZero(&fd);
int listenSocket = nn::htcs::Socket();
NN_ABORT_UNLESS(nn::htcs::Fcntl(listenSocket, nn::htcs::HTCS_F_SETFL, nn::htcs::HTCS_O_NONBLOCK) != -1, "Server: Fcntl failed (%d)\n", nn::htcs::GetLastError());
NN_ABORT_UNLESS(nn::htcs::Bind(listenSocket, &address) != -1, "Server: Bind failed (%d)\n", nn::htcs::GetLastError());
NN_ABORT_UNLESS(nn::htcs::Listen(listenSocket, 1) != -1, "Server: Listen failed (%d)\n", nn::htcs::GetLastError());
NN_LOG("Waiting for the connection from the host.\n");
nn::htcs::FdSetSet(listenSocket, &fd);
NN_ABORT_UNLESS(nn::htcs::Select(listenSocket + 1, &fd, nullptr, nullptr, nullptr) != -1, "Server: Select failed (%d)\n", nn::htcs::GetLastError());
if (nn::htcs::FdSetIsSet(listenSocket, &fd))
{
int socket = nn::htcs::Accept(listenSocket, nullptr);
if (socket < 0)
{
NN_LOG("L%d LastError = %d\n", __LINE__, nn::htcs::GetLastError());
}
NonBlockingEchoBack(socket);
nn::htcs::Close(socket);
}
nn::htcs::Close(listenSocket);
NN_LOG("Closed.\n");
}
void NonBlockingClientSample()
{
// Create the server address information for the host you want to connect to.
nn::htcs::SockAddrHtcs addr;
addr.family = nn::htcs::HTCS_AF_HTCS;
addr.peerName = nn::htcs::GetDefaultHostName();
std::strcpy(addr.portName.name, "ServerInHost");
int sock = nn::htcs::Socket();
nn::htcs::Fcntl(sock, nn::htcs::HTCS_F_SETFL, nn::htcs::HTCS_O_NONBLOCK);
while (true)
{
NN_ABORT_UNLESS(nn::htcs::Connect(sock, &addr) < 0, "Server: Connect unexpected success\n");
if (nn::htcs::GetLastError() == nn::htcs::HTCS_EINPROGRESS
|| nn::htcs::GetLastError() == nn::htcs::HTCS_EISCONN)
{
break;
}
else if (nn::htcs::GetLastError() == nn::htcs::HTCS_EADDRNOTAVAIL
|| nn::htcs::GetLastError() == nn::htcs::HTCS_EAGAIN)
{
NN_LOG("Waiting for the host address to become valid.\n");
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1000));
continue;
}
else
{
NN_LOG("Unexpected connection error (%d)\n", nn::htcs::GetLastError());
return;
}
}
// Echo back test.
NonBlockingEchoBack(sock);
// Disconnect.
nn::htcs::Close(sock);
NN_LOG("Closed.\n");
}
extern "C" void nnMain()
{
#if defined(NN_BUILD_TARGET_PLATFORM_OS_WIN)
// For Windows applications, you must call nn::socket::Initialize() to use htc.
static nn::socket::ConfigDefaultWithMemory s_SocketConfig;
nn::socket::Initialize(s_SocketConfig);
#endif
nn::htcs::Initialize(&Allocate, &Deallocate);
ServerSample();
ClientSample();
NonBlockingServerSample();
NonBlockingClientSample();
// Finalization.
nn::htcs::Finalize();
#if defined(NN_BUILD_TARGET_PLATFORM_OS_WIN)
nn::socket::Finalize();
#endif
NN_LOG("Sample end.\n");
while (NN_STATIC_CONDITION(true))
{
nn::os::SleepThread(nn::TimeSpan::FromMilliSeconds(1000));
}
return;
}
