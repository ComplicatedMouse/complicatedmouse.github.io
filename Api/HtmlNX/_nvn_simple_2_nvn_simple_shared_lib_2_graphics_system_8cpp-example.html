void GraphicsSystem::Initialize(nn::mem::StandardAllocator* pAllocator, const GraphicsSystemParams& params) NN_NOEXCEPT
{
if (!m_IsInitialized)
{
m_pAllocator = pAllocator;
InitializeGraphicsContextAndNvnDevice(&m_Device, &m_NativeWindow);
// initialize nvn queue
int minCommandMemSize = 0;
m_Device.GetInteger(nvn::DeviceInfo::QUEUE_COMMAND_MEMORY_MIN_SIZE, &minCommandMemSize);
nvn::QueueBuilder queueBuilder;
queueBuilder.SetDefaults()
.SetDevice(&m_Device)
.SetComputeMemorySize(0)
.SetCommandMemorySize(minCommandMemSize)
.SetCommandFlushThreshold(minCommandMemSize);
size_t queueMemSize = queueBuilder.GetQueueMemorySize();
if ((queueMemSize % NVN_MEMORY_POOL_STORAGE_GRANULARITY) != 0)
{
NN_ASSERT(0, "Memory size reported for queue is not the proper granularity");
}
m_QueueMemory = m_pAllocator->Allocate(queueMemSize, NVN_MEMORY_POOL_STORAGE_ALIGNMENT);
queueBuilder.SetQueueMemory(m_QueueMemory, queueMemSize);
if (!m_Queue.Initialize(&queueBuilder))
{
NN_ASSERT(false, "nvn::Queue::Initialize");
}
// initialize shared memory pool for shaders.
// CPU_UNCACHED is used so that we can shader code can be loaded from the CPU during runtime without having to flush.
m_ShaderMemoryPool.Initialize(this,
nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_CACHED | nvn::MemoryPoolFlags::SHADER_CODE,
params.GetShaderMemoryPoolSize(), "Shader MemPool");
// obtain resource size/limit/alignment data and configure global command/control memory size
m_Device.GetInteger(nvn::DeviceInfo::COMMAND_BUFFER_COMMAND_ALIGNMENT, &m_ResourceInfo.commandMemAlignment);
m_Device.GetInteger(nvn::DeviceInfo::COMMAND_BUFFER_CONTROL_ALIGNMENT, &m_ResourceInfo.controlMemAlignment);
m_Device.GetInteger(nvn::DeviceInfo::UNIFORM_BUFFER_ALIGNMENT, &m_ResourceInfo.uniformBufferAlignment);
m_Device.GetInteger(nvn::DeviceInfo::TEXTURE_DESCRIPTOR_SIZE, &m_ResourceInfo.textureDescriptorSize);
m_Device.GetInteger(nvn::DeviceInfo::RESERVED_TEXTURE_DESCRIPTORS, &m_ResourceInfo.numReservedTextureDescriptors);
m_Device.GetInteger(nvn::DeviceInfo::MAX_TEXTURE_POOL_SIZE, &m_ResourceInfo.maxTextureDescriptorsInPool);
m_Device.GetInteger(nvn::DeviceInfo::SAMPLER_DESCRIPTOR_SIZE, &m_ResourceInfo.samplerDescriptorSize);
m_Device.GetInteger(nvn::DeviceInfo::RESERVED_SAMPLER_DESCRIPTORS, &m_ResourceInfo.numReservedSamplerDescriptors);
m_Device.GetInteger(nvn::DeviceInfo::MAX_SAMPLER_POOL_SIZE, &m_ResourceInfo.maxSamplerDescriptorsInPool);
m_Device.GetInteger(nvn::DeviceInfo::LINEAR_TEXTURE_STRIDE_ALIGNMENT, &m_ResourceInfo.linearTextureStrideAlignment);
// initialize shared memory pool used for command buffers, and then initialize command buffer used for render target setting
m_CommandBufferMemoryPool.Initialize(this,
nvn::MemoryPoolFlags::CPU_UNCACHED | nvn::MemoryPoolFlags::GPU_UNCACHED,
params.GetCommandBufferMemoryPoolSize(), "Command Buffer MemPool");
m_CommandBufferForRenderTarget.Initialize(this);
// initialize timestamp memory pool used for timestamps, and then initialize command buffers used for writing timestamps.
size_t timestampMemoryPoolSize = 4 * sizeof(uint64_t);
m_TimestampBufferMemoryPool.Initialize(this, nvn::MemoryPoolFlags::CPU_CACHED | nvn::MemoryPoolFlags::GPU_CACHED, timestampMemoryPoolSize, "Timestamp MemPool");
m_CommandBufferForMeasureGpuBegin.Initialize(this);
m_CommandBufferForMeasureGpuEnd.Initialize(this);
m_TimestampBufferMemoryPool.AllocateBuffer(&m_TimestampBuffer, timestampMemoryPoolSize);
// initialize sync objects
if (!m_DisplayReleaseSync.Initialize(&m_Device))
{
NN_ASSERT(0, "Sync object failed to initialize.\n");
}
if (!m_GpuDoneSync.Initialize(&m_Device))
{
NN_ASSERT(0, "Sync object failed to initialize.\n");
}
// set render target and screen resolutions
m_RenderTargetWidth = params.GetRenderTargetWidth();
m_RenderTargetHeight = params.GetRenderTargetHeight();
m_ScreenWidth = m_RenderTargetWidth;
m_ScreenHeight = m_RenderTargetHeight;
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
if (m_RenderTargetWidth == 1920 && m_RenderTargetHeight == 1080)
{
if (nn::oe::GetOperationMode() == nn::oe::OperationMode_Handheld)
{
m_ScreenWidth = 1280;
m_ScreenHeight = 720;
}
}
#endif
// initialize render target buffers and depth/stencil buffer after creating a memory pool for them
nvn::TextureBuilder renderTargetBuilder;
renderTargetBuilder.SetDefaults()
.SetDevice(&m_Device)
.SetSize2D(m_RenderTargetWidth, m_RenderTargetHeight)
.SetTarget(nvn::TextureTarget::TARGET_2D)
.SetFlags(nvn::TextureFlags::DISPLAY | nvn::TextureFlags::COMPRESSIBLE)
.SetFormat(nvn::Format::RGBA8);
nvn::TextureBuilder depthStencilBuilder;
depthStencilBuilder.SetDefaults()
.SetDevice(&m_Device)
.SetSize2D(m_RenderTargetWidth, m_RenderTargetHeight)
.SetTarget(nvn::TextureTarget::TARGET_2D)
.SetFlags(nvn::TextureFlags::COMPRESSIBLE)
.SetFormat(nvn::Format::DEPTH24_STENCIL8);
size_t renderTargetSize = renderTargetBuilder.GetStorageSize();
size_t renderTargetAlignment = renderTargetBuilder.GetStorageAlignment();
size_t depthStencilSize = depthStencilBuilder.GetStorageSize();
size_t depthStencilAlignment = depthStencilBuilder.GetStorageAlignment();
m_RenderTargetMemoryPool.Initialize(
this,
nvn::MemoryPoolFlags::CPU_NO_ACCESS | nvn::MemoryPoolFlags::GPU_CACHED | nvn::MemoryPoolFlags::COMPRESSIBLE,
renderTargetSize * NumRenderTargets + depthStencilSize, "Display Buffer MemPool"
);
for (int i = 0; i < NumRenderTargets; ++i)
{
ptrdiff_t renderTargetOffset = m_RenderTargetMemoryPool.Allocate(renderTargetSize, renderTargetAlignment);
renderTargetBuilder.SetStorage(m_RenderTargetMemoryPool.GetMemoryPool(), renderTargetOffset);
m_RenderTargets[i] = new nvn::Texture;
if (!m_RenderTargets[i]->Initialize(&renderTargetBuilder))
{
NN_ASSERT(0, "nvnTextureInitialize failed for render target.\n");
}
}
ptrdiff_t depthStencilOffset = m_RenderTargetMemoryPool.Allocate(depthStencilSize, depthStencilAlignment);
depthStencilBuilder.SetStorage(m_RenderTargetMemoryPool.GetMemoryPool(), depthStencilOffset);
if (!m_DepthStencilBuffer.Initialize(&depthStencilBuilder))
{
NN_ASSERT(0, "nvnTextureInitialize failed for depth/stencil buffer.\n");
}
// initialize window
nvn::WindowBuilder windowBuilder;
windowBuilder.SetDefaults()
.SetDevice(&m_Device)
.SetNativeWindow(m_NativeWindow)
.SetTextures(NumRenderTargets, m_RenderTargets);
if (!m_Window.Initialize(&windowBuilder))
{
NN_ASSERT(0, "nvnWindowInitialize failed.\n");
}
m_Window.SetCrop(0, 0, m_ScreenWidth, m_ScreenHeight);
m_CurRenderTargetIndex = 0;
m_FrameCount = 0;
m_GpuDrawTime = 0;
m_IsInitialized = true;
}
}//NOLINT(impl/function_size)
void GraphicsSystem::DrawBegin() NN_NOEXCEPT
{
// the render loop framework implemented by DrawBegin() and DrawEnd() is copied from the Immediate execution
// style as described in [NintendoSDK Documents > Features > Graphics > Graphics (NX Add-On) > Rendering Framework].
if (m_FrameCount == 0)
{
nvn::WindowAcquireTextureResult result = m_Window.AcquireTexture(&m_DisplayReleaseSync, &m_CurRenderTargetIndex);
if (result != nvn::WindowAcquireTextureResult::SUCCESS)
{
NN_ASSERT(0, "nvn::Window::AcquireTexture failed.");
}
m_DisplayReleaseSync.Wait(NVN_WAIT_TIMEOUT_MAXIMUM);
}
// record GPU commands for setting render target and submit them
ResetRenderTarget();
// record and execute GPU commands for writing timestamp
nvn::BufferAddress timestampBufferAddress = m_TimestampBuffer.GetAddress();
nvn::CommandBuffer* pCommandBuffer = m_CommandBufferForMeasureGpuBegin.StartRecording();
pCommandBuffer->ReportCounter(nvn::CounterType::TIMESTAMP, timestampBufferAddress);
pCommandBuffer->Barrier(nvn::BarrierBits::ORDER_PRIMITIVES);
m_CommandBufferForMeasureGpuBegin.EndRecordingAndExecute();
m_Queue.Flush();
// applications are free to record and submit commands via other command lists until DrawEnd is subsequently called
}
void GraphicsSystem::DrawEnd() NN_NOEXCEPT
{
// record and execute GPU commands for writing timestamp
nvn::BufferAddress timestampBufferAddress = m_TimestampBuffer.GetAddress() + sizeof(uint64_t) * 2;
nvn::CommandBuffer* pCommandBuffer = m_CommandBufferForMeasureGpuEnd.StartRecording();
pCommandBuffer->ReportCounter(nvn::CounterType::TIMESTAMP, timestampBufferAddress);
pCommandBuffer->Barrier(nvn::BarrierBits::ORDER_PRIMITIVES);
m_CommandBufferForMeasureGpuEnd.EndRecordingAndExecute();
m_Queue.Flush();
// tell the GPU to signal the sync object when it's done executing all the current GPU commands
m_Queue.FenceSync(&m_GpuDoneSync, nvn::SyncCondition::ALL_GPU_COMMANDS_COMPLETE, 0);
// submit GPU command to display the current render texture
m_Queue.PresentTexture(&m_Window, m_CurRenderTargetIndex); // implicitly calls queue.Flush
// obtain texture for next frame
nvn::WindowAcquireTextureResult result = m_Window.AcquireTexture(&m_DisplayReleaseSync, &m_CurRenderTargetIndex);
if (result != nvn::WindowAcquireTextureResult::SUCCESS)
{
NN_ASSERT(0, "nvn::Window::AcquireTexture failed.");
}
// wait until the render target is released by the display (essentially waiting for vsync)
m_DisplayReleaseSync.Wait(NVN_WAIT_TIMEOUT_MAXIMUM);
// wait for the GPU commands to complete before moving onto the next frame
m_GpuDoneSync.Wait(NVN_WAIT_TIMEOUT_MAXIMUM);
++m_FrameCount;
// read timestamp value from buffer
uint64_t* pData = static_cast<uint64_t*>(m_TimestampBuffer.Map());
m_TimestampBuffer.InvalidateMappedRange(0, sizeof(uint64_t) * 4);
uint64_t timestampBegin = pData[1];
uint64_t timestampEnd = pData[3];
#if defined(NN_BUILD_TARGET_PLATFORM_NX)
// On the NX platform, the timestamp is reported in units of 625 / 384 nanoseconds since the NX device was initialized
m_GpuDrawTime = nn::TimeSpan::FromNanoSeconds((timestampEnd - timestampBegin) * 625 / 384);
#else
m_GpuDrawTime = nn::TimeSpan::FromNanoSeconds(timestampEnd - timestampBegin);
#endif
m_IsScreenResolutionChanged = false;
}
void GraphicsSystem::ResetRenderTarget() NN_NOEXCEPT
{
// generate command buffer for setting our target texture to be renderer into and execute it on the GPU
nvn::CommandBuffer* pCommandBuffer = m_CommandBufferForRenderTarget.StartRecording();
{
pCommandBuffer->SetRenderTargets(1, &m_RenderTargets[m_CurRenderTargetIndex], nullptr, &m_DepthStencilBuffer, nullptr);
}
m_CommandBufferForRenderTarget.EndRecordingAndExecute();
}
void GraphicsSystem::SetDefaultViewportAndScissor(nvn::CommandBuffer* pCommandBuffer) NN_NOEXCEPT
{
pCommandBuffer->SetScissor(0, 0, m_ScreenWidth, m_ScreenHeight);
pCommandBuffer->SetViewport(0, 0, m_ScreenWidth, m_ScreenHeight);
}
void GraphicsSystem::SetScreenResolution(int screenWidth, int screenHeight) NN_NOEXCEPT
{
if (m_ScreenWidth != screenWidth || m_ScreenHeight != screenHeight)
{
m_ScreenWidth = screenWidth;
m_ScreenHeight = screenHeight;
m_IsScreenResolutionChanged = true;
m_Window.SetCrop(0, 0, m_ScreenWidth, m_ScreenHeight);
}
}
void GraphicsSystem::Finalize() NN_NOEXCEPT
{
if (m_IsInitialized)
{
m_Queue.Finish();
m_DisplayReleaseSync.Finalize();
m_GpuDoneSync.Finalize();
m_Window.Finalize();
m_NativeWindow = nullptr;
for (int i = 0; i < NumRenderTargets; ++i)
{
m_RenderTargets[i]->Finalize();
delete m_RenderTargets[i];
m_RenderTargets[i] = nullptr;
}
m_DepthStencilBuffer.Finalize();
m_RenderTargetMemoryPool.Finalize();
m_TimestampBuffer.Finalize();
m_CommandBufferForRenderTarget.Finalize();
m_CommandBufferForMeasureGpuBegin.Finalize();
m_CommandBufferForMeasureGpuEnd.Finalize();
m_CommandBufferMemoryPool.Finalize();
m_TimestampBufferMemoryPool.Finalize();
m_ShaderMemoryPool.Finalize();
m_Queue.Finalize();
m_pAllocator->Free(m_QueueMemory);
m_QueueMemory = nullptr;
m_Device.Finalize();
m_pAllocator = nullptr;
m_GpuDrawTime = 0;
m_CurRenderTargetIndex = 0;
m_IsInitialized = false;
}
