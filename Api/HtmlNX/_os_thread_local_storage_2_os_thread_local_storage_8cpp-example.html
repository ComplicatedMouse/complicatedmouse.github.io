#include <stdint.h>
#include <nn/nn_Macro.h>
#include <nn/nn_Log.h>
#include <nn/nn_Abort.h>
#include <nn/os.h>
//-----------------------------------------------------------------------------
namespace {
const size_t ThreadStackSize = 8192; // Stack size of the thread operation thread.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack1[ ThreadStackSize ]; // The first thread stack.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack2[ ThreadStackSize ]; // The second thread stack.
NN_OS_ALIGNAS_THREAD_STACK char g_ThreadStack3[ ThreadStackSize ]; // The third thread stack.
nn::os::TlsSlot g_tlsSlot;
} // Namespace.
//-----------------------------------------------------------------------------
//
// Access to thread local storage.
//
void CommonFunction()
{
// Get the TLS parameters.
auto name = reinterpret_cast<char*>( nn::os::GetTlsValue( g_tlsSlot ) );
// The obtained TLS values are different for each thread.
NN_LOG("%s: Succeeded in getting a TLS value.\n", name);
}
//
// Thread function (shared implementation).
//
void ThreadSample(void *arg)
{
auto name = reinterpret_cast<char*>( arg );
// Set the arg parameter in TLS.
nn::os::SetTlsValue( g_tlsSlot, reinterpret_cast<uintptr_t>( arg ) );
NN_LOG("%s: Succeeded in setting a TLS value.\n", name );
// Call the common function (all threads perform the same process).
CommonFunction();
// Display the message before the thread ends.
NN_LOG("%s: End of thread.\n", name );
}
//
// TLS destructor function.
//
void TlsDestructFunction(uintptr_t value)
{
char* name = reinterpret_cast<char*>( value );
NN_LOG("%s: TLS destructor function is called back.\n", name);
}
//
// The main function.
//
extern "C" void nnMain()
{
nn::Result result;
nn::os::ThreadType thread1;
nn::os::ThreadType thread2;
nn::os::ThreadType thread3;
// Allocate one TLS slot.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::AllocateTlsSlot( &g_tlsSlot, TlsDestructFunction ) );
// Generate threads.
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread1, ThreadSample, const_cast<char*>("Thread1"), g_ThreadStack1, ThreadStackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread2, ThreadSample, const_cast<char*>("Thread2"), g_ThreadStack2, ThreadStackSize, nn::os::DefaultThreadPriority ) );
NN_ABORT_UNLESS_RESULT_SUCCESS( nn::os::CreateThread( &thread3, ThreadSample, const_cast<char*>("Thread3"), g_ThreadStack3, ThreadStackSize, nn::os::DefaultThreadPriority ) );
// Start thread execution.
nn::os::StartThread( &thread1 );
nn::os::StartThread( &thread2 );
nn::os::StartThread( &thread3 );
// Wait for the thread to end.
nn::os::WaitThread( &thread1 );
nn::os::WaitThread( &thread2 );
nn::os::WaitThread( &thread3 );
// Free threads.
nn::os::DestroyThread( &thread1 );
nn::os::DestroyThread( &thread2 );
nn::os::DestroyThread( &thread3 );
// Release the TLS slot.
nn::os::FreeTlsSlot( g_tlsSlot );
}
